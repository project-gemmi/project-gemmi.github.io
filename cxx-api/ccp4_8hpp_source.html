34struct Ccp4Base {
35 DataStats hstats; // data statistics read from / written to ccp4 map
36 // stores raw headers if the grid was read from ccp4 map
37 std::vector<int32_t> ccp4_header;
38 bool same_byte_order = true;
39
40 // methods to access info from ccp4 headers, w is word number from the spec
41 void* header_word(int w) { return &ccp4_header.at(w - 1); }
42 const void* header_word(int w) const { return &ccp4_header.at(w - 1); }
43
44 int32_t header_i32(int w) const {
45 int32_t value = ccp4_header.at(w - 1);
46 if (!same_byte_order)
47 swap_four_bytes(&value);
48 return value;
49 }
50 std::array<int, 3> header_3i32(int w) const {
51 return {{ header_i32(w), header_i32(w+1), header_i32(w+2) }};
52 }
53 float header_float(int w) const {
54 int32_t int_value = header_i32(w);
55 float f;
56 std::memcpy(&f, &int_value, 4);
57 return f;
58 }
59 // ccp4 map header has mostly 80-byte strings
60 std::string header_str(int w, size_t len=80) const {
61 if (4 * ccp4_header.size() < 4 * (w - 1) + len)
62 fail("invalid end of string");
63 return std::string(reinterpret_cast<const char*>(header_word(w)), len);
64 }
65 void set_header_i32(int w, int32_t value) {
66 if (!same_byte_order)
67 swap_four_bytes(&value);
68 ccp4_header.at(w - 1) = value;
69 };
70 void set_header_3i32(int w, int32_t x, int32_t y, int32_t z) {
71 set_header_i32(w, x);
72 set_header_i32(w+1, y);
73 set_header_i32(w+2, z);
74 };
75 void set_header_float(int w, float value) {
76 int32_t int32_value;
77 std::memcpy(&int32_value, &value, 4);
78 set_header_i32(w, int32_value);
79 };
80 void set_header_str(int w, const std::string& str) {
81 std::memcpy(header_word(w), str.c_str(), str.size());
82 }
83
84 std::array<int, 3> axis_positions() const {
85 if (ccp4_header.empty())
86 return {{0, 1, 2}}; // assuming it's X,Y,Z
87 std::array<int, 3> pos{{-1, -1, -1}};
88 for (int i = 0; i != 3; ++i) {
89 int mapi = header_i32(17 + i);
90 if (mapi <= 0 || mapi > 3 || pos[mapi - 1] != -1)
91 fail("Incorrect MAPC/MAPR/MAPS records");
92 pos[mapi - 1] = i;
93 }
94 return pos;
95 }
96
97 double header_rfloat(int w) const { // rounded to 5 digits
98 return std::round(1e5 * header_float(w)) / 1e5;
99 }

101 Box<Fractional> get_extent() const {
102 Box<Fractional> box;
103 // cf. setup()
104 auto pos = axis_positions();
105 std::array<int, 3> start = header_3i32(5);
106 std::array<int, 3> size = header_3i32(1);
107 std::array<int, 3> sampl = header_3i32(8);
108 for (int i = 0; i < 3; ++i) {
109 double scale = 1. / sampl[i];
110 int p = pos[i];
111 box.minimum.at(i) = scale * start[p] - 1e-9;
112 box.maximum.at(i) = scale * (start[p] + size[p] - 1) + 1e-9;
113 }
114 return box;
115 }

117 // Skew transformation (words 25-37) is supported by CCP4 maplib and PyMOL,
118 // but it's not in the MRC format and is not supported by most programs.
119 // From maplib.html: Skew transformation is from standard orthogonal
120 // coordinate frame (as used for atoms) to orthogonal map frame, as
121 // Xo(map) = S * (Xo(atoms) - t)
122 bool has_skew_transformation() const {
123 return header_i32(25) != 0; // LSKFLG should be 0 or 1
124 }
125 Transform get_skew_transformation() const {
126 return {
127 // 26-34 SKWMAT
128 { header_float(26), header_float(27), header_float(28),
129 header_float(29), header_float(30), header_float(31),
130 header_float(32), header_float(33), header_float(34) },
131 // 35-37 SKWTRN
132 { header_float(35), header_float(36), header_float(37) }
133 };
134 }

141 // this function assumes that the whole unit cell is covered with offset 0
142 void prepare_ccp4_header_except_mode_and_stats() {
143 GroupOps ops;
144 if (grid.spacegroup)
145 ops = grid.spacegroup->operations();
146 ccp4_header.clear();
147 ccp4_header.resize(256 + ops.order() * 20, 0);
148 set_header_3i32(1, grid.nu, grid.nv, grid.nw); // NX, NY, NZ
149 set_header_3i32(5, 0, 0, 0); // NXSTART, NYSTART, NZSTART
150 if (grid.axis_order == AxisOrder::XYZ)
151 set_header_3i32(8, grid.nu, grid.nv, grid.nw); // MX, MY, MZ
152 else // grid.axis_order == AxisOrder::ZYX
153 set_header_3i32(8, grid.nw, grid.nv, grid.nu);
154 set_header_float(11, (float) grid.unit_cell.a);
155 set_header_float(12, (float) grid.unit_cell.b);
156 set_header_float(13, (float) grid.unit_cell.c);
157 set_header_float(14, (float) grid.unit_cell.alpha);
158 set_header_float(15, (float) grid.unit_cell.beta);
159 set_header_float(16, (float) grid.unit_cell.gamma);
160 if (grid.axis_order == AxisOrder::XYZ)
161 set_header_3i32(17, 1, 2, 3); // MAPC, MAPR, MAPS
162 else // grid.axis_order == AxisOrder::ZYX
163 set_header_3i32(17, 3, 2, 1);
164 set_header_i32(23, grid.spacegroup ? grid.spacegroup->ccp4 : 1); // ISPG
165 set_header_i32(24, ops.order() * 80); // NSYMBT
166 set_header_str(27, "CCP4"); // EXTTYP
167 set_header_i32(28, 20140); // NVERSION
168 set_header_str(53, "MAP ");
169 set_header_i32(54, is_little_endian() ? 0x00004144 : 0x11110000); // MACHST
170 set_header_i32(56, 1); // labels
171 std::memset(header_word(57), ' ', 800 + ops.order() * 80);
172 set_header_str(57, "written by GEMMI");
173 int n = 257;
174 for (Op op : ops) {
175 set_header_str(n, op.triplet());
176 n += 20;
177 }
178 }

182 void update_ccp4_header(int mode=-1, bool update_stats=true) {
183 if (mode > 2 && mode != 6)
184 fail("Only modes 0, 1, 2 and 6 are supported.");
185 if (grid.point_count() == 0)
186 fail("update_ccp4_header(): set the grid first (it has size 0)");
187 if (grid.axis_order == AxisOrder::Unknown)
188 fail("update_ccp4_header(): run setup() first");
189 if (update_stats)
190 hstats = calculate_data_statistics(grid.data);
191 if (ccp4_header.empty())
192 prepare_ccp4_header_except_mode_and_stats();
193 assert(ccp4_header.size() >= 256);
194 if (mode < 0) {
195 mode = mode_for_data();
196 if (mode < 0)
197 fail("update_ccp4_header: specify map mode explicitely (usually 2)");
198 }
199 set_header_i32(4, mode);
200 set_header_float(20, (float) hstats.dmin);
201 set_header_float(21, (float) hstats.dmax);
202 set_header_float(22, (float) hstats.dmean);
203 set_header_float(55, (float) hstats.rms);
204 // labels could be modified but it's not important
205 }

207 static int mode_for_data() {
208 if (std::is_same<T, std::int8_t>::value)
209 return 0;
210 if (std::is_same<T, std::int16_t>::value)
211 return 1;
212 if (std::is_same<T, float>::value)
213 return 2;
214 if (std::is_same<T, std::uint16_t>::value)
215 return 6;
216 return -1;
217 }

219 bool full_cell() const {
220 if (ccp4_header.empty())
221 return true; // assuming it's full cell
222 return
223 // NXSTART et al. must be 0
224 header_i32(5) == 0 && header_i32(6) == 0 && header_i32(7) == 0 &&
225 // MX == NX
226 header_i32(8) == grid.nu && header_i32(9) == grid.nv && header_i32(10) == grid.nw &&
227 // just in case, check ORIGIN
228 header_i32(50) == 0 && header_i32(51) == 0 && header_i32(52) == 0;
229 }

231 template<typename Stream>
232 void read_ccp4_header(Stream& f, const std::string& path) {
233 const size_t hsize = 256;
234 ccp4_header.resize(hsize);
235 if (!f.read(ccp4_header.data(), 4 * hsize))
236 fail("Failed to read map header: " + path);
237 if (header_str(53, 4) != "MAP ")
238 fail("Not a CCP4 map: " + path);
239 std::string machst = header_str(54, 4);
240 if (machst[0] != 0x44 && machst[0] != 0x11)
241 fail("Unsupported machine stamp (endiannes) in the file?");
242 same_byte_order = machst[0] == (is_little_endian() ? 0x44 : 0x11);
243 grid.unit_cell.set(header_rfloat(11), header_rfloat(12), header_rfloat(13),
244 header_rfloat(14), header_rfloat(15), header_rfloat(16));
245 size_t ext_w = header_i32(24) / 4; // NSYMBT in words
246 if (ext_w != 0) {
247 if (ext_w > 1000000)
248 fail("Unexpectedly long extended header: " + path);
249 ccp4_header.resize(hsize + ext_w);
250 if (!f.read(ccp4_header.data() + hsize, 4 * ext_w))
251 fail("Failed to read extended header: " + path);
252 }
253 grid.nu = header_i32(1);
254 grid.nv = header_i32(2);
255 grid.nw = header_i32(3);
256 for (int i = 0; i < 3; ++i) {
257 int axis = header_i32(17 + i);
258 if (axis < 1 || axis > 3)
259 fail("Unexpected axis value in word " + std::to_string(17 + i)
260 + ": " + std::to_string(axis));
261 }
262 hstats.dmin = header_float(20);
263 hstats.dmax = header_float(21);
264 hstats.dmean = header_float(22);
265 hstats.rms = header_float(55);
266 grid.spacegroup = find_spacegroup_by_number(header_i32(23));
267 auto pos = axis_positions();
268 grid.axis_order = AxisOrder::Unknown;
269 if (pos[0] == 0 && pos[1] == 1 && pos[2] == 2 && full_cell()) {
270 grid.axis_order = AxisOrder::XYZ;
271 grid.calculate_spacing();
272 }
273 }

313template<typename Stream, typename TFile, typename TMem>
314void read_data(Stream& f, std::vector<TMem>& content) {
315 if (std::is_same<TFile, TMem>::value) {
316 size_t len = content.size();
317 if (!f.read(content.data(), sizeof(TMem) * len))
318 fail("Failed to read all the data from the map file.");
319 } else {
320 constexpr size_t chunk_size = 64 * 1024;
321 std::vector<TFile> work(chunk_size);
322 for (size_t i = 0; i < content.size(); i += chunk_size) {
323 size_t len = std::min(chunk_size, content.size() - i);
324 if (!f.read(work.data(), sizeof(TFile) * len))
325 fail("Failed to read all the data from the map file.");
326 for (size_t j = 0; j < len; ++j)
327 content[i+j] = translate_map_point<TFile,TMem>(work[j]);
328 }
329 }
330}
331
332template<typename TFile, typename TMem>
333void write_data(const std::vector<TMem>& content, FILE* f) {
334 if (std::is_same<TMem, TFile>::value) {
335 size_t len = content.size();
336 if (std::fwrite(content.data(), sizeof(TFile), len, f) != len)
337 sys_fail("Failed to write data to the map file");
338 } else {
339 constexpr size_t chunk_size = 64 * 1024;
340 std::vector<TFile> work(chunk_size);
341 for (size_t i = 0; i < content.size(); i += chunk_size) {
342 size_t len = std::min(chunk_size, content.size() - i);
343 for (size_t j = 0; j < len; ++j)
344 work[j] = static_cast<TFile>(content[i+j]);
345 if (std::fwrite(work.data(), sizeof(TFile), len, f) != len)
346 sys_fail("Failed to write data to the map file");
347 }
348 }

354// let us know if you need support for other architectures.
355template<typename T> template<typename Stream>
356void Ccp4<T>::read_ccp4_stream(Stream f, const std::string& path) {
357 read_ccp4_header(f, path);
358 grid.data.resize(grid.point_count());
359 int mode = header_i32(4);
360 if (mode == 0)
361 impl::read_data<Stream, std::int8_t>(f, grid.data);
362 else if (mode == 1)
363 impl::read_data<Stream, std::int16_t>(f, grid.data);
364 else if (mode == 2)
365 impl::read_data<Stream, float>(f, grid.data);
366 else if (mode == 6)
367 impl::read_data<Stream, std::uint16_t>(f, grid.data);
368 else
369 fail("Mode " + std::to_string(mode) + " is not supported "
370 "(only 0, 1, 2 and 6 are supported).");
371 //if (std::fgetc(f) != EOF)
372 // fail("The map file is longer then expected.");
373
374 if (!same_byte_order) {
375 if (sizeof(T) == 2)
376 for (T& value : grid.data)
377 swap_two_bytes(&value);
378 else if (sizeof(T) == 4)
379 for (T& value : grid.data)
380 swap_four_bytes(&value);
381 }

384template<typename T>
385void Ccp4<T>::setup(T default_value, MapSetup mode) {
386 if (grid.axis_order == AxisOrder::XYZ || ccp4_header.empty())
387 return;
388 // cell sampling does not change
389 const std::array<int, 3> sampl = header_3i32(8);
390 // get old metadata
391 const std::array<int, 3> pos = axis_positions();
392 std::array<int, 3> start = header_3i32(5);
393 int end[3] = { start[0] + grid.nu, start[1] + grid.nv, start[2] + grid.nw };
394 // set new metadata
395 if (mode == MapSetup::ReorderOnly) {
396 set_header_3i32(5, start[pos[0]], start[pos[1]], start[pos[2]]);
397 for (int i = 0; i < 3; ++i) {
398 end[i] -= start[i];
399 start[i] = 0;
400 }
401 int crs[3] = { grid.nu, grid.nv, grid.nw };
402 grid.nu = crs[pos[0]];
403 grid.nv = crs[pos[1]];
404 grid.nw = crs[pos[2]];
405 } else {
406 grid.nu = sampl[0];
407 grid.nv = sampl[1];
408 grid.nw = sampl[2];
409 set_header_3i32(5, 0, 0, 0); // start
410 }
411 set_header_3i32(1, grid.nu, grid.nv, grid.nw); // NX, NY, NZ
412 set_header_3i32(17, 1, 2, 3); // axes (MAPC, MAPR, MAPS)
413 grid.axis_order = full_cell() ? AxisOrder::XYZ : AxisOrder::Unknown;
414 if (grid.axis_order == AxisOrder::XYZ)
415 grid.calculate_spacing();
416
417 // now set the data
418 {
419 std::vector<T> full(grid.point_count(), default_value);
420 int it[3];
421 int idx = 0;
422 for (it[2] = start[2]; it[2] < end[2]; it[2]++) // sections
423 for (it[1] = start[1]; it[1] < end[1]; it[1]++) // rows
424 for (it[0] = start[0]; it[0] < end[0]; it[0]++) { // cols
425 T val = grid.data[idx++];
426 size_t new_index = grid.index_s(it[pos[0]], it[pos[1]], it[pos[2]]);
427 full[new_index] = val;
428 }
429 grid.data = std::move(full);
430 }
431
432 if (mode == MapSetup::Full &&
433 // no need to apply symmetry if we started with the whole cell
434 (end[pos[0]] - start[pos[0]] < sampl[0] ||
435 end[pos[1]] - start[pos[1]] < sampl[1] ||
436 end[pos[2]] - start[pos[2]] < sampl[2]))
437 grid.symmetrize_nondefault(default_value);
438}

440template<typename T>
441void Ccp4<T>::set_extent(const Box<Fractional>& box) {
442 if (ccp4_header.empty())
443 fail("set_extent(): no header in the map. Call update_ccp4_header() first");
444 if (!full_cell())
445 fail("Ccp4::set_extent() works only after setup()");
446 if (grid.axis_order != AxisOrder::XYZ)
447 fail("Ccp4::set_extent() works only with XYZ order");
448 int u0 = (int)std::ceil(box.minimum.x * grid.nu);
449 int v0 = (int)std::ceil(box.minimum.y * grid.nv);
450 int w0 = (int)std::ceil(box.minimum.z * grid.nw);
451 int nu = (int)std::floor(box.maximum.x * grid.nu) - u0 + 1;
452 int nv = (int)std::floor(box.maximum.y * grid.nv) - v0 + 1;
453 int nw = (int)std::floor(box.maximum.z * grid.nw) - w0 + 1;
454 // set the data
455 std::vector<T> new_data((size_t)nu * nv * nw);
456 grid.get_subarray(new_data.data(), {u0, v0, w0}, {nu, nv, nw});
457 grid.data.swap(new_data);
458 // and metadata
459 grid.nu = nu;
460 grid.nv = nv;
461 grid.nw = nw;
462 set_header_3i32(1, grid.nu, grid.nv, grid.nw); // NX, NY, NZ
463 set_header_3i32(5, u0, v0, w0);
464 // AxisOrder::XYZ is used only for grid covering full cell
465 grid.axis_order = AxisOrder::Unknown;
466}
