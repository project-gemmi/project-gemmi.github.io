23// CIF files are read in binary mode. It makes difference only for text fields.
24// If the text field with \r\n would be written as is in text mode on Windows
25// \r would get duplicated. As a workaround, here we convert \r\n to \n.
26// Hopefully \r that gets removed here is never meaningful.
27inline void write_text_field(std::ostream& os, const std::string& value) {
28 for (size_t pos = 0, end = 0; end != std::string::npos; pos = end + 1) {
29 end = value.find("\r\n", pos);
30 size_t len = (end == std::string::npos ? value.size() : end) - pos;
31 os.write(value.c_str() + pos, len);
32 }
33}

35inline void write_out_pair(std::ostream& os, const std::string& name,
36 const std::string& value, Style style) {
37 os << name;
38 if (is_text_field(value)) {
39 os.put('\n');
40 write_text_field(os, value);
41 } else {
42 if (name.size() + value.size() > 120)
43 os.put('\n');
44 else if ((style == Style::Indent35 || style == Style::Aligned) && name.size() < 34)
45 os.write(" ", 34 - name.size());
46 else
47 os.put(' ');
48 os << value;
49 }
50 os.put('\n');
51}

53inline void write_out_loop(std::ostream& os, const Loop& loop, Style style) {
54 constexpr size_t max_padding = 30; // if increased, adjust os.write() below
55 if (loop.values.empty())
56 return;
57 if ((style == Style::PreferPairs || style == Style::Pdbx) &&
58 loop.length() == 1) {
59 for (size_t i = 0; i != loop.tags.size(); ++i)
60 write_out_pair(os, loop.tags[i], loop.values[i], style);
61 return;
62 }
63 // tags
64 os << "loop_";
65 for (const std::string& tag : loop.tags)
66 os << '\n' << tag;
67 // values
68 size_t ncol = loop.tags.size();
69
70 std::vector<size_t> col_width;
71 if (style == Style::Aligned) {
72 col_width.resize(ncol, 1);
73 size_t col = 0;
74 for (const std::string& val : loop.values) {
75 col_width[col] = std::max(col_width[col], val.size());
76 if (++col == ncol)
77 col = 0;
78 }
79 for (size_t& w : col_width)
80 w = std::min(w, max_padding);
81 }
82
83 size_t col = 0;
84 for (const std::string& val : loop.values) {
85 bool text_field = is_text_field(val);
86 os.put(col == 0 || text_field ? '\n' : ' ');
87 if (text_field)
88 write_text_field(os, val);
89 else
90 os << val;
91 if (col != ncol - 1) {
92 if (!col_width.empty() && val.size() < col_width[col])
93 os.write(" ", col_width[col] - val.size());
94 ++col;
95 } else {
96 col = 0;
97 }
98 }
99 os.put('\n');
100}

102inline void write_out_item(std::ostream& os, const Item& item, Style style) {
103 switch (item.type) {
104 case ItemType::Pair:
105 write_out_pair(os, item.pair[0], item.pair[1], style);
106 break;
107 case ItemType::Loop:
108 write_out_loop(os, item.loop, style);
109 break;
110 case ItemType::Frame:
111 os << "save_" << item.frame.name << '\n';
112 for (const Item& inner_item : item.frame.items)
113 write_out_item(os, inner_item, style);
114 os << "save_\n";
115 break;
116 case ItemType::Comment:
117 os << item.pair[1] << '\n';
118 break;
119 case ItemType::Erased:
120 break;
121 }
122}

124inline bool should_be_separated_(const Item& a, const Item& b) {
125 if (a.type == ItemType::Comment || b.type == ItemType::Comment)
126 return false;
127 if (a.type != ItemType::Pair || b.type != ItemType::Pair)
128 return true;
129 // check if we have mmcif-like tags from different categories
130 auto adot = a.pair[0].find('.');
131 if (adot == std::string::npos)
132 return false;
133 auto bdot = b.pair[0].find('.');
134 return adot != bdot || a.pair[0].compare(0, adot, b.pair[0], 0, adot) != 0;
135}

137inline void write_cif_block_to_stream(std::ostream& os, const Block& block,
138 Style style=Style::Simple) {
139 os << "data_" << block.name << '\n';
140 if (style == Style::Pdbx)
141 os << "#\n";
142 const Item* prev = nullptr;
143 for (const Item& item : block.items)
144 if (item.type != ItemType::Erased) {
145 if (prev && style != Style::NoBlankLines &&
146 should_be_separated_(*prev, item))
147 os << (style == Style::Pdbx ? "#\n" : "\n");
148 write_out_item(os, item, style);
149 prev = &item;
150 }
151 if (style == Style::Pdbx)
152 os << "#\n";
153}

155inline void write_cif_to_stream(std::ostream& os, const Document& doc,
