30// Information from REMARK 200/230 is significantly expanded in PDBx/mmCIF.
31// These remarks corresponds to data across 12 mmCIF categories
32// including categories _exptl, _reflns, _exptl_crystal, _diffrn and others.
33// _exptl and _reflns seem to be 1:1. Usually we have one experiment (_exptl),
34// except for a joint refinement (e.g. X-ray + neutron data).
35// Both crystal (_exptl_crystal) and reflection statistics (_reflns) can
36// be associated with multiple diffraction sets (_diffrn).
37// But if we use the PDB format, only one diffraction set per method
38// can be described.

260// Secondary structure. PDBx/mmCIF stores helices and sheets separately.
261
262// mmCIF spec defines 32 possible values for _struct_conf.conf_type_id -
263// "the type of the conformation of the backbone of the polymer (whether
264// protein or nucleic acid)". But as of 2019 only HELX_P is used (not counting
265// TURN_P that occurs in only 6 entries). The actual helix type is given
266// by numeric value of _struct_conf.pdbx_PDB_helix_class, which corresponds
267// to helixClass from the PDB HELIX record. These values are in the range 1-10.
268// As of 2019 it's almost only type 1 and 5:
269// 3116566 of 1 - right-handed alpha
270// 16 of 2 - right-handed omega
271// 84 of 3 - right-handed pi
272// 79 of 4 - right-handed gamma
273// 1063337 of 5 - right-handed 3-10
274// 27 of 6 - left-handed alpha
275// 5 of 7 - left-handed omega
276// 2 of 8 - left-handed gamma
277// 8 of 9 - 2-7 ribbon/helix
278// 46 of 10 - polyproline
279struct Helix {
280 enum HelixClass {
281 UnknownHelix, RAlpha, ROmega, RPi, RGamma, R310,
282 LAlpha, LOmega, LGamma, Helix27, HelixPolyProlineNone
283 };
284 AtomAddress start, end;
285 HelixClass pdb_helix_class = UnknownHelix;
286 int length = -1;
287 void set_helix_class_as_int(int n) {
288 if (n >= 1 && n <= 10)
289 pdb_helix_class = static_cast<HelixClass>(n);
290 }
291};
