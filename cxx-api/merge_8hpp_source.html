31inline bool parse_voigt_notation(const char* start, const char* end, SMat33<double>& b) {
32 bool first = true;
33 for (double* u : {&b.u11, &b.u22, &b.u33, &b.u23, &b.u13, &b.u12}) {
34 if (*start != (first ? '(' : ','))
35 return false;
36 first = false;
37 auto result = fast_from_chars(++start, end, *u);
38 if (result.ec != std::errc())
39 return false;
40 start = skip_blank(result.ptr);
41 }
42 return *start == ')';
43}

45inline bool read_staraniso_b_from_mmcif(const cif::Block& block, SMat33<double>& output) {
46 // read what is written by MtzToCif::write_staraniso_b()
47 cif::Table table = const_cast<cif::Block*>(&block)->find(
48 "_reflns.pdbx_aniso_B_tensor_eigen",
49 {"value_1", "value_2", "value_3",
50 "vector_1_ortho[1]", "vector_1_ortho[2]", "vector_1_ortho[3]",
51 "vector_2_ortho[1]", "vector_2_ortho[2]", "vector_2_ortho[3]",
52 "vector_3_ortho[1]", "vector_3_ortho[2]", "vector_3_ortho[3]"});
53 if (!table.ok())
54 return false;
55 cif::Table::Row row = table.one();
56 using cif::as_number;
57 double eigval[3] = {as_number(row[0]), as_number(row[1]), as_number(row[2])};
58 double min_val = std::min(std::min(eigval[0], eigval[1]), eigval[2]);
59 Mat33 mat(as_number(row[3]), as_number(row[6]), as_number(row[9]),
60 as_number(row[4]), as_number(row[7]), as_number(row[10]),
61 as_number(row[5]), as_number(row[8]), as_number(row[11]));
62 Vec3 diag(eigval[0] - min_val, eigval[1] - min_val, eigval[2] - min_val);
63 // If the columns of mat are an orthonomal basis, mat^−1==mat^T.
64 // But just in case it isn't, we use mat^-1 here.
65 Mat33 t = mat.multiply_by_diagonal(diag).multiply(mat.inverse());
66 // t is a symmetric tensor, so we return only 6 numbers
67 output = {t[0][0], t[1][1], t[2][2], t[0][1], t[0][2], t[1][2]};
68 return true;
69}

71// returns STARANISO version or empty string
72inline std::string read_staraniso_b_from_mtz(const Mtz& mtz, SMat33<double>& output) {
73 std::string version;
74 size_t hlen = mtz.history.size();
75 for (size_t i = 0; i != hlen; ++i)
76 if (mtz.history[i].find("STARANISO") != std::string::npos) {
77 size_t version_pos = mtz.history[i].find("version:");
78 if (version_pos != std::string::npos)
79 version = read_word(mtz.history[i].c_str() + version_pos + 8);
80 else
81 version = "?";
82 // StarAniso 2.3.74 (24-Apr-2021) and later write B tensor in history
83 for (size_t j = i+1; j < std::min(i+4, hlen); ++j) {
84 const std::string& line = mtz.history[j];
85 if (starts_with(line, "B=(")) {
86 if (!parse_voigt_notation(line.c_str() + 2,
87 line.c_str() + line.size(),
88 output))
89 fail("failed to parse tensor Voigt notation: " + line);
90 break;
91 }
92 }
93 break;
94 }
95 return version;
96}

173 Correlation calculate_correlation(const Intensities& other) const {
174 Correlation corr;
175 auto r1 = data.begin();
176 auto r2 = other.data.begin();
177 while (r1 != data.end() && r2 != other.data.end()) {
178 if (r1->hkl == r2->hkl && r1->isign == r2->isign) {
179 corr.add_point(r1->value, r2->value);
180 ++r1;

182 } else if (*r1 < *r2) {
183 ++r1;
184 } else {
185 ++r2;

202 void merge_in_place(DataType data_type) {
203 type = data_type;
204 if (data.empty())
205 return;
206 if (data_type == DataType::Mean)
207 // discard signs so that merging produces Imean
208 for (Refl& refl : data)
209 refl.isign = 0;
210 sort();
211 std::vector<Refl>::iterator out = data.begin();
212 double sum_wI = 0.;
213 double sum_w = 0.;
214 int nobs = 0;
215 for (auto in = data.begin(); in != data.end(); ++in) {
216 if (out->hkl != in->hkl || out->isign != in->isign) {
217 out->value = sum_wI / sum_w;
218 out->sigma = 1.0 / std::sqrt(sum_w);
219 out->nobs = nobs;
220 sum_wI = sum_w = 0.;
221 nobs = 0;
222 ++out;
223 out->hkl = in->hkl;
224 out->isign = in->isign;
225 }
226 double w = 1. / (in->sigma * in->sigma);
227 sum_wI += w * in->value;
228 sum_w += w;
229 ++nobs;
230 }
231 out->value = sum_wI / sum_w;
232 out->sigma = 1.0 / std::sqrt(sum_w);
233 out->nobs = nobs;
234 data.erase(++out, data.end());
235 }

237 // for unmerged centric reflections set isign=1.
238 void switch_to_asu_indices(bool merged=false) {
239 GroupOps gops = spacegroup->operations();
240 ReciprocalAsu asu(spacegroup);
241 for (Refl& refl : data) {
242 if (asu.is_in(refl.hkl)) {
243 if (!merged) {
244 // isign is 0 for original hkl (e.g. from XDS file)
245 if (refl.isign == 0)
246 refl.isign = 1; // since it's in asu - I+ or centric
247 // when reading asu hkl from MTZ file - count centrics always as I+
248 else if (refl.isign == -1 && gops.is_reflection_centric(refl.hkl))
249 refl.isign = 1;
250 }
251 continue;
252 }
253 auto hkl_isym = asu.to_asu(refl.hkl, gops);
254 refl.hkl = hkl_isym.first;
255 if (!merged) {
256 if (gops.is_reflection_centric(refl.hkl))
257 refl.isign = 1;
258 else
259 refl.isign = (hkl_isym.second % 2 == 0 ? -1 : 1);
260 }
261 }

264 void read_unmerged_intensities_from_mtz(const Mtz& mtz) {
265 if (mtz.batches.empty())
266 fail("expected unmerged file");
267 const Mtz::Column* isym_col = mtz.column_with_label("M/ISYM");
268 if (!isym_col || isym_col->idx != 3)
269 fail("unmerged file should have M/ISYM as 4th column");
270 const Mtz::Column& col = mtz.get_column_with_label("I");
271 size_t value_idx = col.idx;
272 size_t sigma_idx = mtz.get_column_with_label("SIGI").idx;
273 unit_cell = mtz.get_average_cell_from_batch_headers(unit_cell_rmsd);
274 spacegroup = mtz.spacegroup;
275 if (!spacegroup)
276 fail("unknown space group");
277 wavelength = mtz.dataset(col.dataset_id).wavelength;
278 for (size_t i = 0; i < mtz.data.size(); i += mtz.columns.size()) {
279 short isign = ((int)mtz.data[i + 3] % 2 == 0 ? -1 : 1);
280 add_if_valid(mtz.get_hkl(i), isign, mtz.data[i + value_idx], mtz.data[i + sigma_idx]);
281 }
282 // Aimless >=0.7.6 (from 2021) has an option to output unmerged file
283 // with original indices instead of reduced indices, with all ISYM = 1.
284 switch_to_asu_indices();
285 type = DataType::Unmerged;
286 }

464 for (size_t i = 0; i < proxy.size(); i += proxy.stride())

470 template<typename DataProxy>
471 void read_anomalous_data(const DataProxy& proxy, int mean_idx,
472 size_t (&value_idx)[2], size_t (&sigma_idx)[2]) {
473 GroupOps gops = spacegroup->operations();
474 for (size_t i = 0; i < proxy.size(); i += proxy.stride()) {
475 Miller hkl = proxy.get_hkl(i);
476 bool centric = gops.is_reflection_centric(hkl);
477
478 // sanity check
479 if (mean_idx >= 0 && !std::isnan(proxy.get_num(i + mean_idx)) && !centric) {
480 if (std::isnan(proxy.get_num(i + value_idx[0])) &&
481 std::isnan(proxy.get_num(i + value_idx[1])))
482 fail(miller_str(hkl), " has <I>, but I(+) and I(-) are both null");
483 }
484
485 add_if_valid(hkl, 1, proxy.get_num(i + value_idx[0]),
486 proxy.get_num(i + sigma_idx[0]));
487 if (!centric) // ignore I(-) of centric reflections
488 add_if_valid(hkl, -1, proxy.get_num(i + value_idx[1]),
489 proxy.get_num(i + sigma_idx[1]));
490 }
491 }
