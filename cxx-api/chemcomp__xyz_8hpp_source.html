23inline Residue make_residue_from_chemcomp_block(const cif::Block& block,
24 ChemCompModel kind) {
25 std::array<std::string, 3> xyz_tags;
26 switch (kind) {
27 case ChemCompModel::Xyz:
28 xyz_tags = {{"x", "y", "z"}};
29 break;
30 case ChemCompModel::Example:
31 xyz_tags = {{"model_Cartn_x", "model_Cartn_y", "model_Cartn_z"}};
32 break;
33 case ChemCompModel::Ideal:
34 xyz_tags = {{"pdbx_model_Cartn_x_ideal",
35 "pdbx_model_Cartn_y_ideal",
36 "pdbx_model_Cartn_z_ideal"}};
37 break;
38 }
39 Residue res;
40 cif::Column col =
41 const_cast<cif::Block&>(block).find_values("_chem_comp_atom.comp_id");
42 if (col && col.length() > 0)
43 res.name = col[0];
44 else
45 res.name = block.name.substr(starts_with(block.name, "comp_") ? 5 : 0);
46 cif::Table table = const_cast<cif::Block&>(block).find("_chem_comp_atom.",
47 {"atom_id", "type_symbol", "?charge",
48 xyz_tags[0], xyz_tags[1], xyz_tags[2]});
49 res.atoms.resize(table.length());
50 int n = 0;
51 for (auto row : table) {
52 Atom& atom = res.atoms[n++];
53 atom.name = row.str(0);
54 atom.element = Element(row.str(1));
55 if (row.has2(2))
56 // Charge is defined as integer, but some cif files in the wild have
57 // trailing '.000', so we read it as floating-point number.
58 atom.charge = (signed char) std::round(cif::as_number(row[2]));
59 atom.pos = Position(cif::as_number(row[3]),
60 cif::as_number(row[4]),
61 cif::as_number(row[5]));
62 }
63 return res;
64}

66inline Model make_model_from_chemcomp_block(const cif::Block& block,
67 ChemCompModel kind) {
68 std::string name;
69 switch (kind) {
70 case ChemCompModel::Xyz: name = "xyz"; break;
71 case ChemCompModel::Example: name = "example_xyz"; break;
72 case ChemCompModel::Ideal: name = "ideal_xyz"; break;
73 }
74 Model model(name);
75 model.chains.emplace_back("");
76 model.chains[0].residues.push_back(
77 make_residue_from_chemcomp_block(block, kind));
78 return model;
79}

81// For CCD input - returns a structure with two single-residue models:
82// example (model_Cartn_x) and ideal (pdbx_model_Cartn_x_ideal).
83// For Refmac dictionary (monomer library) files returns structure with
84// a single model.
85inline Structure make_structure_from_chemcomp_block(const cif::Block& block) {
86 Structure st;
87 st.input_format = CoorFormat::ChemComp;
88 if (const std::string* name = block.find_value("_chem_comp.id"))
89 st.name = *name;
90 if (block.has_any_value("_chem_comp_atom.x"))
91 st.models.push_back(
92 make_model_from_chemcomp_block(block, ChemCompModel::Xyz));
93 if (block.has_any_value("_chem_comp_atom.model_Cartn_x"))
94 st.models.push_back(
95 make_model_from_chemcomp_block(block, ChemCompModel::Example));
96 if (block.has_any_value("_chem_comp_atom.pdbx_model_Cartn_x_ideal"))
97 st.models.push_back(
98 make_model_from_chemcomp_block(block, ChemCompModel::Ideal));
99 return st;
100}

102// a helper function for use with make_structure_from_chemcomp_block():
103// int n = check_chemcomp_block_number(doc);
104// if (n != -1)
105// Structure st = make_structure_from_chemcomp_block(doc.blocks[n]);
106inline int check_chemcomp_block_number(const cif::Document& doc) {
107 // monomer library file without global_
108 if (doc.blocks.size() == 2 && doc.blocks[0].name == "comp_list")
109 return 1;
110 // monomer library file with global_
111 if (doc.blocks.size() == 3 && doc.blocks[0].name.empty() &&
112 doc.blocks[1].name == "comp_list")
113 return 2;
114 // CCD file
115 if (doc.blocks.size() == 1 &&
116 !doc.blocks[0].has_tag("_atom_site.id") &&
117 doc.blocks[0].has_tag("_chem_comp_atom.atom_id"))
118 return 0;
119 return -1;
120}

122inline Structure make_structure_from_chemcomp_doc(const cif::Document& doc) {
123 int n = check_chemcomp_block_number(doc);
124 if (n == -1)
125 fail("Not a chem_comp format.");
126 return make_structure_from_chemcomp_block(doc.blocks[n]);
127}
