18// Assumes no hydrogens in the residue.
19// Position and serial number are not assigned for new atoms.
20inline void add_hydrogens_without_positions(const ChemComp& cc, Residue& res) {
21 for (auto it = cc.atoms.begin(); it != cc.atoms.end(); ++it) {
22 if (!it->is_hydrogen())
23 continue;
24 Atom atom = it->to_full_atom();
25 if (const Restraints::AtomId* bonded = cc.rt.first_bonded_atom(atom.name))
26 // Add H atom for each conformation (altloc) of the parent atom.
27 // Avoid range-based-for here because res.atoms may get re-allocated.
28 for (size_t i = 0, size = res.atoms.size(); i != size; ++i)
29 if (res.atoms[i].name == bonded->atom) {
30 const Atom& parent = res.atoms[i];
31 atom.altloc = parent.altloc;
32 atom.occ = parent.occ;
33 atom.b_iso = parent.b_iso;
34 // calc_flag will be changed to Calculated when the position is set
35 atom.calc_flag = CalcFlag::Dummy;
36 res.atoms.push_back(atom);
37 }
38 }
39}

42// Calculate position using one angle (theta) and one dihedral angle (tau).
43// Returns position of x4 in x1-x2-x3-x4, where dist=|x3-x4| and
44// theta is angle(x2, x3, x4).
45// Based on section 3.3 of Paciorek et al, Acta Cryst. A52, 349 (1996).
46inline
47Position position_from_angle_and_torsion(const Position& x1,
48 const Position& x2,
49 const Position& x3,
50 double dist, // |x3-x4|
51 double theta, // angle x2-x3-x4
52 double tau) { // dihedral angle
53 using std::sin;
54 using std::cos;
55 Vec3 u = x2 - x1;
56 Vec3 v = x3 - x2;
57 Vec3 e1 = v.normalized();
58 double delta = u.dot(e1);
59 Vec3 e2 = -(u - delta * e1).normalized();
60 Vec3 e3 = e1.cross(e2);
61 return x3 + Position(dist * (-cos(theta) * e1 +
62 sin(theta) * (cos(tau) * e2 + sin(tau) * e3)));
63}

65// Rodrigues' rotation formula, rotate vector v given axis of rotation
66// (which must be a unit vector) and angle (in radians).
67inline
68Vec3 rotate_about_axis(const Vec3& v, const Vec3& axis, double theta) {
69 double sin_theta = std::sin(theta);
70 double cos_theta = std::cos(theta);
71 return v * cos_theta + axis.cross(v) * sin_theta +
72 axis * (axis.dot(v) * (1 - cos_theta));
73}

75inline Vec3 get_vector_to_line(const Position& point,
76 const Position& point_on_the_line,
77 const Vec3& unit_vector) {
78 // en.wikipedia.org/wiki/Distance_from_a_point_to_a_line#Vector_formulation
79 // the component of a - p perpendicular to the line is: (a-p) - ((a-p).n)n
80 Vec3 ap = point_on_the_line - point;
81 return ap - ap.dot(unit_vector) * unit_vector;
82}

84// Based on https://en.wikipedia.org/wiki/Trilateration
85// If no points satisfy the returns NaNs
86inline
87std::pair<Position, Position> trilaterate(const Position& p1, double r1sq,
88 const Position& p2, double r2sq,
89 const Position& p3, double r3sq) {
90 // variables have the same names as on the Wikipedia Trilateration page
91 Vec3 ex = (p2 - p1).normalized();
92 double i = ex.dot(p3-p1);
93 Vec3 ey = (Vec3(p3) - p1 - i*ex).normalized();
94 Vec3 ez = ex.cross(ey);
95 double d = (p2-p1).length();
96 double j = ey.dot(p3-p1);
97 double x = (r1sq - r2sq + d*d) / (2*d);
98 double y = (r1sq - r3sq + i*i + j*j) / (2*j) - x*i/j;
99 double z2 = r1sq - x*x - y*y;
100 double z = std::sqrt(z2); // may result in NaN
101 return std::make_pair(p1 + Position(x*ex + y*ey + z*ez),
102 p1 + Position(x*ex + y*ey - z*ez));
103}

105// Calculate position using two angles.
106// Returns p4. Topology: p1 is bonded to p2, p3 and p4.
107inline std::pair<Position, Position>
108position_from_two_angles(const Position& p1,
109 const Position& p2,
110 const Position& p3,
111 double dist14, // |p4-p1|
112 double theta214, // angle p2-p1-p4
113 double theta314) { // angle p3-p1-p4
114 double d12sq = p1.dist_sq(p2);
115 double d13sq = p1.dist_sq(p3);
116 double d14sq = dist14 * dist14;
117 // the law of cosines
118 double d24sq = d14sq + d12sq - 2 * std::sqrt(d14sq * d12sq) * cos(theta214);
119 double d34sq = d14sq + d13sq - 2 * std::sqrt(d14sq * d13sq) * cos(theta314);
120 auto t = trilaterate(p1, d14sq, p2, d24sq, p3, d34sq);
121 return t;
122}

125inline void place_hydrogens(const Topo& topo, const Atom& atom) {
126 using Angle = Restraints::Angle;
127 struct BondedAtom {
128 Atom* ptr;
129 Position& pos; // == ptr->pos;
130 double dist;
131 };
132
133 // put atoms bonded to atom into two lists
134 std::vector<BondedAtom> known; // heavy atoms with known positions
135 std::vector<BondedAtom> hs; // H atoms (unknown)
136 known.reserve(3);
137 hs.reserve(4);
138
139 auto range = topo.bond_index.equal_range(&atom);
140 for (auto i = range.first; i != range.second; ++i) {
141 const Topo::Bond* t = i->second;
142 Atom* other = t->atoms[t->atoms[0] == &atom ? 1 : 0];
143 if (other->altloc) {
144 if (atom.altloc && atom.altloc != other->altloc)
145 continue;
146 if (atom.altloc == '\0' &&
147 in_vector_f([&](const BondedAtom& a) { return a.ptr->name == other->name; },
148 known))
149 continue;
150 }
151 auto& atom_list = other->is_hydrogen() ? hs : known;
152 atom_list.push_back({other, other->pos, t->restr->value});
153 }
154
155 if (hs.size() == 0)
156 return;
157
158 auto giveup = [&](const std::string& message) {
159 for (BondedAtom& bonded_h : hs)
160 bonded_h.ptr->occ = 0;
161 fail(message);
162 };
163
164 for (BondedAtom& bonded_h : hs)
165 bonded_h.ptr->calc_flag = CalcFlag::Calculated;
166
167 // ==== only hydrogens ====
168 if (known.size() == 0) {
169 // we can only arbitrarily pick directions of atoms
170 for (BondedAtom& bonded_h : hs)
171 bonded_h.ptr->occ = 0;
172 hs[0].pos = atom.pos + Position(hs[0].dist, 0, 0);
173 if (hs.size() > 1) {
174 double theta = pi();
175 if (const Angle* ang = topo.take_angle(hs[1].ptr, &atom, hs[0].ptr))
176 theta = ang->radians();
177 hs[1].pos = atom.pos + Position(hs[1].dist * cos(theta),
178 hs[1].dist * sin(theta), 0);
179 }
180 if (hs.size() > 2) {
181 if (hs.size() == 3) {
182 // for now only NH3 (NH2.cif and NH3.cif) has such configuration,
183 // so we are cheating here a little.
184 double y = 2 * atom.pos.y - hs[1].pos.y;
185 hs[2].pos = Position(hs[1].pos.x, y, hs[1].pos.z);
186 } else if (hs.size() == 4) {
187 // similarly, only CH4 (CH2.cif) and and NH4 (NH4.cif) are handled here
188 const Angle* ang1 = topo.take_angle(hs[2].ptr, &atom, hs[0].ptr);
189 const Angle* ang2 = topo.take_angle(hs[2].ptr, &atom, hs[1].ptr);
190 double theta1 = rad(ang1 ? ang1->value : 109.47122);
191 double theta2 = rad(ang2 ? ang2->value : 109.47122);
192 auto pos = position_from_two_angles(atom.pos, hs[0].pos, hs[1].pos,
193 hs[2].dist, theta1, theta2);
194 hs[2].pos = pos.first;
195 hs[3].pos = pos.second;
196 }
197 }
198
199 // ==== one heavy atom and hydrogens ====
200 } else if (known.size() == 1) {
201 const BondedAtom& h = hs[0];
202 const BondedAtom& heavy = known[0];
203 const Restraints::Angle* angle = topo.take_angle(h.ptr, &atom, heavy.ptr);
204 if (!angle)
205 giveup("No angle restraint for " + h.ptr->name + ".\n");
206 if (std::abs(angle->value - 180.0) < 0.5) {
207 Vec3 u = atom.pos - h.pos;
208 h.pos = atom.pos + Position(u * (h.dist / u.length()));
209 if (hs.size() > 1)
210 giveup("Unusual: one of two H atoms has angle restraint 180 deg.");
211 return;
212 }
213 double theta = angle->radians();
214 double tau = 0.0;
215 int period = 0;
216 const Atom* tau_end = nullptr;
217 for (const Topo::Plane& plane : topo.planes) {
218 // only Topo::Plane with atoms.size() >= 4 is put into planes
219 if (plane.has(h.ptr) && plane.has(&atom) && plane.has(heavy.ptr)) {
220 for (const Atom* a : plane.atoms) {
221 if (!a->is_hydrogen() && a != &atom && a != heavy.ptr) {
222 tau_end = a;
223 break;
224 }
225 }
226 break;
227 }
228 }
229 Atom* torsion_h = nullptr;
230 if (!tau_end) {
231 // Using one dihedral angle.
232 // We don't check here for which hydrogen the torsion angle is defined.
233 // If an atom has 2 or 3 hydrogens, the torsion angle may not be given
234 // for the first one, but only for the 2nd or 3rd (e.g. HD22 in ASN).
235 for (const Topo::Torsion& tor : topo.torsions) {
236 if (tor.atoms[0]->is_hydrogen() && tor.atoms[1] == &atom &&
237 tor.atoms[2] == heavy.ptr && !tor.atoms[3]->is_hydrogen()) {
238 tau = rad(tor.restr->value);
239 torsion_h = tor.atoms[0];
240 tau_end = tor.atoms[3];
241 period = tor.restr->period;
242 break;
243 } else if (tor.atoms[3]->is_hydrogen() && tor.atoms[2] == &atom &&
244 tor.atoms[1] == heavy.ptr && !tor.atoms[0]->is_hydrogen()) {
245 tau = rad(tor.restr->value);
246 torsion_h = tor.atoms[3];
247 tau_end = tor.atoms[0];
248 period = tor.restr->period;
249 break;
250 }
251 }
252 }
253 h.pos = position_from_angle_and_torsion(
254 tau_end ? tau_end->pos : Position(0, 0, 0),
255 heavy.pos, atom.pos, h.dist, theta, tau);
256 if (hs.size() == 2) {
257 // I think we can assume the two hydrogens are symmetric.
258 Vec3 axis = (heavy.pos - atom.pos).normalized();
259 Vec3 perpendicular = get_vector_to_line(hs[0].pos, atom.pos, axis);
260 hs[1].pos = hs[0].pos + Position(2 * perpendicular);
261 } else if (hs.size() == 3) {
262 // Here we assume the three hydrogens are in the same distance from
263 // the parent atom and that they make an equilateral triangle.
264 // First check for which hydrogen was the torsion restraint (if any).
265 int idx = 0;
266 for (int i : {1, 2})
267 if (torsion_h == hs[i].ptr) {
268 idx = i;
269 hs[i].pos = h.pos;
270 }
271 // Now get the other positions by rotation
272 Vec3 axis = (heavy.pos - atom.pos).normalized();
273 Vec3 v1 = h.pos - atom.pos;
274 Vec3 v2 = rotate_about_axis(v1, axis, rad(120));
275 Vec3 v3 = rotate_about_axis(v1, axis, rad(-120));
276 hs[(idx+1) % 3].pos = atom.pos + Position(v2);
277 hs[(idx+2) % 3].pos = atom.pos + Position(v3);
278 } else if (hs.size() >= 4) {
279 giveup("Unusual: atom bonded to one heavy atoms and 4+ hydrogens.");
280 }
281 // somewhat arbitrary rule
282 if (!tau_end || (hs.size() == 1 && period > 1))
283 for (BondedAtom& bonded_h : hs)
284 bonded_h.ptr->occ = 0;
285 // ==== two heavy atoms and hydrogens ====
286 } else { // known.size() >= 2
287 const Angle* ang1 = topo.take_angle(hs[0].ptr, &atom, known[0].ptr);
288 const Angle* ang2 = topo.take_angle(hs[0].ptr, &atom, known[1].ptr);
289 const Angle* ang3 = topo.take_angle(known[0].ptr, &atom, known[1].ptr);
290
291 if (!ang1 || !ang2)
292 giveup(cat("Missing angle restraint ", hs[0].ptr->name, '-', atom.name,
293 '-', known[ang1 ? 1 : 0].ptr->name, ".\n"));
294 double theta1 = ang1->radians();
295 double theta2 = ang2->radians();
296 if (ang3) {
297 // If all atoms are in the same plane (sum of angles is 360 degree)
298 // the calculations can be simplified.
299 double theta3 = ang3->radians();
300 Vec3 v12 = known[0].pos - atom.pos;
301 Vec3 v13 = known[1].pos - atom.pos;
302 // theta3 is the ideal restraint value, cur_theta3 is the current value
303 double cur_theta3 = v12.angle(v13);
304 constexpr double two_pi = 2 * pi();
305 if (theta1 + theta2 + std::max(theta3, cur_theta3) + 0.01 > two_pi) {
306 double ratio = (two_pi - cur_theta3) / (theta1 + theta2);
307 Vec3 axis = v13.cross(v12).normalized();
308 Vec3 v14 = rotate_about_axis(v12, axis, theta1 * ratio);
309 hs[0].pos = atom.pos + Position(hs[0].dist / v14.length() * v14);
310 return;
311 }
312 }
313 auto pos = position_from_two_angles(atom.pos, known[0].pos, known[1].pos,
314 hs[0].dist, theta1, theta2);
315 switch (hs.size()) {
316 case 1:
317 if (known.size() == 2) {
318 const Topo::Chirality* chir = topo.get_chirality(&atom);
319 if (chir && chir->restr->sign != ChiralityType::Both) {
320 hs[0].pos = chir->check() ? pos.first : pos.second;
321 } else {
322 hs[0].pos = pos.first;
323 hs[0].ptr->occ = 0;
324 }
325 } else { // known.size() > 2
326 const Atom* a3 = known[2].ptr;
327 if (const Angle* a = topo.take_angle(a3, &atom, hs[0].ptr)){
328 double val1 = calculate_angle(a3->pos, atom.pos, pos.first);
329 double val2 = calculate_angle(a3->pos, atom.pos, pos.second);
330 double diff1 = angle_abs_diff(val1, a->value);
331 double diff2 = angle_abs_diff(val2, a->value);
332 hs[0].pos = diff1 < diff2 ? pos.first : pos.second;
333 } else {
334 hs[0].pos = pos.first;
335 hs[0].ptr->occ = 0;
336 }
337 }
338 break;
339 case 2:
340 hs[0].pos = pos.first;
341 hs[1].pos = pos.second;
342 break;
343 default:
344 giveup("Unusual: atom bonded to 2+ heavy atoms and 3+ hydrogens.");
345 }
346 }
347}

349inline void adjust_hydrogen_distances(Topo& topo, Restraints::DistanceOf of,
350 double default_scale=1.) {
351 for (const Topo::Bond& t : topo.bonds) {
352 assert(t.atoms[0] != nullptr && t.atoms[1] != nullptr);
353 if (t.atoms[0]->is_hydrogen() || t.atoms[1]->is_hydrogen()) {
354 Position u = t.atoms[1]->pos - t.atoms[0]->pos;
355 double scale = t.restr->distance(of) / u.length();
356 if (std::isnan(scale))
357 scale = default_scale;
358 if (t.atoms[1]->is_hydrogen())
359 t.atoms[1]->pos = t.atoms[0]->pos + u * scale;
360 else
361 t.atoms[0]->pos = t.atoms[1]->pos - u * scale;
362 }
363 }

366inline void place_hydrogens_on_all_atoms(Topo& topo) {
367 for (Topo::ChainInfo& chain_info : topo.chain_infos)
368 for (Topo::ResInfo& ri : chain_info.res_infos)
369 for (Atom& atom : ri.res->atoms)
370 if (!atom.is_hydrogen()) {
371 try {
372 place_hydrogens(topo, atom);
373 } catch (const std::runtime_error& e) {
374 topo.err("Placing of hydrogen bonded to "
375 + atom_str(chain_info.chain_ref, *ri.res, atom)
376 + " failed:\n " + e.what());
377 }
378 }
379}

383inline std::unique_ptr<Topo>
384prepare_topology(Structure& st, MonLib& monlib, size_t model_index,
385 HydrogenChange h_change, bool reorder,
386 std::ostream* warnings=nullptr, bool ignore_unknown_links=false) {
387 std::unique_ptr<Topo> topo(new Topo);
388 topo->warnings = warnings;
389 if (model_index >= st.models.size())
390 fail("no such model index: " + std::to_string(model_index));
391 topo->initialize_refmac_topology(st, st.models[model_index], monlib, ignore_unknown_links);
392
393 bool keep = (h_change == HydrogenChange::NoChange || h_change == HydrogenChange::Shift);
394 if (!keep || reorder) {
395 // remove/add hydrogens, sort atoms in residues, assign serial numbers
396 int serial = 0;
397 for (Topo::ChainInfo& chain_info : topo->chain_infos)
398 for (Topo::ResInfo& ri : chain_info.res_infos) {
399 const ChemComp& cc = ri.chemcomp;
400 Residue& res = *ri.res;
401 if (!keep) {
402 remove_hydrogens(res);
403 if (h_change == HydrogenChange::ReAdd ||
404 (h_change == HydrogenChange::ReAddButWater && !res.is_water())) {
405 add_hydrogens_without_positions(cc, res);
406 if (h_change == HydrogenChange::ReAddButWater) {
407 // a special handling of HIS for compatibility with Refmac
408 if (cc.name == "HIS") {
409 for (gemmi::Atom& atom : ri.res->atoms)
410 if (atom.name == "HD1" || atom.name == "HE2")
411 atom.occ = 0;
412 }
413 }
414 }
415 }
416 if (reorder) {
417 for (Atom& atom : res.atoms) {
418 auto it = cc.find_atom(atom.name);
419 if (it == cc.atoms.end())
420 topo->err("definition not found for " +
421 atom_str(chain_info.chain_ref, *ri.res, atom));
422 atom.serial = int(it - cc.atoms.begin()); // temporary, for sorting only
423 }
424 std::sort(res.atoms.begin(), res.atoms.end(), [](const Atom& a, const Atom& b) {
425 return a.serial != b.serial ? a.serial < b.serial
426 : a.altloc < b.altloc;
427 });
428 }
429 for (Atom& atom : res.atoms)
430 atom.serial = ++serial;
431 }
432 }
433
434 topo->finalize_refmac_topology(monlib);
435
436 // the hydrogens added previously have positions not set
437 if (h_change != HydrogenChange::NoChange)
438 place_hydrogens_on_all_atoms(*topo);
439
440 return topo;
441}
