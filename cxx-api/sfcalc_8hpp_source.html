20// calculate part of the structure factor: exp(2 pi i r * s)
21inline std::complex<double> calculate_sf_part(const Fractional& fpos,
22 const Miller& hkl) {
23 double arg = 2 * pi() * (hkl[0]*fpos.x + hkl[1]*fpos.y + hkl[2]*fpos.z);
24 return std::complex<double>{std::cos(arg), std::sin(arg)};
25}

32 void set_stol2_and_scattering_factors(const Miller& hkl) {
33 stol2_ = cell_.calculate_stol_sq(hkl);
34 scattering_factors_.clear();
35 scattering_factors_.resize(addends.size(), 0.);

38 double get_scattering_factor(Element element) {
39 double& sfactor = scattering_factors_[element.ordinal()];
40 if (sfactor == 0.) {
41 if (!Table::has(element.elem))
42 fail("Missing scattering factor for ", element.name());
43 sfactor = Table::get(element.elem).calculate_sf(stol2_) + addends.get(element);
44 }
45 return sfactor;
46 }

48 // Calculation of Debye-Waller factor with isotropic ADPs
49 double dwf_iso(const SmallStructure::Site& site) const {
50 return std::exp(-u_to_b() * stol2_ * site.u_iso);
51 }
52 double dwf_iso(const Atom& atom) const {
53 return std::exp(-stol2_ * atom.b_iso);
54 }
55
56 // Calculation of Debye-Waller factor exp(-2 pi^2 s.U.s)
57 // cf. B. Rupp's book, p. 641 or RWGK & Adams 2002, J. Appl. Cryst. 35, 477
58 // Small molecule and macromolecular anisotropic U's are defined differently,
59 // so we have two functions.
60 double dwf_aniso(const SmallStructure::Site& site, const Vec3& hkl) const {
61 Vec3 arh(cell_.ar * hkl.x, cell_.br * hkl.y, cell_.cr * hkl.z);
62 return std::exp(-2 * pi() * pi() * site.aniso.r_u_r(arh));
63 }
64 double dwf_aniso(const Atom& atom, const Vec3& hkl) const {
65 return std::exp(-2 * pi() * pi() *
66 atom.aniso.transformed_by<>(cell_.frac.mat).r_u_r(hkl));
67 }

69 template<typename Site>
70 std::complex<double> calculate_sf_from_atom_sf(const Fractional& fract,
71 const Site& site,
72 const Miller& hkl,
73 double sf) {
74 double oc_sf = site.occ * sf;
75 std::complex<double> sum = calculate_sf_part(fract, hkl);
76 if (!site.aniso.nonzero()) {
77 for (const FTransform& image : cell_.images)
78 sum += calculate_sf_part(image.apply(fract), hkl);
79 return oc_sf * dwf_iso(site) * sum;
80 }
81 Vec3 vhkl(hkl[0], hkl[1], hkl[2]);
82 sum *= dwf_aniso(site, vhkl);
83 for (const FTransform& image : cell_.images)
84 sum += calculate_sf_part(image.apply(fract), hkl) *
85 dwf_aniso(site, image.mat.left_multiply(vhkl));
86 return oc_sf * sum;
87 }

96 std::complex<double> calculate_sf_from_model(const Model& model, const Miller& hkl) {
97 std::complex<double> sf = 0.;
98 set_stol2_and_scattering_factors(hkl);
99 for (const Chain& chain : model.chains)
100 for (const Residue& res : chain.residues)
101 for (const Atom& site : res.atoms)
102 sf += calculate_sf_from_atom(cell_.fractionalize(site.pos), site, hkl);
103 return sf;
104 }

106 // Z part of Mott-Bethe formula (when need to use different model)
107 std::complex<double> calculate_mb_z(const Model& model, const Miller& hkl, bool only_h) {
108 std::complex<double> sf = 0.;
109 stol2_ = cell_.calculate_stol_sq(hkl);
110 for (const Chain& chain : model.chains)
111 for (const Residue& res : chain.residues)
112 for (const Atom& site : res.atoms)
113 if (!only_h || site.element.is_hydrogen())
114 sf += calculate_sf_from_atom_sf(cell_.fractionalize(site.pos), site, hkl, -1.*site.element.atomic_number());
115 return sf;
116 }
117
118 double mott_bethe_factor() const {
119 return -1. / (8 * pi() * pi() * bohrradius()) / stol2_;
120 }

122 // The occupancy is assumed to take into account symmetry,
123 // i.e. to be fractional if the atom is on special position.
124 std::complex<double> calculate_sf_from_small_structure(const SmallStructure& small,
125 const Miller& hkl) {
126 std::complex<double> sf = 0.;
127 set_stol2_and_scattering_factors(hkl);
128 for (const SmallStructure::Site& site : small.sites)
129 sf += calculate_sf_from_atom(site.fract, site, hkl);
