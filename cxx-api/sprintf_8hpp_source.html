48template<int Prec>
49std::string to_str_prec(double d) {
50 static_assert(Prec >= 0 && Prec < 7, "unsupported precision");
51 char buf[16];
52 int len = d > -1e8 && d < 1e8 ? gstb_sprintf(buf, "%.*f", Prec, d)
53 : gstb_sprintf(buf, "%g", d);
54 return std::string(buf, len > 0 ? len : 0);
