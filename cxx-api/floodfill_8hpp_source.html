52 // Find all connected points with value Land. Change them to this_island().
53 Result find_all_connected_points(int u, int v, int w) {
54 Result r;
55 T* ptr = &mask.data[mask.index_q(u, v, w)];
56 r.lines.push_back(line_from_point(u, v, w, ptr));
57 set_line_values(r.lines.back(), this_island());
58 for (size_t i = 0; i < r.lines.size()/*increasing!*/; ++i) {
59 int u_ = r.lines[i].u;
60 int v_ = r.lines[i].v;
61 int w_ = r.lines[i].w;
62 int ustart = u_ != 0 ? u_ - 1 : mask.nu - 1;
63 int ulen = std::min(mask.nu, r.lines[i].ulen + 2);
64 // add adjacent lines
65 int vl = v_ != 0 ? v_ - 1 : mask.nv - 1;
66 int vh = v_ + 1 != mask.nv ? v_ + 1 : 0;
67 int wl = w_ != 0 ? w_ - 1 : mask.nw - 1;
68 int wh = w_ + 1 != mask.nw ? w_ + 1 : 0;
69 add_lines(ustart, vl, wl, ulen, r);
70 add_lines(ustart, vl, w_, ulen, r);
71 add_lines(ustart, vl, wh, ulen, r);
72 add_lines(ustart, v_, wl, ulen, r);
73 add_lines(ustart, v_, wh, ulen, r);
74 add_lines(ustart, vh, wl, ulen, r);
75 add_lines(ustart, vh, w_, ulen, r);
76 add_lines(ustart, vh, wh, ulen, r);
77 }
78 return r;
79 }

81 template<typename Func>
82 void for_each_islands(Func func) {
83 size_t idx = 0;
84 for (int w = 0; w != mask.nw; ++w)
85 for (int v = 0; v != mask.nv; ++v)
86 for (int u = 0; u != mask.nu; ++u, ++idx) {
87 assert(idx == mask.index_q(u, v, w));
88 if (mask.data[idx] == Land) {
89 // it temporarily marks current island as this_island()
90 Result r = find_all_connected_points(u, v, w);
91 func(r);
92 }
93 }
94 // set big islands (continents) back to Land
95 for (T& p : mask.data)
96 p = T((int)p & 1);
97 }

99private:
100 void add_lines(int u, int v, int w, int ulen, Result& r) {
101 T* ptr = &mask.data[mask.index_q(u, v, w)];
102 for (int i = 0; i < std::min(ulen, mask.nu - u); ++i)
103 if (ptr[i] == Land) {
104 r.lines.push_back(line_from_point(u + i, v, w, ptr + i));
105 set_line_values(r.lines.back(), this_island());
106 }
107 for (int i = -u; i < ulen - mask.nu; ++i)
108 if (ptr[i] == Land) {
109 r.lines.push_back(line_from_point(u + i, v, w, ptr + i));
110 set_line_values(r.lines.back(), this_island());
111 }
112 }

114 Line line_from_point(int u, int v, int w, T* ptr) const {
115 int len = 1;
116 while (u + len < mask.nu && ptr[len] == Land)
117 ++len;
118 if (u + len == mask.nu)
119 while (len < mask.nu && ptr[len - mask.nu] == Land)
120 ++len;
121 for (int i = 0; i > -u; --i)
122 if (ptr[i-1] != Land)
123 return {u + i, v, w, len - i, ptr + i};
124 if (ptr[mask.nu-1-u] != Land)
125 return {0, v, w, len + u, ptr - u};
126 for (int i = mask.nu - 1 - u; i > 1; --i)
127 if (ptr[i-1] != Land)
128 return {u + i, v, w, len + mask.nu - 2 - i, ptr + i};
129 return {u, v, w, mask.nu, ptr};
130 }

133inline void mask_nodes_above_threshold(Grid<std::int8_t>& mask, const Grid<float>& grid,
134 double threshold, bool negate=false) {
135 mask.copy_metadata_from(grid);
136 mask.data.resize(grid.data.size());
137 size_t n = 0;
138 for (float d : grid.data)
139 mask.data[n++] = std::int8_t((negate ? -d : d) > threshold);
140}

142inline Grid<std::int8_t> flood_fill_above(const Grid<float>& grid,
143 const std::vector<Position>& seeds,
144 double threshold,
145 bool negate=false) {
146 grid.check_not_empty();
147 Grid<std::int8_t> mask;
148 mask_nodes_above_threshold(mask, grid, threshold, negate);
149 FloodFill<std::int8_t, 1> flood_fill{mask};
150 for (const Position& pos : seeds) {
151 auto point = mask.get_nearest_point(pos);
152 if (*point.value == 1)
153 flood_fill.find_all_connected_points(point.u, point.v, point.w);
154 }
155 for (std::int8_t& d : mask.data)
156 d = std::int8_t(d == flood_fill.this_island());
157 return mask;
158}
