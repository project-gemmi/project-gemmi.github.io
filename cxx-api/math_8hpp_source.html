101struct Mat33 {
102 double a[3][3] = { {1.,0.,0.}, {0.,1.,0.}, {0.,0.,1.} };
103
104 // make it accessible with ".a"
105 typedef double row_t[3];
106 const row_t& operator[](int i) const { return a[i]; }
107 row_t& operator[](int i) { return a[i]; }
108
109 Mat33() = default;
110 explicit Mat33(double d) : a{{d, d, d}, {d, d, d}, {d, d, d}} {}
111 Mat33(double a1, double a2, double a3, double b1, double b2, double b3,
112 double c1, double c2, double c3)
113 : a{{a1, a2, a3}, {b1, b2, b3}, {c1, c2, c3}} {}

115 Vec3 row_copy(int i) const {
116 if (i < 0 || i > 2)
117 throw std::out_of_range("Mat33 row index must be 0, 1 or 2.");
118 return Vec3(a[i][0], a[i][1], a[i][2]);
119 }
120
121 Vec3 column_copy(int i) const {
122 if (i < 0 || i > 2)
123 throw std::out_of_range("Mat33 column index must be 0, 1 or 2.");
124 return Vec3(a[0][i], a[1][i], a[2][i]);
125 }
126
127 Vec3 multiply(const Vec3& p) const {
128 return {a[0][0] * p.x + a[0][1] * p.y + a[0][2] * p.z,
129 a[1][0] * p.x + a[1][1] * p.y + a[1][2] * p.z,
130 a[2][0] * p.x + a[2][1] * p.y + a[2][2] * p.z};
131 }
132 Vec3 left_multiply(const Vec3& p) const {
133 return {a[0][0] * p.x + a[1][0] * p.y + a[2][0] * p.z,
134 a[0][1] * p.x + a[1][1] * p.y + a[2][1] * p.z,
135 a[0][2] * p.x + a[1][2] * p.y + a[2][2] * p.z};
136 }
137 // p has elements from the main diagonal of a 3x3 diagonal matrix
138 Mat33 multiply_by_diagonal(const Vec3& p) const {
139 return Mat33(a[0][0] * p.x, a[0][1] * p.y, a[0][2] * p.z,
140 a[1][0] * p.x, a[1][1] * p.y, a[1][2] * p.z,
141 a[2][0] * p.x, a[2][1] * p.y, a[2][2] * p.z);
142 }
143 Mat33 multiply(const Mat33& b) const {
144 Mat33 r;
145 for (int i = 0; i != 3; ++i)
146 for (int j = 0; j != 3; ++j)
147 r[i][j] = a[i][0] * b[0][j] + a[i][1] * b[1][j] + a[i][2] * b[2][j];
148 return r;
149 }
150 Mat33 transpose() const {
151 return Mat33(a[0][0], a[1][0], a[2][0],
152 a[0][1], a[1][1], a[2][1],
153 a[0][2], a[1][2], a[2][2]);
154 }
155 double trace() const { return a[0][0] + a[1][1] + a[2][2]; }
156
157 bool approx(const Mat33& other, double epsilon) const {
158 for (int i = 0; i < 3; ++i)
159 for (int j = 0; j < 3; ++j)
160 if (std::fabs(a[i][j] - other.a[i][j]) > epsilon)
161 return false;
162 return true;
163 }
164 double determinant() const {
165 return a[0][0] * (a[1][1]*a[2][2] - a[2][1]*a[1][2]) +
166 a[0][1] * (a[1][2]*a[2][0] - a[2][2]*a[1][0]) +
167 a[0][2] * (a[1][0]*a[2][1] - a[2][0]*a[1][1]);
168 }
169 Mat33 inverse() const {
170 Mat33 inv;
171 double inv_det = 1.0 / determinant();
172 inv[0][0] = inv_det * (a[1][1] * a[2][2] - a[2][1] * a[1][2]);
173 inv[0][1] = inv_det * (a[0][2] * a[2][1] - a[0][1] * a[2][2]);
174 inv[0][2] = inv_det * (a[0][1] * a[1][2] - a[0][2] * a[1][1]);
175 inv[1][0] = inv_det * (a[1][2] * a[2][0] - a[1][0] * a[2][2]);
176 inv[1][1] = inv_det * (a[0][0] * a[2][2] - a[0][2] * a[2][0]);
177 inv[1][2] = inv_det * (a[1][0] * a[0][2] - a[0][0] * a[1][2]);
178 inv[2][0] = inv_det * (a[1][0] * a[2][1] - a[2][0] * a[1][1]);
179 inv[2][1] = inv_det * (a[2][0] * a[0][1] - a[0][0] * a[2][1]);
180 inv[2][2] = inv_det * (a[0][0] * a[1][1] - a[1][0] * a[0][1]);
181 return inv;
182 }
183 bool is_identity() const {
184 return a[0][0] == 1 && a[0][1] == 0 && a[0][2] == 0 &&
185 a[1][0] == 0 && a[1][1] == 1 && a[1][2] == 0 &&
186 a[2][0] == 0 && a[2][1] == 0 && a[2][2] == 1;
187 }
188
189 double column_dot(int i, int j) const {
190 return a[0][i] * a[0][j] + a[1][i] * a[1][j] + a[2][i] * a[2][j];
191 }

194// Symmetric matrix 3x3. Used primarily for an ADP tensor.
195template<typename T> struct SMat33 {
196 T u11, u22, u33, u12, u13, u23;
197
198 // The PDB ANISOU record has the above order, but in a different context
199 // (such as metric tensor) the order of Voigt notation may be preferred.
200 std::array<T, 6> elements_pdb() const { return {{u11, u22, u33, u12, u13, u23}}; }
201 std::array<T, 6> elements_voigt() const { return {{u11, u22, u33, u23, u13, u12}}; }
202
203 Mat33 as_mat33() const {
204 return Mat33(u11, u12, u13, u12, u22, u23, u13, u23, u33);
205 }
206
207 T trace() const { return u11 + u22 + u33; }
208 bool nonzero() const { return trace() != 0; }
209
210 bool all_zero() const {
211 return u11 == 0 && u22 == 0 && u33 == 0 && u12 == 0 && u13 == 0 && u23 == 0;
212 }
213
214 void scale(T s) const {
215 u11 *= s; u22 *= s; u33 *= s; u12 *= s; u13 *= s; u23 *= s;
216 };
217
218 template<typename Real>
219 SMat33<Real> scaled(Real s) const {
220 return SMat33<Real>{u11*s, u22*s, u33*s, u12*s, u13*s, u23*s};
221 }
222
223 // returns U + kI
224 SMat33<T> added_kI(T k) const {
225 return {u11+k, u22+k, u33+k, u12, u13, u23};
226 }
227
228 // returns squared norm r^T U r where U is this matrix and vector r is arg
229 double r_u_r(const Vec3& r) const {
230 return r.x * r.x * u11 + r.y * r.y * u22 + r.z * r.z * u33 +
231 2 * (r.x * r.y * u12 + r.x * r.z * u13 + r.y * r.z * u23);
232 }
233 double r_u_r(const std::array<int,3>& h) const {
234 // it's faster to first convert ints to doubles (Vec3)
235 return r_u_r(Vec3(h));
236 }
237
238 Vec3 multiply(const Vec3& p) const {
239 return {u11 * p.x + u12 * p.y + u13 * p.z,
240 u12 * p.x + u22 * p.y + u23 * p.z,
241 u13 * p.x + u23 * p.y + u33 * p.z};
242 }
243
244 SMat33 operator-(const SMat33& o) const {
245 return {u11-o.u11, u22-o.u22, u33-o.u33, u12-o.u12, u13-o.u13, u23-o.u23};
246 }
247 SMat33 operator+(const SMat33& o) const {
248 return {u11+o.u11, u22+o.u22, u33+o.u33, u12+o.u12, u13+o.u13, u23+o.u23};
249 }

251 // return M U M^T
252 template<typename Real=double>
253 SMat33<Real> transformed_by(const Mat33& m) const {
254 // slightly faster than m.multiply(as_mat33()).multiply(m.transpose());
255 auto elem = [&](int i, int j) {
256 return static_cast<Real>(
257 m[i][0] * (m[j][0] * u11 + m[j][1] * u12 + m[j][2] * u13) +
258 m[i][1] * (m[j][0] * u12 + m[j][1] * u22 + m[j][2] * u23) +
259 m[i][2] * (m[j][0] * u13 + m[j][1] * u23 + m[j][2] * u33));
260 };
261 return SMat33<Real>{elem(0, 0), elem(1, 1), elem(2, 2),
262 elem(0, 1), elem(0, 2), elem(1, 2)};
263 }
264
265 T determinant() const {
266 return u11 * (u22*u33 - u23*u23) +
267 u12 * (u23*u13 - u33*u12) +
268 u13 * (u12*u23 - u13*u22);
269 }
270
271 SMat33 inverse() const {
272 SMat33 inv;
273 T inv_det = 1.0f / determinant();
274 inv.u11 = inv_det * (u22 * u33 - u23 * u23);
275 inv.u22 = inv_det * (u11 * u33 - u13 * u13);
276 inv.u33 = inv_det * (u11 * u22 - u12 * u12);
277 inv.u12 = inv_det * (u13 * u23 - u12 * u33);
278 inv.u13 = inv_det * (u12 * u23 - u13 * u22);
279 inv.u23 = inv_det * (u12 * u13 - u11 * u23);
280 return inv;
281 }

283 // Based on https://en.wikipedia.org/wiki/Eigenvalue_algorithm
284 std::array<double, 3> calculate_eigenvalues() const {
285 double p1 = u12*u12 + u13*u13 + u23*u23;
286 if (p1 == 0)
287 return {{u11, u22, u33}};
288 double q = (1./3.) * trace();
289 SMat33<double> b{u11 - q, u22 - q, u33 - q, u12, u13, u23};
290 double p2 = sq(b.u11) + sq(b.u22) + sq(b.u33) + 2 * p1;
291 double p = std::sqrt((1./6.) * p2);
292 double r = b.determinant() / ((1./3.) * p2 * p);
293 double phi = 0;
294 if (r <= -1)
295 phi = (1./3.) * pi();
296 else if (r < 1)
297 phi = (1./3.) * std::acos(r);
298 double eig1 = q + 2 * p * std::cos(phi);
299 double eig3 = q + 2 * p * std::cos(phi + 2./3.*pi());
300 return {{eig1, 3 * q - eig1 - eig3, eig3}};
301 }
302
303 // Assumes one of the eigenvalue calculate above.
304 // May not work if eigenvalues are not distinct.
305 Vec3 calculate_eigenvector(double eigenvalue) const {
306 Vec3 r0(u11 - eigenvalue, u12, u13);
307 Vec3 r1(u12, u22 - eigenvalue, u23);
308 Vec3 r2(u13, u23, u33 - eigenvalue);
309 Vec3 cr[3] = {r0.cross(r1), r0.cross(r2), r1.cross(r2)};
310 int idx = 0;
311 double lensq = 0;
312 for (int i = 0; i < 3; ++i) {
313 double tmp = cr[i].length_sq();
314 if (tmp > lensq) {
315 idx = i;
316 lensq = tmp;
317 }
318 }
319 if (lensq == 0)
320 return Vec3(0, 0, 1); // an arbitrary choice for the special case
321 return cr[idx] / std::sqrt(lensq);
322 }

367// popular single-pass algorithm for calculating variance and mean
368struct Variance {
369 int n = 0;
370 double sum_sq = 0.;
371 double mean_x = 0.;
372
373 Variance() = default;
374 template <typename T> Variance(T begin, T end) : Variance() {
375 for (auto i = begin; i != end; ++i)
376 add_point(*i);
377 }
378 void add_point(double x) {
379 ++n;
380 double dx = x - mean_x;
381 mean_x += dx / n;
382 sum_sq += dx * (x - mean_x);
383 }
384 double for_sample() const { return sum_sq / (n - 1); }
385 double for_population() const { return sum_sq / n; }
386};
387
388struct Covariance : Variance {
389 double mean_y = 0.;
390 void add_point(double x, double y) {
391 ++n;
392 double dx = x - mean_x;
393 mean_x += dx / n;
394 mean_y += (y - mean_y) / n;
395 sum_sq += dx * (y - mean_y);
396 }
397};

399struct Correlation {
400 int n = 0;
401 double sum_xx = 0.;
402 double sum_yy = 0.;
403 double sum_xy = 0.;
404 double mean_x = 0.;
405 double mean_y = 0.;
406 void add_point(double x, double y) {
407 ++n;
408 double weight = (double)(n - 1) / n;
409 double dx = x - mean_x;
410 double dy = y - mean_y;
411 sum_xx += weight * dx * dx;
412 sum_yy += weight * dy * dy;
413 sum_xy += weight * dx * dy;
414 mean_x += dx / n;
415 mean_y += dy / n;
416 }
417 double coefficient() const { return sum_xy / std::sqrt(sum_xx * sum_yy); }
418 double x_variance() const { return sum_xx / n; }
419 double y_variance() const { return sum_yy / n; }
420 double covariance() const { return sum_xy / n; }
421 double mean_ratio() const { return mean_y / mean_x; }
422 // the regression line
423 double slope() const { return sum_xy / sum_xx; }
424 double intercept() const { return mean_y - slope() * mean_x; }
425};

436template<typename T>
437DataStats calculate_data_statistics(const std::vector<T>& data) {
438 DataStats stats;
439 double sum = 0;
440 double sq_sum = 0;
441 stats.dmin = INFINITY;
442 stats.dmax = -INFINITY;
443 for (double d : data) {
444 if (std::isnan(d)) {
445 stats.nan_count++;
446 continue;
447 }
448 sum += d;
449 sq_sum += d * d;
450 if (d < stats.dmin)
451 stats.dmin = d;
452 if (d > stats.dmax)
453 stats.dmax = d;
454 }
455 if (stats.nan_count != data.size()) {
456 stats.dmean = sum / (data.size() - stats.nan_count);
457 stats.rms = std::sqrt(sq_sum / (data.size() - stats.nan_count) - stats.dmean * stats.dmean);
458 } else {
459 stats.dmin = NAN;
460 stats.dmax = NAN;
461 }
462 return stats;
463}
