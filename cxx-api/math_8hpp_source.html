37inline double log_cosh(double x) {
38 // cosh(x) would overflow for x > 710.5, so we calculate:
39 // ln(cosh(x)) = ln(e^x + e^-x) - ln(2) = ln(e^x * (1 + e^-2x)) - ln(2)
40 x = std::abs(x);
41 return x - std::log(2) + std::log1p(std::exp(-2 * x));
42}

46inline double angle_abs_diff(double a, double b, double full=360.0) {
47 double d = std::fabs(a - b);
48 if (d > full)
49 d -= std::floor(d / full) * full;
50 return std::min(d, full - d);
51}

208 double determinant() const {
209 return a[0][0] * (a[1][1]*a[2][2] - a[2][1]*a[1][2]) +
210 a[0][1] * (a[1][2]*a[2][0] - a[2][2]*a[1][0]) +

213 Mat33 inverse() const {
214 Mat33 inv;
215 double inv_det = 1.0 / determinant();
216 inv[0][0] = inv_det * (a[1][1] * a[2][2] - a[2][1] * a[1][2]);
217 inv[0][1] = inv_det * (a[0][2] * a[2][1] - a[0][1] * a[2][2]);
218 inv[0][2] = inv_det * (a[0][1] * a[1][2] - a[0][2] * a[1][1]);
219 inv[1][0] = inv_det * (a[1][2] * a[2][0] - a[1][0] * a[2][2]);
220 inv[1][1] = inv_det * (a[0][0] * a[2][2] - a[0][2] * a[2][0]);
221 inv[1][2] = inv_det * (a[1][0] * a[0][2] - a[0][0] * a[1][2]);
222 inv[2][0] = inv_det * (a[1][0] * a[2][1] - a[2][0] * a[1][1]);
223 inv[2][1] = inv_det * (a[2][0] * a[0][1] - a[0][0] * a[2][1]);
224 inv[2][2] = inv_det * (a[0][0] * a[1][1] - a[1][0] * a[0][1]);
225 return inv;
226 }

267// Symmetric matrix 3x3. Used primarily for an ADP tensor.
268template<typename T> struct SMat33 {
269 T u11, u22, u33, u12, u13, u23;
270
271 // The PDB ANISOU record has the above order, but in a different context
272 // (such as metric tensor) the order of Voigt notation may be preferred.
273 std::array<T, 6> elements_pdb() const { return {{u11, u22, u33, u12, u13, u23}}; }
274 std::array<T, 6> elements_voigt() const { return {{u11, u22, u33, u23, u13, u12}}; }
275
276 Mat33 as_mat33() const {
277 return Mat33(u11, u12, u13, u12, u22, u23, u13, u23, u33);
278 }
279
280 // the arguments i and j must be in [0,2], i.e. 0, 1 or 2.
281 T& unchecked_ref(int i, int j) {
282 T* ptrs[9] = {&u11, &u12, &u13, &u12, &u22, &u23, &u13, &u23, &u33};
283 return *ptrs[3 * i + j];
284 }

286 T trace() const { return u11 + u22 + u33; }
287 bool nonzero() const { return trace() != 0; }

308 template<typename VT>

310 return r.x * r.x * u11 + r.y * r.y * u22 + r.z * r.z * u33 +

331 // return M U M^T
332 template<typename Real=double>
333 SMat33<Real> transformed_by(const Mat33& m) const {
334 // slightly faster than m.multiply(as_mat33()).multiply(m.transpose());
335 auto elem = [&](int i, int j) {
336 return static_cast<Real>(
337 m[i][0] * (m[j][0] * u11 + m[j][1] * u12 + m[j][2] * u13) +
338 m[i][1] * (m[j][0] * u12 + m[j][1] * u22 + m[j][2] * u23) +
339 m[i][2] * (m[j][0] * u13 + m[j][1] * u23 + m[j][2] * u33));
340 };
341 return SMat33<Real>{elem(0, 0), elem(1, 1), elem(2, 2),
342 elem(0, 1), elem(0, 2), elem(1, 2)};
343 }

345 T determinant() const {
346 return u11 * (u22*u33 - u23*u23) +
347 u12 * (u23*u13 - u33*u12) +
348 u13 * (u12*u23 - u13*u22);
349 }
350
351 SMat33 inverse_(T det) const {
352 SMat33 inv;
353 T inv_det = 1.0f / det;
354 inv.u11 = inv_det * (u22 * u33 - u23 * u23);
355 inv.u22 = inv_det * (u11 * u33 - u13 * u13);
356 inv.u33 = inv_det * (u11 * u22 - u12 * u12);
357 inv.u12 = inv_det * (u13 * u23 - u12 * u33);
358 inv.u13 = inv_det * (u12 * u23 - u13 * u22);
359 inv.u23 = inv_det * (u12 * u13 - u11 * u23);
360 return inv;
361 }
362 SMat33 inverse() const {
363 return inverse_(determinant());
364 }

368 std::array<double, 3> calculate_eigenvalues() const {
369 double p1 = u12*u12 + u13*u13 + u23*u23;
370 if (p1 == 0)
371 return {{u11, u22, u33}};
372 double q = (1./3.) * trace();
373 SMat33<double> b{u11 - q, u22 - q, u33 - q, u12, u13, u23};
374 double p2 = sq(b.u11) + sq(b.u22) + sq(b.u33) + 2 * p1;
375 double p = std::sqrt((1./6.) * p2);
376 double r = b.determinant() / ((1./3.) * p2 * p);
377 double phi = 0;
378 if (r <= -1)
379 phi = (1./3.) * pi();
380 else if (r < 1)
381 phi = (1./3.) * std::acos(r);
382 double eig1 = q + 2 * p * std::cos(phi);
383 double eig3 = q + 2 * p * std::cos(phi + 2./3.*pi());
384 return {{eig1, 3 * q - eig1 - eig3, eig3}};
385 }
