22// strip directory and suffixes from filename
23inline std::string path_basename(const std::string& path,
24 std::initializer_list<const char*> exts) {
25 size_t pos = path.find_last_of("\\/");
26 std::string basename = pos == std::string::npos ? path : path.substr(pos + 1);
27 for (const char* ext : exts) {
28 size_t len = std::strlen(ext);
29 if (basename.size() > len &&
30 basename.compare(basename.length() - len, len, ext, len) == 0)
31 basename.resize(basename.length() - len);
32 }
33 return basename;
34}

39struct needs_fclose {
40 bool use_fclose;
41 void operator()(std::FILE* f) const noexcept {

49inline fileptr_t file_open(const char* path, const char* mode) {
50 std::FILE* file;
51#if defined(_WIN32) && !defined(GEMMI_USE_FOPEN)
52 std::wstring wpath = UTF8_to_wchar(path);
53 std::wstring wmode = UTF8_to_wchar(mode);
54 if ((file = ::_wfopen(wpath.c_str(), wmode.c_str())) == nullptr)
55#else
56 if ((file = std::fopen(path, mode)) == nullptr)
57#endif
58 sys_fail(std::string("Failed to open ") + path +
59 (*mode == 'w' ? " for writing" : ""));
60 return fileptr_t(file, needs_fclose{true});
61}

71inline std::size_t file_size(std::FILE* f, const std::string& path) {
72 if (std::fseek(f, 0, SEEK_END) != 0)
73 sys_fail(path + ": fseek failed");
74 long length = std::ftell(f);
75 if (length < 0)
76 sys_fail(path + ": ftell failed");
77 if (std::fseek(f, 0, SEEK_SET) != 0)
78 sys_fail(path + ": fseek failed");
79 return length;
80}

107// reading file into a memory buffer
108inline CharArray read_file_into_buffer(const std::string& path) {
109 fileptr_t f = file_open(path.c_str(), "rb");
110 size_t size = file_size(f.get(), path);
111 CharArray buffer(size);
112 if (std::fread(buffer.data(), size, 1, f.get()) != 1)
113 sys_fail(path + ": fread failed");
114 return buffer;
115}

117inline CharArray read_stdin_into_buffer() {
118 size_t n = 0;
119 CharArray buffer(16 * 1024);
120 for (;;) {
121 n += std::fread(buffer.data() + n, 1, buffer.size() - n, stdin);
122 if (n != buffer.size()) {
123 buffer.set_size(n);
124 break;
125 }
126 buffer.resize(2*n);
127 }
128 return buffer;
129}
