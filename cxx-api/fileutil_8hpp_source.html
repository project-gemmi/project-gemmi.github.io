25// strip directory and suffixes from filename
26inline std::string path_basename(const std::string& path,
27 std::initializer_list<const char*> exts) {
28 size_t pos = path.find_last_of("\\/");
29 std::string basename = pos == std::string::npos ? path : path.substr(pos + 1);
30 for (const char* ext : exts) {
31 size_t len = std::strlen(ext);
32 if (basename.size() > len &&
33 basename.compare(basename.length() - len, len, ext, len) == 0)
34 basename.resize(basename.length() - len);
35 }
36 return basename;
37}

40typedef std::unique_ptr<std::FILE, decltype(&std::fclose)> fileptr_t;

42inline fileptr_t file_open(const char* path, const char* mode) {
43 std::FILE* file;
44#if defined(_WIN32) && !defined(GEMMI_USE_FOPEN)
45 std::wstring wpath = UTF8_to_wchar(path);
46 std::wstring wmode = UTF8_to_wchar(mode);
47 if ((file = ::_wfopen(wpath.c_str(), wmode.c_str())) == nullptr)
48#else
49 if ((file = std::fopen(path, mode)) == nullptr)
50#endif
51 sys_fail(std::string("Failed to open ") + path +
52 (*mode == 'w' ? " for writing" : ""));
53 return fileptr_t(file, &std::fclose);
54}

64inline std::size_t file_size(std::FILE* f, const std::string& path) {
65 if (std::fseek(f, 0, SEEK_END) != 0)
66 sys_fail(path + ": fseek failed");
67 long length = std::ftell(f);
68 if (length < 0)
69 sys_fail(path + ": ftell failed");
70 if (std::fseek(f, 0, SEEK_SET) != 0)
71 sys_fail(path + ": fseek failed");
72 return length;
73}

80// Call it after checking the code with gemmi::is_pdb_code(code).
81// The convention for $PDB_DIR is the same as in BioJava, see the docs.
82// type is the requested file type: 'M' for mmCIF or 'P' for PDB, 'S' for SF-mmCIF.
83inline std::string expand_pdb_code_to_path(const std::string& code, char type) {
84 std::string path;
85 if (const char* pdb_dir = std::getenv("PDB_DIR")) {
86 int n = 0;
87 if (type == 'M')
88 n = 1;
89 else if (type == 'S')
90 n = 2;
91 std::string lc = to_lower(code);
92 path = pdb_dir;
93 path += "/structures/divided/";
94 const char* dir[] = {"pdb/", "mmCIF/", "structure_factors/"};
95 path += dir[n];
96 path += lc.substr(1, 2);
97 const char* prefix[] = {"/pdb", "/", "/r"};
98 path += prefix[n];
99 path += lc;
100 const char* suffix[] = {".ent.gz", ".cif.gz", "sf.ent.gz"};
101 path += suffix[n];
102 }
103 return path;
104}

106// type must be 'M' for mmCIF or 'P' for PDB
107inline std::string expand_if_pdb_code(const std::string& input, char type='M') {
108 std::string path;
109 if (is_pdb_code(input)) {
110 path = gemmi::expand_pdb_code_to_path(input, type);
111 if (path.empty())
112 fail(input + " is a PDB code, but $PDB_DIR is not set.");
113 } else {
114 path = input;
115 }
116 return path;

144// reading file into a memory buffer
145inline CharArray read_file_into_buffer(const std::string& path) {
146 fileptr_t f = file_open(path.c_str(), "rb");
147 size_t size = file_size(f.get(), path);
148 CharArray buffer(size);
149 if (std::fread(buffer.data(), size, 1, f.get()) != 1)
150 sys_fail(path + ": fread failed");
151 return buffer;
152}

154inline CharArray read_stdin_into_buffer() {
155 size_t n = 0;
156 CharArray buffer(16 * 1024);
157 for (;;) {
158 n += std::fread(buffer.data() + n, 1, buffer.size() - n, stdin);
159 if (n != buffer.size()) {
160 buffer.set_size(n);
161 break;
162 }
163 buffer.resize(2*n);
164 }
165 return buffer;
166}
