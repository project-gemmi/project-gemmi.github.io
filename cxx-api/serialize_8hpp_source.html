160template <typename Archive>
161void serialize(Archive& archive, Item& o) {
162 archive(o.type, o.line_number);
163 switch (o.type) {
164 case ItemType::Pair:
165 case ItemType::Comment: new(&o.pair) cif::Pair; archive(o.pair); break;
166 case ItemType::Loop: new(&o.loop) cif::Loop; archive(o.loop); break;
167 case ItemType::Frame: new(&o.frame) cif::Block; archive(o.frame); break;
168 case ItemType::Erased: break;
169 }
170}
171template <typename Archive>
172void serialize(Archive& archive, const Item& o) {
173 archive(o.type, o.line_number);
174 switch (o.type) {
175 case ItemType::Pair:
176 case ItemType::Comment: archive(o.pair); break;
177 case ItemType::Loop: archive(o.loop); break;
178 case ItemType::Frame: archive(o.frame); break;
179 case ItemType::Erased: break;
180 }
181}
