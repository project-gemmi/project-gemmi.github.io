48#define WRITE(...) do { \
49 gf_snprintf(buf, 82, __VA_ARGS__); \
50 buf[80] = '\n'; \
51 os.write(buf, 81); \
52 } while(0)
53
54#define WRITEU(...) do { \
55 gf_snprintf(buf, 82, __VA_ARGS__); \
56 buf[80] = '\n'; \
57 for (int i_ = 0; i_ != 80; i_++) \
58 if (buf[i_] >= 'a' && buf[i_] <= 'z') buf[i_] -= 0x20; \
59 os.write(buf, 81); \
60 } while(0)
61
62#define WRITELN(...) do { \
63 int length__ = gf_snprintf(buf, 82, __VA_ARGS__); \
64 if (length__ < 80) \
65 std::memset(buf + length__, ' ', 80 - length__); \
66 buf[80] = '\n'; \
67 os.write(buf, 81); \
68 } while(0)

72bool use_hetatm(const Residue& res) {
73 if (res.het_flag == 'H')
74 return true;
75 if (res.het_flag == 'A')
76 return false;
77 if (res.entity_type == EntityType::Branched ||
78 res.entity_type == EntityType::NonPolymer ||
79 res.entity_type == EntityType::Water)
80 return true;
81 return !find_tabulated_residue(res.name).is_standard();
82}

85inline char *base36_encode(char* buffer, int width, int value) {
86 const char base36[] = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ";
87 buffer[width] = '\0';
88 do {
89 buffer[--width] = base36[value % 36];
90 value /= 36;
91 } while (value != 0 && width != 0);
92 while (width != 0)
93 buffer[--width] = ' ';
94 return buffer;
95}

97// based on http://cci.lbl.gov/hybrid_36/
98inline char* encode_serial_in_hybrid36(char* str, int serial) {
99 assert(serial >= 0);
100 if (serial < 100000) {
101 gstb_sprintf(str, "%5d", serial);
102 return str;
103 }
104 return base36_encode(str, 5, serial - 100000 + 10 * 36 * 36 * 36 * 36);
105}

107// based on http://cci.lbl.gov/hybrid_36/
108inline char* encode_seq_num_in_hybrid36(char* str, int seq_id) {
109 if (seq_id > -1000 && seq_id < 10000) {
110 gstb_sprintf(str, "%4d", seq_id);
111 return str;
112 }
113 return base36_encode(str, 4, seq_id - 10000 + 10 * 36 * 36 * 36);
114}

116inline char* write_seq_id(char* str, const SeqId& seqid) {
117 encode_seq_num_in_hybrid36(str, *seqid.num);
118 str[4] = seqid.icode;
119 str[5] = '\0';
120 return str;
121}

123inline const char* find_last_break(const char *str, int max_len) {
124 int last_break = 0;
125 for (int i = 0; i < max_len; i++) {
126 if (str[i] == '\0')
127 return str + i;
128 if (str[i] == ' ' || str[i] == '-')
129 last_break = i + 1;
130 }
131 return str + (last_break != 0 ? last_break : max_len);
132}

134// Write record with possible continuation lines, with the format:
135// 1-6 record name, 8-10 continuation, 11-lastcol string.
136inline void write_multiline(std::ostream& os, const char* record_name,
137 const std::string& text, int lastcol) {
138 if (text.empty())
139 return;
140 char buf[88]; // a few bytes extra, just in case
141 const char *start = text.c_str();
142 const char *end = find_last_break(start, lastcol-10);
143 WRITEU("%-6s %-70.*s", record_name, static_cast<int>(end-start), start);
144 for (int n = 2; n < 1000 && *end != '\0'; ++n) {
145 start = end;
146 end = find_last_break(start, lastcol-11);
147 int len = int(end - start);
148 WRITEU("%-6s %3d %-69.*s", record_name, n, len, start);
149 }
150}

181inline void write_remarks(const Structure& st, std::ostream& os) {
182 char buf[88];
183 if (st.resolution > 0) {
184 WRITE("%-80s", "REMARK 2");
185 WRITE("REMARK 2 RESOLUTION. %7.2f %-49s", st.resolution, "ANGSTROMS.");
186 }
187 if (!st.assemblies.empty()) {
188 const char* preface[] = {
189 "REMARK 350",
190 "REMARK 350 COORDINATES FOR A COMPLETE MULTIMER REPRESENTING THE KNOWN",
191 "REMARK 350 BIOLOGICALLY SIGNIFICANT OLIGOMERIZATION STATE OF THE",
192 "REMARK 350 MOLECULE CAN BE GENERATED BY APPLYING BIOMT TRANSFORMATIONS",
193 "REMARK 350 GIVEN BELOW. BOTH NON-CRYSTALLOGRAPHIC AND",
194 "REMARK 350 CRYSTALLOGRAPHIC OPERATIONS ARE GIVEN."
195 };
196 for (const char* line : preface)
197 WRITE("%-80s", line);
198 int counter = 0;
199 for (const Assembly& assem : st.assemblies) {
200 WRITE("%-80s", "REMARK 350");
201 WRITE("REMARK 350 BIOMOLECULE: %-56d", ++counter);
202 if (assem.author_determined)
203 WRITEU("REMARK 350 AUTHOR DETERMINED BIOLOGICAL UNIT: %-34s",
204 assem.oligomeric_details.c_str());
205 if (assem.software_determined) {
206 WRITEU("REMARK 350 SOFTWARE DETERMINED QUATERNARY STRUCTURE: %-27s",
207 assem.oligomeric_details.c_str());
208 if (!assem.software_name.empty())
209 WRITEU("REMARK 350 SOFTWARE USED: %-54s",
210 assem.software_name.c_str());
211 if (!std::isnan(assem.absa))
212 WRITELN("REMARK 350 TOTAL BURIED SURFACE AREA: %.0f ANGSTROM**2",
213 assem.absa);
214 if (!std::isnan(assem.ssa))
215 WRITELN("REMARK 350 SURFACE AREA OF THE COMPLEX: %.0f ANGSTROM**2",
216 assem.ssa);
217 if (!std::isnan(assem.more))
218 WRITELN("REMARK 350 CHANGE IN SOLVENT FREE ENERGY: %.1f KCAL/MOL",
219 assem.more);
220 }
221 int oper_cnt = 0;
222 for (const Assembly::Gen& gen : assem.generators) {
223 std::string chains_str;
224 if (!gen.chains.empty()) {
225 chains_str = join_str(gen.chains, ", ");
226 } else {
227 // subchains -> chains
228 std::vector<std::string> chains;
229 for (const Chain& ch : st.first_model().chains)
230 if (!ch.residues.empty() && !in_vector(ch.name, chains) &&
231 in_vector(ch.residues[0].subchain, gen.subchains))
232 chains.push_back(ch.name);
233 chains_str = join_str(chains, ", ");
234 }
235 size_t end = chains_str.length();
236 if (end >= 30)
237 end = chains_str.rfind(' ', 29);
238 WRITELN("REMARK 350 APPLY THE FOLLOWING TO CHAINS: %s",
239 chains_str.substr(0, end).c_str());
240 while (end < chains_str.length()) {
241 size_t begin = end + 1;
242 end = chains_str.length();
243 if (end - begin >= 30)
244 end = chains_str.rfind(' ', begin + 29);
245 WRITELN("REMARK 350 AND CHAINS: %s",
246 chains_str.substr(begin, end - begin).c_str());
247 }
248 for (const Assembly::Operator& oper : gen.operators) {
249 ++oper_cnt;
250 const Transform& tr = oper.transform;
251 for (int i = 0; i < 3; ++i)
252 WRITE("REMARK 350 "
253 "BIOMT%d %3d%10.6f%10.6f%10.6f %14.5f ",
254 i+1, oper_cnt,
255 tr.mat[i][0], tr.mat[i][1], tr.mat[i][2], tr.vec.at(i));
256 }
257 }
258 }
259 }

262inline void write_chain_atoms(const Chain& chain, std::ostream& os,
263 int& serial, PdbWriteOptions opt) {
264 char buf[88];
265 char buf8[8];
266 char buf8a[8];
267 buf[0] = '\0';
268 if (chain.name.length() > 2)
269 fail("long chain name: " + chain.name);
270 for (const Residue& res : chain.residues) {
271 bool as_het = use_hetatm(res);
272 for (const Atom& a : res.atoms) {
273 // 1- 6 6s record name
274 // 7-11 5d integer serial
275 // 12 1 -
276 // 13-16 4s atom name (from 13 only if 4-char or 2-char symbol)
277 // 17 1c altloc
278 // 18-20 3s residue name
279 // 21 1 -
280 // 22 1s chain
281 // 23-26 4d integer residue sequence number
282 // 27 1c insertion code
283 // 28-30 3 -
284 // 31-38 8f x (8.3)
285 // 39-46 8f y
286 // 47-54 8f z
287 // 55-60 6f occupancy (6.2)
288 // 61-66 6f temperature factor (6.2)
289 // 67-76 6 -
290 // 73-76 segment identifier, left-justified (non-standard)
291 // 77-78 2s element symbol, right-justified
292 // 79-80 2s charge
293 WRITE("%-6s%5s %-4.4s%c%3s"
294 "%2s%5s %8.3f%8.3f%8.3f"
295 "%6.2f%6.2f %-4.4s%2s%c%c",
296 as_het ? "HETATM" : "ATOM",
297 impl::encode_serial_in_hybrid36(buf8, ++serial),
298 a.padded_name().c_str(),
299 a.altloc ? std::toupper(a.altloc) : ' ',
300 res.name.c_str(),
301 chain.name.c_str(),
302 impl::write_seq_id(buf8a, res.seqid),
303 // We want to avoid negative zero and round the numbers up
304 // if they originally had one digit more and that digit was 5.
305 a.pos.x > -5e-4 && a.pos.x < 0 ? 0 : a.pos.x + 1e-10,
306 a.pos.y > -5e-4 && a.pos.y < 0 ? 0 : a.pos.y + 1e-10,
307 a.pos.z > -5e-4 && a.pos.z < 0 ? 0 : a.pos.z + 1e-10,
308 // Occupancy is stored as single prec, but we know it's <= 1,
309 // so no precision is lost even if it had 6 digits after dot.
310 a.occ + 1e-6,
311 // B is harder to get rounded right. It is stored as float,
312 // and may be given with more than single precision in mmCIF
313 // If it was originally %.5f (5TIS) we need to add 0.5 * 10^-5.
314 std::min(a.b_iso + 0.5e-5, 999.99),
315 res.segment.c_str(),
316 a.element.uname(),
317 // Charge is written as 1+ or 2-, etc, or just empty space.
318 // Sometimes PDB files have explicit 0s (5M05); we ignore them.
319 a.charge ? a.charge > 0 ? '0'+a.charge : '0'-a.charge : ' ',
320 a.charge ? a.charge > 0 ? '+' : '-' : ' ');
321 if (a.aniso.nonzero()) {
322 // re-using part of the buffer
323 std::memcpy(buf, "ANISOU", 6);
324 const double eps = 1e-6;
325 gf_snprintf(buf+28, 43, "%7.0f%7.0f%7.0f%7.0f%7.0f%7.0f",
326 a.aniso.u11*1e4 + eps, a.aniso.u22*1e4 + eps,
327 a.aniso.u33*1e4 + eps, a.aniso.u12*1e4 + eps,
328 a.aniso.u13*1e4 + eps, a.aniso.u23*1e4 + eps);
329 buf[28+42] = ' ';
330 buf[80] = '\n';
331 os.write(buf, 81);
332 }
333 }
334 if (opt.ter_records && buf[0] != '\0' &&
335 (opt.ter_ignores_type ? &res == &chain.residues.back()
336 : (res.entity_type == EntityType::Polymer &&
337 (&res == &chain.residues.back() ||
338 (&res + 1)->entity_type != EntityType::Polymer)))) {
339 if (opt.numbered_ter) {
340 // re-using part of the buffer in the middle, e.g.:
341 // TER 4153 LYS B 286
342 gf_snprintf(buf, 82, "TER %5s",
343 impl::encode_serial_in_hybrid36(buf8, ++serial));
344 std::memset(buf+11, ' ', 6);
345 std::memset(buf+28, ' ', 52);
346 buf[80] = '\n';
347 os.write(buf, 81);
348 } else {
349 WRITE("%-80s", "TER");
350 }
351 }
352 }
353}

355inline void write_atoms(const Structure& st, std::ostream& os,
356 PdbWriteOptions opt) {
357 char buf[88];
358 for (const Model& model : st.models) {
359 int serial = 0;
360 if (st.models.size() > 1) {
361 // according to the spec model name in mmCIF may not be numeric
362 std::string name = model.name;
363 for (char c : name)
364 if (!std::isdigit(c)) {
365 name = std::to_string(&model - &st.models[0] + 1);
366 break;
367 }
368 WRITE("MODEL %8s %65s", name.c_str(), "");
369 }
370 for (const Chain& chain : model.chains)
371 write_chain_atoms(chain, os, serial, opt);
372 if (st.models.size() > 1)
373 WRITE("%-80s", "ENDMDL");
374 }
375}

377inline void write_header(const Structure& st, std::ostream& os,
378 PdbWriteOptions opt) {
379 const std::string& entry_id = st.get_info("_entry.id");
380 char buf[88];
381 { // header line
382 const char* months = "JANFEBMARAPRMAYJUNJULAUGSEPOCTNOVDEC???";
383 const std::string& date =
384 st.get_info("_pdbx_database_status.recvd_initial_deposition_date");
385 std::string pdb_date;
386 if (date.size() == 10) {
387 unsigned month_idx = 10 * (date[5] - '0') + date[6] - '0' - 1;
388 std::string month(months + 3 * std::min(month_idx, 13u), 3);
389 pdb_date = date.substr(8, 2) + "-" + month + "-" + date.substr(2, 2);
390 }
391 // "classification" in PDB == _struct_keywords.pdbx_keywords in mmCIF
392 const std::string& keywords = st.get_info("_struct_keywords.pdbx_keywords");
393 if (!pdb_date.empty() || !keywords.empty() || !entry_id.empty())
394 WRITEU("HEADER %-40.40s%-9s %-18.18s",
395 keywords.c_str(), pdb_date.c_str(), entry_id.c_str());
396 }
397 write_multiline(os, "TITLE", st.get_info("_struct.title"), 80);
398 write_multiline(os, "KEYWDS", st.get_info("_struct_keywords.text"), 79);
399 std::string expdta = st.get_info("_exptl.method");
400 if (expdta.empty())
401 expdta = join_str(st.meta.experiments, "; ",
402 [](const ExperimentInfo& e) { return e.method; });
403 write_multiline(os, "EXPDTA", expdta, 79);
404 if (st.models.size() > 1)
405 WRITE("NUMMDL %-6zu %63s", st.models.size(), "");
406
407 if (!st.raw_remarks.empty()) {
408 for (const std::string& line : st.raw_remarks) {
409 os << line;
410 if (line.empty() || line.back() != '\n')
411 os << '\n';
412 }
413 } else {
414 write_remarks(st, os);
415 }
416
417 // DBREF[12], SEQRES
418 if (!st.models.empty() && opt.seqres_records) {
419 std::vector<const Entity*> entity_list;
420 entity_list.reserve(st.models[0].chains.size());
421 for (const Chain& ch : st.models[0].chains) {
422 const Entity* entity = st.get_entity_of(ch.get_polymer());
423 // If the input pdb file has no TER records the subchains and entities
424 // are not setup automatically. In such case it is possible to call
425 // setup_entities() to use heuristic to split polymer and ligands
426 // and assign entities. But if it was not called, we may still find
427 // the original SEQRES in the entity named after the chain name.
428 if (!entity && st.input_format == CoorFormat::Pdb &&
429 !ch.residues.empty() && ch.residues[0].subchain.empty()) {
430 entity = st.get_entity(ch.name);
431 if (entity && !entity->subchains.empty())
432 entity = nullptr;
433 }
434 entity_list.push_back(entity);
435 }
436 // DBREF / DBREF1 / DBREF2
437 for (size_t i = 0; i != entity_list.size(); ++i)
438 if (const Entity* entity = entity_list[i]) {
439 const Chain& ch = st.models[0].chains[i];
440 const char* dbref_entry_id = entry_id.size() <= 4 ? entry_id.c_str() : "";
441 for (const Entity::DbRef& dbref : entity->dbrefs) {
442 bool short_record = *dbref.db_end.num < 100000 &&
443 dbref.accession_code.size() < 9 &&
444 dbref.id_code.size() < 13;
445 SeqId begin = dbref.seq_begin;
446 SeqId end = dbref.seq_end;
447 if (!begin.num || !end.num)
448 if (ConstResidueGroup polymer = ch.get_polymer()) {
449 begin = polymer.label_seq_id_to_auth(dbref.label_seq_begin);
450 end = polymer.label_seq_id_to_auth(dbref.label_seq_end);
451 }
452 char buf8[8];
453 char buf8a[8];
454 gf_snprintf(buf, 82, "DBREF %4s%2s %5s %5s %-6s ",
455 dbref_entry_id, ch.name.c_str(),
456 impl::write_seq_id(buf8, begin),
457 impl::write_seq_id(buf8a, end),
458 dbref.db_name.c_str());
459 if (dbref.db_name == "PDB" && dbref.id_code == entry_id) {
460 // PDB uses self-reference for fragments that don't have real
461 // reference. No idea why. In such case the same begin/end is used.
462 } else {
463 begin = dbref.db_begin;
464 end = dbref.db_end;
465 }
466 if (short_record) {
467 gf_snprintf(buf+33, 82-33, "%-8s %-12s %5d%c %5d%c \n",
468 dbref.accession_code.c_str(), dbref.id_code.c_str(),
469 *begin.num, begin.icode, *end.num, end.icode);
470 } else {
471 buf[5] = '1'; // -> DBREF1
472 gf_snprintf(buf+33, 82-33, " %-33s\n",
473 dbref.id_code.c_str());
474 }
475 buf[80] = '\n';
476 os.write(buf, 81);
477 if (!short_record)
478 WRITE("DBREF2 %4s%2s %-22s %10d %10d ",
479 dbref_entry_id, ch.name.c_str(),
480 dbref.accession_code.c_str(), *begin.num, *end.num);
481 }
482 }
483 // SEQRES
484 for (size_t i = 0; i != entity_list.size(); ++i)
485 if (const Entity* entity = entity_list[i]) {
486 const Chain& ch = st.models[0].chains[i];
487 int row = 0;
488 int col = 0;
489 for (const std::string& monomers : entity->full_sequence) {
490 if (col == 0)
491 gf_snprintf(buf, 82, "SEQRES%4d%2s%5zu %62s\n",
492 ++row, ch.name.c_str(),
493 entity->full_sequence.size(), "");
494 size_t end = monomers.find(',');
495 if (end == std::string::npos)
496 end = monomers.length();
497 std::memcpy(buf + 18 + 4*col + 4-end, monomers.c_str(), end);
498 if (++col == 13) {
499 os.write(buf, 81);
500 col = 0;
501 }
502 }
503 if (col != 0) {
504 buf[80] = '\n';
505 os.write(buf, 81);
506 }
507 }
508 }
509
510 if (!st.helices.empty()) {
511 char buf8[8];
512 char buf8a[8];
513 int counter = 0;
514 for (const Helix& helix : st.helices) {
515 if (++counter == 10000)
516 counter = 0;
517 // According to the PDB spec serial number can be from 1 to 999.
518 // Here, we allow for up to 9999 helices by using columns 7 and 11.
519 gf_snprintf(buf, 82, "HELIX %4d%4d %3s%2s %5s %3s%2s %5s%2d %35d \n",
520 counter, counter,
521 helix.start.res_id.name.c_str(), helix.start.chain_name.c_str(),
522 write_seq_id(buf8, helix.start.res_id.seqid),
523 helix.end.res_id.name.c_str(), helix.end.chain_name.c_str(),
524 write_seq_id(buf8a, helix.end.res_id.seqid),
525 (int) helix.pdb_helix_class, helix.length);
526 if (helix.length < 0) // make 72-76 blank if the length is not given
527 std::memset(buf+71, ' ', 5);
528 buf[80] = '\n';
529 os.write(buf, 81);
530 }
531 }
532
533 if (!st.sheets.empty()) {
534 char buf8a[8], buf8b[8], buf8c[8], buf8d[8];
535 for (const Sheet& sheet : st.sheets) {
536 int strand_counter = 0;
537 for (const Sheet::Strand& strand : sheet.strands) {
538 const AtomAddress& a2 = strand.hbond_atom2;
539 const AtomAddress& a1 = strand.hbond_atom1;
540 // H-bond atom names are expected to be O and N
541 WRITE("SHEET%5d %3.3s%2zu %3s%2s%5s %3s%2s%5s%2d %-3s%3s%2s%5s "
542 " %-3s%3s%2s%5s ",
543 ++strand_counter, sheet.name.c_str(), sheet.strands.size(),
544 strand.start.res_id.name.c_str(), strand.start.chain_name.c_str(),
545 write_seq_id(buf8a, strand.start.res_id.seqid),
546 strand.end.res_id.name.c_str(), strand.end.chain_name.c_str(),
547 write_seq_id(buf8b, strand.end.res_id.seqid), strand.sense,
548 a2.atom_name.c_str(), a2.res_id.name.c_str(),
549 a2.chain_name.c_str(),
550 a2.res_id.seqid.num ? write_seq_id(buf8c, a2.res_id.seqid) : "",
551 a1.atom_name.c_str(), a1.res_id.name.c_str(),
552 a1.chain_name.c_str(),
553 a1.res_id.seqid.num ? write_seq_id(buf8d, a1.res_id.seqid) : "");
554 }
555 }
556 }
557
558 if (!st.models.empty()) {
559 char buf8[8];
560 char buf8a[8];
561 // SSBOND (note: uses only the first model and primary conformation)
562 if (opt.ssbond_records) {
563 int counter = 0;
564 for (const Connection& con : st.connections)
565 if (con.type == Connection::Disulf) {
566 const_CRA cra1 = st.models[0].find_cra(con.partner1, true);
567 const_CRA cra2 = st.models[0].find_cra(con.partner2, true);
568 if (!cra1.atom || !cra2.atom)
569 continue;
570 NearestImage im = st.cell.find_nearest_image(cra1.atom->pos,
571 cra2.atom->pos, con.asu);
572 if (++counter == 10000)
573 counter = 0;
574 WRITE("SSBOND%4d %3s%2s %5s %5s%2s %5s %28s %6s %5.2f ",
575 counter,
576 cra1.residue->name.c_str(), cra1.chain->name.c_str(),
577 write_seq_id(buf8, cra1.residue->seqid),
578 cra2.residue->name.c_str(), cra2.chain->name.c_str(),
579 write_seq_id(buf8a, cra2.residue->seqid),
580 "1555", im.symmetry_code(false).c_str(), im.dist());
581 }
582 }
583
584 // LINK (note: uses only the first model and primary conformation)
585 if (!st.models.empty() && opt.link_records) {
586 for (const Connection& con : st.connections)
587 if (con.type == Connection::Covale || con.type == Connection::MetalC ||
588 con.type == Connection::Unknown) {
589 const_CRA cra1 = st.models[0].find_cra(con.partner1, true);
590 const_CRA cra2 = st.models[0].find_cra(con.partner2, true);
591 // In special cases (LINKR gap) atoms are not there.
592 if (!cra1.residue || !cra2.residue)
593 continue;
594 std::string im_pdb_symbol, im_dist_str;
595 bool im_same_asu = true;
596 if (cra1.atom && cra2.atom) {
597 NearestImage im = st.cell.find_nearest_image(cra1.atom->pos,
598 cra2.atom->pos, con.asu);
599 im_pdb_symbol = im.symmetry_code(false);
600 im_dist_str = to_str_prec<2>(im.dist());
601 im_same_asu = im.same_asu();
602 }
603 // Pdb spec: "sym1 and sym2 are right justified and are given as
604 // blank when the identity operator (and no cell translation) is
605 // to be applied to the atom." But all files from wwPDB have
606 // 1555 not blank, so here we also write 1555,
607 // except for LINKR (Refmac variant of LINK).
608 gf_snprintf(buf, 82, "LINK %-4s%c%3s%2s%5s "
609 " %-4s%c%3s%2s%5s %6s %6s %5s \n",
610 cra1.atom ? cra1.atom->padded_name().c_str() : "",
611 cra1.atom && cra1.atom->altloc ? std::toupper(cra1.atom->altloc) : ' ',
612 cra1.residue->name.c_str(),
613 con.partner1.chain_name.c_str(),
614 write_seq_id(buf8, cra1.residue->seqid),
615 cra2.atom ? cra2.atom->padded_name().c_str() : "",
616 cra2.atom && cra2.atom->altloc ? std::toupper(cra2.atom->altloc) : ' ',
617 cra2.residue->name.c_str(),
618 con.partner2.chain_name.c_str(),
619 write_seq_id(buf8a, cra2.residue->seqid),
620 "1555", im_pdb_symbol.c_str(), im_dist_str.c_str());
621 if (opt.use_linkr && !con.link_id.empty()) {
622 buf[4] = 'R'; // LINK -> LINKR
623 if (im_same_asu)
624 std::memset(buf+58, ' ', 14); // erase symmetry
625 // overwrite distance with link_id
626 gf_snprintf(buf+72, 82-72, "%-8s\n", con.link_id.c_str());
627 }
628 buf[80] = '\n';
629 os.write(buf, 81);
630 }
631
632 }
633
634 // CISPEP (note: uses only the first conformation)
635 if (!st.models.empty() && opt.cispep_records) {
636 int counter = 0;
637 for (const Model& model : st.models)
638 for (const Chain& chain : model.chains)
639 for (const Residue& res : chain.residues)
640 if (res.is_cis) {
641 const Residue* next = chain.next_residue(res);
642 if (next && are_connected(res, *next, PolymerType::PeptideL)) {
643 if (++counter == 10000)
644 counter = 0;
645 WRITE("CISPEP%4d %3s%2s %5s %3s%2s %5s %9s %12.2f %20s",
646 counter,
647 res.name.c_str(), chain.name.c_str(),
648 write_seq_id(buf8, res.seqid),
649 next->name.c_str(), chain.name.c_str(),
650 write_seq_id(buf8a, next->seqid),
651 st.models.size() > 1 ? model.name.c_str() : "0",
652 deg(calculate_omega(res, *next)),
653 "");
654 }
655 }
656 }
657 }
658
659 write_cryst1(st, os);
660 if (st.has_origx && !st.origx.is_identity()) {
661 for (int i = 0; i < 3; ++i)
662 WRITE("ORIGX%d %13.6f%10.6f%10.6f %14.5f %24s", i+1,
663 st.origx.mat[i][0], st.origx.mat[i][1], st.origx.mat[i][2],
664 st.origx.vec.at(i), "");
665 }
666 if (st.cell.explicit_matrices) {
667 for (int i = 0; i < 3; ++i)
668 // We add a small number to avoid negative 0.
669 WRITE("SCALE%d %13.6f%10.6f%10.6f %14.5f %24s", i+1,
670 st.cell.frac.mat[i][0] + 1e-15, st.cell.frac.mat[i][1] + 1e-15,
671 st.cell.frac.mat[i][2] + 1e-15, st.cell.frac.vec.at(i) + 1e-15, "");
672 }
673 write_ncs(st, os);
674}
