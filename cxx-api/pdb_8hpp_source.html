59// A function for transparent reading of stdin and/or gzipped files.
60template<typename T>
61inline Structure read_pdb(T&& input, PdbReadOptions options=PdbReadOptions()) {
62 if (input.is_stdin()) {
63 return read_pdb_from_stream(LineReader<FileStream>{stdin}, "stdin", options);
64 }
65 if (input.is_compressed()) {
66 using LR = LineReader<decltype(input.get_uncompressing_stream())>;
67 return read_pdb_from_stream(LR{input.get_uncompressing_stream()}, input.path(), options);
68 }
69 return read_pdb_file(input.path(), options);
70}
