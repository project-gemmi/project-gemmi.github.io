36 template<typename Stream>
37 Structure from_stream(Stream&& stream, const std::string& source) {
38 Structure st;
39 st.input_format = CoorFormat::Pdb;
40 st.name = path_basename(source, {".gz", ".pdb"});
41 char line[122] = {0};
42 while (size_t len = copy_line_from_stream(line, options.max_line_length+1, stream)) {
43 ++line_num;
44 read_pdb_line(line, len, st, source);
45 if (is_end)
46 break;
47 }
48 finalize_structure_after_reading_pdb(st);
