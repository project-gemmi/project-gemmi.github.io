129// Obliquity calculated here is the same as Le Page delta in cctbx.
130// (tested against output of lebedev_2005_perturbation.py from cctbx)
131inline double calculate_cos_obliquity(const UnitCell& reduced_cell,
132 const Vec3& d_axis, const Vec3& r_axis) {
133 // From the Le Page paper:
134 // tan(delta) = |t x tau| / |t . tau| (i.e. delta is angle 0...90deg)
135 // t = L U where U is direct-space axis
136 // tau = (L^T)^-1 h where h is reciprocal-space axis
137 // L is orthogonalization matrix, although in a different order
138 Vec3 t = reduced_cell.orth.mat.multiply(d_axis);
139 Vec3 tau = reduced_cell.frac.mat.left_multiply(r_axis);
140 // it's faster to calculate cos(delta) than tan(delta)
141 return std::min(1.0, std::fabs(t.cos_angle(tau)));
142}

144// Reduced cell can be from GruberVector::get_cell() after Niggli reduction.
145// max_obliq is max obliquity (delta) in degrees as defined in Le Page (1982).
146inline std::vector<OpObliquity> find_lattice_2fold_ops(const UnitCell& reduced_cell,
147 double max_obliq) {
148 std::vector<OpObliquity> ret;
149 const double cos_max_obliq = std::cos(rad(max_obliq));
150 for (const impl::TwoFoldData& row : impl::TwoFold::table) {
151 Vec3 d_axis(row.ds_axis[0], row.ds_axis[1], row.ds_axis[2]);
152 Vec3 r_axis(row.rs_axis[0], row.rs_axis[1], row.rs_axis[2]);
153 double cos_delta = calculate_cos_obliquity(reduced_cell, d_axis, r_axis);
154 if (cos_delta > cos_max_obliq) {
155 constexpr int D = Op::DEN;
156 Op op{{row.matrix[0] * D, row.matrix[1] * D, row.matrix[2] * D,
157 row.matrix[3] * D, row.matrix[4] * D, row.matrix[5] * D,
158 row.matrix[6] * D, row.matrix[7] * D, row.matrix[8] * D},
159 {0, 0, 0}};
160 ret.emplace_back(op, deg(std::acos(cos_delta)));
161 }
162 }
163 std::sort(ret.begin(), ret.end(),
164 [](const OpObliquity& a, const OpObliquity& b) { return a.second < b.second; });
165 return ret;
166}

168// Reduced cell can be from GruberVector::get_cell() after Niggli reduction.
169// max_obliq is max obliquity (delta) in degrees as defined in Le Page (1982).
170// Returns lattice symmetry except inversion.
171inline GroupOps find_lattice_symmetry_r(const UnitCell& reduced_cell, double max_obliq) {
172 std::vector<OpObliquity> gen = find_lattice_2fold_ops(reduced_cell, max_obliq);
173 std::vector<Op> genops;
174 genops.reserve(gen.size());
175 for (const OpObliquity& op_obl : gen)
176 genops.push_back(op_obl.first);
177 GroupOps go;
178 go.sym_ops.push_back(Op::identity());
179 go.cen_ops.push_back({0,0,0});
180 // cf. GroupOps::add_missing_elements()
181 if (!gen.empty())
182 go.sym_ops.push_back(gen[0].first);
183 // no need to try operator^2, we know it must be identity
184 go.add_missing_elements_part2(genops, 24, true);
185 return go;
186}

188// Returns lattice symmetry, but without inversion.
189inline GroupOps find_lattice_symmetry(const UnitCell& cell, char centring,
190 double max_obliq) {
191 GruberVector gv(cell, centring, true);
192 gv.niggli_reduce();
193 UnitCell reduced = gv.get_cell();
194 GroupOps gops = find_lattice_symmetry_r(reduced, max_obliq);
195 gops.change_basis_forward(*gv.change_of_basis);
196 return gops;
197}

199// Determine potential twinning operators.
200// Returns all operators or only unique ones (coset representatives).
201inline std::vector<Op> find_twin_laws(const UnitCell& cell,
202 const SpaceGroup* sg,
203 double max_obliq,
204 bool all_ops) {
205 if (sg == nullptr)
206 sg = &get_spacegroup_p1();
207 GroupOps go = sg->operations();
208 GroupOps lat_go = find_lattice_symmetry(cell, sg->centring_type(), max_obliq);
209 if (!go.has_same_centring(lat_go))
210 fail("find_twin_laws(): internal error");
211 std::vector<Op> ops;
212 size_t sg_symop_count = go.sym_ops.size();
213 for (const Op& op : lat_go.sym_ops)
214 if (!go.find_by_rotation(op.rot) &&
215 !go.find_by_rotation(op.negated_rot())) {
216 ops.push_back(op);
217 if (!all_ops)
218 for (size_t i = 1; i < sg_symop_count; ++i)
219 go.sym_ops.push_back(op * go.sym_ops[i]);
220 }
221 return ops;
222}
