18inline std::vector<bool> prepare_free_gapo(const ConstResidueSpan& polymer,
19 PolymerType polymer_type) {
20 std::vector<bool> gaps;
21 gaps.reserve(polymer.size());
22 gaps.push_back(true); // free gap opening at the beginning of sequence
23 if (!is_polypeptide(polymer_type) && !is_polynucleotide(polymer_type))
24 return gaps;
25 auto first_conformer = polymer.first_conformer();
26 auto res = first_conformer.begin();
27 for (auto next_res = res; ++next_res != first_conformer.end(); res = next_res)
28 gaps.push_back(!are_connected3(*res, *next_res, polymer_type));
29 return gaps;
30}

33inline AlignmentResult align_sequence_to_polymer(
34 const std::vector<std::string>& full_seq,
35 const ConstResidueSpan& polymer,
36 PolymerType polymer_type,
37 const AlignmentScoring& scoring) {
38 std::map<std::string, std::uint8_t> encoding;
39 for (const std::string& res_name : scoring.matrix_encoding)
40 encoding.emplace(res_name, (std::uint8_t)encoding.size());
41 for (const Residue& res : polymer)
42 encoding.emplace(res.name, (std::uint8_t)encoding.size());
43 for (const std::string& mon_list : full_seq)
44 encoding.emplace(Entity::first_mon(mon_list), (std::uint8_t)encoding.size());
45 if (encoding.size() > 255)
46 return AlignmentResult();
47
48 std::vector<std::uint8_t> encoded_full_seq(full_seq.size());
49 for (size_t i = 0; i != full_seq.size(); ++i)
50 encoded_full_seq[i] = encoding.at(Entity::first_mon(full_seq[i]));
51
52 std::vector<std::uint8_t> encoded_model_seq;
53 encoded_model_seq.reserve(polymer.size());
54 for (const Residue& res : polymer.first_conformer())
55 encoded_model_seq.push_back(encoding.at(res.name));
56
57 return align_sequences(encoded_full_seq, encoded_model_seq,
58 prepare_free_gapo(polymer, polymer_type),
59 (std::uint8_t)encoding.size(), scoring);
60}

62// check for exact match between model sequence and full sequence (SEQRES)
63inline bool seqid_matches_seqres(const ConstResidueSpan& polymer,
64 const Entity& ent) {
65 if (ent.full_sequence.size() != polymer.size())
66 return false;
67 int idx = 0;
68 for (const Residue& res : polymer) {
69 if (ent.full_sequence[idx] != res.name ||
70 ++idx != *res.seqid.num || res.seqid.has_icode())
71 return false;
72 }
73 return true;
74}

76// Uses sequence alignment (model to SEQRES) to assign label_seq.
77// force: assign label_seq even if full sequence is not known (assumes no gaps)
78inline void assign_label_seq_to_polymer(ResidueSpan& polymer,
79 const Entity* ent, bool force) {
80 AlignmentResult result;
81
82 // sequence not known
83 if (!ent || ent->full_sequence.empty()) {
84 if (!force)
85 return;
86 PolymerType ptype = (ent && ent->polymer_type != PolymerType::Unknown
87 ? ent->polymer_type
88 : check_polymer_type(polymer));
89 const Residue* prev = nullptr;
90 for (const Residue& res : polymer.first_conformer()) {
91 if (prev && !are_connected3(*prev, res, ptype))
92 result.push_cigar(1, 1); // assume a single insertion
93 result.push_cigar(0, 1);
94 prev = &res;
95 }
96
97 // exact match - common case that doesn't require alignment
98 } else if (seqid_matches_seqres(polymer, *ent)) {
99 result.push_cigar(0, (int)ent->full_sequence.size());
100
101 // sequence alignment
102 } else {
103 AlignmentScoring scoring;
104 result = align_sequence_to_polymer(ent->full_sequence, polymer,
105 ent->polymer_type, scoring);
106 }
107
108 auto res_group = polymer.first_conformer().begin();
109 int id = 1;
110 for (AlignmentResult::Item item : result.cigar) {
111 switch (item.op()) {
112 case 'I':
113 id += item.len();
114 break;
115 case 'D': // leaving label_seq as it is
116 for (uint32_t i = 0; i < item.len(); ++i)
117 res_group++;
118 break;
119 case 'M': // not checking for mismatches
120 for (uint32_t i = 0; i < item.len(); ++i, ++id)
121 for (Residue* res = &*res_group++; res != &*res_group; ++res)
122 res->label_seq = id;
123 break;
124 }
125 }
126}

154inline void prepare_positions_for_superposition(std::vector<Position>& pos1,
155 std::vector<Position>& pos2,
156 ConstResidueSpan fixed,
157 ConstResidueSpan movable,
158 PolymerType ptype,
159 SupSelect sel,
160 char altloc='\0',
161 std::vector<int>* ca_offsets=nullptr) {
162 AlignmentScoring scoring;
163 AlignmentResult result = align_sequence_to_polymer(fixed.extract_sequence(),
164 movable, ptype, scoring);
165 auto it1 = fixed.first_conformer().begin();
166 auto it2 = movable.first_conformer().begin();
167 std::vector<AtomNameElement> used_atoms;
168 bool is_na = is_polynucleotide(ptype);
169 const AtomNameElement* ca_p = nullptr;
170 if (sel == SupSelect::CaP) {
171 used_atoms.push_back({is_na ? "P" : "CA", is_na ? El::P : El::C});
172 } else if (sel == SupSelect::MainChain) {
173 used_atoms = get_mainchain_atoms(ptype);
174 ca_p = &used_atoms[is_na ? 0 : 1];
175 }
176 for (AlignmentResult::Item item : result.cigar) {
177 char op = item.op();
178 for (uint32_t i = 0; i < item.len(); ++i) {
179 int ca_offset = -1;
180 if (op == 'M' && it1->name == it2->name) {
181 if (!used_atoms.empty()) {
182 for (const AtomNameElement& ane : used_atoms) {
183 const Atom* a1 = it1->find_atom(ane.atom_name, altloc, ane.el);
184 const Atom* a2 = it2->find_atom(ane.atom_name, altloc, ane.el);
185 if (a1 && a2) {
186 if (&ane == ca_p)
187 ca_offset = (int)pos1.size();
188 pos1.push_back(a1->pos);
189 pos2.push_back(a2->pos);
190 }
191 }
192 } else {
193 for (const Atom& a1 : it1->atoms)
194 if (a1.altloc_matches(altloc))
195 if (const Atom* a2 = it2->find_atom(a1.name, altloc, a1.element)) {
196 pos1.push_back(a1.pos);
197 pos2.push_back(a2->pos);
198 }
199 }
200 }
201 if (op == 'M' || op == 'I') {
202 ++it1;
203 if (ca_offsets)
204 ca_offsets->push_back(ca_offset);
205 }
206 if (op == 'M' || op == 'D')
207 ++it2;
208 }
209 }
210}

212inline SupResult calculate_current_rmsd(ConstResidueSpan fixed,
213 ConstResidueSpan movable,
214 PolymerType ptype,
215 SupSelect sel,
216 char altloc='\0') {
217 std::vector<Position> pos1, pos2;
218 prepare_positions_for_superposition(pos1, pos2, fixed, movable, ptype, sel, altloc);
219 SupResult r;
220 r.count = pos1.size();
221 double sd = 0;
222 for (size_t i = 0; i != pos1.size(); ++i)
223 sd += pos1[i].dist_sq(pos2[i]);
224 r.rmsd = std::sqrt(sd / r.count);
225 return r;
226}

228inline SupResult calculate_superposition(ConstResidueSpan fixed,
229 ConstResidueSpan movable,
230 PolymerType ptype,
231 SupSelect sel,
232 int trim_cycles=0,
233 double trim_cutoff=2.0,
234 char altloc='\0') {
235 std::vector<Position> pos1, pos2;
236 prepare_positions_for_superposition(pos1, pos2, fixed, movable, ptype, sel, altloc);
237 const double* weights = nullptr;
238 size_t len = pos1.size();
239 SupResult sr = superpose_positions(pos1.data(), pos2.data(), len, weights);
240
241 for (int n = 0; n < trim_cycles; ++n) {
242 double max_dist_sq = sq(trim_cutoff * sr.rmsd);
243 size_t p = 0;
244 for (size_t i = 0; i != len; ++i) {
245 Vec3 m2 = sr.transform.apply(pos2[i]);
246 if (m2.dist_sq(pos1[i]) <= max_dist_sq) {
247 if (i != p) {
248 pos1[p] = pos1[i];
249 pos2[p] = pos2[i];
250 }
251 ++p;
252 }
253 }
254 if (p == len)
255 break;
256 len = p;
257 if (len < 3)
258 fail("in calculate_superposition(): only ", std::to_string(len),
259 " atoms after trimming");
260 sr = superpose_positions(pos1.data(), pos2.data(), len, weights);
261 }
262
263 return sr;
264}

266// Returns superpositions for all residues in fixed.first_conformer(),
267// performed by superposing backbone in radius=10.0 from residue's Ca.
268inline std::vector<SupResult> calculate_superpositions_in_moving_window(
269 ConstResidueSpan fixed,
270 ConstResidueSpan movable,
271 PolymerType ptype,
272 double radius=10.0) {
273 const double radius_sq = radius * radius;
274 std::vector<Position> pos1, pos2;
275 char altloc = '\0';
276 SupSelect sel = SupSelect::MainChain;
277 std::vector<int> ca_offsets;
278 prepare_positions_for_superposition(pos1, pos2, fixed, movable, ptype,
279 sel, altloc, &ca_offsets);
280 const double* weights = nullptr;
281 std::vector<SupResult> result;
282 for (int offset : ca_offsets) {
283 if (offset == -1) {
284 result.push_back(SupResult{NAN, 0, {}, {}, {}});
285 continue;
286 }
287 const Position& ca_pos = pos1[offset];
288 int a = offset;
289 while (a > 0 && ca_pos.dist_sq(pos1[a-1]) < radius_sq)
290 --a;
291 int b = offset;
292 while (b+1 < (int)pos1.size() && ca_pos.dist_sq(pos1[b+1]) < radius_sq)
293 ++b;
294 result.push_back(superpose_positions(&pos1[a], &pos2[a], b-a+1, weights));
295 }
296 return result;
297}
