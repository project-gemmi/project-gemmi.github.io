3// Classes for iterating files in a directory tree, top-down,
4// in an alphabetical order. It wraps the tinydir library (as we cannot
5// depend on C++17 <filesystem> yet).
6
7// DirWalk<> iterates through all files and directories.
8// CifWalk yields only cif files (either files that end with .cif or .cif.gz,
9// or files that look like SF mmCIF files from wwPDB, e.g. r3aaasf.ent.gz).
10// It's good for traversing a local copy of the wwPDB archive.
11// PdbWalk: .pdb or .ent (optionally with .gz) except r????sf.ent
12// CoorFileWalk: .cif, .pdb or .ent (optionally with .gz)
13// except r????sf.ent and *-sf.cif
14//
15// Usage:
16// for (const std::string& file : gemmi::DirWalk<>(top_dir))
17// do_something(file);
18// or
19// for (const std::string& file : gemmi::CifWalk(top_dir))
20// do_something(file);
21// You should also catch std::runtime_error.

51// linear-time glob matching: https://research.swtch.com/glob
52inline bool glob_match(const std::string& pattern, const std::string& str) {
53 size_t pat_next = 0;
54 size_t str_next = std::string::npos;
55 size_t pat_pos = 0;
56 size_t str_pos = 0;
57 while (pat_pos < pattern.size() || str_pos < str.size()) {
58 if (pat_pos < pattern.size()) {
59 char c = pattern[pat_pos];
60 if (c == '*') {
61 pat_next = pat_pos;
62 str_next = str_pos + 1;
63 pat_pos++;
64 continue;
65 } else if (str_pos < str.size() && (c == '?' || c == str[str_pos])) {
66 pat_pos++;
67 str_pos++;
68 continue;
69 }
70 }
71 if (str_next > str.size())
72 return false;
73 pat_pos = pat_next;
74 str_pos = str_next;
75 }
76 return true;
77}

129template<bool FileOnly=true, typename Filter=impl::IsAnyFile>
130class DirWalk {
131public:
132 explicit DirWalk(const char* path) {
133#if defined(_WIN32) && defined(_UNICODE)
134 std::wstring str = UTF8_to_wchar(path);
135 const _tinydir_char_t* xpath = str.c_str();
136#else
137 const char* xpath = path;
138#endif
139 if (tinydir_file_open(&top_, xpath) == -1)
140 sys_fail("Cannot open " + std::string(path));
141 }
142 explicit DirWalk(const std::string& path) : DirWalk(path.c_str()) {}
143 ~DirWalk() {
144 for (auto& d : dirs_)
145 tinydir_close(&d.second);
146 }
147 void push_dir(size_t cur_pos, const _tinydir_char_t* path) {
148 dirs_.emplace_back();
149 dirs_.back().first = cur_pos;
150 if (tinydir_open_sorted(&dirs_.back().second, path) == -1)
151 sys_fail("Cannot open directory " + as_utf8(path));
152 }
153 size_t pop_dir() {
154 assert(!dirs_.empty());
155 size_t old_pos = dirs_.back().first;
156 tinydir_close(&dirs_.back().second);
157 dirs_.pop_back();
158 return old_pos;
159 }

161 struct Iter {
162 DirWalk& walk;
163 size_t cur;
164
165 const tinydir_dir& get_dir() const { return walk.dirs_.back().second; }
166
167 const tinydir_file& get() const {
168 if (walk.dirs_.empty())
169 return walk.top_;
170 assert(cur < get_dir().n_files);
171 return get_dir()._files[cur];
172 }
173
174 std::string operator*() const { return as_utf8(get().path); }
175
176 // checks for "." and ".."
177 bool is_special(const _tinydir_char_t* name) const {
178 return name[0] == '.' && (name[1] == '\0' ||
179 (name[1] == '.' && name[2] == '\0'));
180 }
181
182 size_t depth() const { return walk.dirs_.size(); }
183
184 void next() { // depth first
185 const tinydir_file& tf = get();
186 if (tf.is_dir) {
187 walk.push_dir(cur, tf.path);
188 cur = 0;
189 } else {
190 cur++;
191 }
192 while (!walk.dirs_.empty()) {
193 if (cur == get_dir().n_files)
194 cur = walk.pop_dir() + 1;
195 else if (is_special(get_dir()._files[cur].name))
196 cur++;
197 else
198 break;
199 }
200 }
201
202 void operator++() {
203 for (;;) {
204 next();
205 const tinydir_file& f = get();
206 if ((!FileOnly && f.is_dir)
207 || (!f.is_dir && walk.filter.check(as_utf8(f.name)))
208 || walk.is_single_file()
209 || (depth() == 0 && cur == 1))
210 break;
211 }
212 }
213
214 // == and != is used only to compare with end()
215 bool operator==(const Iter& o) const { return depth()==0 && cur == o.cur; }
216 bool operator!=(const Iter& o) const { return !operator==(o); }
217 };
218
219 Iter begin() {
220 Iter it{*this, 0};
221 if (FileOnly && !is_single_file()) // i.e. the top item is a directory
222 ++it;
223 return it;
224 }
225
226 Iter end() { return Iter{*this, 1}; }
227 bool is_single_file() { return !top_.is_dir; }
228
229private:
230 friend struct Iter;
231 tinydir_file top_;
232 std::vector<std::pair<size_t, tinydir_dir>> dirs_;
233protected:
234 Filter filter;
235};
