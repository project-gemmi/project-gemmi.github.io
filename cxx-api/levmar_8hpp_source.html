28inline void jordan_solve(std::vector<double>& a, std::vector<double>& b) {
29 assert(a.size() == b.size() * b.size());
30 int n = (int) b.size();
31 for (int i = 0; i < n; i++) {
32 // looking for a pivot element
33 int maxnr = -1;
34 double amax = 0;
35 for (int j = i; j < n; j++) {
36 double aji = std::fabs(a[n * j + i]);
37 if (aji > amax) {
38 maxnr = j;
39 amax = aji;
40 }
41 }
42 // handle singular matrix
43 if (maxnr == -1) {
44 // i-th column has only zeros.
45 // If it's the same about i-th row, and b[i]==0, let x[i]==0.
46 for (int j = i; j < n; j++)
47 if (a[n * i + j] != 0. || b[i] != 0.)
48 fail("Trying to reverse singular matrix. Column ", std::to_string(i), " is zeroed.");
49 continue; // x[i]=b[i], b[i]==0
50 }
51 // interchanging rows
52 if (maxnr != i) {
53 for (int j = i; j < n; j++)
54 std::swap(a[n * maxnr + j], a[n * i + j]);
55 std::swap(b[i], b[maxnr]);
56 }
57 // divide by a_ii -- to get a_ii=1
58 double c = 1.0 / a[i * n + i];
59 for (int j = i; j < n; j++)
60 a[i * n + j] *= c;
61 b[i] *= c;
62 // subtract -- to zero all remaining elements of this row
63 for (int k = 0; k < n; k++)
64 if (k != i) {
65 double d = a[k * n + i];
66 for (int j = i; j < n; j++)
67 a[k * n + j] -= a[i * n + j] * d;
68 b[k] -= b[i] * d;
69 }
70 }
71}

87 int eval_limit = 100;
88 double lambda_limit = 1e+15;
89 double stop_rel_change = 1e-5;

92 double lambda_up_factor = 10;
93 double lambda_down_factor = 0.1;
94 double lambda_start = 0.001;

106 template<typename Target>
107 double fit(Target& target) {
108 eval_count = 0;
109 std::vector<double> initial_a = target.get_parameters();
110 debug_print("ini", initial_a);
111 initial_wssr = this->compute_wssr(target.compute_values(), target.points);
112 std::vector<double> best_a = initial_a;
113 size_t na = initial_a.size();
114
115 double lambda = lambda_start;
116 alpha.resize(na * na);
117 beta.resize(na);
118
119 double wssr = initial_wssr;
120 this->compute_derivatives(target);
121
122 int small_change_counter = 0;
123 for (int iter = 0; ; iter++) {
124 if (eval_limit > 0 && eval_count >= eval_limit)
125 break;
126
127 // prepare next parameters -> temp_beta
128 temp_alpha = alpha;
129 for (size_t j = 0; j < na; j++)
130 temp_alpha[na * j + j] *= (1.0 + lambda);
131 temp_beta = beta;
132
133 // Matrix solution (Ax=b) temp_alpha * da == temp_beta
134 jordan_solve(temp_alpha, temp_beta);
135
136 for (size_t i = 0; i < na; i++)
137 // put new a[] into temp_beta[]
138 temp_beta[i] += best_a[i];
139
140 target.set_parameters(temp_beta);
141 double new_wssr = this->compute_wssr(target.compute_values(), target.points);
142#ifdef GEMMI_DEBUG_LEVMAR
143 fprintf(stderr, " #%d WSSR=%.8g %+g%% (%+.4g%%) lambda=%g\n",
144 iter, new_wssr, 100. * (new_wssr / initial_wssr - 1.),
145 100. * (new_wssr / wssr - 1.), lambda);
146 if (new_wssr < wssr)
147 debug_print("", temp_beta);
148#endif
149 if (new_wssr < wssr) {
150 double rel_change = (wssr - new_wssr) / wssr;
151 wssr = new_wssr;
152 best_a = temp_beta;
153
154 if (wssr == 0)
155 break;
156 // termination criterion: negligible change of wssr
157 if (rel_change < stop_rel_change) {
158 if (++small_change_counter >= 2)
159 break;
160 } else {
161 small_change_counter = 0;
162 }
163 this->compute_derivatives(target);
164 lambda *= lambda_down_factor;
165 } else { // worse fitting
166 if (lambda > lambda_limit) // termination criterion: large lambda
167 break;
168 lambda *= lambda_up_factor;
169 }
170 }
171
172 target.set_parameters(wssr < initial_wssr ? best_a : initial_a);
173 return wssr;
174 }

177 template<typename Target>
178 void compute_derivatives(const Target& target) {
179 assert(alpha.size() == beta.size() * beta.size());
180 int na = (int)beta.size();
181 assert(na != 0);
182#ifdef GEMMI_DEBUG_LEVMAR
183 check_derivatives(const_cast<Target&>(target));
184#endif
185 std::fill(alpha.begin(), alpha.end(), 0.0);
186 std::fill(beta.begin(), beta.end(), 0.0);
187 // Iterating over points is tiled to limit memory usage. It's also a little
188 // faster than a single loop over all points for large number of points.
189 const size_t kMaxTileSize = 1024;
190 std::vector<double> dy_da;
191 size_t n = target.points.size();
192 for (size_t tstart = 0; tstart < n; tstart += kMaxTileSize) {
193 size_t tsize = std::min(n - tstart, kMaxTileSize);
194 std::vector<double> yy(tsize, 0.);
195 dy_da.resize(tsize * na);
196 std::fill(dy_da.begin(), dy_da.end(), 0.);
197 target.compute_values_and_derivatives(tstart, tsize, yy, dy_da);
198 for (size_t i = 0; i != tsize; ++i) {
199 double weight = target.points[tstart + i].get_weight();
200 double dy_sig = weight * (target.points[tstart + i].get_y() - yy[i]);
201 double* t = &dy_da[i * na];
202 for (int j = 0; j != na; ++j) {
203 if (t[j] != 0) {
204 t[j] *= weight;
205 for (int k = j; k != -1; --k)
206 alpha[na * j + k] += t[j] * t[k];
207 beta[j] += dy_sig * t[j];
208 }
209 }
210 }
211 }
212
213 // Only half of the alpha matrix was filled above. Fill the rest.
214 for (int j = 1; j < na; j++)
215 for (int k = 0; k < j; k++)
216 alpha[na * k + j] = alpha[na * j + k];
217 }

220 template<typename Target>
221 void check_derivatives(Target& target) {
222 assert(!beta.empty());
223 assert(alpha.size() == beta.size() * beta.size());
224 std::vector<double> yy(target.points.size(), 0.);
225 std::vector<double> dy_da(target.points.size() * beta.size());
226 target.compute_values_and_derivatives(0, target.points.size(), yy, dy_da);
227 std::vector<double> yy2 = target.compute_values();
228 assert(yy.size() == yy2.size());
229 for (size_t i = 0; i != yy.size(); ++i) {
230 double m = std::max(std::fabs(yy[i]), std::fabs(yy2[i]));
231 if (m > 1e-5 && std::fabs(yy[i] - yy2[i]) > 1e-6 * m)
232 fprintf(stderr, "!! value %zu: %g != %g\n", i, yy[i], yy2[i]);
233 }
234 const double numerical_h = 1e-3;
235 const double small_number = 1e-6; // prevents h==0
236 std::vector<double> aa = target.get_parameters();
237 assert(aa.size() == beta.size());
238 for (size_t k = 0; k < aa.size(); k++) {
239 double acopy = aa[k];
240 double h = std::max(std::fabs(acopy), small_number) * numerical_h;
241 aa[k] -= h;
242 target.set_parameters(aa);
243 std::vector<double> y_left = target.compute_values();
244 aa[k] = acopy + h;
245 target.set_parameters(aa);
246 std::vector<double> y_right = target.compute_values();
247 aa[k] = acopy;
248 for (size_t i = 0; i != target.points.size(); ++i) {
249 double symbolic = dy_da[i * aa.size() + k];
250 double numeric = (y_right[i] - y_left[i]) / (2 * h);
251 double m = std::max(std::fabs(symbolic), std::fabs(numeric));
252 if (m > 1e-3 && std::fabs(symbolic - numeric) > 0.02 * m)
253 fprintf(stderr, "!! dy[%zu]/da[%zu]: %g != %g\n", i, k, symbolic, numeric);
254 if (i == 30)
255 break;
256 }
257 }
258 target.set_parameters(aa);
259 }

262 template<typename Point>
263 double compute_wssr(const std::vector<double>& yy, const std::vector<Point>& data) {
264 // long double here notably increases the accuracy of calculations
265 long double wssr = 0;
266 for (int j = 0; j < (int)yy.size(); j++) {
267 double dy = data[j].get_weight() * (data[j].get_y() - yy[j]);
268 wssr += dy * dy;
269 }
270 ++eval_count;
271 return (double) wssr;
272 }
