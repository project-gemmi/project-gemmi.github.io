39 std::string cigar_str() const {
40 std::string s;
41 for (Item item : cigar) {
42 s += std::to_string(item.len());
43 s += item.op();
44 }
45 return s;

48 // 1=query, 2=target, other=shorter
49 std::size_t input_length(int which) const {
50 std::size_t counters[3] = {0, 0, 0};
51 for (Item item : cigar)
52 counters[item.value & 0xf] += item.len();
53 if (which == 1 || which == 2)
54 return counters[0] + counters[which];
55 return counters[0] + std::min(counters[1], counters[2]);
56 }
57 double calculate_identity(int which=0) const {
58 return 100. * match_count / input_length(which);
59 }

61 // In the backtrack matrix, value p[] has the following structure:
62 // bit 0-2: which type gets the max - 0 for H, 1 for E, 2 for F
63 // bit 3/0x08: 1 if a continuation on the E state
64 // bit 4/0x10: 1 if a continuation on the F state
65 void backtrack_to_cigar(const std::uint8_t *p, int i, int j) {
66 i--;
67 int j0 = j--;
68 int state = 0;
69 while (i >= 0 && j >= 0) {
70 // at the beginning of the loop, _state_ tells us which state to check
71 // if requesting the H state, find state one maximizes it.
72 uint32_t tmp = p[(std::size_t)i * j0 + j];
73 if (state == 0 || (tmp & (1 << (state + 2))) == 0)
74 state = tmp & 7;
75 if (state == 0) { // match
76 push_cigar(0, 1);
77 --i;
78 --j;
79 } else if (state == 1) { // deletion
80 push_cigar(2, 1);
81 --i;
82 } else { // insertion
83 push_cigar(1, 1);
84 --j;
85 }
86 }
87 if (i >= 0)
88 push_cigar(2, i + 1); // first deletion
89 else if (j >= 0)
90 push_cigar(1, j + 1); // first insertion
91 std::reverse(cigar.begin(), cigar.end());
92 }

95 void count_matches(const std::vector<std::uint8_t>& query,
96 const std::vector<std::uint8_t>& target) {
97 match_count = 0;
98 size_t pos1 = 0, pos2 = 0;
99 for (Item item : cigar)
100 if (item.op() == 'M') {
101 for (uint32_t i = 0; i < item.len(); ++i)
102 if (query[pos1++] == target[pos2++]) {
103 ++match_count;
104 match_string += '|';
105 } else {
106 match_string += '.';
107 }
108 } else if (item.op() == 'I') {
109 pos1 += item.len();
110 match_string.append(item.len(), ' ');
111 } else /*item.op() == 'D'*/ {
112 pos2 += item.len();
113 match_string.append(item.len(), ' ');
114 }
115 }

117 std::string add_gaps(const std::string& s, unsigned which) const {
118 std::string out;
119 size_t pos = 0;
120 for (Item item : cigar) {
121 bool show = (item.value & 0xf) == 0 || (item.value & 0xf) == which;
122 for (uint32_t i = 0; i < item.len(); ++i)
123 out += show ? s.at(pos++) : '-';
124 }
125 return out;
126 }

128 std::string formatted(const std::string& a, const std::string& b) const {
129 std::string r;
130 r.reserve((match_string.size() + 1) * 3);
131 r += add_gaps(a, 1);
132 r += '\n';
133 r += match_string;
134 r += '\n';
135 r += add_gaps(b, 2);
136 r += '\n';
137 return r;
138 }

140 // op: 0=match/mismatch, 1=insertion, 2=deletion
141 void push_cigar(std::uint32_t op, int len) {
142 if (cigar.empty() || op != (cigar.back().value & 0xf))
143 cigar.push_back({len<<4 | op});
144 else
145 cigar.back().value += len<<4;
146 }

149// All values in query and target must be less then m.
150// free_gapo marks positions in target where gap opening is free.
151inline
152AlignmentResult align_sequences(const std::vector<std::uint8_t>& query,
153 const std::vector<std::uint8_t>& target,
154 const std::vector<bool>& free_gapo,
155 std::uint8_t m,
156 const AlignmentScoring& scoring) {
157 // generate the query profile
158 std::int8_t *query_profile = new std::int8_t[query.size() * m];
159 {
160 std::uint32_t mat_size = (std::uint32_t) scoring.matrix_encoding.size();
161 std::int32_t i = 0;
162 for (std::uint8_t k = 0; k < m; ++k)
163 for (std::uint8_t q : query)
164 if (k < mat_size && q < mat_size)
165 query_profile[i++] = scoring.score_matrix[k * m + q];
166 else
167 query_profile[i++] = (k == q ? scoring.match : scoring.mismatch);
168 }
169
170 struct eh_t { std::int32_t h, e; };
171 eh_t *eh = new eh_t[query.size() + 1];
172 std::int32_t gape = scoring.gape;
173 std::int32_t gapoe = scoring.gapo + gape;
174
175 // fill the first row
176 {
177 std::int32_t gap0 = !free_gapo.empty() && free_gapo[0] ? gape : gapoe;
178 eh[0].h = 0;
179 eh[0].e = gap0 + gapoe;
180 for (std::int32_t j = 1; j <= (std::int32_t)query.size(); ++j) {
181 eh[j].h = gap0 + gape * (j - 1);
182 eh[j].e = gap0 + gapoe + gape * j;
183 }
184 }
185
186 // backtrack matrix; in each cell: f<<4|e<<2|h
187 std::uint8_t *z = new std::uint8_t[query.size() * target.size()];
188 // DP loop
189 for (std::int32_t i = 0; i < (std::int32_t)target.size(); ++i) {
190 std::uint8_t target_item = target[i];
191 std::int8_t *scores = &query_profile[target_item * query.size()];
192 std::uint8_t *zi = &z[i * query.size()];
193 std::int32_t h1 = gapoe + gape * i;
194 std::int32_t f = gapoe + gapoe + gape * i;
195 std::int32_t gapx = i+1 < (std::int32_t)free_gapo.size() && free_gapo[i+1]
196 ? gape : gapoe;
197 for (std::size_t j = 0; j < query.size(); ++j) {
198 // At the beginning of the loop:
199 // eh[j] = { H(i-1,j-1), E(i,j) }, f = F(i,j) and h1 = H(i,j-1)
200 // Cells are computed in the following order:
201 // H(i,j) = max{H(i-1,j-1) + S(i,j), E(i,j), F(i,j)}
202 // E(i+1,j) = max{H(i,j)+gapo, E(i,j)} + gape
203 // F(i,j+1) = max{H(i,j)+gapo, F(i,j)} + gape
204 eh_t *p = &eh[j];
205 std::int32_t h = p->h;
206 std::int32_t e = p->e;
207 p->h = h1;
208 h += scores[j];
209 std::uint8_t direction = 0;
210 if (h < e) {
211 direction = 1; // deletion
212 h = e;
213 }
214 if (h <= f) {
215 direction = 2; // insertion
216 h = f;
217 }
218 h1 = h;
219
220 h += gapoe;
221 e += gape;
222 if (e > h)
223 direction |= 0x08;
224 else
225 e = h;
226
227 h = h1 + gapx;
228 p->e = e;
229 f += gape;
230 if (f > h)
231 direction |= 0x10;
232 else
233 f = h;
234
235 // z[i,j] keeps h for the current cell and e/f for the next cell
236 zi[j] = direction;
237 }
238 eh[query.size()].h = h1;
239 eh[query.size()].e = -0x40000000; // -infinity
240 }
241
242 AlignmentResult result;
243 result.score = eh[query.size()].h;
244 delete [] query_profile;
245 delete [] eh;
246 result.backtrack_to_cigar(z, (int)target.size(), (int)query.size());
247 delete [] z;
248 result.count_matches(query, target);
249 return result;
250}

252inline
253AlignmentResult align_string_sequences(const std::vector<std::string>& query,
254 const std::vector<std::string>& target,
255 const std::vector<bool>& free_gapo,
256 const AlignmentScoring& scoring) {
257 std::map<std::string, std::uint8_t> encoding;
258 for (const std::string& res_name : scoring.matrix_encoding)
259 encoding.emplace(res_name, (std::uint8_t)encoding.size());
260 for (const std::string& s : query)
261 encoding.emplace(s, (std::uint8_t)encoding.size());
262 for (const std::string& s : target)
263 encoding.emplace(s, (std::uint8_t)encoding.size());
264 if (encoding.size() > 255)
265 return AlignmentResult();
266 std::vector<std::uint8_t> encoded_query(query.size());
267 for (size_t i = 0; i != query.size(); ++i)
268 encoded_query[i] = encoding.at(query[i]);
269 std::vector<std::uint8_t> encoded_target(target.size());
270 for (size_t i = 0; i != target.size(); ++i)
271 encoded_target[i] = encoding.at(target[i]);
272 return align_sequences(encoded_query, encoded_target,
273 free_gapo, (std::uint8_t)encoding.size(), scoring);
274}
