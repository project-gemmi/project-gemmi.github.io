15struct ComplexCorrelation {
16 int n = 0;
17 double sum_xx = 0.;
18 double sum_yy = 0.;
19 std::complex<double> sum_xy = 0.;
20 std::complex<double> mean_x = 0.;
21 std::complex<double> mean_y = 0.;
22 void add_point(std::complex<double> x, std::complex<double> y) {
23 ++n;
24 double inv_n = 1.0 / n;
25 double weight = (n - 1.0) * inv_n;
26 std::complex<double> dx = x - mean_x;
27 std::complex<double> dy = y - mean_y;
28 sum_xx += weight * std::norm(dx);
29 sum_yy += weight * std::norm(dy);
30 sum_xy += weight * (dx * std::conj(dy));
31 mean_x += dx * inv_n;
32 mean_y += dy * inv_n;
33 }
34 void add_point(std::complex<float> x, std::complex<float> y) {
35 add_point(std::complex<double>(x), std::complex<double>(y));
36 }
37 std::complex<double> coefficient() const { return sum_xy / std::sqrt(sum_xx * sum_yy); }
38 double mean_ratio() const { return std::abs(mean_y) / std::abs(mean_x); }
39};

43template<typename Func, typename T>
44void for_matching_reflections(const std::vector<T>& a,
45 const std::vector<T>& b,
46 const Func& func) {
47 auto r1 = a.begin();
48 auto r2 = b.begin();
49 while (r1 != a.end() && r2 != b.end()) {
50 if (r1->hkl == r2->hkl) {
51 func(*r1, *r2);
52 ++r1;
53 ++r2;
54 } else if (r1->hkl < r2->hkl) {
55 ++r1;
56 } else {
57 ++r2;
58 }
59 }
60}

120template<typename R>
121void move_to_asu(const GroupOps& gops, const Miller& hkl, int isym,
122 HklValue<std::complex<R>>& hkl_value) {
123 hkl_value.hkl = hkl;
124 // cf. Mtz::ensure_asu()
125 const Op& op = gops.sym_ops[(isym - 1) / 2];
126 double shift = op.phase_shift(hkl);
127 if (shift != 0) {
128 if (isym % 2 == 0)
129 shift = -shift;
130 double phase = std::arg(hkl_value.value) + shift;
131 hkl_value.value = std::polar(std::abs(hkl_value.value), (R)phase);
132 }
133}

136template<typename T>
137struct AsuData {
138 std::vector<HklValue<T>> v;
139 UnitCell unit_cell_;
140 const SpaceGroup* spacegroup_ = nullptr;
141 // function defining FPhiProxy interface
142 size_t stride() const { return 1; }
143 size_t size() const { return v.size(); }
144 Miller get_hkl(size_t n) const { return v[n].hkl; }
145 double get_f(size_t n) const { return std::abs(v[n].value); }
146 double get_phi(size_t n) const { return std::arg(v[n].value); }
147 const UnitCell& unit_cell() const { return unit_cell_; }
148 const SpaceGroup* spacegroup() const { return spacegroup_; }
149 void ensure_sorted() {
150 if (!std::is_sorted(v.begin(), v.end()))
151 std::sort(v.begin(), v.end());
152 }
153
154 void ensure_asu() {
155 if (!spacegroup_)
156 fail("AsuData::ensure_asu(): space group not set");
157 GroupOps gops = spacegroup_->operations();
158 ReciprocalAsu asu(spacegroup_);
159 for (HklValue<T>& hkl_value : v) {
160 const Miller& hkl = hkl_value.hkl;
161 if (asu.is_in(hkl))
162 continue;
163 auto result = asu.to_asu(hkl, gops);
164 impl::move_to_asu(gops, result.first, result.second, hkl_value);
165 }
166 }

168 // load values from one column
169 template<typename DataProxy>
170 void load_values(const DataProxy& proxy, const std::string& label,
171 bool as_is=false) {
172 std::size_t col = proxy.column_index(label);
173 unit_cell_ = proxy.unit_cell();
174 spacegroup_ = proxy.spacegroup();
175 for (size_t i = 0; i < proxy.size(); i += proxy.stride()) {
176 auto num = proxy.get_num(i + col);
177 if (!std::isnan(num))
178 v.push_back({proxy.get_hkl(i), (T)num});
179 }
180 if (!as_is) {
181 ensure_asu();
182 ensure_sorted();
183 }
184 }

186 // load values from two or more columns
187 template<int N, typename DataProxy>
188 void load_values(const DataProxy& proxy, const std::array<std::string,N>& labels,
189 bool as_is=false) {
190 std::array<std::size_t, N> cols;
191 for (int i = 0; i < N; ++i)
192 cols[i] = proxy.column_index(labels[i]);
193 unit_cell_ = proxy.unit_cell();
194 spacegroup_ = proxy.spacegroup();
195 using Val = typename T::value_type;
196 for (size_t i = 0; i < proxy.size(); i += proxy.stride()) {
197 std::array<Val, N> nums;
198 for (int j = 0; j < N; ++j)
199 nums[j] = (Val) proxy.get_num(i + cols[j]);
200 if (!std::any_of(nums.begin(), nums.end(), [](Val f) { return std::isnan(f); })) {
201 v.emplace_back();
202 v.back().hkl = proxy.get_hkl(i);
203 set_value_from_array(v.back().value, nums);
204 }
205 }
206 if (!as_is) {
207 ensure_asu();
208 ensure_sorted();
209 }
210 }
