176 // BFS
177 std::vector<AtomId> find_shortest_path(const AtomId& a, const AtomId& b,
178 std::vector<AtomId> visited) const {
179 int start = (int) visited.size();
180 int end = -1;
181 visited.push_back(b);
182 std::vector<int> parent(visited.size(), -1);
183 for (int n = start; end == -1 && n != (int) visited.size(); ++n) {
184 for (const Bond& bond : bonds)
185 if (const AtomId* id = bond.other(visited[n])) {
186 if (*id == a)
187 end = (int) visited.size();
188 if (!in_vector(*id, visited)) {
189 visited.push_back(*id);
190 parent.push_back(n);
191 }
192 }
193 }
194 std::vector<AtomId> path;
195 for (int n = end; n != -1; n = parent[n])
196 path.push_back(visited[n]);
197 return path;
198 }

268inline double chiral_abs_volume(double bond1, double bond2, double bond3,
269 double angle1, double angle2, double angle3) {
270 double mult = bond1 * bond2 * bond3;
271 double x = 1;
272 double y = 2;
273 for (double a : {angle1, angle2, angle3}) {
274 double cosine = a == 90. ? 0. : std::cos(rad(a));
275 x -= cosine * cosine;
276 y *= cosine;
277 }
278 return mult * std::sqrt(x + y);
279}

350 void remove_nonmatching_restraints() {
351 vector_remove_if(rt.bonds, [&](const Restraints::Bond& x) {
352 return !has_atom(x.id1.atom) ||
353 !has_atom(x.id2.atom);
354 });

358 !has_atom(x.id3.atom);

362 !has_atom(x.id2.atom) ||
363 !has_atom(x.id3.atom) ||

365 });

367 return !has_atom(x.id_ctr.atom) ||
368 !has_atom(x.id1.atom) ||
369 !has_atom(x.id2.atom) ||

371 });
372 for (Restraints::Plane& plane : rt.planes)
373 vector_remove_if(plane.ids, [&](const Restraints::AtomId& x) {

387inline BondType bond_type_from_string(const std::string& s) {
388 if (istarts_with(s, "sing"))
389 return BondType::Single;
390 if (istarts_with(s, "doub"))
391 return BondType::Double;
392 if (istarts_with(s, "trip"))
393 return BondType::Triple;
394 if (istarts_with(s, "arom"))
395 return BondType::Aromatic;
396 if (istarts_with(s, "metal"))
397 return BondType::Metal;
398 if (istarts_with(s, "delo") || s == "1.5")
399 return BondType::Deloc;
400 if (cif::is_null(s))
401 return BondType::Unspec;
402 // program PDB2TNT produces a restraint file with bond type 'coval'
403 if (s == "coval")
404 return BondType::Unspec;
405 throw std::out_of_range("Unexpected bond type: " + s);
406}

464inline ChemComp make_chemcomp_from_block(const cif::Block& block_) {
465 ChemComp cc;
466 cc.name = block_.name.substr(starts_with(block_.name, "comp_") ? 5 : 0);
467 cif::Block& block = const_cast<cif::Block&>(block_);
468 // CCD uses _chem_comp.type, monomer libraries use .group in separate block,
469 // but just in case check for the presence of _chem_comp.group here.
470 cif::Column group_col = block.find_values("_chem_comp.group");
471 if (!group_col)
472 group_col = block.find_values("_chem_comp.type");
473 if (group_col)
474 cc.group = group_col.str(0);
475 for (auto row : block.find("_chem_comp_atom.",
476 {"atom_id", "type_symbol", "?type_energy",
477 "?charge", "?partial_charge"}))
478 cc.atoms.push_back({row.str(0), Element(row.str(1)),
479 (float) cif::as_number(row.one_of(3, 4), 0.0),
480 row.has(2) ? row.str(2) : ""});
481 for (auto row : block.find("_chem_comp_bond.",
482 {"atom_id_1", "atom_id_2", // 0, 1
483 "?type", "?value_order", // 2, 3
484 "?aromatic", "?pdbx_aromatic_flag", // 4, 5
485 "?value_dist", "?value_dist_esd", // 6, 7
486 "?value_dist_nucleus", "?value_dist_nucleus_esd"})) { // 8, 9
487 bool aromatic_flag = (row.one_of(4, 5)[0] | 0x20) == 'y';
488 double dist = row.has(6) ? cif::as_number(row[6]) : NAN;
489 double esd = row.has(7) ? cif::as_number(row[7]) : NAN;
490 double dist_nucl = row.has(8) ? cif::as_number(row[8]) : NAN;
491 double esd_nucl = row.has(9) ? cif::as_number(row[9]) : NAN;
492 cc.rt.bonds.push_back({{1, row.str(0)}, {1, row.str(1)},
493 bond_type_from_string(row.one_of(2, 3)),
494 aromatic_flag, dist, esd, dist_nucl, esd_nucl});
495 }
496 for (auto row : block.find("_chem_comp_angle.",
497 {"atom_id_1", "atom_id_2", "atom_id_3",
498 "value_angle", "value_angle_esd"}))
499 cc.rt.angles.push_back({{1, row.str(0)}, {1, row.str(1)}, {1, row.str(2)},
500 cif::as_number(row[3]), cif::as_number(row[4])});
501 for (auto row : block.find("_chem_comp_tor.",
502 {"id",
503 "atom_id_1", "atom_id_2",
504 "atom_id_3", "atom_id_4",
505 "value_angle", "value_angle_esd",
506 "period"}))
507 cc.rt.torsions.push_back({row.str(0),
508 {1, row.str(1)}, {1, row.str(2)},
509 {1, row.str(3)}, {1, row.str(4)},
510 cif::as_number(row[5]), cif::as_number(row[6]),
511 cif::as_int(row[7])});
512 for (auto row : block.find("_chem_comp_chir.",
513 {"atom_id_centre",
514 "atom_id_1", "atom_id_2", "atom_id_3",
515 "volume_sign"}))
516 if (row[4][0] != 'c') // ignore crossN
517 cc.rt.chirs.push_back({{1, row.str(0)},
518 {1, row.str(1)}, {1, row.str(2)}, {1, row.str(3)},
519 chirality_from_string(row[4])});
520 // mmCIF compliant
521 cif::Table chir_tab = block.find("_chem_comp_chir_atom.",
522 {"chir_id", "atom_id"});
523 if (chir_tab.ok()) {
524 std::map<std::string, std::vector<std::string>> chir_atoms;
525 for (auto chir : chir_tab)
526 chir_atoms[chir[0]].push_back(chir[1]);
527 for (auto row : block.find("_chem_comp_chir.",
528 {"id", "atom_id", "volume_flag", "volume_three"})) {
529 auto atoms = chir_atoms.find(row.str(0));
530 if (atoms != chir_atoms.end() && atoms->second.size() == 3)
531 cc.rt.chirs.push_back({{1, row.str(1)},
532 {1, atoms->second[0]}, {1, atoms->second[1]},
533 {1, atoms->second[2]},
534 chirality_from_flag_and_volume(row[2],
535 cif::as_number(row[3]))});
536 }
537 }
538 for (auto row : block.find("_chem_comp_plane_atom.",
539 {"plane_id", "atom_id" , "dist_esd"})) {
540 Restraints::Plane& plane = cc.rt.get_or_add_plane(row.str(0));
541 if (plane.esd == 0.0)
542 plane.esd = cif::as_number(row[2]);
543 plane.ids.push_back({1, row.str(1)});
544 }
545 return cc;
546}

548inline ChemComp make_chemcomp_from_cif(const std::string& name,
549 const cif::Document& doc) {
550 const cif::Block* block = doc.find_block("comp_" + name);
551 if (!block)
552 block = doc.find_block(name);
553 if (!block)
554 throw std::runtime_error("data_comp_" + name + " not in the cif file");
555 ChemComp cc = make_chemcomp_from_block(*block);
556 if (cc.group.empty())
557 if (const cif::Block* list = doc.find_block("comp_list")) {
558 cif::Table tab =
559 const_cast<cif::Block*>(list)->find("_chem_comp.", {"id", "group"});
560 if (tab.ok())
561 cc.group = tab.find_row(name).str(1);
562 }
563 return cc;
564}
