20inline CoorFormat coor_format_from_ext(const std::string& path) {
21 if (iends_with(path, ".pdb") || iends_with(path, ".ent"))
22 return CoorFormat::Pdb;
23 if (iends_with(path, ".cif") || iends_with(path, ".mmcif"))
24 return CoorFormat::Mmcif;
25 if (iends_with(path, ".json"))
26 return CoorFormat::Mmjson;
27 return CoorFormat::Unknown;
28}

30// If it's neither CIF nor JSON nor almost empty - we assume PDB.
31inline CoorFormat coor_format_from_content(const char* buf, const char* end) {
32 while (buf < end - 8) {
33 if (std::isspace(*buf)) {
34 ++buf;
35 } else if (*buf == '#') {
36 while (buf < end - 8 && *buf != '\n')
37 ++buf;
38 } else if (*buf == '{') {
39 return CoorFormat::Mmjson;
40 } else if (ialpha4_id(buf) == ialpha4_id("data") && buf[4] == '_') {
41 return CoorFormat::Mmcif;
42 } else {
43 return CoorFormat::Pdb;
44 }
45 }
46 return CoorFormat::Unknown;
47}

59inline Structure read_structure_from_char_array(char* data, size_t size,
60 const std::string& path) {
61 CoorFormat format = coor_format_from_content(data, data + size);
62 if (format == CoorFormat::Pdb)
63 return read_pdb_from_memory(data, size, path);
64 if (format == CoorFormat::Mmcif)
65 return make_structure_from_doc(cif::read_memory(data, size, path.c_str()), true);
66 if (format == CoorFormat::Mmjson)
67 return make_structure(cif::read_mmjson_insitu(data, size, path));
68 fail("wrong format of coordinate file " + path);
69}

71template<typename T>
72Structure read_structure(T&& input, CoorFormat format=CoorFormat::Unknown) {
73 if (format == CoorFormat::Detect) {
74 CharArray mem = read_into_buffer(input);
75 return read_structure_from_char_array(mem.data(), mem.size(), input.path());
76 }
77 if (format == CoorFormat::Unknown)
78 format = coor_format_from_ext(input.basepath());
79 switch (format) {
80 case CoorFormat::Pdb:
81 return read_pdb(input);
82 case CoorFormat::Mmcif:
83 return make_structure(cif::read(input));
84 case CoorFormat::Mmjson:
85 return make_structure(cif::read_mmjson(input));
86 case CoorFormat::ChemComp:
87 return make_structure_from_chemcomp_doc(cif::read(input));
88 case CoorFormat::Unknown:
89 case CoorFormat::Detect:
90 fail("Unknown format of " +
91 (input.path().empty() ? "coordinate file" : input.path()) + ".");
92 }
93 unreachable();
94}
