17SmallStructure make_small_structure_from_block(const cif::Block& block_) {
18 using cif::as_number;
19 using cif::as_string;
20 cif::Block& block = const_cast<cif::Block&>(block_);
21 SmallStructure st;
22 st.name = block.name;
23
24 // unit cell and symmetry
25 cif::Table cell = block.find("_cell_",
26 {"length_a", "length_b", "length_c",
27 "angle_alpha", "angle_beta", "angle_gamma"});
28 if (cell.ok()) {
29 auto c = cell.one();
30 if (!cif::is_null(c[0]) && !cif::is_null(c[1]) && !cif::is_null(c[2]))
31 st.cell.set(as_number(c[0]), as_number(c[1]), as_number(c[2]),
32 as_number(c[3]), as_number(c[4]), as_number(c[5]));
33 }
34 for (const char* tag : {"_space_group_name_H-M_alt",
35 "_symmetry_space_group_name_H-M"})
36 if (const std::string* val = block.find_value(tag)) {
37 st.spacegroup_hm = as_string(*val);
38 break;
39 }
40
41 enum { kLabel, kSymbol, kX, kY, kZ, kUiso, kBiso, kOcc, kDisorderGroup };
42 cif::Table atom_table = block.find("_atom_site_",
43 {"label",
44 "?type_symbol",
45 "?fract_x",
46 "?fract_y",
47 "?fract_z",
48 "?U_iso_or_equiv",
49 "?B_iso_or_equiv",
50 "?occupancy",
51 "?disorder_group"});
52 for (auto row : atom_table) {
53 SmallStructure::Site site;
54 site.label = as_string(row[kLabel]);
55 if (row.has(kSymbol))
56 site.type_symbol = as_string(row[kSymbol]);
57 else
58 site.type_symbol = site.label;
59 if (row.has(kX))
60 site.fract.x = as_number(row[kX]);
61 if (row.has(kY))
62 site.fract.y = as_number(row[kY]);
63 if (row.has(kZ))
64 site.fract.z = as_number(row[kZ]);
65 if (row.has(kUiso))
66 site.u_iso = as_number(row[kUiso], 0.0);
67 else if (row.has(kBiso))
68 site.u_iso = as_number(row[kBiso], 0.0) / u_to_b();
69 if (row.has(kOcc))
70 site.occ = as_number(row[kOcc], 1.0);
71 if (row.has2(kDisorderGroup))
72 // ignore non-integer _atom_site_disorder_group
73 site.disorder_group = string_to_int(row[kDisorderGroup], false);
74 split_element_and_charge(site.type_symbol, &site);
75 st.sites.push_back(site);
76 }
77
78 std::vector<SmallStructure::Site>::iterator aniso_site = st.sites.begin();
79 for (auto row : block.find("_atom_site_aniso_",
80 {"label", "U_11", "U_22", "U_33",
81 "U_12", "U_13", "U_23"})) {
82 std::string aniso_label = row.str(0);
83 if (aniso_site == st.sites.end() || aniso_site->label != aniso_label) {
84 aniso_site = std::find_if(st.sites.begin(), st.sites.end(),
85 [&](SmallStructure::Site& x) { return x.label == aniso_label; });
86 if (aniso_site == st.sites.end())
87 continue;
88 }
89 aniso_site->aniso.u11 = as_number(row[1], 0.0);
90 aniso_site->aniso.u22 = as_number(row[2], 0.0);
91 aniso_site->aniso.u33 = as_number(row[3], 0.0);
92 aniso_site->aniso.u12 = as_number(row[4], 0.0);
93 aniso_site->aniso.u13 = as_number(row[5], 0.0);
94 aniso_site->aniso.u23 = as_number(row[6], 0.0);
95 ++aniso_site;
96 }
97
98 for (auto row : block.find("_atom_type_",
99 {"symbol",
100 "scat_dispersion_real",
101 "scat_dispersion_imag"})) {
102 SmallStructure::AtomType atom_type;
103 atom_type.symbol = row.str(0);
104 atom_type.dispersion_real = cif::as_number(row[1]);
105 atom_type.dispersion_imag = cif::as_number(row[2]);
106 split_element_and_charge(atom_type.symbol, &atom_type);
107 st.atom_types.push_back(atom_type);
108 }
109 if (cif::Column w_col = block.find_values("_diffrn_radiation_wavelength"))
110 st.wavelength = cif::as_number(w_col.at(0));
111 st.setup_cell_images();
112
113 return st;
114}

116inline cif::Block make_cif_block_from_small_structure(const SmallStructure& st) {
117 cif::Block block;
118 block.name = st.name;
119 if (st.cell.is_crystal()) {
120 block.set_pair("_cell_length_a", to_str(st.cell.a));
121 block.set_pair("_cell_length_b", to_str(st.cell.b));
122 block.set_pair("_cell_length_c", to_str(st.cell.c));
123 block.set_pair("_cell_angle_alpha", to_str(st.cell.alpha));
124 block.set_pair("_cell_angle_beta", to_str(st.cell.beta));
125 block.set_pair("_cell_angle_gamma", to_str(st.cell.gamma));
126 }
127 if (!st.spacegroup_hm.empty())
128 block.set_pair("_symmetry_space_group_name_H-M", cif::quote(st.spacegroup_hm));
129 if (st.wavelength != 0)
130 block.set_pair("_diffrn_radiation_wavelength", to_str(st.wavelength));
131
132 cif::Loop& atom_loop = block.init_loop("_atom_site_", {"label",
133 "type_symbol",
134 "fract_x",
135 "fract_y",
136 "fract_z",
137 "U_iso_or_equiv",
138 "occupancy",
139 "disorder_group"});
140 for (const SmallStructure::Site& site: st.sites) {
141 atom_loop.add_row({
142 cif::quote(site.label),
143 site.type_symbol.empty() ? site.element_and_charge_symbol()
144 : cif::quote(site.type_symbol),
145 to_str(site.fract.x),
146 to_str(site.fract.y),
147 to_str(site.fract.z),
148 to_str(site.u_iso),
149 to_str(site.occ),
150 site.disorder_group == 0 ? "." : std::to_string(site.disorder_group)
151 });
152 }
153
154 bool has_aniso = false;
155 for (const SmallStructure::Site& site: st.sites)
156 if (site.aniso.nonzero()) {
157 has_aniso = true;
158 break;
159 }
160 if (has_aniso) {
161 cif::Loop& aniso_loop = block.init_loop("_atom_site_aniso_", {
162 "label", "U_11", "U_22", "U_33", "U_12", "U_13", "U_23"});
163 for (const SmallStructure::Site& site: st.sites)
164 if (site.aniso.nonzero())
165 aniso_loop.add_row({
166 cif::quote(site.label),
167 to_str(site.aniso.u11), to_str(site.aniso.u22), to_str(site.aniso.u33),
168 to_str(site.aniso.u12), to_str(site.aniso.u13), to_str(site.aniso.u23)
169 });
170 }
171
172 return block;
173}
