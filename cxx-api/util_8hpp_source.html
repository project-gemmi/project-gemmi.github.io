105inline std::string trim_str(const std::string& str) {
106 const std::string ws = " \r\n\t";
107 std::string::size_type first = str.find_first_not_of(ws);
108 if (first == std::string::npos)
109 return std::string{};
110 std::string::size_type last = str.find_last_not_of(ws);
111 return str.substr(first, last - first + 1);
112}

114inline std::string rtrim_str(const std::string& str) {
115 std::string::size_type last = str.find_last_not_of(" \r\n\t");
116 return str.substr(0, last == std::string::npos ? 0 : last + 1);
117}

119// end is after the last character of the string (typically \0)
120inline const char* rtrim_cstr(const char* start, const char* end=nullptr) {
121 if (!start)
122 return nullptr;
123 if (!end) {
124 end = start;
125 while (*end != '\0')
126 ++end;
127 }
128 while (end > start && std::isspace(end[-1]))
129 --end;
130 return end;
131}

138// takes a single separator (usually char or string);
139// may return empty fields
140template<typename S>
141void split_str_into(const std::string& str, S sep,
142 std::vector<std::string>& result) {
143 std::size_t start = 0, end;
144 while ((end = str.find(sep, start)) != std::string::npos) {
145 result.emplace_back(str, start, end - start);
146 start = end + impl::length(sep);
147 }
148 result.emplace_back(str, start);
149}

151template<typename S>
152std::vector<std::string> split_str(const std::string& str, S sep) {
153 std::vector<std::string> result;
154 split_str_into(str, sep, result);
155 return result;
156}

158// _multi variants takes multiple 1-char separators as a string;
159// discards empty fields
160inline void split_str_into_multi(const std::string& str, const char* seps,
161 std::vector<std::string>& result) {
162 std::size_t start = str.find_first_not_of(seps);
163 while (start != std::string::npos) {
164 std::size_t end = str.find_first_of(seps, start);
165 result.emplace_back(str, start, end - start);
166 start = str.find_first_not_of(seps, end);
167 }
168}

177template<typename T, typename S, typename F>
178std::string join_str(T begin, T end, const S& sep, const F& getter) {
179 std::string r;
180 bool first = true;
181 for (T i = begin; i != end; ++i) {
182 if (!first)
183 r += sep;
184 r += getter(*i);

212inline void replace_all(std::string &s,

214 std::string::size_type pos = 0;

221// list is a comma separated string
222inline bool is_in_list(const std::string& name, const std::string& list,
223 char sep=',') {
224 if (name.length() >= list.length())
225 return name == list;
226 for (size_t start=0, end=0; end != std::string::npos; start=end+1) {
227 end = list.find(sep, start);
228 if (list.compare(start, end - start, name) == 0)
229 return true;
230 }
231 return false;
232}

269// Numeric ID used for case-insensitive comparison of 4 letters.
270// s must have 4 chars or 3 chars + NUL, ' ' and NUL are equivalent in s.
271constexpr int ialpha4_id(const char* s) {
272 return (s[0] << 24 | s[1] << 16 | s[2] << 8 | s[3]) & ~0x20202020;
273}
