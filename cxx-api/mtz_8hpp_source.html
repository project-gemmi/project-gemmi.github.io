31template <typename T, typename FP=typename std::iterator_traits<T>::value_type>
32std::array<FP,2> calculate_min_max_disregarding_nans(T begin, T end) {
33 std::array<FP,2> minmax = {{NAN, NAN}};
34 T i = begin;
35 while (i != end && std::isnan(*i))
36 ++i;
37 if (i != end) {
38 minmax[0] = minmax[1] = *i;
39 while (++i != end) {
40 if (*i < minmax[0])
41 minmax[0] = *i;
42 else if (*i > minmax[1])
43 minmax[1] = *i;
44 }
45 }
46 return minmax;
47}

261 UnitCell get_average_cell_from_batch_headers(double* rmsd) const {
262 if (rmsd)
263 for (int i = 0; i < 6; ++i)
264 rmsd[i] = 0.;
265 double avg[6] = {0., 0., 0., 0., 0., 0.};
266 for (const Batch& batch : batches)
267 for (int i = 0; i < 6; ++i) {
268 // if batch headers are not set correctly, return global cell
269 if (batch.floats[i] <= 0)
270 return cell;
271 avg[i] += batch.floats[i];
272 }
273 if (avg[0] <= 0 || avg[1] <= 0 || avg[2] <= 0 ||
274 avg[3] <= 0 || avg[4] <= 0 || avg[5] <= 0)
275 return UnitCell();
276 size_t n = batches.size();
277 for (int i = 0; i < 6; ++i)
278 avg[i] /= n;
279 if (rmsd) {
280 for (const Batch& batch : batches)
281 for (int i = 0; i < 6; ++i)
282 rmsd[i] += sq(avg[i] - batch.floats[i]);
283 for (int i = 0; i < 6; ++i)
284 rmsd[i] = std::sqrt(rmsd[i] / n);
285 }
286 return UnitCell(avg[0], avg[1], avg[2], avg[3], avg[4], avg[5]);
287 }

462 std::array<double,2> calculate_min_max_1_d2() const {
463 if (!has_data() || columns.size() < 3)
464 fail("No data.");
465 double min_value = INFINITY;
466 double max_value = 0.;
467 if (cell.is_crystal() && cell.a > 0)
468 extend_min_max_1_d2(cell, min_value, max_value);
469 const UnitCell* prev_cell = nullptr;
470 for (const Dataset& ds : datasets)
471 if (ds.cell.is_crystal() && ds.cell.a > 0 && ds.cell != cell &&
472 (!prev_cell || ds.cell != *prev_cell)) {
473 extend_min_max_1_d2(ds.cell, min_value, max_value);
474 prev_cell = &ds.cell;
475 }
476 if (min_value == INFINITY)
477 min_value = 0;
478 return {{min_value, max_value}};
479 }

494 template<typename Stream>
495 void read_first_bytes(Stream& stream) {
496 char buf[20] = {0};
497
498 if (!stream.read(buf, 20))
499 fail("Could not read the MTZ file (is it empty?)");
500 if (buf[0] != 'M' || buf[1] != 'T' || buf[2] != 'Z' || buf[3] != ' ')
501 fail("Not an MTZ file - it does not start with 'MTZ '");
502
503 // Bytes 9-12 have so-called machine stamp:
504 // "The first 4 half-bytes represent the real, complex, integer and
505 // character formats".
506 // We don't try to handle all the combinations here, only the two most
507 // common: big endian (for all types) and little endian (for all types).
508 // BE is denoted by 1 and LE by 4.
509 // If we get a value different than 1 and 4 we assume the native byte order.
510 if ((buf[9] & 0xf0) == (is_little_endian() ? 0x10 : 0x40))
511 toggle_endianness();
512
513 std::int32_t tmp_header_offset;
514 std::memcpy(&tmp_header_offset, buf + 4, 4);
515 if (!same_byte_order)
516 swap_four_bytes(&tmp_header_offset);
517
518 if (tmp_header_offset == -1) {
519 std::memcpy(&header_offset, buf + 12, 8);
520 if (!same_byte_order) {
521 swap_eight_bytes(&header_offset);
522 }
523 } else {
524 header_offset = (int64_t) tmp_header_offset;
525 }
526 }

553 // read headers until END
554 template<typename Stream>
555 void read_main_headers(Stream& stream) {
556 char line[81] = {0};
557 seek_headers(stream);
558 int ncol = 0;
559 bool has_batch = false;
560 while (stream.read(line, 80) && ialpha3_id(line) != ialpha3_id("END")) {
561 const char* args = skip_word(line);
562 switch (ialpha4_id(line)) {
563 case ialpha4_id("VERS"):
564 version_stamp = rtrim_str(args);
565 break;
566 case ialpha4_id("TITL"):
567 title = rtrim_str(args);
568 break;
569 case ialpha4_id("NCOL"): {
570 ncol = simple_atoi(args, &args);
571 nreflections = simple_atoi(args, &args);
572 int nbatches = simple_atoi(args);
573 if (nbatches < 0 || nbatches > 10000000) // sanity check
574 fail("Wrong NCOL header");
575 batches.resize(nbatches);
576 break;
577 }
578 case ialpha4_id("CELL"):
579 cell = read_cell_parameters(args);
580 break;
581 case ialpha4_id("SORT"):
582 for (int& n : sort_order)
583 n = simple_atoi(args, &args);
584 break;
585 case ialpha4_id("SYMI"): {
586 nsymop = simple_atoi(args, &args);
587 symops.reserve(nsymop);
588 simple_atoi(args, &args); // ignore number of primitive operations
589 args = skip_word(skip_blank(args)); // ignore lattice type
590 spacegroup_number = simple_atoi(args, &args);
591 args = skip_blank(args);
592 if (*args != '\'')
593 spacegroup_name = read_word(args);
594 else if (const char* end = std::strchr(++args, '\''))
595 spacegroup_name.assign(args, end);
596 // ignore point group which is at the end of args
597 break;
598 }
599 case ialpha4_id("SYMM"):
600 symops.push_back(parse_triplet(args));
601 break;
602 case ialpha4_id("RESO"):
603 min_1_d2 = fast_atof(args, &args);
604 max_1_d2 = fast_atof(args, &args);
605 break;
606 case ialpha4_id("VALM"):
607 if (*args != 'N') {
608 const char* endptr;
609 float v = (float) fast_atof(args, &endptr);
610 if (*endptr == '\0' || is_space(*endptr))
611 valm = v;
612 else
613 logger.note("Unexpected VALM value: " + rtrim_str(args));
614 }
615 break;
616 case ialpha4_id("COLU"): {
617 columns.emplace_back();
618 Column& col = columns.back();
619 col.label = read_word(args, &args);
620 col.type = read_word(args, &args)[0];
621 col.min_value = (float) fast_atof(args, &args);
622 col.max_value = (float) fast_atof(args, &args);
623 col.dataset_id = simple_atoi(args);
624 col.parent = this;
625 col.idx = columns.size() - 1;
626 break;
627 }
628 case ialpha4_id("COLS"):
629 // COLSRC is undocumented. CMTZ (libccp4) adds it after COLUMN:
630 // COLUMN IMEAN J -300.600006 4619 1
631 // COLSRC IMEAN CREATED_07/08/2019_11:00:23 1
632 if (!columns.empty() && columns.back().label == read_word(args, &args))
633 columns.back().source = read_word(args);
634 else
635 logger.note("MTZ: COLSRC is not after matching COLUMN");
636 break;
637 case ialpha4_id("COLG"):
638 // Column group - not used.
639 break;
640 case ialpha4_id("NDIF"):
641 datasets.reserve(simple_atoi(args));
642 break;
643 case ialpha4_id("PROJ"):
644 datasets.emplace_back();
645 datasets.back().id = simple_atoi(args, &args);
646 datasets.back().project_name = read_word(skip_word(args));
647 datasets.back().wavelength = 0.0;
648 break;
649 case ialpha4_id("CRYS"):
650 if (simple_atoi(args, &args) == last_dataset().id)
651 datasets.back().crystal_name = read_word(args);
652 else
653 logger.note("MTZ CRYSTAL line: unusual numbering.");
654 break;
655 case ialpha4_id("DATA"):
656 if (simple_atoi(args, &args) == last_dataset().id)
657 datasets.back().dataset_name = read_word(args);
658 else
659 logger.note("MTZ DATASET line: unusual numbering.");
660 break;
661 case ialpha4_id("DCEL"):
662 if (simple_atoi(args, &args) == last_dataset().id)
663 datasets.back().cell = read_cell_parameters(args);
664 else
665 logger.note("MTZ DCELL line: unusual numbering.");
666 break;
667 // case("DRES"): not in use yet
668 case ialpha4_id("DWAV"):
669 if (simple_atoi(args, &args) == last_dataset().id)
670 datasets.back().wavelength = fast_atof(args);
671 else
672 logger.note("MTZ DWAV line: unusual numbering.");
673 break;
674 case ialpha4_id("BATCH"):
675 // We take number of batches from the NCOL record and serial numbers
676 // from BH. This header could be used only to check consistency.
677 has_batch = true;
678 break;
679 default:
680 logger.note("Unknown header: " + rtrim_str(line));
681 }
682 }
683 if (ncol != (int) columns.size())
684 fail("Number of COLU records inconsistent with NCOL record.");
685 if (has_batch != !batches.empty())
686 fail("BATCH header inconsistent with NCOL record.");
687 }

689 // read the part between END and MTZENDOFHEADERS
690 template<typename Stream>
691 void read_history_and_batch_headers(Stream& stream) {
692 char buf[81] = {0};
693 int n_headers = 0;
694 while (stream.read(buf, 80) && ialpha4_id(buf) != ialpha4_id("MTZE")) {
695 if (n_headers != 0) {
696 const char* start = skip_blank(buf);
697 const char* end = rtrim_cstr(start, start+80);
698 history.emplace_back(start, end);
699 --n_headers;
700 } else if (ialpha4_id(buf) == ialpha4_id("MTZH")) {
701 n_headers = simple_atoi(skip_word(buf));
702 if (n_headers < 0 || n_headers > 30) {
703 logger.note("Wrong MTZ: number of headers should be between 0 and 30");
704 return;
705 }
706 history.reserve(n_headers);
707 } else if (ialpha4_id(buf) == ialpha4_id("MTZB")) {
708 for (Batch& batch : batches) {
709 stream.read(buf, 80);
710 if (ialpha3_id(buf) != ialpha3_id("BH "))
711 fail("Missing BH header");
712 const char* args = skip_word(buf);
713 batch.number = simple_atoi(args, &args);
714 int total_words = simple_atoi(args, &args);
715 int int_words = simple_atoi(args, &args);
716 int float_words = simple_atoi(args);
717 if (total_words != int_words + float_words || total_words > 1000)
718 fail("Wrong BH header");
719 stream.read(buf, 80); // TITLE
720 const char* end = rtrim_cstr(buf + 6, buf+76);
721 batch.title.assign(buf, end - buf);
722 batch.ints.resize(int_words);
723 stream.read(batch.ints.data(), int_words * 4);
724 batch.floats.resize(float_words);
725 stream.read(batch.floats.data(), float_words * 4);
726 stream.read(buf, 80);
727 if (ialpha4_id(buf) != ialpha4_id("BHCH"))
728 fail("Missing BHCH header");
729 split_str_into_multi(buf + 5, " \t", batch.axes);
730 }
731 }
732 }
733 appended_text = stream.read_rest();
734 }

858 template <typename Func>
859 void remove_rows_if(Func condition) {
860 if (!has_data())
861 fail("No data.");
862 auto out = data.begin();
863 size_t width = columns.size();
864 for (auto r = data.begin(); r < data.end(); r += width)
865 if (!condition(&*r)) {
866 if (r != out)
867 std::copy(r, r + width, out);
868 out += width;
869 }
870 data.erase(out, data.end());
871 nreflections = int(data.size() / width);
872 }

876 if (data.size() != old_row_size * nreflections)
877 fail("Internal error");

879 if (pos > old_row_size)
880 fail("expand_data_rows(): pos out of range");
881 vector_insert_columns(data, old_row_size, (size_t)nreflections, added, pos, NAN);
882 }
