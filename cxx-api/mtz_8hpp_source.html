31template <typename T, typename FP=typename std::iterator_traits<T>::value_type>
32std::array<FP,2> calculate_min_max_disregarding_nans(T begin, T end) {
33 std::array<FP,2> minmax = {{NAN, NAN}};
34 T i = begin;
35 while (i != end && std::isnan(*i))
36 ++i;
37 if (i != end) {
38 minmax[0] = minmax[1] = *i;
39 while (++i != end) {
40 if (*i < minmax[0])
41 minmax[0] = *i;
42 else if (*i > minmax[1])
43 minmax[1] = *i;
44 }
45 }
46 return minmax;
47}

249 UnitCell get_average_cell_from_batch_headers(double* rmsd) const {
250 if (rmsd)
251 for (int i = 0; i < 6; ++i)
252 rmsd[i] = 0.;
253 double avg[6] = {0., 0., 0., 0., 0., 0.};
254 for (const Batch& batch : batches)
255 for (int i = 0; i < 6; ++i) {
256 // if batch headers are not set correctly, return global cell
257 if (batch.floats[i] <= 0)
258 return cell;
259 avg[i] += batch.floats[i];
260 }
261 if (avg[0] <= 0 || avg[1] <= 0 || avg[2] <= 0 ||
262 avg[3] <= 0 || avg[4] <= 0 || avg[5] <= 0)
263 return UnitCell();
264 size_t n = batches.size();
265 for (int i = 0; i < 6; ++i)
266 avg[i] /= n;
267 if (rmsd) {
268 for (const Batch& batch : batches)
269 for (int i = 0; i < 6; ++i)
270 rmsd[i] += sq(avg[i] - batch.floats[i]);
271 for (int i = 0; i < 6; ++i)
272 rmsd[i] = std::sqrt(rmsd[i] / n);
273 }
274 return UnitCell(avg[0], avg[1], avg[2], avg[3], avg[4], avg[5]);
275 }

438 if (!has_data() || columns.size() < 3)

469 template<typename Stream>
470 void read_first_bytes(Stream& stream) {
471 char buf[20] = {0};
472
473 if (!stream.read(buf, 20))
474 fail("Could not read the MTZ file (is it empty?)");
475 if (buf[0] != 'M' || buf[1] != 'T' || buf[2] != 'Z' || buf[3] != ' ')
476 fail("Not an MTZ file - it does not start with 'MTZ '");
477
478 // Bytes 9-12 have so-called machine stamp:
479 // "The first 4 half-bytes represent the real, complex, integer and
480 // character formats".
481 // We don't try to handle all the combinations here, only the two most
482 // common: big endian (for all types) and little endian (for all types).
483 // BE is denoted by 1 and LE by 4.
484 // If we get a value different than 1 and 4 we assume the native byte order.
485 if ((buf[9] & 0xf0) == (is_little_endian() ? 0x10 : 0x40))
486 toggle_endiannes();
487
488 std::int32_t tmp_header_offset;
489 std::memcpy(&tmp_header_offset, buf + 4, 4);
490 if (!same_byte_order)
491 swap_four_bytes(&tmp_header_offset);
492
493 if (tmp_header_offset == -1) {
494 std::memcpy(&header_offset, buf + 12, 8);
495 if (!same_byte_order) {
496 swap_eight_bytes(&header_offset);
497 }
498 } else {
499 header_offset = (int64_t) tmp_header_offset;
500 }
501 }

533 // read headers until END
534 template<typename Stream>
535 void read_main_headers(Stream& stream) {
536 char line[81] = {0};
537 seek_headers(stream);
538 int ncol = 0;
539 bool has_batch = false;
540 while (stream.read(line, 80) && ialpha3_id(line) != ialpha3_id("END")) {
541 const char* args = skip_word(line);
542 switch (ialpha4_id(line)) {
543 case ialpha4_id("VERS"):
544 version_stamp = rtrim_str(args);
545 break;
546 case ialpha4_id("TITL"):
547 title = rtrim_str(args);
548 break;
549 case ialpha4_id("NCOL"): {
550 ncol = simple_atoi(args, &args);
551 nreflections = simple_atoi(args, &args);
552 int nbatches = simple_atoi(args);
553 if (nbatches < 0 || nbatches > 10000000) // sanity check
554 fail("Wrong NCOL header");
555 batches.resize(nbatches);
556 break;
557 }
558 case ialpha4_id("CELL"):
559 cell = read_cell_parameters(args);
560 break;
561 case ialpha4_id("SORT"):
562 for (int& n : sort_order)
563 n = simple_atoi(args, &args);
564 break;
565 case ialpha4_id("SYMI"): {
566 nsymop = simple_atoi(args, &args);
567 symops.reserve(nsymop);
568 simple_atoi(args, &args); // ignore number of primitive operations
569 args = skip_word(skip_blank(args)); // ignore lattice type
570 spacegroup_number = simple_atoi(args, &args);
571 args = skip_blank(args);
572 if (*args != '\'')
573 spacegroup_name = read_word(args);
574 else if (const char* end = std::strchr(++args, '\''))
575 spacegroup_name.assign(args, end);
576 // ignore point group which is at the end of args
577 break;
578 }
579 case ialpha4_id("SYMM"):
580 symops.push_back(parse_triplet(args));
581 break;
582 case ialpha4_id("RESO"):
583 min_1_d2 = fast_atof(args, &args);
584 max_1_d2 = fast_atof(args, &args);
585 break;
586 case ialpha4_id("VALM"):
587 if (*args != 'N') {
588 const char* endptr;
589 float v = (float) fast_atof(args, &endptr);
590 if (*endptr == '\0' || is_space(*endptr))
591 valm = v;
592 else
593 warn("Unexpected VALM value: " + rtrim_str(args));
594 }
595 break;
596 case ialpha4_id("COLU"): {
597 columns.emplace_back();
598 Column& col = columns.back();
599 col.label = read_word(args, &args);
600 col.type = read_word(args, &args)[0];
601 col.min_value = (float) fast_atof(args, &args);
602 col.max_value = (float) fast_atof(args, &args);
603 col.dataset_id = simple_atoi(args);
604 col.parent = this;
605 col.idx = columns.size() - 1;
606 break;
607 }
608 case ialpha4_id("COLS"):
609 if (columns.empty())
610 fail("MTZ: COLSRC before COLUMN?");
611 args = skip_word(args);
612 columns.back().source = read_word(args);
613 break;
614 case ialpha4_id("COLG"):
615 // Column group - not used.
616 break;
617 case ialpha4_id("NDIF"):
618 datasets.reserve(simple_atoi(args));
619 break;
620 case ialpha4_id("PROJ"):
621 datasets.emplace_back();
622 datasets.back().id = simple_atoi(args, &args);
623 datasets.back().project_name = read_word(skip_word(args));
624 datasets.back().wavelength = 0.0;
625 break;
626 case ialpha4_id("CRYS"):
627 if (simple_atoi(args, &args) == last_dataset().id)
628 datasets.back().crystal_name = read_word(args);
629 else
630 warn("MTZ CRYSTAL line: unusual numbering.");
631 break;
632 case ialpha4_id("DATA"):
633 if (simple_atoi(args, &args) == last_dataset().id)
634 datasets.back().dataset_name = read_word(args);
635 else
636 warn("MTZ DATASET line: unusual numbering.");
637 break;
638 case ialpha4_id("DCEL"):
639 if (simple_atoi(args, &args) == last_dataset().id)
640 datasets.back().cell = read_cell_parameters(args);
641 else
642 warn("MTZ DCELL line: unusual numbering.");
643 break;
644 // case("DRES"): not in use yet
645 case ialpha4_id("DWAV"):
646 if (simple_atoi(args, &args) == last_dataset().id)
647 datasets.back().wavelength = fast_atof(args);
648 else
649 warn("MTZ DWAV line: unusual numbering.");
650 break;
651 case ialpha4_id("BATCH"):
652 // We take number of batches from the NCOL record and serial numbers
653 // from BH. This header could be used only to check consistency.
654 has_batch = true;
655 break;
656 default:
657 warn("Unknown header: " + rtrim_str(line));
658 }
659 }
660 if (ncol != (int) columns.size())
661 fail("Number of COLU records inconsistent with NCOL record.");
662 if (has_batch != !batches.empty())
663 fail("BATCH header inconsistent with NCOL record.");
664 }

666 // read the part between END and MTZENDOFHEADERS
667 template<typename Stream>
668 void read_history_and_batch_headers(Stream& stream) {
669 char buf[81] = {0};
670 int n_headers = 0;
671 while (stream.read(buf, 80) && ialpha4_id(buf) != ialpha4_id("MTZE")) {
672 if (n_headers != 0) {
673 const char* start = skip_blank(buf);
674 const char* end = rtrim_cstr(start, start+80);
675 history.emplace_back(start, end);
676 --n_headers;
677 } else if (ialpha4_id(buf) == ialpha4_id("MTZH")) {
678 n_headers = simple_atoi(skip_word(buf));
679 if (n_headers < 0 || n_headers > 30) {
680 warn("Wrong MTZ: number of headers should be between 0 and 30");
681 return;
682 }
683 history.reserve(n_headers);
684 } else if (ialpha4_id(buf) == ialpha4_id("MTZB")) {
685 for (Batch& batch : batches) {
686 stream.read(buf, 80);
687 if (ialpha3_id(buf) != ialpha3_id("BH "))
688 fail("Missing BH header");
689 const char* args = skip_word(buf);
690 batch.number = simple_atoi(args, &args);
691 int total_words = simple_atoi(args, &args);
692 int int_words = simple_atoi(args, &args);
693 int float_words = simple_atoi(args);
694 if (total_words != int_words + float_words || total_words > 1000)
695 fail("Wrong BH header");
696 stream.read(buf, 80); // TITLE
697 const char* end = rtrim_cstr(buf + 6, buf+76);
698 batch.title.assign(buf, end - buf);
699 batch.ints.resize(int_words);
700 stream.read(batch.ints.data(), int_words * 4);
701 batch.floats.resize(float_words);
702 stream.read(batch.floats.data(), float_words * 4);
703 stream.read(buf, 80);
704 if (ialpha4_id(buf) != ialpha4_id("BHCH"))
705 fail("Missing BHCH header");
706 split_str_into_multi(buf + 5, " \t", batch.axes);
707 }
708 }
709 }
710 appended_text = stream.read_rest();
711 }

817 // (for merged MTZ only) change HKL to ASU equivalent, adjust phases
818 void ensure_asu(bool tnt_asu=false) {
819 if (!is_merged())
820 fail("Mtz::ensure_asu() is for merged MTZ only");
821 if (!spacegroup)
822 return;
823 GroupOps gops = spacegroup->operations();
824 ReciprocalAsu asu(spacegroup, tnt_asu);
825 std::vector<int> phase_columns = positions_of_columns_with_type('P');
826 std::vector<int> abcd_columns = positions_of_columns_with_type('A');
827 std::vector<int> dano_columns = positions_of_columns_with_type('D');
828 std::vector<std::pair<int,int>> plus_minus_columns = positions_of_plus_minus_columns();
829 bool no_special_columns = phase_columns.empty() && abcd_columns.empty() &&
830 plus_minus_columns.empty() && dano_columns.empty();
831 bool centric = no_special_columns || gops.is_centrosymmetric();
832 for (size_t n = 0; n < data.size(); n += columns.size()) {
833 Miller hkl = get_hkl(n);
834 if (asu.is_in(hkl))
835 continue;
836 auto result = asu.to_asu(hkl, gops);
837 // cf. impl::move_to_asu() in asudata.hpp
838 set_hkl(n, result.first);
839 if (no_special_columns)
840 continue;
841 int isym = result.second;
842 if (!phase_columns.empty() || !abcd_columns.empty()) {
843 const Op& op = gops.sym_ops[(isym - 1) / 2];
844 double shift = op.phase_shift(hkl);
845 if (shift != 0) {
846 if (isym % 2 == 0)
847 shift = -shift;
848 double shift_deg = deg(shift);
849 for (int col : phase_columns)
850 data[n + col] = float(data[n + col] + shift_deg);
851 for (auto i = abcd_columns.begin(); i+3 < abcd_columns.end(); i += 4) {
852 double sinx = std::sin(shift);
853 double cosx = std::cos(shift);
854 double sin2x = 2 * sinx * cosx;
855 double cos2x = sq(cosx)- sq(sinx);
856 double a = data[n + *(i+0)];
857 double b = data[n + *(i+1)];
858 double c = data[n + *(i+2)];
859 double d = data[n + *(i+3)];
860 // a sin(x+y) + b cos(x+y) = a sin(x) cos(y) - b sin(x) sin(y)
861 // + a cos(x) sin(y) + b cos(x) cos(y)
862 data[n + *(i+0)] = float(a * cosx - b * sinx);
863 data[n + *(i+1)] = float(a * sinx + b * cosx);
864 data[n + *(i+2)] = float(c * cos2x - d * sin2x);
865 data[n + *(i+3)] = float(c * sin2x + d * cos2x);
866 }
867 }
868 }
869 if (isym % 2 == 0 && !centric) {
870 for (std::pair<int,int> cols : plus_minus_columns)
871 std::swap(data[n + cols.first], data[n + cols.second]);
872 for (int col : dano_columns)
873 data[n + col] = -data[n + col];
874 }
875 }
876 }

878 // (for unmerged MTZ only) change HKL according to M/ISYM
879 bool switch_to_original_hkl() {
880 if (indices_switched_to_original)
881 return false;
882 if (!has_data())
883 fail("switch_to_original_hkl(): data not read yet");
884 const Column* col = column_with_label("M/ISYM");
885 if (col == nullptr || col->type != 'Y' || col->idx < 3)
886 return false;
887 std::vector<Op> inv_symops;
888 inv_symops.reserve(symops.size());
889 for (const Op& op : symops)
890 inv_symops.push_back(op.inverse());
891 for (size_t n = 0; n + col->idx < data.size(); n += columns.size()) {
892 int isym = static_cast<int>(data[n + col->idx]) & 0xFF;
893 const Op& op = inv_symops.at((isym - 1) / 2);
894 Miller hkl = op.apply_to_hkl(get_hkl(n));
895 int sign = (isym & 1) ? 1 : -1;
896 for (int i = 0; i < 3; ++i)
897 data[n+i] = static_cast<float>(sign * hkl[i]);
898 }
899 indices_switched_to_original = true;
900 return true;
901 }

903 // (for unmerged MTZ only) change HKL to ASU equivalent and set ISYM
904 bool switch_to_asu_hkl() {
905 if (!indices_switched_to_original)
906 return false;
907 if (!has_data())
908 fail("switch_to_asu_hkl(): data not read yet");
909 const Column* col = column_with_label("M/ISYM");
910 if (col == nullptr || col->type != 'Y' || col->idx < 3 || !spacegroup)
911 return false;
912 size_t misym_idx = col->idx;
913 UnmergedHklMover hkl_mover(spacegroup);
914 for (size_t n = 0; n + col->idx < data.size(); n += columns.size()) {
915 Miller hkl = get_hkl(n);
916 int isym = hkl_mover.move_to_asu(hkl); // modifies hkl
917 set_hkl(n, hkl);
918 float& misym = data[n + misym_idx];
919 misym = float(((int)misym & ~0xff) | isym);
920 }
921 indices_switched_to_original = false;
922 return true;
923 }

978 void do_replace_column(size_t dest_idx, const Column& src_col,
979 const std::vector<std::string>& trailing_cols) {
980 const Mtz* src_mtz = src_col.parent;
981 for (size_t i = 0; i <= trailing_cols.size(); ++i) {
982 Column& dst = columns[dest_idx + i];
983 const Column& src = src_mtz->columns[src_col.idx + i];
984 dst.type = src.type;
985 dst.label = src.label;
986 dst.min_value = src.min_value;
987 dst.max_value = src.max_value;
988 dst.source = src.source;
989 if (src_mtz == this)
990 dst.dataset_id = src.dataset_id;
991 }
992 if (src_mtz == this) {
993 // internal copying
994 for (size_t n = 0; n < data.size(); n += columns.size())
995 for (size_t i = 0; i <= trailing_cols.size(); ++i)
996 data[n + dest_idx + i] = data[n + src_col.idx + i];
997 } else {
998 // external copying - need to match indices
999 std::vector<int> dst_indices = sorted_row_indices();
1000 std::vector<int> src_indices = src_mtz->sorted_row_indices();
1001 // cf. for_matching_reflections()
1002 size_t dst_stride = columns.size();
1003 size_t src_stride = src_mtz->columns.size();
1004 auto dst = dst_indices.begin();
1005 auto src = src_indices.begin();
1006 while (dst != dst_indices.end() && src != src_indices.end()) {
1007 Miller dst_hkl = get_hkl(*dst * dst_stride);
1008 Miller src_hkl = src_mtz->get_hkl(*src * src_stride);
1009 if (dst_hkl == src_hkl) {
1010 // copy values
1011 for (size_t i = 0; i <= trailing_cols.size(); ++i)
1012 data[*dst * dst_stride + dest_idx + i] =
1013 src_mtz->data[*src * src_stride + src_col.idx + i];
1014 ++dst;
1015 ++src;
1016 } else if (dst_hkl < src_hkl) {
1017 ++dst;
1018 } else {
1019 ++src;
1020 }
1021 }
1022 }
1023 }
