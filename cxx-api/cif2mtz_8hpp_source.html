22template<typename DataProxy>
23DataType check_data_type_under_symmetry(const DataProxy& proxy) {
24 const SpaceGroup* sg = proxy.spacegroup();
25 if (!sg)
26 return DataType::Unknown;
27 std::unordered_map<Op::Miller, int, MillerHash> seen;
28 ReciprocalAsu asu(sg);
29 GroupOps gops = sg->operations();
30 bool centric = gops.is_centrosymmetric();
31 DataType data_type = DataType::Mean;
32 for (size_t i = 0; i < proxy.size(); i += proxy.stride()) {
33 auto hkl_isym = asu.to_asu(proxy.get_hkl(i), gops);
34 int sign = hkl_isym.second % 2 + 1; // 2=positive, 1=negative
35 auto r = seen.emplace(hkl_isym.first, sign);
36 if (!r.second) {
37 if ((r.first->second & sign) != 0 || centric)
38 return DataType::Unmerged;
39 r.first->second |= sign;
40 data_type = DataType::Anomalous;
41 }
42 }
43 return data_type;
44}

102 Entry(const std::string& line) {
103 std::vector<std::string> tokens;
104 tokens.reserve(4);
105 split_str_into_multi(line, " \t\r\n", tokens);
106 if (tokens.size() != 4)
107 fail("line should have 4 words: " + line);
108 if (tokens[2].size() != 1 || tokens[3].size() != 1 ||
109 (tokens[3][0] != '0' && tokens[3][0] != '1'))
110 fail("incorrect line: " + line);
111 refln_tag = tokens[0];
112 col_label = tokens[1];
113 col_type = tokens[2][0];
114 dataset_id = tokens[3][0] - '0';
115 }
116 };

124 Mtz convert_block_to_mtz(const ReflnBlock& rb, std::ostream& out) const {
125 Mtz mtz;
126 mtz.title = title.empty() ? "Converted from mmCIF block " + rb.block.name : title;
127 if (!history.empty()) {
128 mtz.history.reserve(mtz.history.size() + history.size());
129 mtz.history.insert(mtz.history.end(), history.begin(), history.end());
130 }
131 mtz.cell = rb.cell;
132 mtz.spacegroup = rb.spacegroup;
133 mtz.add_dataset("HKL_base");
134 const cif::Loop* loop = rb.refln_loop ? rb.refln_loop : rb.diffrn_refln_loop;
135 if (!loop)
136 fail("_refln category not found in mmCIF block: " + rb.block.name);
137 if (verbose)
138 out << "Searching tags with known MTZ equivalents ...\n";
139 bool uses_status = false;
140 std::vector<int> indices;
141 std::string tag = loop->tags[0];
142 const size_t tag_offset = rb.tag_offset();
143
144 bool unmerged = force_unmerged || !rb.refln_loop;
145
146 std::vector<Entry> spec_entries;
147 if (!spec_lines.empty()) {
148 spec_entries.reserve(spec_lines.size());
149 for (const std::string& line : spec_lines)
150 spec_entries.emplace_back(line);
151 } else {
152 const char** line = default_spec(!unmerged);
153 for (; *line != nullptr; ++line)
154 spec_entries.emplace_back(*line);
155 }
156
157 // always start with H, K, L
158 tag.replace(tag_offset, std::string::npos, "index_h");
159 for (char c : {'h', 'k', 'l'}) {
160 tag.back() = c;
161 int index = loop->find_tag(tag);
162 if (index == -1)
163 fail("Miller index tag not found: " + tag);
164 indices.push_back(index);
165 auto col = mtz.columns.emplace(mtz.columns.end());
166 col->dataset_id = 0;
167 col->type = 'H';
168 col->label = alpha_up(c);
169 }
170
171 // M/ISYM and BATCH
172 if (unmerged) {
173 auto col = mtz.columns.emplace(mtz.columns.end());
174 col->dataset_id = 0;
175 col->type = 'Y';
176 col->label = "M/ISYM";
177
178 col = mtz.columns.emplace(mtz.columns.end());
179 col->dataset_id = 0;
180 col->type = 'B';
181 col->label = "BATCH";
182 }
183
184 if (!unmerged)
185 mtz.add_dataset("unknown").wavelength = rb.wavelength;
186
187 // other columns according to the spec
188 bool column_added = false;
189 for (const Entry& entry : spec_entries) {
190 tag.replace(tag_offset, std::string::npos, entry.refln_tag);
191 int index = loop->find_tag(tag);
192 if (index == -1)
193 continue;
194 if (mtz.column_with_label(entry.col_label))
195 continue;
196 column_added = true;
197 indices.push_back(index);
198 auto col = mtz.columns.emplace(mtz.columns.end());
199 // dataset_id is meaningless in unmerged MTZ files
200 col->dataset_id = unmerged ? 0 : entry.dataset_id;
201 col->type = entry.col_type;
202 if (col->type == 's') {
203 col->type = 'I';
204 uses_status = true;
205 }
206 col->label = entry.col_label;
207 if (verbose)
208 out << " " << tag << " -> " << col->label << '\n';
209 }
210 if (!column_added)
211 fail(force_unmerged ? "Unmerged d" : "D", "ata not found in block ", rb.block.name);
212
213 for (size_t i = 0; i != mtz.columns.size(); ++i) {
214 mtz.columns[i].parent = &mtz;
215 mtz.columns[i].idx = i;
216 }
217 mtz.nreflections = (int) loop->length();
218
219 std::unique_ptr<UnmergedHklMover> hkl_mover;
220 struct BatchInfo {
221 int sweep_id;
222 int frame_id;
223 };
224 std::vector<BatchInfo> batch_nums;
225 if (unmerged) {
226 hkl_mover.reset(new UnmergedHklMover(mtz.spacegroup));
227 tag.replace(tag_offset, std::string::npos, "diffrn_id");
228 int sweep_id_index = loop->find_tag(tag);
229 if (sweep_id_index == -1 && verbose)
230 out << "No diffrn_id. Assuming a single sweep.\n";
231 tag.replace(tag_offset, std::string::npos, "pdbx_image_id");
232 int image_id_index = loop->find_tag(tag);
233 if (verbose) {
234 if (image_id_index == -1)
235 out << "No pdbx_image_id, setting BATCH to a dummy value.\n";
236 else
237 out << " " << tag << " & diffrn_id -> BATCH\n";
238 }
239 struct SweepInfo {
240 std::set<int> frame_ids;
241 int offset;
242 float wavelength = 0.f;
243 std::string crystal_id = "unknown";
244 int dataset_id;
245 };
246 std::map<int, SweepInfo> sweeps;
247 cif::Block& block = const_cast<cif::Block&>(rb.block);
248 cif::Table tab_w0 = block.find("_diffrn.", {"id", "crystal_id"});
249 cif::Table tab_w1 = block.find("_diffrn_radiation.",
250 {"diffrn_id", "wavelength_id"});
251 cif::Table tab_w2 = block.find("_diffrn_radiation_wavelength.",
252 {"id", "wavelength"});
253 // store sweep and frame numbers corresponding to reflections
254 batch_nums.reserve(loop->length());
255 for (size_t i = 0; i < loop->values.size(); i += loop->tags.size()) {
256 int sweep_id = 1;
257 if (sweep_id_index >= 0)
258 sweep_id = cif::as_int(loop->values[i + sweep_id_index], 1);
259 int frame = 1;
260 if (image_id_index >= 0) {
261 double d = cif::as_number(loop->values[i + image_id_index], 1.);
262 frame = (int) std::ceil(d);
263 }
264 batch_nums.push_back({sweep_id, frame});
265 if (frame >= 0) {
266 SweepInfo& sweep = sweeps[sweep_id];
267 // if new sweep was added - try to set crystal_id and wavelength
268 if (sweep.frame_ids.empty() && sweep_id_index >= 0) {
269 const std::string& sweep_str = loop->values[i + sweep_id_index];
270 try {
271 sweep.crystal_id = tab_w0.find_row(sweep_str).str(1);
272 } catch(std::exception&) {}
273 try {
274 const std::string& wave_id = tab_w1.find_row(sweep_str)[1];
275 const std::string& wavelen = tab_w2.find_row(wave_id)[1];
276 sweep.wavelength = (float) cif::as_number(wavelen, 0.);
277 } catch(std::exception&) {}
278 }
279 sweep.frame_ids.insert(frame);
280 }
281 }
282
283 // add datasets and set SweepInfo::dataset_id
284 for (auto& sweep_pair : sweeps) {
285 SweepInfo& si = sweep_pair.second;
286 Mtz::Dataset* ds = mtz.dataset_with_name(si.crystal_id);
287 if (ds == nullptr) {
288 ds = &mtz.add_dataset(si.crystal_id);
289 // both crystal name and dataset name are set to crystal_id
290 ds->project_name = "unknown";
291 ds->wavelength = si.wavelength;
292 }
293 si.dataset_id = ds->id;
294 }
295
296 // determine offset that makes frame numbers unique
297 int cap = 0;
298 for (auto& it : sweeps) {
299 it.second.offset = cap;
300 cap += *--it.second.frame_ids.end() + 1100;
301 cap -= cap % 1000;
302 }
303 // add offset to BatchInfo::frame_id
304 for (BatchInfo& p : batch_nums)
305 if (p.sweep_id >= 0 && p.frame_id >= 0)
306 p.frame_id += sweeps.at(p.sweep_id).offset;
307
308 // add MTZ batches
309 for (const auto& sweep_pair : sweeps) {
310 const SweepInfo& sweep_info = sweep_pair.second;
311 Mtz::Batch batch;
312 batch.set_dataset_id(sweep_info.dataset_id);
313 batch.set_cell(mtz.cell);
314 batch.set_wavelength(sweep_info.wavelength);
315 // FIXME should we set more properties in BATCH header?
316 int min_frame = *sweep_info.frame_ids.begin();
317 int max_frame = *--sweep_info.frame_ids.end();
318 if (2 * sweep_info.frame_ids.size() > size_t(max_frame - min_frame)) {
319 // probably consecutive range, even if some frames are missing
320 for (int n = min_frame; n <= max_frame; ++n) {
321 batch.number = n + sweep_info.offset;
322 mtz.batches.push_back(batch);
323 }
324 } else {
325 for (int n : sweep_info.frame_ids) {
326 batch.number = n + sweep_info.offset;
327 mtz.batches.push_back(batch);
328 }
329 }
330 }
331 } // - if (unmerged)
332
333 // fill in the data
334 mtz.data.resize(mtz.columns.size() * mtz.nreflections);
335 size_t k = 0, row = 0;
336 for (size_t i = 0; i < loop->values.size(); i += loop->tags.size()) {
337 if (unmerged) {
338 std::array<int, 3> hkl;
339 for (int ii = 0; ii != 3; ++ii)
340 hkl[ii] = cif::as_int(loop->values[i + indices[ii]]);
341 int isym = hkl_mover->move_to_asu(hkl);
342 for (int j = 0; j != 3; ++j)
343 mtz.data[k++] = (float) hkl[j];
344 mtz.data[k++] = (float) isym;
345 mtz.data[k++] = batch_nums.empty() ? 1.f : (float) batch_nums[row++].frame_id;
346 } else {
347 for (int j = 0; j != 3; ++j)
348 mtz.data[k++] = (float) cif::as_int(loop->values[i + indices[j]]);
349 }
350 size_t j = 3;
351 if (uses_status)
352 mtz.data[k++] = status_to_freeflag(loop->values[i + indices[j++]]);
353 for (; j != indices.size(); ++j) {
354 const std::string& v = loop->values[i + indices[j]];
355 if (cif::is_null(v)) {
356 mtz.data[k] = (float) NAN;
357 } else {
358 mtz.data[k] = (float) cif::as_number(v);
359 if (std::isnan(mtz.data[k]))
360 out << "Value #" << i + indices[j] << " in the loop is not a number: "
361 << v << '\n';
362 }
363 ++k;
364 }
365 }
366 return mtz;
367 }

370 static float status_to_freeflag(const std::string& str) {
371 char c = str[0];

375 return 1.f;
376 if (c == 'f')
377 return 0.f;
378 return NAN;
379 }
