22template<typename DataProxy>
23std::pair<DataType, size_t> check_data_type_under_symmetry(const DataProxy& proxy) {
24 const SpaceGroup* sg = proxy.spacegroup();
25 if (!sg)
26 return {DataType::Unknown, 0};
27 std::unordered_map<Op::Miller, int, MillerHash> seen;
28 ReciprocalAsu asu(sg);
29 GroupOps gops = sg->operations();
30 bool centric = gops.is_centrosymmetric();
31 DataType data_type = DataType::Mean;
32 for (size_t i = 0; i < proxy.size(); i += proxy.stride()) {
33 auto hkl_sign = asu.to_asu_sign(proxy.get_hkl(i), gops);
34 int sign = hkl_sign.second ? 2 : 1; // 2=positive, 1=negative
35 auto r = seen.emplace(hkl_sign.first, sign);
36 if (data_type != DataType::Unmerged && !r.second) {
37 if ((r.first->second & sign) != 0 || centric) {
38 data_type = DataType::Unmerged;
39 } else {
40 r.first->second |= sign;
41 data_type = DataType::Anomalous;
42 }
43 }
44 }
45 return {data_type, seen.size()};
46}

48// "Old-style" anomalous or unmerged data is expected to have only these tags.
49inline bool possible_old_style(const ReflnBlock& rb, DataType data_type) {
50 if (rb.refln_loop == nullptr)
51 return false;
52 for (const std::string& tag : rb.refln_loop->tags) {
53 if (tag.size() < 7 + 6)
54 return false;
55 int tag_id = ialpha4_id(tag.c_str() + 7);
56 if (tag_id != ialpha4_id("inde") && // index_[hkl]
57 tag_id != ialpha4_id("wave") && // wavelength_id
58 tag_id != ialpha4_id("crys") && // crystal_id
59 tag_id != ialpha4_id("scal") && // scale_group_code
60 tag_id != ialpha4_id("stat") && // status
61 tag_id != ialpha4_id("inte") && // intensity_meas, intensity_sigma
62 (data_type == DataType::Unmerged ||
63 (tag_id != ialpha4_id("F_me") && // F_meas_au, F_meas_sigma_au
64 tag != "_refln.pdbx_r_free_flag")))
65 return false;
66 }
67 return true;
68}

78inline cif::Loop transcript_old_anomalous_to_standard(const cif::Loop& loop,
79 const SpaceGroup* sg) {
80 std::vector<int> positions;
81 positions.reserve(13); // usually less, but it doesn't matter
82 for (const char* tag : {"_refln.index_h", "_refln.index_k", "_refln.index_l"}) {
83 int pos = loop.find_tag_lc(tag);
84 if (pos == -1)
85 fail("while reading old anomalous: _refln.index_{h,k,l} not found");
86 positions.push_back(pos);
87 }
88 for (const char* tag : {"_refln.status", "_refln.pdbx_r_free_flag"}) {
89 int pos = loop.find_tag_lc(tag);
90 if (pos != -1)
91 positions.push_back(pos);
92 }
93
94 cif::Loop ret;
95 ret.tags.reserve(positions.size());
96 for (int p : positions)
97 ret.tags.push_back(loop.tags[p]);
98 const char* old_labels[2][2] = {
99 {"_refln.F_meas_au", "_refln.F_meas_sigma_au"},
100 {"_refln.intensity_meas", "_refln.intensity_sigma"}
101 };
102 const char* new_labels[2][4] = {
103 {"_refln.pdbx_F_plus", "_refln.pdbx_F_plus_sigma",
104 "_refln.pdbx_F_minus", "_refln.pdbx_F_minus_sigma"},
105 {"_refln.pdbx_I_plus", "_refln.pdbx_I_plus_sigma",
106 "_refln.pdbx_I_minus", "_refln.pdbx_I_minus_sigma"}
107 };
108 size_t common_tags = positions.size();
109 for (int n = 0; n < 2; ++n) {
110 int idx = loop.find_tag(old_labels[n][0]);
111 if (idx >= 0 && idx+1 < (int)loop.width() && loop.tags[idx+1] == old_labels[n][1]) {
112 positions.insert(positions.end(), {idx, idx+1, idx, idx+1});
113 ret.tags.insert(ret.tags.end(), new_labels[n], new_labels[n] + 4);
114 }
115 }
116 if (common_tags == positions.size())
117 fail("while reading old anomalous: _refln has neither F_meas_au nor intensity_meas");
118
119 ret.values.reserve(loop.length() * ret.width()); // upper bound
120 std::unordered_map<Miller, std::string*, MillerHash> seen;
121 if (!sg)
122 sg = &get_spacegroup_p1();
123 ReciprocalAsu asu(sg);
124 GroupOps gops = sg->operations();
125 for (size_t i = 0; i < loop.values.size(); i += loop.width()) {
126 const std::string* row = &loop.values[i];
127 Miller hkl;
128 for (size_t j = 0; j < 3; ++j)
129 hkl[j] = cif::as_int(row[positions[j]]);
130 auto hkl_sign = asu.to_asu_sign(hkl, gops);
131 // pointers don't change, .reserve() above prevents re-allocations
132 std::string* new_row = ret.values.data() + ret.values.size();
133 auto r = seen.emplace(hkl_sign.first, new_row);
134 bool sign = hkl_sign.second;
135 if (r.second) { // adding a new row
136 for (int p : positions)
137 ret.values.push_back(row[p]);
138 // Don't move hkl to asu here, only change the sign if F- is before F+.
139 if (!sign) // negative sign
140 for (int j = 0; j < 3; ++j)
141 new_row[j] = std::to_string(-hkl[j]);
142 size_t first_absent = common_tags + (sign ? 2 : 0);
143 for (size_t j = first_absent; j < ret.width(); j += 4) {
144 new_row[j] = ".";
145 new_row[j+1] = ".";
146 }
147 } else { // modifying existing row
148 std::string* modified_row = r.first->second;
149 if (sign) // positive sign - this hkl might be better
150 for (int j = 0; j < 3; ++j)
151 modified_row[j] = row[positions[j]];
152 // if a status or free flag value differs, set it to null
153 for (size_t j = 3; j < common_tags; ++j)
154 if (modified_row[j] != row[positions[j]])
155 modified_row[j] = ".";
156 for (size_t j = common_tags + (sign ? 0 : 2); j < ret.width(); j += 4) {
157 modified_row[j] = row[positions[j]];
158 modified_row[j+1] = row[positions[j+1]];
159 }
160 }
161 }
162
163 return ret;
164}

220 struct Entry {
221 std::string refln_tag;
222 std::string col_label;
223 char col_type;
224 int dataset_id;
225 std::vector<std::pair<std::string, float>> code_to_number;
226
227 Entry(const std::string& line) {
228 std::vector<std::string> tokens;
229 tokens.reserve(4);
230 split_str_into_multi(line, " \t\r\n", tokens);
231 if (tokens.size() != 4 && tokens.size() != 5)
232 fail("line should have 4 or 5 words: " + line);
233 if (tokens[2].size() != 1 || tokens[3].size() != 1 ||
234 (tokens[3][0] != '0' && tokens[3][0] != '1'))
235 fail("incorrect line: " + line);
236 refln_tag = tokens[0];
237 col_label = tokens[1];
238 col_type = tokens[2][0];
239 dataset_id = tokens[3][0] - '0';
240 // for compatibility with older spec
241 if (col_type == 's' && tokens.size() == 4) {
242 col_type = 'I';
243 tokens.push_back("o=1,f=0");
244 }
245 if (tokens.size() == 5) {
246 std::vector<std::string> items = split_str(tokens[4], ',');
247 code_to_number.reserve(items.size());
248 for (const std::string& item : items) {
249 size_t pos = item.find('=');
250 if (pos == std::string::npos)
251 fail("wrong mapping (", item, ") in: ", line);
252 float f;
253 auto result = fast_float::from_chars(item.c_str() + pos + 1,
254 item.c_str() + item.size(), f);
255 if (result.ec != std::errc())
256 fail("failed to parse value in ", item, " in: ", line);
257 code_to_number.emplace_back(item.substr(0, pos), f);
258 }
259 }
260 }
261
262 float translate_code_to_number(const std::string& v) const {
263 if (v.size() == 1) {
264 for (const auto& c2n : code_to_number)
265 if (c2n.first.size() == 1 && c2n.first[0] == v[0])
266 return c2n.second;
267 } else {
268 std::string s = cif::as_string(v);
269 for (const auto& c2n : code_to_number)
270 if (c2n.first == s)
271 return c2n.second;
272 }
273 return NAN;
274 }

284 Mtz convert_block_to_mtz(const ReflnBlock& rb, std::ostream& out) const {
285 Mtz mtz;
286 mtz.title = title.empty() ? "Converted from mmCIF block " + rb.block.name : title;
287 if (!history.empty()) {
288 mtz.history.reserve(mtz.history.size() + history.size());
289 mtz.history.insert(mtz.history.end(), history.begin(), history.end());
290 }
291 mtz.cell = rb.cell;
292 mtz.spacegroup = rb.spacegroup;
293 mtz.add_dataset("HKL_base");
294
295 const cif::Loop* loop = rb.refln_loop ? rb.refln_loop : rb.diffrn_refln_loop;
296 if (!loop)
297 fail("_refln category not found in mmCIF block: " + rb.block.name);
298 bool unmerged = force_unmerged || !rb.refln_loop;
299
300 if (!unmerged) {
301 Mtz::Dataset& ds = mtz.add_dataset("unknown");
302 if (!std::isnan(wavelength))
303 ds.wavelength = wavelength;
304 else if (rb.wavelength_count > 1)
305 out << "Warning: ignoring wavelengths, " << rb.wavelength_count
306 << " are present in block " << rb.block.name << ".\n";
307 else
308 ds.wavelength = rb.wavelength;
309 }
310
311 if (verbose)
312 out << "Searching tags with known MTZ equivalents ...\n";
313 std::vector<int> indices;
314 std::vector<const Entry*> entries; // used for code_to_number only
315 std::string tag = loop->tags[0];
316 const size_t tag_offset = rb.tag_offset();
317 // NOLINTNEXTLINE(readability-suspicious-call-argument): clang-tidy finds npos suspicious
318 auto set_tag = [&](const std::string& t) { tag.replace(tag_offset, std::string::npos, t); };
319
320 std::vector<Entry> spec_entries;
321 if (!spec_lines.empty()) {
322 spec_entries.reserve(spec_lines.size());
323 for (const std::string& line : spec_lines)
324 spec_entries.emplace_back(line);
325 } else {
326 const char** line = default_spec(!unmerged);
327 for (; *line != nullptr; ++line)
328 spec_entries.emplace_back(*line);
329 }
330
331 // always start with H, K, L
332 set_tag("index_h");
333 for (char c : {'h', 'k', 'l'}) {
334 tag.back() = c;
335 int index = loop->find_tag(tag);
336 if (index == -1)
337 fail("Miller index tag not found: " + tag);
338 indices.push_back(index);
339 entries.push_back(nullptr);
340 auto col = mtz.columns.emplace(mtz.columns.end());
341 col->dataset_id = 0;
342 col->type = 'H';
343 col->label = alpha_up(c);
344 }
345
346 // M/ISYM and BATCH
347 if (unmerged) {
348 auto col = mtz.columns.emplace(mtz.columns.end());
349 col->dataset_id = 0;
350 col->type = 'Y';
351 col->label = "M/ISYM";
352
353 col = mtz.columns.emplace(mtz.columns.end());
354 col->dataset_id = 0;
355 col->type = 'B';
356 col->label = "BATCH";
357 }
358
359 // other columns according to the spec
360 bool column_added = false;
361 for (const Entry& entry : spec_entries) {
362 set_tag(entry.refln_tag);
363 int index = loop->find_tag(tag);
364 if (index == -1)
365 continue;
366 if (mtz.column_with_label(entry.col_label))
367 continue;
368 column_added = true;
369 indices.push_back(index);
370 entries.push_back(entry.code_to_number.empty() ? nullptr : &entry);
371 auto col = mtz.columns.emplace(mtz.columns.end());
372 // dataset_id is meaningless in unmerged MTZ files
373 col->dataset_id = unmerged ? 0 : entry.dataset_id;
374 col->type = entry.col_type;
375 col->label = entry.col_label;
376 if (verbose)
377 out << " " << tag << " -> " << col->label << '\n';
378 }
379 if (!column_added)
380 fail(force_unmerged ? "Unmerged d" : "D", "ata not found in block ", rb.block.name);
381
382 for (size_t i = 0; i != mtz.columns.size(); ++i) {
383 mtz.columns[i].parent = &mtz;
384 mtz.columns[i].idx = i;
385 }
386 mtz.nreflections = (int) loop->length();
387
388 std::unique_ptr<UnmergedHklMover> hkl_mover;
389 struct BatchInfo {
390 int sweep_id;
391 int frame_id;
392 };
393 std::vector<BatchInfo> batch_nums;
394 if (unmerged) {
395 hkl_mover.reset(new UnmergedHklMover(mtz.spacegroup));
396 set_tag("diffrn_id");
397 int sweep_id_index = loop->find_tag(tag);
398 if (sweep_id_index == -1 && verbose)
399 out << "No diffrn_id. Assuming a single sweep.\n";
400 set_tag("pdbx_image_id");
401 int image_id_index = loop->find_tag(tag);
402 if (verbose) {
403 if (image_id_index == -1)
404 out << "No pdbx_image_id, setting BATCH to a dummy value.\n";
405 else
406 out << " " << tag << " & diffrn_id -> BATCH\n";
407 }
408 struct SweepInfo {
409 std::set<int> frame_ids;
410 int offset;
411 float wavelength = 0.f;
412 std::string crystal_id = "unknown";
413 int dataset_id;
414 };
415 std::map<int, SweepInfo> sweeps;
416 cif::Block& block = const_cast<cif::Block&>(rb.block);
417 cif::Table tab_w0 = block.find("_diffrn.", {"id", "crystal_id"});
418 cif::Table tab_w1 = block.find("_diffrn_radiation.",
419 {"diffrn_id", "wavelength_id"});
420 cif::Table tab_w2 = block.find("_diffrn_radiation_wavelength.",
421 {"id", "wavelength"});
422 // store sweep and frame numbers corresponding to reflections
423 batch_nums.reserve(loop->length());
424 for (size_t i = 0; i < loop->values.size(); i += loop->tags.size()) {
425 int sweep_id = 1;
426 if (sweep_id_index >= 0)
427 sweep_id = cif::as_int(loop->values[i + sweep_id_index], 1);
428 int frame = 1;
429 if (image_id_index >= 0) {
430 double d = cif::as_number(loop->values[i + image_id_index], 1.);
431 frame = (int) std::ceil(d);
432 }
433 batch_nums.push_back({sweep_id, frame});
434 if (frame >= 0) {
435 SweepInfo& sweep = sweeps[sweep_id];
436 // if new sweep was added - try to set crystal_id and wavelength
437 if (sweep.frame_ids.empty() && sweep_id_index >= 0) {
438 const std::string& sweep_str = loop->values[i + sweep_id_index];
439 try {
440 sweep.crystal_id = tab_w0.find_row(sweep_str).str(1);
441 } catch(std::exception&) {}
442 try {
443 const std::string& wave_id = tab_w1.find_row(sweep_str)[1];
444 const std::string& wavelen = tab_w2.find_row(wave_id)[1];
445 sweep.wavelength = (float) cif::as_number(wavelen, 0.);
446 } catch(std::exception&) {}
447 }
448 sweep.frame_ids.insert(frame);
449 }
450 }
451
452 // add datasets and set SweepInfo::dataset_id
453 for (auto& sweep_pair : sweeps) {
454 SweepInfo& si = sweep_pair.second;
455 Mtz::Dataset* ds = mtz.dataset_with_name(si.crystal_id);
456 if (ds == nullptr) {
457 ds = &mtz.add_dataset(si.crystal_id);
458 // both crystal name and dataset name are set to crystal_id
459 ds->project_name = "unknown";
460 ds->wavelength = si.wavelength;
461 }
462 si.dataset_id = ds->id;
463 }
464
465 // determine offset that makes frame numbers unique
466 int cap = 0;
467 for (auto& it : sweeps) {
468 it.second.offset = cap;
469 cap += *--it.second.frame_ids.end() + 1100;
470 cap -= cap % 1000;
471 }
472 // add offset to BatchInfo::frame_id
473 for (BatchInfo& p : batch_nums)
474 if (p.sweep_id >= 0 && p.frame_id >= 0)
475 p.frame_id += sweeps.at(p.sweep_id).offset;
476
477 // add MTZ batches
478 for (const auto& sweep_pair : sweeps) {
479 const SweepInfo& sweep_info = sweep_pair.second;
480 Mtz::Batch batch;
481 batch.set_dataset_id(sweep_info.dataset_id);
482 batch.set_cell(mtz.cell);
483 batch.set_wavelength(sweep_info.wavelength);
484 // FIXME should we set more properties in BATCH header?
485 int min_frame = *sweep_info.frame_ids.begin();
486 int max_frame = *--sweep_info.frame_ids.end();
487 if (2 * sweep_info.frame_ids.size() > size_t(max_frame - min_frame)) {
488 // probably consecutive range, even if some frames are missing
489 for (int n = min_frame; n <= max_frame; ++n) {
490 batch.number = n + sweep_info.offset;
491 mtz.batches.push_back(batch);
492 }
493 } else {
494 for (int n : sweep_info.frame_ids) {
495 batch.number = n + sweep_info.offset;
496 mtz.batches.push_back(batch);
497 }
498 }
499 }
500 } // - if (unmerged)
501
502 // fill in the data
503 mtz.data.resize(mtz.columns.size() * mtz.nreflections);
504 size_t k = 0, row = 0;
505 for (size_t i = 0; i < loop->values.size(); i += loop->tags.size()) {
506 if (unmerged) {
507 std::array<int, 3> hkl;
508 for (size_t ii = 0; ii != 3; ++ii)
509 hkl[ii] = cif::as_int(loop->values[i + indices[ii]]);
510 int isym = hkl_mover->move_to_asu(hkl);
511 for (size_t j = 0; j != 3; ++j)
512 mtz.data[k++] = (float) hkl[j];
513 mtz.data[k++] = (float) isym;
514 mtz.data[k++] = batch_nums.empty() ? 1.f : (float) batch_nums[row++].frame_id;
515 } else {
516 for (size_t j = 0; j != 3; ++j)
517 mtz.data[k++] = (float) cif::as_int(loop->values[i + indices[j]]);
518 }
519 for (size_t j = 3; j != indices.size(); ++j) {
520 const std::string& v = loop->values[i + indices[j]];
521 if (cif::is_null(v)) {
522 mtz.data[k] = (float) NAN;
523 } else if (entries[j] != nullptr) {
524 mtz.data[k] = entries[j]->translate_code_to_number(v);
525 } else {
526 mtz.data[k] = (float) cif::as_number(v);
527 if (std::isnan(mtz.data[k]))
528 out << "Value #" << i + indices[j] << " in the loop is not a number: "
529 << v << '\n';
530 }
531 ++k;
532 }
533 }
534 return mtz;
535 }

537 Mtz auto_convert_block_to_mtz(ReflnBlock& rb, std::ostream& out, char mode) const {
538 if (mode == 'f' && possible_old_style(rb, DataType::Anomalous))
539 *rb.refln_loop = transcript_old_anomalous_to_standard(*rb.refln_loop, rb.spacegroup);
540 Mtz mtz = convert_block_to_mtz(rb, out);
541 if (mtz.is_merged() && mode == 'a') {
542 auto type_unique = check_data_type_under_symmetry(MtzDataProxy{mtz});
543 if (type_unique.first == DataType::Anomalous) {
544 if (possible_old_style(rb, DataType::Anomalous)) {
545 out << "NOTE: data in " << rb.block.name
546 << " is read as \"old-style\" anomalous (" << rb.refln_loop->length()
547 << " -> " << type_unique.second << " rows).\n";
548 *rb.refln_loop = transcript_old_anomalous_to_standard(*rb.refln_loop,
549 rb.spacegroup);
550 // this is rare, so it's OK to run the conversion twice
551 mtz = convert_block_to_mtz(rb, out);
552 } else {
553 out << "WARNING: in " << rb.block.name << ", out of "
554 << rb.refln_loop->length() << " HKLs, only " << type_unique.second
555 << " are unique under symmetry; the rest are equivalent to Friedel mates\n";
556 }
557 } else if (type_unique.first == DataType::Unmerged) {
558 out << "WARNING: in " << rb.block.name << ", out of "
559 << rb.refln_loop->length() << " HKLs, only " << type_unique.second
560 << " are unique under symmetry\n";
561 if (possible_old_style(rb, gemmi::DataType::Unmerged))
562 out << "Possibly unmerged data - you may use option --refln-to=unmerged\n";
563 }
564 }
565 return mtz;
566 }

569 static float status_to_freeflag(const std::string& str) {
570 char c = str[0];
