90 struct AtomInequality {
91 char property;
92 int relation;
93 double value;
94
95 bool matches(const Atom& a) const {
96 double atom_value = 0.;
97 if (property == 'q')
98 atom_value = a.occ;
99 else if (property == 'b')
100 atom_value = a.b_iso;
101 if (relation < 0)
102 return atom_value < value;
103 if (relation > 0)
104 return atom_value > value;
105 return atom_value == value;
106 }
107
108 std::string str() const {
109 std::string r = ";";
110 r += property;
111 r += relation == 0 ? '=' : relation < 0 ? '<' : '>';
112 r += to_str(value);
113 return r;
114 }
115 };

166 bool matches(const Model& model) const {
167 return mdl == 0 || std::to_string(mdl) == model.name;
168 }
169 bool matches(const Chain& chain) const {
170 return chain_ids.has(chain.name);
171 }
172 bool matches(const Residue& res) const {
173 return residue_names.has(res.name) &&
174 from_seqid.compare(res.seqid) <= 0 &&
175 to_seqid.compare(res.seqid) >= 0 &&
176 residue_flags.has(res.flag);
177 }
178 bool matches(const Atom& a) const {
179 return atom_names.has(a.name) &&
180 (elements.all || elements.has(a.element.uname())) &&
181 (altlocs.all || altlocs.has(std::string(a.altloc ? 1 : 0, a.altloc))) &&
182 atom_flags.has(a.flag) &&
183 std::all_of(atom_inequalities.begin(), atom_inequalities.end(),
184 [&](const AtomInequality& i) { return i.matches(a); });
185 }
186 bool matches(const CRA& cra) const {
187 return (cra.chain == nullptr || matches(*cra.chain)) &&
188 (cra.residue == nullptr || matches(*cra.residue)) &&
189 (cra.atom == nullptr || matches(*cra.atom));

205 CRA first_in_model(Model& model) const {
206 if (matches(model))
207 for (Chain& chain : model.chains) {
208 if (matches(chain))
209 for (Residue& res : chain.residues) {
210 if (matches(res))
211 for (Atom& atom : res.atoms) {
212 if (matches(atom))
213 return {&chain, &res, &atom};
214 }

217 return {nullptr, nullptr, nullptr};

282inline int determine_omitted_cid_fields(const std::string& cid) {
283 if (cid[0] == '/')
284 return 0; // model
285 if (std::isdigit(cid[0]) || cid[0] == '.' || cid[0] == '(' || cid[0] == '-')
286 return 2; // residue
287 size_t sep = cid.find_first_of("/([:;");
288 if (sep == std::string::npos || cid[sep] == '/')
289 return 1; // chain
290 if (cid[sep] == '(')
291 return 2; // residue
292 return 3; // atom
293}

295inline Selection::List make_cid_list(const std::string& cid, size_t pos, size_t end) {
296 Selection::List list;
297 list.all = (cid[pos] == '*');
298 if (cid[pos] == '!') {
299 list.inverted = true;
300 ++pos;
301 }
302 list.list = cid.substr(pos, end - pos);
303 return list;
304}

306inline Selection::SequenceId parse_cid_seqid(const std::string& cid, size_t& pos,
307 int default_seqnum) {
308 size_t initial_pos = pos;
309 int seqnum = default_seqnum;
310 char icode = ' ';
311 if (cid[pos] == '*') {
312 ++pos;
313 icode = '*';
314 } else if (std::isdigit(cid[pos])) {
315 char* endptr;
316 seqnum = std::strtol(&cid[pos], &endptr, 10);
317 pos = endptr - &cid[0];
318 }
319 if (cid[pos] == '.')
320 ++pos;
321 if (initial_pos != pos && (std::isalpha(cid[pos]) || cid[pos] == '*'))
322 icode = cid[pos++];
323 return {seqnum, icode};
324}

326inline Selection::AtomInequality parse_atom_inequality(const std::string& cid,
327 size_t pos, size_t end) {
328 Selection::AtomInequality r;
329 while (cid[pos] == ' ')
330 ++pos;
331 if (cid[pos] != 'q' && cid[pos] != 'b')
332 fail("Invalid selection syntax (at ", cid[pos], "): ", cid);
333 r.property = cid[pos];
334 ++pos;
335 while (cid[pos] == ' ')
336 ++pos;
337 if (cid[pos] == '<')
338 r.relation = -1;
339 else if (cid[pos] == '>')
340 r.relation = 1;
341 else if (cid[pos] == '=')
342 r.relation = 0;
343 else
344 fail("Invalid selection syntax (at ", cid[pos], "): ", cid);
345 ++pos;
346 auto result = fast_from_chars(cid.c_str() + pos, r.value);
347 if (result.ec != std::errc())
348 fail("Invalid selection syntax (number expected at '", cid.substr(pos), "'): ", cid);
349 pos = size_t(result.ptr - cid.c_str());
350 while (cid[pos] == ' ')
351 ++pos;
352 if (pos != std::min(end, cid.size()))
353 fail("Invalid selection syntax (at ", cid[pos], "): ", cid);
354 return r;
355}

357inline void parse_cid(const std::string& cid, Selection& sel) {
358 if (cid.empty() || (cid.size() == 1 && cid[0] == '*'))
359 return;
360 int omit = determine_omitted_cid_fields(cid);
361 size_t sep = 0;
362 // model
363 if (omit == 0) {
364 sep = cid.find('/', 1);
365 if (sep != 1 && cid[1] != '*') {
366 char* endptr;
367 sel.mdl = std::strtol(&cid[1], &endptr, 10);
368 size_t end_pos = endptr - &cid[0];
369 if (end_pos != sep && end_pos != cid.length())
370 fail("Expected model number first: " + cid);
371 }
372 }
373
374 // chain
375 if (omit <= 1 && sep != std::string::npos) {
376 size_t pos = (sep == 0 ? 0 : sep + 1);
377 sep = cid.find('/', pos);
378 sel.chain_ids = make_cid_list(cid, pos, sep);
379 }
380
381 // residue; MMDB CID syntax: s1.i1-s2.i2 or *(res).ic
382 // In gemmi both 14.a and 14a are accepted.
383 // *(ALA). and *(ALA) and (ALA). can be used instead of (ALA) for
384 // compatibility with MMDB.
385 if (omit <= 2 && sep != std::string::npos) {
386 size_t pos = (sep == 0 ? 0 : sep + 1);
387 if (cid[pos] != '(')
388 sel.from_seqid = parse_cid_seqid(cid, pos, INT_MIN);
389 if (cid[pos] == '(') {
390 ++pos;
391 size_t right_br = cid.find(')', pos);
392 sel.residue_names = make_cid_list(cid, pos, right_br);
393 pos = right_br + 1;
394 }
395 // allow "(RES)." and "(RES).*" and "(RES)*"
396 if (cid[pos] == '.')
397 ++pos;
398 if (cid[pos] == '*')
399 ++pos;
400 if (cid[pos] == '-') {
401 ++pos;
402 sel.to_seqid = parse_cid_seqid(cid, pos, INT_MAX);
403 }
404 sep = pos;
405 }
406
407 // atom; at[el]:aloc
408 if (sep < cid.size()) {
409 if (sep != 0 && cid[sep] != '/')
410 fail("Invalid selection syntax: " + cid);
411 size_t pos = (sep == 0 ? 0 : sep + 1);
412 size_t end = cid.find_first_of("[:;", pos);
413 if (end != pos) {
414 sel.atom_names = make_cid_list(cid, pos, end);
415 // Chain name can be empty, but not atom name,
416 // so we interpret empty atom name as *.
417 if (!sel.atom_names.inverted && sel.atom_names.list.empty())
418 sel.atom_names.all = true;
419 }
420 if (end != std::string::npos) {
421 if (cid[end] == '[') {
422 pos = end + 1;
423 end = cid.find(']', pos);
424 if (end == std::string::npos)
425 fail("Invalid selection syntax (no matching ']'): " + cid);
426 sel.elements = make_cid_list(cid, pos, end);
427 sel.elements.list = to_upper(sel.elements.list);
428 ++end;
429 }
430 if (cid[end] == ':') {
431 pos = end + 1;
432 end = cid.find(';', pos);
433 sel.altlocs = make_cid_list(cid, pos, end);
434 }
435 while (end != std::string::npos && cid[end] == ';') {
436 pos = end + 1;
437 end = cid.find(';', pos);
438 sel.atom_inequalities.push_back(parse_atom_inequality(cid, pos, end));
439 }
440 if (end < cid.length())
441 fail("Invalid selection syntax (atom properties): " + cid);
442 }
443 }
