116 return s.empty() ? "." : cif::quote(s);

119 return s.empty() ? "?" : cif::quote(s);

130void add_cif_atoms(const Structure& st, cif::Block& block, bool use_group_pdb) {
131 // atom list
132 cif::Loop& atom_loop = block.init_mmcif_loop("_atom_site.", {
133 "id",
134 "type_symbol",
135 "label_atom_id",
136 "label_alt_id",
137 "label_comp_id",
138 "label_asym_id",
139 "label_entity_id",
140 "label_seq_id",
141 "pdbx_PDB_ins_code",
142 "Cartn_x",
143 "Cartn_y",
144 "Cartn_z",
145 "occupancy",
146 "B_iso_or_equiv",
147 "pdbx_formal_charge",
148 "auth_seq_id",
149 "auth_asym_id",
150 "pdbx_PDB_model_num"});
151 if (use_group_pdb)
152 atom_loop.tags.emplace(atom_loop.tags.begin(), "_atom_site.group_PDB");
153 bool has_calc_flag = false;
154 bool has_tls_group_id = false;
155 size_t atom_site_count = 0;
156 for (const Model& model : st.models)
157 for (const Chain& chain : model.chains)
158 for (const Residue& res : chain.residues)
159 for (const Atom& atom : res.atoms) {
160 ++atom_site_count;
161 if (atom.calc_flag != CalcFlag::NotSet)
162 has_calc_flag = true;
163 if (atom.tls_group_id >= 0)
164 has_tls_group_id = true;
165 }
166 if (has_calc_flag)
167 atom_loop.tags.emplace_back("_atom_site.calc_flag");
168 if (has_tls_group_id)
169 atom_loop.tags.emplace_back("_atom_site.pdbx_tls_group_id");
170
171 std::vector<std::string>& vv = atom_loop.values;
172 vv.reserve(atom_site_count * atom_loop.tags.size());
173 std::vector<std::pair<int, const Atom*>> aniso;
174 int serial = 0;
175 for (const Model& model : st.models) {
176 for (const Chain& chain : model.chains) {
177 for (const Residue& res : chain.residues) {
178 std::string label_seq_id = res.label_seq.str('.');
179 std::string auth_seq_id = res.seqid.num.str();
180 std::string entity_id;
181 if (const Entity* ent = gemmi::find_entity_of_subchain(res.subchain, st.entities))
182 entity_id = cif::quote(ent->name);
183 else
184 entity_id = string_or_dot(res.entity_id);
185 for (const Atom& atom : res.atoms) {
186 if (use_group_pdb)
187 vv.emplace_back(res.het_flag != 'H' ? "ATOM" : "HETATM");
188 vv.emplace_back(std::to_string(++serial));
189 vv.emplace_back(atom.element.uname());
190 vv.emplace_back(cif::quote(atom.name));
191 vv.emplace_back(1, atom.altloc_or('.'));
192 vv.emplace_back(cif::quote(res.name));
193 vv.emplace_back(subchain_or_dot(res));
194 vv.emplace_back(entity_id);
195 vv.emplace_back(label_seq_id);
196 vv.emplace_back(pdbx_icode(res));
197 vv.emplace_back(to_str(atom.pos.x));
198 vv.emplace_back(to_str(atom.pos.y));
199 vv.emplace_back(to_str(atom.pos.z));
200 vv.emplace_back(to_str(atom.occ));
201 vv.emplace_back(to_str(atom.b_iso));
202 vv.emplace_back(atom.charge == 0 ? "?" : std::to_string(atom.charge));
203 vv.emplace_back(auth_seq_id);
204 vv.emplace_back(impl::qchain(chain.name));
205 vv.emplace_back(string_or_qmark(model.name));
206 if (has_calc_flag)
207 vv.emplace_back(&".\0d\0c\0dum"[2 * (int) atom.calc_flag]);
208 if (has_tls_group_id)
209 vv.emplace_back(int_or_qmark(atom.tls_group_id));
210 if (atom.aniso.nonzero())
211 aniso.emplace_back(serial, &atom);
212 }
213 }
214 }
215 }
216 if (aniso.empty()) {
217 block.find_mmcif_category("_atom_site_anisotrop.").erase();
218 } else {
219 cif::Loop& aniso_loop = block.init_mmcif_loop("_atom_site_anisotrop.", {
220 "id", "type_symbol", "U[1][1]", "U[2][2]",
221 "U[3][3]", "U[1][2]", "U[1][3]", "U[2][3]"});
222 std::vector<std::string>& aniso_val = aniso_loop.values;
223 aniso_val.reserve(aniso_loop.tags.size() * aniso.size());
224 for (const auto& a : aniso) {
225 aniso_val.emplace_back(std::to_string(a.first));
226 aniso_val.emplace_back(a.second->element.uname());
227 aniso_val.emplace_back(to_str(a.second->aniso.u11));
228 aniso_val.emplace_back(to_str(a.second->aniso.u22));
229 aniso_val.emplace_back(to_str(a.second->aniso.u33));
230 aniso_val.emplace_back(to_str(a.second->aniso.u12));
231 aniso_val.emplace_back(to_str(a.second->aniso.u13));
232 aniso_val.emplace_back(to_str(a.second->aniso.u23));
233 }
234 }

237// the names are: monomeric, dimeric, ...meric, 21-meric, 22-meric, ...
238int xmeric_to_number(const std::string& oligomeric) {
239 static const char names[20][10] = {
240 "mono", "di", "tri", "tetra", "penta",
241 "hexa", "hepta", "octa", "nona", "deca",
242 "undeca", "dodeca", "trideca", "tetradeca", "pentadeca",
243 "hexadeca", "heptadeca", "octadeca", "nonadeca", "eicosa"
244 };
245 size_t len = oligomeric.length();
246 const char* p = oligomeric.c_str();
247 for (int i = 0; i != 20; ++i)
248 if (len == std::strlen(names[i]) + 5 && strncmp(p, names[i], len-5) == 0)
249 return i + 1;
250 return no_sign_atoi(p);
251}

253void write_assemblies(const Structure& st, cif::Block& block) {
254 block.items.reserve(block.items.size() + 4); // avoid re-allocation
255 cif::Loop& a_loop = block.init_mmcif_loop("_pdbx_struct_assembly.",
256 {"id", "details", "method_details",
257 "oligomeric_details", "oligomeric_count"});
258 cif::Loop& prop_loop = block.init_mmcif_loop("_pdbx_struct_assembly_prop.",
259 {"biol_id", "type", "value"});
260 cif::Loop& gen_loop = block.init_mmcif_loop("_pdbx_struct_assembly_gen.",
261 {"assembly_id", "oper_expression", "asym_id_list"});
262 cif::Loop& oper_loop = block.init_mmcif_loop("_pdbx_struct_oper_list.",
263 {"id", "type",
264 "matrix[1][1]", "matrix[1][2]", "matrix[1][3]", "vector[1]",
265 "matrix[2][1]", "matrix[2][2]", "matrix[2][3]", "vector[2]",
266 "matrix[3][1]", "matrix[3][2]", "matrix[3][3]", "vector[3]"});
267 std::vector<const Assembly::Operator*> distinct_oper;
268 for (const Assembly& as : st.assemblies) {
269 std::string how_defined = "?";
270 if (as.author_determined && as.software_determined)
271 how_defined = "author_and_software_defined_assembly";
272 else if (as.author_determined)
273 how_defined = "author_defined_assembly";
274 else if (as.software_determined)
275 how_defined = "software_defined_assembly";
276 else if (as.special_kind == Assembly::SpecialKind::CompleteIcosahedral)
277 how_defined = "'complete icosahedral assembly'";
278 else if (as.special_kind == Assembly::SpecialKind::RepresentativeHelical)
279 how_defined = "'representative helical assembly'";
280 else if (as.special_kind == Assembly::SpecialKind::CompletePoint)
281 how_defined = "'complete point assembly'";
282 std::string oligomer = to_lower(as.oligomeric_details);
283 int nmer = as.oligomeric_count != 0 ? as.oligomeric_count
284 : xmeric_to_number(oligomer);
285 // _pdbx_struct_assembly
286 a_loop.add_row({as.name,
287 how_defined,
288 impl::string_or_qmark(as.software_name),
289 impl::string_or_qmark(oligomer),
290 nmer == 0 ? "?" : std::to_string(nmer)});
291
292 // _pdbx_struct_assembly_prop
293 if (!std::isnan(as.absa))
294 prop_loop.add_row({as.name, "'ABSA (A^2)'", to_str(as.absa)});
295 if (!std::isnan(as.ssa))
296 prop_loop.add_row({as.name, "'SSA (A^2)'", to_str(as.ssa)});
297 if (!std::isnan(as.more))
298 prop_loop.add_row({as.name, "MORE", to_str(as.more)});
299
300 // _pdbx_struct_assembly_gen and _pdbx_struct_oper_list
301 for (const Assembly::Gen& gen : as.generators) {
302 std::string subchain_str;
303 for (const std::string& name : gen.subchains)
304 string_append_sep(subchain_str, ',', name);
305 if (subchain_str.empty()) // chain names to subchain names
306 for (const Chain& chain : st.models[0].chains)
307 if (in_vector(chain.name, gen.chains))
308 for (const auto& sub : chain.subchains())
309 string_append_sep(subchain_str, ',', sub.front().subchain);
310 std::string oper_str;
311 for (const Assembly::Operator& oper : gen.operators) {
312 size_t k = 0;
313 for (; k != distinct_oper.size(); ++k)
314 if (distinct_oper[k]->transform.approx(oper.transform, 1e-9))
315 break;
316 string_append_sep(oper_str, ',', std::to_string(k+1));
317 if (k != distinct_oper.size())
318 continue;
319 distinct_oper.emplace_back(&oper);
320 oper_loop.values.emplace_back(std::to_string(k+1));
321 if (!oper.type.empty()) {
322 oper_loop.values.emplace_back(cif::quote(oper.type));
323 } else if (oper.transform.is_identity()) {
324 oper_loop.values.emplace_back("'identity operation'");
325 } else if (as.author_determined || as.software_determined) {
326 oper_loop.values.emplace_back("'crystal symmetry operation'");
327 } else {
328 oper_loop.values.emplace_back(".");
329 }
330 for (int i = 0; i < 3; ++i) {
331 for (int j = 0; j < 3; ++j)
332 oper_loop.values.emplace_back(to_str(oper.transform.mat[i][j]));
333 oper_loop.values.emplace_back(to_str(oper.transform.vec.at(i)));
334 }
335 }
336 gen_loop.add_row({as.name,
337 oper_str.empty() ? "." : oper_str,
338 subchain_str.empty() ? "?" : subchain_str});
339 }
340 }

343void write_struct_conn(const Structure& st, cif::Block& block) {
344 // example:
345 // disulf1 disulf A CYS 3 SG ? 3 ? 1_555 A CYS 18 SG ? 18 ? 1_555 ? 2.045
346 std::array<bool,(int)Connection::Type::Unknown+1> type_ids{};
347 bool use_ccp4_link_id = false;
348 for (const Connection& con : st.connections)
349 if (!con.link_id.empty())
350 use_ccp4_link_id = true;
351 cif::Loop& conn_loop = block.init_mmcif_loop("_struct_conn.",
352 {"id", "conn_type_id",
353 "ptnr1_auth_asym_id", "ptnr1_label_asym_id", "ptnr1_label_comp_id",
354 "ptnr1_label_seq_id", "ptnr1_label_atom_id", "pdbx_ptnr1_label_alt_id",
355 "ptnr1_auth_seq_id", "pdbx_ptnr1_PDB_ins_code", "ptnr1_symmetry",
356 "ptnr2_auth_asym_id", "ptnr2_label_asym_id", "ptnr2_label_comp_id",
357 "ptnr2_label_seq_id", "ptnr2_label_atom_id", "pdbx_ptnr2_label_alt_id",
358 "ptnr2_auth_seq_id", "pdbx_ptnr2_PDB_ins_code", "ptnr2_symmetry",
359 "details", "pdbx_dist_value"});
360 if (use_ccp4_link_id)
361 conn_loop.tags.push_back("_struct_conn.ccp4_link_id");
362 for (const Connection& con : st.connections) {
363 const_CRA cra1 = st.models[0].find_cra(con.partner1, true);
364 const_CRA cra2 = st.models[0].find_cra(con.partner2, true);
365 if (!cra1.residue || !cra2.residue)
366 continue;
367 const Atom* at1 = cra1.atom;
368 const Atom* at2 = cra2.atom;
369 std::string im_pdb_symbol = "?", im_dist_str = "?";
370 if (at1 && at2) {
371 NearestImage im = st.cell.find_nearest_image(at1->pos, at2->pos, con.asu);
372 im_pdb_symbol = im.symmetry_code(true);
373 im_dist_str = to_str_prec<4>(im.dist());
374 }
375 auto& v = conn_loop.values;
376 v.emplace_back(impl::string_or_qmark(con.name)); // id
377 v.emplace_back(connection_type_to_string(con.type)); // conn_type_id
378 v.emplace_back(impl::qchain(con.partner1.chain_name)); // ptnr1_auth_asym_id
379 v.emplace_back(subchain_or_dot(*cra1.residue)); // ptnr1_label_asym_id
380 v.emplace_back(cra1.residue->name); // ptnr1_label_comp_id
381 v.emplace_back(cra1.residue->label_seq.str('.')); // ptnr1_label_seq_id
382 v.emplace_back(at1 ? cif::quote(at1->name) : "?"); // ptnr1_label_atom_id
383 v.emplace_back(1, at1 ? at1->altloc_or('?') : '?'); // pdbx_ptnr1_label_alt_id
384 v.emplace_back(cra1.residue->seqid.num.str()); // ptnr1_auth_seq_id
385 v.emplace_back(pdbx_icode(con.partner1.res_id)); // ptnr1_PDB_ins_code
386 v.emplace_back("1_555"); // ptnr1_symmetry
387 v.emplace_back(impl::qchain(con.partner2.chain_name)); // ptnr2_auth_asym_id
388 v.emplace_back(subchain_or_dot(*cra2.residue)); // ptnr2_label_asym_id
389 v.emplace_back(cra2.residue->name); // ptnr2_label_comp_id
390 v.emplace_back(cra2.residue->label_seq.str('.')); // ptnr2_label_seq_id
391 v.emplace_back(at2 ? cif::quote(at2->name) : "?"); // ptnr2_label_atom_id
392 v.emplace_back(1, at2 ? at2->altloc_or('?') : '?'); // pdbx_ptnr2_label_alt_id
393 v.emplace_back(cra2.residue->seqid.num.str()); // ptnr2_auth_seq_id
394 v.emplace_back(pdbx_icode(con.partner2.res_id)); // ptnr2_PDB_ins_code
395 v.emplace_back(im_pdb_symbol); // ptnr2_symmetry
396 v.emplace_back("?"); // details
397 v.emplace_back(im_dist_str); // pdbx_dist_value
398 if (use_ccp4_link_id)
399 v.emplace_back(impl::string_or_qmark(con.link_id)); // ccp4_link_id
400 type_ids[int(con.type)] = true;
401 }
402
403 cif::Loop& type_loop = block.init_mmcif_loop("_struct_conn_type.", {"id"});
404 for (int i = 0; i < (int)type_ids.size() - 1; ++i)
405 if (type_ids[i])
406 type_loop.add_row({connection_type_to_string((Connection::Type)i)});
407}

418void write_ncs_oper(const Structure& st, cif::Block& block) {
419 // _struct_ncs_oper (MTRIX)
420 if (st.ncs.empty())
421 return;
422 cif::Loop& ncs_oper = block.init_mmcif_loop("_struct_ncs_oper.",
423 {"id", "code",
424 "matrix[1][1]", "matrix[1][2]", "matrix[1][3]", "vector[1]",
425 "matrix[2][1]", "matrix[2][2]", "matrix[2][3]", "vector[2]",
426 "matrix[3][1]", "matrix[3][2]", "matrix[3][3]", "vector[3]"});
427 auto add_op = [&ncs_oper](const NcsOp& op) {
428 ncs_oper.values.emplace_back(op.id);
429 ncs_oper.values.emplace_back(op.given ? "given" : "generate");
430 for (int i = 0; i < 3; ++i) {
431 for (int j = 0; j < 3; ++j)
432 ncs_oper.values.emplace_back(to_str(op.tr.mat[i][j]));
433 ncs_oper.values.emplace_back(to_str(op.tr.vec.at(i)));
434 }
435 };
436 auto identity = st.info.find("_struct_ncs_oper.id");
437 if (identity != st.info.end() &&
438 !in_vector_f([&](const NcsOp& op) { return op.id == identity->second; }, st.ncs))
439 add_op(NcsOp{identity->second, true, {}});
440 for (const NcsOp& op : st.ncs)
441 add_op(op);
442}

451void update_mmcif_block(const Structure& st, cif::Block& block, MmcifOutputGroups groups) {
452 if (st.models.empty())
453 return;
454
455 if (groups.block_name)
456 block.name = impl::is_valid_block_name(st.name) ? st.name : "model";
457
458 auto e_id = st.info.find("_entry.id");
459 std::string id = cif::quote(e_id != st.info.end() ? e_id->second : block.name);
460 if (groups.entry)
461 block.set_pair("_entry.id", id);
462 else if (const std::string* val = block.find_value("_entry.id"))
463 id = *val;
464
465 if (groups.database_status) {
466 auto initial_date = st.info.find("_pdbx_database_status.recvd_initial_deposition_date");
467 if (initial_date != st.info.end() && !initial_date->second.empty()) {
468 cif::ItemSpan span(block.items, "_pdbx_database_status.");
469 span.set_pair("_pdbx_database_status.entry_id", id);
470 span.set_pair(initial_date->first, initial_date->second);
471 }
472 }
473
474 if (groups.author && !st.meta.authors.empty()) {
475 cif::Loop& loop = block.init_mmcif_loop("_audit_author.", {"pdbx_ordinal", "name"});
476 int n = 0;
477 for (const std::string& author : st.meta.authors)
478 loop.add_row({std::to_string(++n), cif::quote(author)});
479 }
480
481 if (groups.cell) {
482 cif::ItemSpan cell_span(block.items, "_cell.");
483 cell_span.set_pair("_cell.entry_id", id);
484 impl::write_cell_parameters(st.cell, cell_span);
485 auto z_pdb = st.info.find("_cell.Z_PDB");
486 if (z_pdb != st.info.end())
487 cell_span.set_pair(z_pdb->first, z_pdb->second);
488 }
489
490 if (groups.symmetry) {
491 cif::ItemSpan span(block.items, "_symmetry.");
492 span.set_pair("_symmetry.entry_id", id);
