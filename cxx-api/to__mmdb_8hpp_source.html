38inline mmdb::Manager* copy_to_mmdb(const Structure& st, mmdb::Manager* manager) {
39 for (const std::string& s : st.raw_remarks) {
40 std::string line = rtrim_str(s);
41 manager->PutPDBString(line.c_str());
42 }
43
44 for (int imodel = 0; imodel < (int) st.models.size(); ++imodel) {
45 const Model& model = st.models[imodel];
46 mmdb::PModel model2 = mmdb::newModel();
47 model2->SetMMDBManager(manager, imodel);
48
49 for (const Chain& chain : model.chains) {
50 mmdb::PChain chain2 = model2->CreateChain(chain.name.c_str());
51 for (const Residue& res : chain.residues) {
52 const char icode[2] = {res.seqid.icode, '\0'};
53 mmdb::PResidue res2 = chain2->GetResidueCreate(res.name.c_str(),
54 *res.seqid.num,
55 icode,
56 true);
57 if (res.is_cis) {
58 if (const Residue* next = chain.next_residue(res)) {
59 mmdb::CisPep* cispep = new mmdb::CisPep();
60 cispep->serNum = model2->GetCisPeps()->Length() + 1;
61 strcpy_to_mmdb(cispep->chainID1, chain.name);
62 strcpy_to_mmdb(cispep->pep1, res.name);
63 set_seqid_in_mmdb(&cispep->seqNum1, cispep->icode1, res.seqid);
64 strcpy_to_mmdb(cispep->chainID2, chain.name);
65 strcpy_to_mmdb(cispep->pep2, next->name);
66 set_seqid_in_mmdb(&cispep->seqNum2, cispep->icode2, next->seqid);
67 cispep->modNum = imodel;
68 model2->AddCisPep(cispep);
69 }
70 }
71 for (const Atom& atom : res.atoms) {
72 mmdb::PAtom atom2 = mmdb::newAtom();
73 const char altloc[2] = {atom.altloc, '\0'};
74 atom2->SetAtomName(0, atom.serial, atom.padded_name().c_str(),
75 altloc, res.segment.c_str(), atom.element.uname());
76 atom2->Het = res.het_flag == 'H';
77 atom2->SetCharge(atom.charge);
78 atom2->SetCoordinates(atom.pos.x, atom.pos.y, atom.pos.z,
79 atom.occ, atom.b_iso);
80 if (atom.aniso.nonzero()) {
81 atom2->su11 = atom.aniso.u11;
82 atom2->su22 = atom.aniso.u22;
83 atom2->su33 = atom.aniso.u33;
84 atom2->su12 = atom.aniso.u12;
85 atom2->su13 = atom.aniso.u13;
86 atom2->su23 = atom.aniso.u23;
87 atom2->WhatIsSet |= mmdb::ASET_Anis_tFSigma;
88 }
89 res2->AddAtom(atom2);
90 }
91 // TER
92 if (res.entity_type == EntityType::Polymer &&
93 (&res == &chain.residues.back() ||
94 (&res + 1)->entity_type != EntityType::Polymer)) {
95 mmdb::PAtom atom2 = mmdb::newAtom();
96 atom2->MakeTer();
97 atom2->serNum = res.atoms.back().serial + 1;
98 res2->AddAtom(atom2);
99 }
100 }
101 }
102 int n_model = manager->AddModel(model2);
103 manager->GetModel(n_model)->CopyCisPeps(model2);
104 manager->PutCell(st.cell.a, st.cell.b, st.cell.c,
105 st.cell.alpha, st.cell.beta, st.cell.gamma, 1);
106 manager->SetSpaceGroup(st.spacegroup_hm.c_str());
107 mmdb::Cryst* cryst = manager->GetCrystData();
108 auto z = st.info.find("_cell.Z_PDB");
109 if (z != st.info.end() && !z->second.empty()) {
110 cryst->Z = std::atoi(z->second.c_str());
111 cryst->WhatIsSet |= mmdb::CSET_ZValue;
112 }
113 if (st.has_origx && !st.origx.is_identity()) {
114 copy_transform_to_mmdb(st.origx, cryst->o, cryst->t);
115 cryst->WhatIsSet |= mmdb::CSET_OrigMatrix;
116 }
117 if (st.cell.explicit_matrices) {
118 copy_transform_to_mmdb(st.cell.frac, cryst->s, cryst->u);
119 cryst->WhatIsSet |= mmdb::CSET_ScaleMatrix;
120 }
121 if (!st.ncs.empty()) {
122 mmdb::mat33 m;
123 mmdb::vect3 v;
124 if (st.info.find("_struct_ncs_oper.id") != st.info.end()) {
125 copy_transform_to_mmdb(Transform{}, m, v);
126 cryst->AddNCSMatrix(m, v, 1);
127 }
128 for (const NcsOp& op : st.ncs) {
129 copy_transform_to_mmdb(op.tr, m, v);
130 cryst->AddNCSMatrix(m, v, op.given);
131 }
132 }
133 }
134 return manager;
135}
