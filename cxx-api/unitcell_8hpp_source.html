80struct NearestImage {

84
85 double dist() const { return std::sqrt(dist_sq); }
86 bool same_asu() const {

88 }
89
90 // return string such as 1555 or 1_555
91 std::string symmetry_code(bool underscore) const {
92 std::string s = std::to_string(sym_idx + 1);
93 if (underscore)
94 s += '_';
95 if (unsigned(5 + pbc_shift[0]) <= 9 &&
96 unsigned(5 + pbc_shift[1]) <= 9 &&
97 unsigned(5 + pbc_shift[2]) <= 9) { // normal, quick path
98 for (int shift : pbc_shift)
99 s += char('5' + shift);
100 } else { // problematic, non-standard path
101 for (int i = 0; i < 3; ++i) {
102 if (i != 0 && underscore)
103 s += '_';
104 s += std::to_string(5 + pbc_shift[i]);
105 }
106 }
107 return s;
108 }

148 double a = 1.0, b = 1.0, c = 1.0;
149 double alpha = 90.0, beta = 90.0, gamma = 90.0;

152 // volume and reciprocal parameters a*, b*, c*, alpha*, beta*, gamma*
153 double volume = 1.0;
154 double ar = 1.0, br = 1.0, cr = 1.0;
155 double cos_alphar = 0.0, cos_betar = 0.0, cos_gammar = 0.0;
156 bool explicit_matrices = false;

160 // Non-crystalline (for example NMR) structures are supposed to use fake
161 // unit cell 1x1x1, but sometimes they don't. A number of non-crystalline
162 // entries in the PDB has incorrectly set unit cell or fract. matrix,
163 // that is why we check both.
164 bool is_crystal() const { return a != 1.0 && frac.mat[0][0] != 1.0; }

167 return a == o.a && b == o.b && c == o.c &&
168 alpha == o.alpha && beta == o.beta && gamma == o.gamma;
169 }
170 bool operator!=(const UnitCell& o) const { return !operator==(o); }

172 bool approx(const UnitCell& o, double epsilon) const {
173 auto eq = [&](double x, double y) { return std::fabs(x - y) < epsilon; };
174 return eq(a, o.a) && eq(b, o.b) && eq(c, o.c) &&
175 eq(alpha, o.alpha) && eq(beta, o.beta) && eq(gamma, o.gamma);

178 void calculate_properties() {
179 // ensure exact values for right angles
180 double cos_alpha = alpha == 90. ? 0. : std::cos(rad(alpha));
181 double cos_beta = beta == 90. ? 0. : std::cos(rad(beta));
182 double cos_gamma = gamma == 90. ? 0. : std::cos(rad(gamma));
183 double sin_alpha = alpha == 90. ? 1. : std::sin(rad(alpha));
184 double sin_beta = beta == 90. ? 1. : std::sin(rad(beta));
185 double sin_gamma = gamma == 90. ? 1. : std::sin(rad(gamma));
186 if (sin_alpha == 0 || sin_beta == 0 || sin_gamma == 0)
187 fail("Impossible angle - N*180deg.");
188
189 // volume - formula from Giacovazzo p.62
190 volume = a * b * c * std::sqrt(1 - cos_alpha * cos_alpha
191 - cos_beta * cos_beta - cos_gamma * cos_gamma
192 + 2 * cos_alpha * cos_beta * cos_gamma);
193
194 // reciprocal parameters a*, b*, ... (Giacovazzo, p. 64)
195 ar = b * c * sin_alpha / volume;
196 br = a * c * sin_beta / volume;
197 cr = a * b * sin_gamma / volume;
198 double cos_alphar_sin_beta = (cos_beta * cos_gamma - cos_alpha) / sin_gamma;
199 cos_alphar = cos_alphar_sin_beta / sin_beta;
200 //cos_alphar = (cos_beta * cos_gamma - cos_alpha) / (sin_beta * sin_gamma);
201 cos_betar = (cos_alpha * cos_gamma - cos_beta) / (sin_alpha * sin_gamma);
202 cos_gammar = (cos_alpha * cos_beta - cos_gamma) / (sin_alpha * sin_beta);
203
204 if (explicit_matrices)
205 return;
206
207 // The orthogonalization matrix we use is described in ITfC B p.262:
208 // "An alternative mode of orthogonalization, used by the Protein
209 // Data Bank and most programs, is to align the a1 axis of the unit
210 // cell with the Cartesian X_1 axis, and to align the a*_3 axis with the
211 // Cartesian X_3 axis."
212 double sin_alphar = std::sqrt(1.0 - cos_alphar * cos_alphar);
213 orth.mat = {a, b * cos_gamma, c * cos_beta,
214 0., b * sin_gamma, -c * cos_alphar_sin_beta,
215 0., 0. , c * sin_beta * sin_alphar};
216 orth.vec = {0., 0., 0.};
217
218 double o12 = -cos_gamma / (sin_gamma * a);
219 double o13 = -(cos_gamma * cos_alphar_sin_beta + cos_beta * sin_gamma)
220 / (sin_alphar * sin_beta * sin_gamma * a);
221 double o23 = cos_alphar / (sin_alphar * sin_gamma * b);
222 frac.mat = {1 / a, o12, o13,
223 0., 1 / orth.mat[1][1], o23,
224 0., 0., 1 / orth.mat[2][2]};
225 frac.vec = {0., 0., 0.};
226 }
227
228 double cos_alpha() const { return alpha == 90. ? 0. : std::cos(rad(alpha)); }

230 // B matrix following convention from Busing & Levy (1967), not from cctbx.
231 // Cf. https://dials.github.io/documentation/conventions.html
232 Mat33 calculate_matrix_B() const {
233 double sin_gammar = std::sqrt(1 - cos_gammar * cos_gammar);
234 double sin_betar = std::sqrt(1 - cos_betar * cos_betar);
235 return Mat33(ar, br * cos_gammar, cr * cos_betar,
236 0., br * sin_gammar, -cr * sin_betar * cos_alpha(),
237 0., 0., 1.0 / c);
238 }

240 // based on Fischer & Tillmanns (1988). Acta Cryst. C44, 775-776.
241 // "The equivalent isotropic displacement factor."
242 // The argument is a non-orthogonalized tensor U,
243 // i.e. the one from SmallStructure::Site, but not from Atom.
244 double calculate_u_eq(const SMat33<double>& ani) const {
245 double aar = a * ar;
246 double bbr = b * br;
247 double ccr = c * cr;
248 // it could be optimized using orth.mat[0][1] and orth.mat[0][2]
249 double cos_beta = beta == 90. ? 0. : std::cos(rad(beta));
250 double cos_gamma = gamma == 90. ? 0. : std::cos(rad(gamma));
251 return 1/3. * (sq(aar) * ani.u11 + sq(bbr) * ani.u22 + sq(ccr) * ani.u33 +
252 2 * (aar * bbr * cos_gamma * ani.u12 +
253 aar * ccr * cos_beta * ani.u13 +
254 bbr * ccr * cos_alpha() * ani.u23));
255 }

257 void set_matrices_from_fract(const Transform& f) {
258 // mmCIF _atom_sites.fract_transf_* and PDB SCALEn records usually
259 // have less significant digits than unit cell parameters, and should
260 // be ignored unless we have non-standard settings.
261 if (f.mat.approx(frac.mat, 5e-6) && f.vec.approx(frac.vec, 1e-6))
262 return;
263 // The SCALE record is sometimes incorrect. Here we only catch cases
264 // when CRYST1 is set as for non-crystal and SCALE is very suspicious.
265 if (frac.mat[0][0] == 1.0 && (f.mat[0][0] == 0.0 || f.mat[0][0] > 1.0))
266 return;
267 frac = f;
268 orth = f.inverse();
269 explicit_matrices = true;
270 }

290 UnitCell changed_basis_backward(const Op& op, bool set_images) {
291 Mat33 mat = orth.mat.multiply(rot_as_mat33(op));
292 UnitCell new_cell;
293 new_cell.set_from_vectors(mat.column_copy(0),

310 bool is_compatible_with_groupops(const GroupOps& gops, double eps=1e-3) {
311 std::array<double,6> metric = metric_tensor().elements_voigt();
312 for (const Op& op : gops.sym_ops) {
313 Mat33 m = orth.mat.multiply(rot_as_mat33(op));
314 std::array<double,6> other = {{
315 m.column_dot(0,0), m.column_dot(1,1), m.column_dot(2,2),
316 m.column_dot(1,2), m.column_dot(0,2), m.column_dot(0,1)
317 }};
318 for (int i = 0; i < 6; ++i)
319 if (std::fabs(metric[i] - other[i]) > eps)
320 return false;
321 }
322 return true;
323 }

386 // Helper function. PBC = periodic boundary conditions.
387 bool search_pbc_images(Fractional&& diff, NearestImage& image) const {
388 int neg_shift[3] = { iround(diff.x), iround(diff.y), iround(diff.z) };
389 diff.x -= neg_shift[0];
390 diff.y -= neg_shift[1];
391 diff.z -= neg_shift[2];
392 Position orth_diff = orthogonalize_difference(diff);
393 double dsq = orth_diff.length_sq();
394 if (dsq < image.dist_sq) {
395 image.dist_sq = dsq;
396 for (int j = 0; j < 3; ++j)
397 image.pbc_shift[j] = -neg_shift[j];
398 return true;
399 }
400 return false;
401 }

403 NearestImage find_nearest_image(const Position& ref, const Position& pos, Asu asu) const {
404 NearestImage image;
405 if (asu == Asu::Different)
406 image.dist_sq = INFINITY;
407 else
408 image.dist_sq = ref.dist_sq(pos);
409 if (asu == Asu::Same || !is_crystal())
410 return image;
411 Fractional fpos = fractionalize(pos);
412 Fractional fref = fractionalize(ref);
413 search_pbc_images(fpos - fref, image);
414 if (asu == Asu::Different &&
415 image.pbc_shift[0] == 0 && image.pbc_shift[1] == 0 && image.pbc_shift[2] == 0)
416 image.dist_sq = INFINITY;
417 for (int n = 0; n != static_cast<int>(images.size()); ++n)
418 if (search_pbc_images(images[n].apply(fpos) - fref, image))
419 image.sym_idx = n + 1;
420 return image;
421 }

479 // Calculate 1/d^2 for specified hkl reflection.
480 // 1/d^2 = (2*sin(theta)/lambda)^2
481 // The indices are integers, but they may be stored as floating-point
482 // numbers (MTZ format) so we use double to avoid conversions.
483 double calculate_1_d2_double(double h, double k, double l) const {
484 double arh = ar * h;
485 double brk = br * k;
486 double crl = cr * l;
487 return arh * arh + brk * brk + crl * crl + 2 * (arh * brk * cos_gammar +
488 arh * crl * cos_betar +
489 brk * crl * cos_alphar);
490 }
491 double calculate_1_d2(const Miller& hkl) const {
492 return calculate_1_d2_double(hkl[0], hkl[1], hkl[2]);
493 }
494
495 // Calculate d-spacing.
496 // d = lambda/(2*sin(theta))
497 double calculate_d(const Miller& hkl) const {
498 return 1.0 / std::sqrt(calculate_1_d2(hkl));
499 }
500
501 // Calculate (sin(theta)/lambda)^2 = d*^2/4
502 double calculate_stol_sq(const Miller& hkl) const {
503 return 0.25 * calculate_1_d2(hkl);
504 }
