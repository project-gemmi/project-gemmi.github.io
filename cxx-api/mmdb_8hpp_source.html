16inline void copy_transform_to_mmdb(const Transform& tr,
17 mmdb::mat33& mat, mmdb::vect3& vec) {
18 for (int i = 0; i < 3; ++i) {
19 for (int j = 0; j < 3; ++j)
20 mat[i][j] = tr.mat[i][j];
21 vec[i] = tr.vec.at(i);
22 }
23}

42inline mmdb::CisPep* cispep_to_mmdb(const CisPep& g, int ser_num, int model_num) {
43 mmdb::CisPep* m = new mmdb::CisPep();
44 m->serNum = ser_num;
45 strcpy_to_mmdb(m->pep1, g.partner_c.res_id.name);
46 strcpy_to_mmdb(m->chainID1, g.partner_c.chain_name);
47 set_seqid_in_mmdb(&m->seqNum1, m->icode1, g.partner_c.res_id.seqid);
48 strcpy_to_mmdb(m->pep2, g.partner_n.res_id.name);
49 strcpy_to_mmdb(m->chainID2, g.partner_n.chain_name);
50 set_seqid_in_mmdb(&m->seqNum2, m->icode2, g.partner_n.res_id.seqid);
51 m->modNum = model_num;
52 m->measure = g.reported_angle;
53 return m;
54}

69inline mmdb::Manager* copy_to_mmdb(const Structure& st, mmdb::Manager* manager) {
70 for (const std::string& s : st.raw_remarks) {
71 std::string line = rtrim_str(s);
72 manager->PutPDBString(line.c_str());
73 }
74
75 for (int imodel = 0; imodel < (int) st.models.size(); ++imodel) {
76 const Model& model = st.models[imodel];
77 mmdb::Model* model2 = mmdb::newModel();
78 manager->AddModel(model2);
79
80 for (const Chain& chain : model.chains) {
81 mmdb::Chain* chain2 = model2->CreateChain(chain.name.c_str());
82 for (const Residue& res : chain.residues) {
83 const char icode[2] = {res.seqid.icode, '\0'};
84 mmdb::Residue* res2 = chain2->GetResidueCreate(res.name.c_str(),
85 *res.seqid.num,
86 icode,
87 true);
88 for (const Atom& atom : res.atoms) {
89 mmdb::Atom* atom2 = mmdb::newAtom();
90 const char altloc[2] = {atom.altloc, '\0'};
91 std::string padded_name = atom.padded_name();
92 // padded_name() is padding from the left; MMDB from both sides
93 if (padded_name.size() < 4)
94 padded_name.resize(4, ' ');
95 atom2->SetAtomName(0, atom.serial, padded_name.c_str(),
96 altloc, res.segment.c_str(), atom.element.uname());
97 atom2->Het = res.het_flag == 'H';
98 atom2->SetCharge(atom.charge);
99 atom2->SetCoordinates(atom.pos.x, atom.pos.y, atom.pos.z,
100 atom.occ, atom.b_iso);
101 if (atom.aniso.nonzero()) {
102 atom2->u11 = atom.aniso.u11;
103 atom2->u22 = atom.aniso.u22;
104 atom2->u33 = atom.aniso.u33;
105 atom2->u12 = atom.aniso.u12;
106 atom2->u13 = atom.aniso.u13;
107 atom2->u23 = atom.aniso.u23;
108 atom2->WhatIsSet |= mmdb::ASET_Anis_tFSigma;
109 }
110 res2->AddAtom(atom2);
111 }
112 // TER
113 if (res.entity_type == EntityType::Polymer &&
114 (&res == &chain.residues.back() ||
115 (&res + 1)->entity_type != EntityType::Polymer)) {
116 mmdb::Atom* atom2 = mmdb::newAtom();
117 atom2->MakeTer();
118 atom2->serNum = res.atoms.back().serial + 1;
119 res2->AddAtom(atom2);
120 }
121 }
122 }
123 manager->PutCell(st.cell.a, st.cell.b, st.cell.c,
124 st.cell.alpha, st.cell.beta, st.cell.gamma, 1);
125 manager->SetSpaceGroup(st.spacegroup_hm.c_str());
126 mmdb::Cryst* cryst = manager->GetCrystData();
127 auto z = st.info.find("_cell.Z_PDB");
128 if (z != st.info.end() && !z->second.empty()) {
129 cryst->Z = std::atoi(z->second.c_str());
130 cryst->WhatIsSet |= mmdb::CSET_ZValue;
131 }
132 if (st.has_origx && !st.origx.is_identity()) {
133 copy_transform_to_mmdb(st.origx, cryst->o, cryst->t);
134 cryst->WhatIsSet |= mmdb::CSET_OrigMatrix;
135 }
136 if (st.cell.explicit_matrices) {
137 copy_transform_to_mmdb(st.cell.frac, cryst->s, cryst->u);
138 cryst->WhatIsSet |= mmdb::CSET_ScaleMatrix;
139 }
140 if (!st.ncs.empty()) {
141 mmdb::mat33 m;
142 mmdb::vect3 v;
143 if (st.info.find("_struct_ncs_oper.id") != st.info.end()) {
144 copy_transform_to_mmdb(Transform{}, m, v);
145 cryst->AddNCSMatrix(m, v, 1);
146 }
147 for (const NcsOp& op : st.ncs) {
148 copy_transform_to_mmdb(op.tr, m, v);
149 cryst->AddNCSMatrix(m, v, op.given);
150 }
151 }
152 }
153 if (!st.cispeps.empty() && !st.models.empty()) {
154 int ser_num = 0;
155 for (const CisPep& cispep : st.cispeps) {
156 // In the PDB, CISPEP records have modNum=0 if there is only one model
157 int modnum = st.models.size() > 1 ? std::stoi(cispep.model_str) : 0;
158 int model_no = std::max(1, modnum); // GetModel() takes 1-based index
159 if (mmdb::Model* m_model = manager->GetModel(model_no))
160 m_model->AddCisPep(cispep_to_mmdb(cispep, ++ser_num, modnum));
161 }
162 }
163 return manager;
164}

168inline Atom copy_atom_from_mmdb(mmdb::Atom& m_atom) {
169 Atom atom;
170 atom.name = m_atom.label_atom_id;
171 atom.altloc = m_atom.altLoc[0];
172 atom.charge = (signed char) m_atom.charge;
173 atom.element = Element(m_atom.element);
174 atom.serial = m_atom.serNum;
175 atom.pos = Position(m_atom.x, m_atom.y, m_atom.z);
176 atom.occ = (float) m_atom.occupancy;
177 atom.b_iso = (float) m_atom.tempFactor;
178 if (m_atom.WhatIsSet & mmdb::ASET_Anis_tFSigma) {
179 atom.aniso.u11 = (float) m_atom.u11;
180 atom.aniso.u22 = (float) m_atom.u22;
181 atom.aniso.u33 = (float) m_atom.u33;
182 atom.aniso.u12 = (float) m_atom.u12;
183 atom.aniso.u13 = (float) m_atom.u13;
184 atom.aniso.u23 = (float) m_atom.u23;
185 }
186 return atom;

189inline Residue copy_residue_from_mmdb(mmdb::Residue& m_res) {
190 Residue res;
191 res.name = m_res.name;
192 res.seqid = seqid_from_mmdb(m_res.seqNum, m_res.insCode);
193 int n = m_res.GetNumberOfAtoms();
194 res.atoms.reserve(n);
195 bool first = true;
196 for (int i = 0; i < n; ++i)
197 if (mmdb::Atom* m_atom = m_res.GetAtom(i)) {
198 if (m_atom->isTer()) {
199 res.entity_type = EntityType::Polymer;
200 } else {
201 res.atoms.push_back(copy_atom_from_mmdb(*m_atom));
202 if (first) {
203 res.het_flag = m_atom->Het ? 'H' : 'A';
204 res.segment = m_atom->segID;
205 first = false;
206 }
207 }
208 }
209 return res;
210}

212inline Chain copy_chain_from_mmdb(mmdb::Chain& m_chain) {
213 Chain chain(m_chain.GetChainID());
214 int n = m_chain.GetNumberOfResidues();
215 chain.residues.reserve(n);
216 for (int i = 0; i < n; ++i)
217 if (mmdb::Residue* m_res = m_chain.GetResidue(i))
218 chain.residues.push_back(copy_residue_from_mmdb(*m_res));
219 // in MMDB we may have pseudo-atom TER that marks polymer end
220 for (auto i = chain.residues.begin(); i != chain.residues.end(); ++i)
221 if (i->entity_type == EntityType::Polymer) { // residue before TER
222 for (auto j = chain.residues.begin(); j != i; ++j)
223 j->entity_type = EntityType::Polymer;
224 for (auto j = i + 1; j != chain.residues.end(); ++j)
225 j->entity_type = j->is_water() ? EntityType::Water
226 : EntityType::NonPolymer;
227 break;
228 }
229 return chain;
230}

242inline Structure copy_from_mmdb(mmdb::Manager* manager) {
243 Structure st;
244 const mmdb::Cryst& cryst = *manager->GetCrystData();
245 st.cell.set(cryst.a, cryst.b, cryst.c, cryst.alpha, cryst.beta, cryst.gamma);
246 if (cryst.WhatIsSet & mmdb::CSET_ZValue)
247 st.info.emplace("_cell.Z_PDB", std::to_string(cryst.Z));
248 st.spacegroup_hm = cryst.spaceGroup;
249 int n = manager->GetNumberOfModels();
250 st.models.reserve(n);
251 for (int i = 1; i <= n; ++i)
252 if (mmdb::Model* m_model = manager->GetModel(i)) {
253 st.models.push_back(copy_model_from_mmdb(*m_model));
254 const std::string& model_name = st.models.back().name;
255 for (int j = 1; j <= m_model->GetNumberOfCisPeps(); ++j)
256 if (const mmdb::CisPep* m_cispep = m_model->GetCisPep(j))
257 st.cispeps.push_back(cispep_from_mmdb(*m_cispep, model_name));
258 }
259 return st;
260}
