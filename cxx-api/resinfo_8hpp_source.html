13struct ResidueInfo {
14 // Simple approximate classification.
15 // AA - aminoacid
16 // AAD - D-aminoacid
17 // PAA - proline-like aminoacid
18 // MAA - methylated aminoacid
19 // RNA, DNA - nucleic acids
20 // HOH - water or heavy water
21 // PYR - pyranose according to the refmac dictionary
22 // KET - ketopyranose according to the refmac dictionary
23 // BUF - agent from crystallization buffer according to PISA agents.dat
24 // ELS - something else (ligand).
25 enum Kind : unsigned char {
26 // when changing this list update check_polymer_type()
27 UNKNOWN=0, AA, AAD, PAA, MAA, RNA, DNA, BUF, HOH, PYR, KET, ELS
28 };
29 Kind kind;
30 // one-letter code or space (uppercase iff it is a standard residues)
31 char one_letter_code;
32 // rough count of hydrogens used to estimate mass with implicit hydrogens
33 std::uint8_t hydrogen_count;
34 // molecular weight
35 float weight;
36
37 bool found() const { return kind != UNKNOWN; }
38 bool is_water() const { return kind == HOH; }
39 bool is_dna() const { return kind == DNA; }
40 bool is_rna() const { return kind == RNA; }
41 bool is_nucleic_acid() const { return is_dna() || is_rna(); }
42 bool is_amino_acid() const {
43 return kind == AA || kind == AAD || kind == PAA || kind == MAA;
44 }
45 bool is_buffer_or_water() const { return kind == HOH || kind == BUF; }
46 // PDB format has non-standard residues (modified AA) marked as HETATM.
47 bool is_standard() const { return (one_letter_code & 0x20) == 0; }
48 char fasta_code() const { return is_standard() ? one_letter_code : 'X'; }
49};
