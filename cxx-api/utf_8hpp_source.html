13inline std::wstring UTF8_to_wchar(const char* in) {
14 std::wstring out;
15 unsigned int codepoint = 0;
16 while (*in != 0) {
17 unsigned char ch = static_cast<unsigned char>(*in);
18 if (ch <= 0x7f)
19 codepoint = ch;
20 else if (ch <= 0xbf)
21 codepoint = (codepoint << 6) | (ch & 0x3f);
22 else if (ch <= 0xdf)
23 codepoint = ch & 0x1f;
24 else if (ch <= 0xef)
25 codepoint = ch & 0x0f;
26 else
27 codepoint = ch & 0x07;
28 ++in;
29 if ((*in & 0xc0) != 0x80 && codepoint <= 0x10ffff) {
30 if (sizeof(wchar_t) > 2) {
31 out.append(1, static_cast<wchar_t>(codepoint));
32 } else if (codepoint > 0xffff) {
33 out.append(1, static_cast<wchar_t>(0xd800 + (codepoint >> 10)));
34 out.append(1, static_cast<wchar_t>(0xdc00 + (codepoint & 0x03ff)));
35 } else if (codepoint < 0xd800 || codepoint >= 0xe000) {
36 out.append(1, static_cast<wchar_t>(codepoint));
37 }
38 }
39 }
40 return out;

43inline std::string wchar_to_UTF8(const wchar_t* in) {
44 std::string out;
45 unsigned int codepoint = 0;
46 while (*in != 0) {
47 if (*in >= 0xd800 && *in <= 0xdbff) {
48 codepoint = ((*in - 0xd800) << 10) + 0x10000;
49 } else {
50 if (*in >= 0xdc00 && *in <= 0xdfff)
51 codepoint |= *in - 0xdc00;
52 else
53 codepoint = *in;
54 if (codepoint <= 0x7f) {
55 out += static_cast<char>(codepoint);
56 } else if (codepoint <= 0x7ff) {
57 out += static_cast<char>(0xc0 | ((codepoint >> 6) & 0x1f));
58 out += static_cast<char>(0x80 | (codepoint & 0x3f));
59 } else if (codepoint <= 0xffff) {
60 out += static_cast<char>(0xe0 | ((codepoint >> 12) & 0x0f));
61 out += static_cast<char>(0x80 | ((codepoint >> 6) & 0x3f));
62 out += static_cast<char>(0x80 | (codepoint & 0x3f));
63 } else {
64 out += static_cast<char>(0xf0 | ((codepoint >> 18) & 0x07));
65 out += static_cast<char>(0x80 | ((codepoint >> 12) & 0x3f));
66 out += static_cast<char>(0x80 | ((codepoint >> 6) & 0x3f));
67 out += static_cast<char>(0x80 | (codepoint & 0x3f));
68 }

72 }
