14inline SmallStructure::Site atom_to_site(const Atom& atom, const UnitCell& cell) {
15 SmallStructure::Site site;
16 site.label = atom.name;
17 site.type_symbol = atom.element.name();
18 site.fract = cell.fractionalize(atom.pos);
19 site.occ = atom.occ;
20 // occupancy may need to be adjusted if the atom is on special position
21 if (atom.occ <= 0.5) {
22 int n_mates = cell.is_special_position(atom.pos);
23 if (n_mates > 0 && atom.occ * (n_mates + 1) <= 1.0)
24 site.occ = atom.occ * (n_mates + 1);
25 }
26 site.u_iso = atom.b_iso / u_to_b();
27 if (atom.aniso.nonzero()) {
28 if (cell.alpha == 90. || cell.beta == 90. || cell.gamma == 90.) {
29 site.aniso = atom.aniso.scaled(1.0);
30 } else {
31 SMat33<double> t = atom.aniso.transformed_by<>(cell.frac.mat);
32 Vec3 v = {1.0 / cell.ar, 1.0 / cell.br, 1.0 / cell.cr};
33 site.aniso = {t.u11 * v.x * v.x,
34 t.u22 * v.y * v.y,
35 t.u33 * v.z * v.z,
36 t.u12 * v.x * v.y,
37 t.u13 * v.x * v.z,
38 t.u23 * v.y * v.z};
39 }
40 }
41 site.element = atom.element;
42 site.charge = atom.charge;
43 return site;
44}

46inline SmallStructure mx_to_sx_structure(const Structure& st, int n=0) {
47 const Model& model = st.models.at(n);
48 SmallStructure small;
49 small.name = st.name;
50 small.cell = st.cell;
51 small.spacegroup_hm = st.spacegroup_hm;
52 small.sites.reserve(count_atom_sites(model));
53 for (const Chain& chain : model.chains)
54 for (const Residue& residue : chain.residues)
55 for (const Atom& atom : residue.atoms)
56 small.sites.push_back(atom_to_site(atom, st.cell));
57 return small;
58}
