17template<int N, typename Real>
18Real determine_cutoff_radius(Real x1, const ExpSum<N, Real>& precal, Real cutoff_level) {
19 Real y1, dy;
20 std::tie(y1, dy) = precal.calculate_with_derivative(x1);
21 // Generally, density is supposed to decrease with radius.
22 // But if we have addends (in particular -Z for Mott-Bothe),
23 // it can first rise, then decrease. We want to be after the maximum.
24 while (dy > 0) { // unlikely
25 x1 += 1.0f;
26 std::tie(y1, dy) = precal.calculate_with_derivative(x1);
27 }
28 Real x2 = x1;
29 Real y2 = y1;
30 if (y1 < cutoff_level) {
31 while (y1 < cutoff_level) {
32 x2 = x1;
33 y2 = y1;
34 x1 -= 0.5f;
35 std::tie(y1, dy) = precal.calculate_with_derivative(x1);
36 // with addends it's possible to land on the left side of the maximum
37 if (dy > 0) { // unlikely
38 while (dy > 0 && x1 + 0.1f < x2) {
39 x1 += 0.1f;
40 std::tie(y1, dy) = precal.calculate_with_derivative(x1);
41 }
42 if (y1 < cutoff_level)
43 return x1;
44 break;
45 }
46 if (x1 < 0) { // unlikely
47 x1 = 0;
48 y1 = precal.calculate(x1 * x1);
49 break;
50 }
51 }
52 } else {
53 while (y2 > cutoff_level) {
54 x1 = x2;
55 y1 = y2;
56 x2 += 0.5f;
57 y2 = precal.calculate(x2 * x2);
58 }
59 }
60
61 return x1 + (x1 - x2) / (y1 - y2) * (cutoff_level - y1);
62}

64// approximated radius of electron density (IT92) above cutoff=1e-5 for C
65template <typename Real>
66Real it92_radius_approx(Real b) {
67 return (8.5f + 0.075f * b) / (2.4f + 0.0045f * b);
68}

95 void set_refmac_compatible_blur(const Model& model, bool allow_negative=false) {
96 double spacing = requested_grid_spacing();

99 double b_min = calculate_b_aniso_range(model).first;

117 template<int N>
118 CReal estimate_radius(const ExpSum<N, CReal>& precal, CReal b) const {
119 if (N == 1)
120 return std::sqrt(std::log(cutoff / std::abs(precal.a[0])) / precal.b[0]);
121 CReal x1 = it92_radius_approx(b);
122 return determine_cutoff_radius(x1, precal, (CReal)cutoff);
123 }

125 template<typename Coef>
126 void do_add_atom_density_to_grid(const Atom& atom, const Coef& coef, float addend) {
127#if GEMMI_COUNT_DC
128 ++atoms_added;
129#endif
130 Fractional fpos = grid.unit_cell.fractionalize(atom.pos);
131 if (!atom.aniso.nonzero()) {
132 // isotropic
133 CReal b = static_cast<CReal>(atom.b_iso + blur);
134 auto precal = coef.precalculate_density_iso(b, addend);
135 CReal radius = estimate_radius(precal, b);
136 grid.template use_points_around<true>(fpos, radius, [&](GReal& point, double r2) {
137 point += GReal(atom.occ * precal.calculate((CReal)r2));
138#if GEMMI_COUNT_DC
139 ++density_computations;
140#endif
141 }, /*fail_on_too_large_radius=*/false);
142 } else {
143 // anisotropic
144 auto aniso_b = atom.aniso.scaled(CReal(u_to_b())).added_kI(CReal(blur));
145 // rough estimate, so we don't calculate eigenvalues
146 CReal b_max = std::max(std::max(aniso_b.u11, aniso_b.u22), aniso_b.u33);
147 auto precal_iso = coef.precalculate_density_iso(b_max, addend);
148 double radius = estimate_radius(precal_iso, b_max);
149 auto precal = coef.precalculate_density_aniso_b(aniso_b, addend);
150 int du = (int) std::ceil(radius / grid.spacing[0]);
151 int dv = (int) std::ceil(radius / grid.spacing[1]);
152 int dw = (int) std::ceil(radius / grid.spacing[2]);
153 grid.template use_points_in_box<true>(
154 fpos, du, dv, dw,
155 [&](GReal& point, double, const Position& delta, int, int, int) {
156 point += GReal(atom.occ * precal.calculate(delta));
157#if GEMMI_COUNT_DC
158 ++density_computations;
159#endif
160 },
161 false, radius);
162 }
163 }

165 void initialize_grid() {
166 grid.data.clear();
167 double spacing = requested_grid_spacing();
168 if (spacing > 0)
169 grid.set_size_from_spacing(spacing, GridSizeRounding::Up);
170 else if (grid.point_count() > 0)
171 // d_min not set, but a custom grid has been setup by the user
172 grid.fill(0.);
173 else
174 fail("initialize_grid(): d_min is not set");
175 }

177 void add_model_density_to_grid(const Model& model) {
178 grid.check_not_empty();
179 for (const Chain& chain : model.chains)
180 for (const Residue& res : chain.residues)
181 for (const Atom& atom : res.atoms)
182 add_atom_density_to_grid(atom);
183 }

196 // The argument is 1/d^2 - as outputted by unit_cell.calculate_1_d2(hkl).
197 double reciprocal_space_multiplier(double inv_d2) const {
198 return std::exp(blur * 0.25 * inv_d2);
199 }

201 double mott_bethe_factor(const Miller& hkl) const {
202 double inv_d2 = grid.unit_cell.calculate_1_d2(hkl);
203 double factor = -mott_bethe_const() / inv_d2;
204 return blur == 0 ? factor : factor * reciprocal_space_multiplier(inv_d2);
