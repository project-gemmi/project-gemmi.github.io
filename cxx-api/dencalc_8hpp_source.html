16template <typename PrecalExpSum>
17double determine_cutoff_radius(double x1, const PrecalExpSum& precal, double cutoff_level) {
18 double y1, dy;
19 std::tie(y1, dy) = precal.calculate_with_derivative(x1);
20 // Generally, density is supposed to decrease with radius.
21 // But if we have addends (in particular -Z for Mott-Bothe),
22 // it can first rise, then decrease. We want to be after the maximum.
23 while (dy > 0) { // unlikely
24 x1 += 1.0;
25 std::tie(y1, dy) = precal.calculate_with_derivative(x1);
26 }
27 double x2 = x1;
28 double y2 = y1;
29 if (y1 < cutoff_level) {
30 while (y1 < cutoff_level) {
31 x2 = x1;
32 y2 = y1;
33 x1 -= 0.5f;
34 std::tie(y1, dy) = precal.calculate_with_derivative(x1);
35 // with addends it's possible to land on the left side of the maximum
36 if (dy > 0) { // unlikely
37 while (dy > 0 && x1 + 0.1 < x2) {
38 x1 += 0.1;
39 std::tie(y1, dy) = precal.calculate_with_derivative(x1);
40 }
41 if (y1 < cutoff_level)
42 return x1;
43 break;
44 }
45 if (x1 < 0) { // unlikely
46 x1 = 0;
47 y1 = precal.calculate(x1 * x1);
48 break;
49 }
50 }
51 } else {
52 while (y2 > cutoff_level) {
53 x1 = x2;
54 y1 = y2;
55 x2 += 0.5f;
56 y2 = precal.calculate(x2 * x2);
57 }
58 }
59
60 return x1 + (x1 - x2) / (y1 - y2) * (cutoff_level - y1);
61}

63// approximated radius of electron density (IT92) above cutoff=1e-5 for C
64inline double it92_radius_approx(double b) {
65 return (8.5 + 0.075 * b) / (2.4 + 0.0045 * b);
66}

68inline double get_minimum_b_iso(const Model& model) {
69 double b_min = 1000.;
70 for (const Chain& chain : model.chains)
71 for (const Residue& residue : chain.residues)
72 for (const Atom& atom : residue.atoms)
73 if (atom.b_iso < b_min)
74 b_min = atom.b_iso;
75 return b_min;
76}

119 template<int N>
120 double estimate_radius(const ExpSum<N, coef_type>& precal, double b) const {
121 if (N == 1)
122 return std::sqrt(std::log(cutoff / std::abs(precal.a[0])) / precal.b[0]);
123 double x1 = it92_radius_approx(b);
124 return determine_cutoff_radius(x1, precal, cutoff);
125 }

127 template<typename Coef>
128 void do_add_atom_density_to_grid(const Atom& atom, const Coef& coef, float addend) {
129 Fractional fpos = grid.unit_cell.fractionalize(atom.pos);
130 if (!atom.aniso.nonzero()) {
131 // isotropic
132 double b = atom.b_iso + blur;
133 auto precal = coef.precalculate_density_iso(b, addend);
134 double radius = estimate_radius(precal, b);
135 grid.template use_points_around<true>(fpos, radius, [&](Real& point, double r2) {
136 point += Real(atom.occ * precal.calculate((Real)r2));
137 }, /*fail_on_too_large_radius=*/false);
138 } else {
139 // anisotropic
140 SMat33<double> aniso_b = atom.aniso.scaled(u_to_b()).added_kI(blur);
141 // rough estimate, so we don't calculate eigenvalues
142 double b_max = std::max(std::max(aniso_b.u11, aniso_b.u22), aniso_b.u33);
143 auto precal_iso = coef.precalculate_density_iso(b_max, addend);
144 double radius = estimate_radius(precal_iso, b_max);
145 auto precal = coef.precalculate_density_aniso_b(aniso_b, addend);
146 int du = (int) std::ceil(radius / grid.spacing[0]);
147 int dv = (int) std::ceil(radius / grid.spacing[1]);
148 int dw = (int) std::ceil(radius / grid.spacing[2]);
149 grid.template use_points_in_box<true>(fpos, du, dv, dw,
150 [&](Real& point, const Position& delta) {
151 if (delta.length_sq() < radius * radius)
152 point += Real(atom.occ * precal.calculate(delta));
153 }, false);
154 }
155 }

168 void add_model_density_to_grid(const Model& model) {
169 grid.check_not_empty();
170 for (const Chain& chain : model.chains)
171 for (const Residue& res : chain.residues)
172 for (const Atom& atom : res.atoms)
173 add_atom_density_to_grid(atom);
174 }

187 // The argument is 1/d^2 - as outputted by unit_cell.calculate_1_d2(hkl).
188 double reciprocal_space_multiplier(double inv_d2) const {
189 return std::exp(blur * 0.25 * inv_d2);
190 }
191
192 double mott_bethe_factor(const Miller& hkl) const {
193 double inv_d2 = grid.unit_cell.calculate_1_d2(hkl);
194 double factor = -1. / (2 * pi() * pi() * bohrradius()) / inv_d2;
195 return blur == 0 ? factor : factor * reciprocal_space_multiplier(inv_d2);
196 }
