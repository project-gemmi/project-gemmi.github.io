46 // pre: calc and obs are sorted
47 Scaling(const UnitCell& cell_, const SpaceGroup* sg)
48 : cell(cell_) {
49 if (sg) {
50 crystal_system = sg->crystal_system();
51 if (crystal_system == CrystalSystem::Monoclinic) {
52 // take first letter in "c" or "c1", but second in "-c1"
53 char letter = sg->qualifier[sg->qualifier[0] == '-'];
54 if (letter == 'a') {
55 monoclinic_angle_idx = 5;
56 monoclinic_angle = &SMat33<double>::u23;
57 } else if (letter == 'c') {
58 monoclinic_angle_idx = 3;
59 monoclinic_angle = &SMat33<double>::u12;
60 } else {
61 monoclinic_angle_idx = 4;
62 monoclinic_angle = &SMat33<double>::u13;
63 }
64 } else if (crystal_system == CrystalSystem::Trigonal) {
65 if (sg->ext != 'R')
66 crystal_system = CrystalSystem::Hexagonal;

80 // Scale data, optionally adding bulk solvent correction.
81 void scale_data(AsuData<std::complex<Real>>& asu_data,
82 const AsuData<std::complex<Real>>& mask_data) const {
83 if (use_solvent && mask_data.size() != asu_data.size())
84 fail("scale_data(): mask data not prepared");
85 bool use_scaling = (k_overall != 1 || !b_star.all_zero());
86 for (size_t i = 0; i != asu_data.v.size(); ++i) {
87 HklValue<std::complex<Real>>& hv = asu_data.v[i];
88 if (use_solvent) {
89 if (hv.hkl != mask_data.v[i].hkl)
90 fail("scale_data(): data arrays don't match");
91 double stol2 = cell.calculate_stol_sq(hv.hkl);
92 hv.value += (Real)get_solvent_scale(stol2) * mask_data.v[i].value;
93 }
94 if (use_scaling)
95 hv.value *= (Real) get_overall_scale_factor(hv.hkl);
96 }
97 }

99 std::complex<Real> scale_value(const Miller& hkl, std::complex<Real> f_value,
100 std::complex<Real> mask_value) {
101 if (use_solvent) {
102 double stol2 = cell.calculate_stol_sq(hkl);
103 f_value += (Real)get_solvent_scale(stol2) * mask_value;
104 }
105 return f_value * (Real) get_overall_scale_factor(hkl);
106 }

108 std::vector<double> get_parameters() const {
109 std::vector<double> ret;
110 ret.push_back(k_overall);
111 if (use_solvent) {
112 if (!fix_k_sol)
113 ret.push_back(k_sol);
114 if (!fix_b_sol)
115 ret.push_back(b_sol);
116 }
117 switch (crystal_system) {
118 case CrystalSystem::Triclinic:
119 ret.insert(ret.end(), {b_star.u11, b_star.u22, b_star.u33,
120 b_star.u12, b_star.u13, b_star.u23});
121 break;
122 case CrystalSystem::Monoclinic:
123 ret.insert(ret.end(), {b_star.u11, b_star.u22, b_star.u33,
124 b_star.*monoclinic_angle});
125 break;
126 case CrystalSystem::Orthorhombic:
127 ret.insert(ret.end(), {b_star.u11, b_star.u22, b_star.u33});
128 break;
129 case CrystalSystem::Tetragonal:
130 ret.insert(ret.end(), {b_star.u11, b_star.u33});
131 break;
132 case CrystalSystem::Trigonal:
133 ret.insert(ret.end(), {b_star.u11, b_star.u12});
134 break;
135 case CrystalSystem::Hexagonal:
136 ret.insert(ret.end(), {b_star.u11, b_star.u33});
137 break;
138 case CrystalSystem::Cubic:
139 ret.push_back(b_star.u11);
140 break;
141 }
142 return ret;
143 }

145 void set_parameters(const std::vector<double>& p) {
146 k_overall = p[0];
147 int n = 0;
148 if (use_solvent) {
149 if (!fix_k_sol)
150 k_sol = p[++n];
151 if (!fix_b_sol)
152 b_sol = p[++n];
153 }
154 switch (crystal_system) {
155 case CrystalSystem::Triclinic:
156 b_star = {p[n+1], p[n+2], p[n+3], p[n+4], p[n+5], p[n+6]}; break;
157 case CrystalSystem::Monoclinic:
158 b_star = {p[n+1], p[n+2], p[n+3], 0, 0, 0};
159 b_star.*monoclinic_angle = p[n+4];
160 break;
161 case CrystalSystem::Orthorhombic:
162 b_star = {p[n+1], p[n+2], p[n+3], 0., 0., 0.}; break;
163 case CrystalSystem::Tetragonal:
164 b_star = {p[n+1], p[n+1], p[n+2], 0., 0., 0.}; break;
165 case CrystalSystem::Trigonal:
166 b_star = {p[n+1], p[n+1], p[n+1], p[n+2], p[n+2], p[n+2]}; break;
167 case CrystalSystem::Hexagonal:
168 b_star = {p[n+1], p[n+1], p[n+2], 0.5*p[n+1], 0., 0.}; break;
169 case CrystalSystem::Cubic:
170 b_star = {p[n+1], p[n+1], p[n+1], 0., 0., 0.}; break;
171 }
172 }

174 void prepare_points(const AsuData<std::complex<Real>>& calc,
175 const AsuData<ValueSigma<Real>>& obs,
176 const AsuData<std::complex<Real>>& mask_data) {
177 if (use_solvent && mask_data.size() != calc.size())
178 fail("prepare_points(): mask data not prepared");
179 std::complex<Real> fmask;
180 points.reserve(std::min(calc.size(), obs.size()));
181 auto c = calc.v.begin();
182 for (const HklValue<ValueSigma<Real>>& o : obs.v) {
183 if (c->hkl != o.hkl) {
184 while (*c < o.hkl) {
185 ++c;
186 if (c == calc.v.end())
187 return;
188 }
189 if (c->hkl != o.hkl)
190 continue;
191 }
192 if (use_solvent) {
193 const HklValue<std::complex<Real>>& m = mask_data.v[c - calc.v.begin()];
194 if (m.hkl != c->hkl)
195 fail("prepare_points(): unexpected data");
196 fmask = m.value;
197 }
198 double stol2 = cell.calculate_stol_sq(o.hkl);
199 if (!std::isnan(o.value.value) && !std::isnan(o.value.sigma))
200 points.push_back({o.hkl, stol2, c->value, fmask, o.value.value, o.value.sigma});
201 ++c;
202 if (c == calc.v.end())
203 break;
204 }
205 }

208 double get_solvent_scale(double stol2) const {
209 return k_sol * std::exp(-b_sol * stol2);

212 double get_overall_isotropic_scale_factor(const Miller& hkl) const {
213 return k_overall * std::exp(-b_star.u11 * Vec3(hkl).length_sq());
214 }

216 double get_overall_scale_factor(const Miller& hkl) const {
217 return k_overall * std::exp(-0.25 * b_star.r_u_r(hkl));

220 // quick linear fit (ignoring sigma) to get initial parameters
221 void fit_isotropic_b_approximately() {
222 double sx = 0, sy = 0, sxx = 0, sxy = 0;
223 int n = 0;
224 for (const Point& p : points) {
225 if (p.fobs < 1 || p.fobs < p.sigma) // skip weak reflections
226 continue;
227 double x = p.stol2;
228 double fcalc = std::abs(use_solvent ? p.fcmol + (Real)get_solvent_scale(x) * p.fmask
229 : p.fcmol);
230 double y = std::log(static_cast<float>(p.fobs / fcalc));
231 sx += x;
232 sy += y;
233 sxx += x * x;
234 sxy += x * y;
235 n += 1;
236 }
237 if (n <= 5) // this is not expected to happen
238 return;
239 double slope = (n * sxy - sx * sy) / (n * sxx - sx * sx);
240 double intercept = (sy - slope * sx) / n;
241 double b_iso = -slope;
242 k_overall = std::exp(intercept);
243 set_b_overall({b_iso, b_iso, b_iso, 0, 0, 0});
244 }

252 // interface for fitting
253 std::vector<double> compute_values() const {
254 std::vector<double> values;
255 values.reserve(points.size());
256 for (const Point& p : points) {
257 double fcalc = std::abs(p.fcmol + (Real)get_solvent_scale(p.stol2) * p.fmask);
258 values.push_back(fcalc * (Real) get_overall_scale_factor(p.hkl));
259 }
260 return values;

263 // the tile_* parameters allow tiling: computing derivatives from a span
264 // of points at one time, which limits memory usage.
265 void compute_values_and_derivatives(size_t tile_start, size_t tile_size,
266 std::vector<double>& yy,
267 std::vector<double>& dy_da) const {
268 assert(tile_size == yy.size());
269 size_t npar = dy_da.size() / tile_size;
270 assert(dy_da.size() == npar * tile_size);
271 int n = 1;
272 if (use_solvent)
273 n += int(!fix_k_sol) + int(!fix_b_sol);
274 for (size_t i = 0; i != tile_size; ++i) {
275 const Point& pt = points[tile_start+i];
276 Vec3 h(pt.hkl);
277 double kaniso = std::exp(-0.25 * b_star.r_u_r(h));
278 double fcalc_abs;
279 if (use_solvent) {
280 double solv_b = std::exp(-b_sol * pt.stol2);
281 double solv_scale = k_sol * solv_b;
282 auto fcalc = pt.fcmol + (Real)solv_scale * pt.fmask;
283 fcalc_abs = std::abs(fcalc);
284 size_t offset = i * npar + 1;
285 double dy_dsol = (fcalc.real() * pt.fmask.real() +
286 fcalc.imag() * pt.fmask.imag()) / fcalc_abs * k_overall * kaniso;
287 if (!fix_k_sol)
288 dy_da[offset++] = solv_b * dy_dsol;
289 if (!fix_b_sol)
290 dy_da[offset] = -pt.stol2 * solv_scale * dy_dsol;
291 } else {
292 fcalc_abs = std::abs(pt.fcmol);
293 }
294 double fe = fcalc_abs * kaniso;
295 yy[i] = k_overall * fe;
296 dy_da[i * npar + 0] = fe; // dy/d k_overall
297 double du[6] = {
298 -0.25 * yy[i] * (h.x * h.x),
299 -0.25 * yy[i] * (h.y * h.y),
300 -0.25 * yy[i] * (h.z * h.z),
301 -0.5 * yy[i] * (h.x * h.y),
302 -0.5 * yy[i] * (h.x * h.z),
303 -0.5 * yy[i] * (h.y * h.z),
304 };
305 double* dy_db = &dy_da[i * npar + n];
306 switch (crystal_system) {
307 case CrystalSystem::Triclinic:
308 for (int j = 0; j < 6; ++j)
309 dy_db[j] = du[j];
310 break;
311 case CrystalSystem::Monoclinic:
312 for (int j = 0; j < 3; ++j)
313 dy_db[j] = du[j];
314 dy_db[3] = du[monoclinic_angle_idx];
315 break;
316 case CrystalSystem::Orthorhombic:
317 for (int j = 0; j < 3; ++j)
318 dy_db[j] = du[j];
319 break;
320 case CrystalSystem::Tetragonal:
321 dy_db[0] = du[0] + du[1];
322 dy_db[1] = du[2];
323 break;
324 case CrystalSystem::Trigonal:
325 dy_db[0] = du[0] + du[1] + du[2];
326 dy_db[1] = du[3] + du[4] + du[5];
327 break;
328 case CrystalSystem::Hexagonal:
329 dy_db[0] = du[0] + du[1] + 0.5 * du[3];
330 dy_db[1] = du[2];
331 break;
332 case CrystalSystem::Cubic:
333 dy_db[0] = du[0] + du[1] + du[2];
334 break;
335 }
336 }
337 }
