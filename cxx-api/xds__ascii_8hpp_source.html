17// from Pointless docs: likely in-house source, in which case
18// the unpolarised value is left unchanged (recognised wavelengths
19// are CuKalpha 1.5418 +- 0.0019, Mo 0.7107 +- 0.0002, Cr 2.29 +- 0.01)
20inline bool likely_in_house_source(double wavelength) {
21 return std::fabs(wavelength - 1.5418) < 0.0019 ||
22 std::fabs(wavelength - 0.7107) < 0.0002 ||
23 std::fabs(wavelength - 2.29) < 0.01;
24}

137 Mat33 calculate_conversion_from_cambridge() const {
138 // Cambridge z direction is along the principal rotation axis
139 Vec3 z = get_rotation_axis();
140 // Cambridge z direction is along beam
141 Vec3 x = get_s0_direction();
142 Vec3 y = z.cross(x).normalized();
143 // beam and rotation axis may not be orthogonal
144 x = y.cross(z).normalized();
145 return Mat33::from_columns(x, y, z);
146 }
147
148 Mat33 get_orientation() const {
149 if (!has_cell_axes())
150 fail("unknown unit cell axes");
151 Vec3 a = cell_axes.row_copy(0);
152 Vec3 b = cell_axes.row_copy(1);
153 Vec3 c = cell_axes.row_copy(2);
154 Vec3 ar = b.cross(c).normalized();
155 Vec3 br = c.cross(a);
156 Vec3 cr = ar.cross(br).normalized();
157 br = cr.cross(ar);
158 return Mat33::from_columns(ar, br, cr);
159 }
