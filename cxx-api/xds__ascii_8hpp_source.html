73 void gather_iset_statistics() {
74 for (Iset& iset : isets) {
75 double min_zd = 1e6;
76 double max_zd = 0.;
77 for (const XdsAscii::Refl& refl : data)
78 if (refl.iset == iset.id) {
79 ++iset.reflection_count;
80 min_zd = std::min(min_zd, refl.zd);
81 max_zd = std::max(max_zd, refl.zd);
82 }
83 iset.frame_number_min = (int)std::ceil(min_zd);
84 iset.frame_number_max = (int)std::ceil(max_zd);
85 std::vector<uint8_t> frames(iset.frame_number_max - iset.frame_number_min + 1);
86 for (const XdsAscii::Refl& refl : data)
87 if (refl.iset == iset.id)
88 frames[(int)std::ceil(refl.zd) - iset.frame_number_min] = 1;
89 iset.frame_count = 0;
90 for (uint8_t f : frames)
91 iset.frame_count += f;
92 }
93 }

114template<typename Stream>
115void XdsAscii::read_stream(Stream&& stream, const std::string& source) {
116 static const char* expected_columns[8] = {
117 "H=1", "K=2", "L=3", "IOBS=4", "SIGMA(IOBS)=5", "XD=6", "YD=7", "ZD=8",
118 };
119 source_path = source;
120 char line[256];
121 size_t len0 = copy_line_from_stream(line, 255, stream);
122 int iset_col = 0;
123 if (len0 == 0 || !starts_with(line, "!FORMAT=XDS_ASCII MERGE=FALSE"))
124 fail("not an unmerged XDS_ASCII file: " + source_path);
125 const char* rhs;
126 while (size_t len = copy_line_from_stream(line, 255, stream)) {
127 if (line[0] == '!') {
128 if (starts_with_ptr(line+1, "Generated by ", &rhs)) {
129 generated_by = read_word(rhs);
130 } else if (starts_with_ptr(line+1, "SPACE_GROUP_NUMBER=", &rhs)) {
131 spacegroup_number = simple_atoi(rhs);
132 } else if (starts_with_ptr(line+1, "UNIT_CELL_CONSTANTS=", &rhs)) {
133 double par[6];
134 xds_parse_cell_constants(rhs, line+len, par, line);
135 unit_cell.set(par[0], par[1], par[2], par[3], par[4], par[5]);
136 } else if (starts_with_ptr(line+1, "X-RAY_WAVELENGTH=", &rhs)) {
137 auto result = fast_from_chars(rhs, line+len, wavelength);
138 if (result.ec != std::errc())
139 fail("failed to parse wavelength:\n", line);
140 } else if (starts_with_ptr(line+1, "OSCILLATION_RANGE=", &rhs)) {
141 auto result = fast_from_chars(rhs, line+len, oscillation_range);
142 if (result.ec != std::errc())
143 fail("failed to parse:\n", line);
144 } else if (starts_with_ptr(line+1, "STARTING_ANGLE=", &rhs)) {
145 auto result = fast_from_chars(rhs, line+len, starting_angle);
146 if (result.ec != std::errc())
147 fail("failed to parse:\n", line);
148 } else if (starts_with_ptr(line+1, "STARTING_FRAME=", &rhs)) {
149 starting_frame = simple_atoi(rhs);
150 } else if (starts_with_ptr(line+1, " ISET= ", &rhs)) {
151 const char* endptr;
152 int id = simple_atoi(rhs, &endptr);
153 XdsAscii::Iset& iset = find_or_add_iset(id);
154 endptr = skip_blank(endptr);
155 if (starts_with_ptr(endptr, "INPUT_FILE=", &rhs)) {
156 iset.input_file = read_word(rhs);
157 } else if (starts_with_ptr(endptr, "X-RAY_WAVELENGTH=", &rhs)) {
158 double w;
159 auto result = fast_from_chars(rhs, line+len, w);
160 if (result.ec != std::errc())
161 fail("failed to parse iset wavelength:\n", line);
162 iset.wavelength = w;
163 } else if (starts_with_ptr(endptr, "UNIT_CELL_CONSTANTS=", &rhs)) {
164 xds_parse_cell_constants(rhs, line+len, iset.cell_constants, line);
165 }
166 } else if (starts_with_ptr(line+1, "NUMBER_OF_ITEMS_IN_EACH_DATA_RECORD=", &rhs)) {
167 int num = simple_atoi(rhs);
168 if (num < 8)
169 fail("expected 8+ columns, got:\n", line);
170 for (const char* col : expected_columns) {
171 copy_line_from_stream(line, 40, stream);
172 if (std::strncmp(line, "!ITEM_", 6) != 0 ||
173 std::strncmp(line+6, col, std::strlen(col)) != 0)
174 fail("column !ITEM_" + std::string(col), " not found.");
175 }
176 } else if (starts_with_ptr(line+1, "ITEM_ISET=", &rhs)) {
177 iset_col = simple_atoi(rhs);
178 } else if (starts_with(line+1, "END_OF_DATA")) {
179 if (isets.empty()) {
180 for (XdsAscii::Refl& refl : data) {
181 if (refl.iset != 0 && refl.iset != 1)
182 fail("unexpected ITEM_ISET " + std::to_string(refl.iset));
183 refl.iset = 1;
184 }
185 isets.emplace_back(1);
186 isets.back().wavelength = wavelength;
187 }
188 return;
189 }
190 } else {
191 data.emplace_back();
192 XdsAscii::Refl& r = data.back();
193 const char* p = line;
194 for (int i = 0; i < 3; ++i)
195 r.hkl[i] = simple_atoi(p, &p);
196 auto result = fast_from_chars(p, line+len, r.iobs); // 4
197 result = fast_from_chars(result.ptr, line+len, r.sigma); // 5
198 result = fast_from_chars(result.ptr, line+len, r.xd); // 6
199 result = fast_from_chars(result.ptr, line+len, r.yd); // 7
200 result = fast_from_chars(result.ptr, line+len, r.zd); // 8
201 if (result.ec != std::errc())
202 fail("failed to parse data line:\n", line);
203 if (iset_col > 8) {
204 const char* iset_ptr = result.ptr;
205 for (int j = 9; j < iset_col; ++j)
206 iset_ptr = skip_word(skip_blank(iset_ptr));
207 r.iset = simple_atoi(iset_ptr);
208 }
209 }
210 }
211 fail("incorrect or unfinished file: " + source_path);
212}
