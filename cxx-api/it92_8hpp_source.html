3// X-ray scattering factor coefficients from International Tables
4// for Crystallography Volume C, edition from 1992 or later.
5//
6// The same data is in CCP4 (file atomsf.lib) and in cctbx (table="it1992").
7// But here we have only one entry per element. The original table also
8// lists ions, for example Fe2+ and Fe3+ in addition to Fe.
9// Scattering factors from ITC are approximated by four Gaussians
10// and a constant value - together 9 coefficients known as Cromer-Mann
11// coefficients. The approximation holds for sin(theta)/lambda < 2 A^-1.

26template<class Real>
27struct IT92 {
28 using Coef = GaussianCoef<4, 1, Real>;
29 static Coef data[99+112];
30 static std::pair<El, signed char> ion_list[112];
31 static bool ignore_charge;
32
33 static bool has(El el) {
34 return el <= El::Cf || el == El::D;
35 }
36
37 static Coef& get(El el, signed char charge) {
38 // ordinal for X, H, ... Cf; H=1 for D; X=0 for Es, ... Og
39 int pos = el <= El::Cf ? (int)el : (int)(el == El::D);
40 if (charge != 0 && !ignore_charge) {
41 std::pair<El, signed char> p{el, charge};
42 int start = std::max(0, (int)el - 8); // optimization based on plot of index(El)
43 for (int i = start; i < 112; ++i) {
44 if (ion_list[i] == p) {
45 pos = 99 + i;
46 break;
47 }
48 if (ion_list[i] > p)
49 break;
50 }
51 }
52 return data[pos];
53 }
54
55 static Coef* get_exact(El el, signed char charge) {
56 if (has(el)) {
57 Coef* coef = &get(el, charge);
58 if (charge == 0 || coef - data > int(El::Cf))
59 return coef;
60 }
61 return nullptr;
62 }
63
64 // Make a1+a2+a3+a4+c equal exactly to the number of electrons.
65 // It changes the values from ITC only slightly (by not more than 0.12%).
66 static void normalize_one(Coef& f, int n) {
67 double factor = n / (f.a(0) + f.a(1) + f.a(2) + f.a(3) + f.c());
68 for (int j = 0; j < 4; ++j)
69 f.coefs[j] *= factor;
70 f.coefs[8] *= factor;
71 }
72 static void normalize() {
73 for (int i = 1; i < 99; ++i)
74 normalize_one(data[i], i);
75 for (int i = 0; i < 112; ++i) {
76 int n = (int)ion_list[i].first - ion_list[i].second;
77 normalize_one(data[99+i], n);
78 }
79 }
80};
