3// X-ray scattering factor coefficients from International Tables
4// for Crystallography Volume C, edition from 1992 or later.
5//
6// The same data is in CCP4 (file atomsf.lib) and in cctbx (table="it1992").
7// But here we have only one entry per element. The original table also
8// lists ions, for example Fe2+ and Fe3+ in addition to Fe.
9// Scattering factors from ITC are approximated by four Gaussians
10// and a constant value - together 9 coefficients known as Cromer-Mann
11// coefficients. The approximation holds for sin(theta)/lambda < 2 A^-1.

26template<class Real>
27struct IT92 {
28 using Coef = GaussianCoef<4, 1, Real>;
29 static Coef data[99];
30
31 static bool has(El el) {
32 return (unsigned char)el < 99 || el == El::D;
33 }
34
35 static Coef& get(El el) {
36 return data[it92_pos(el)];
37 }
38
39 static Coef* get_ptr(El el) {
40 return has(el) ? &get(el) : nullptr;
41 }
42
43 // Make a1+a2+a3+a4+c equal exactly to Z (number of electrons).
44 // It changes the values from ITC only slightly (by not more than 0.12%).
45 static void normalize() {
46 for (int i = 1; i < 99; ++i) {
47 Coef& f = data[i];
48 double factor = i / (f.a(0) + f.a(1) + f.a(2) + f.a(3) + f.c());
49 for (int j = 0; j < 4; ++j)
50 f.coefs[j] *= factor;
51 f.coefs[8] *= factor;
52 }
53 }
54};
