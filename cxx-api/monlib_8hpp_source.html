32struct ChemLink {
33 enum class Group {
34 // _chem_link.group_comp_N is one of:
35 // "peptide", "P-peptide", "M-peptide", "pyranose", "DNA/RNA" or null
36 // (we ignore "polymer")
37 Peptide, PPeptide, MPeptide, Pyranose, DnaRna, Null
38 };
39 struct Side {
40 std::string comp;
41 std::string mod;
42 Group group;
43 bool matches_group(Group res) const {
44 return group != Group::Null &&
45 (res == group ||
46 (group == Group::Peptide && (int) res <= (int) Group::MPeptide));
47 }
48 int specificity() const {
49 if (!comp.empty())
50 return 3;
51 return group == Group::PPeptide || group == Group::MPeptide ? 1 : 0;
52 };
53 };
54 std::string id;
55 std::string name;
56 Side side1;
57 Side side2;
58 Restraints rt;

60 static Group read_group(const std::string& str) {
61 if (str.size() >= 4) {
62 const char* cstr = str.c_str();
63 if ((str[0] == '\'' || str[0] == '"') && str.size() >= 6)
64 ++cstr;
65 switch (ialpha4_id(cstr)) {
66 case ialpha4_id("pept"): return Group::Peptide;
67 case ialpha4_id("l-pe"): return Group::Peptide;
68 case ialpha4_id("p-pe"): return Group::PPeptide;
69 case ialpha4_id("m-pe"): return Group::MPeptide;
70 case ialpha4_id("pyra"): return Group::Pyranose;
71 case ialpha4_id("dna/"): return Group::DnaRna;
72 }
73 }
74 return Group::Null;
75 }
76
77 static const char* group_str(Group g) {
78 switch (g) {
79 case Group::Peptide: return "peptide";
80 case Group::PPeptide: return "P-peptide";
81 case Group::MPeptide: return "M-peptide";
82 case Group::Pyranose: return "pyranose";
83 case Group::DnaRna: return "DNA/RNA";
84 case Group::Null: return ".";
85 }
86 unreachable();
87 }
88
89 static Group group_from_residue_info(const ResidueInfo& ri) {
90 switch (ri.kind) {
91 case ResidueInfo::UNKNOWN: return Group::Null;
92 case ResidueInfo::AA: return Group::Peptide;
93 case ResidueInfo::AAD: return Group::Peptide;
94 case ResidueInfo::PAA: return Group::PPeptide;
95 case ResidueInfo::MAA: return Group::MPeptide;
96 case ResidueInfo::RNA: return Group::DnaRna;
97 case ResidueInfo::DNA: return Group::DnaRna;
98 case ResidueInfo::BUF: return Group::Null;
99 case ResidueInfo::HOH: return Group::Null;
100 case ResidueInfo::PYR: return Group::Pyranose;
101 case ResidueInfo::ELS: return Group::Null;
102 }
103 unreachable();
104 }

106 // If multiple ChemLinks match a bond, the scores can pick the best match.
107 int calculate_score(const Residue& res1, const Residue* res2,
108 char alt) const {
109 int link_score = side1.specificity() + side2.specificity();
110 // check chirality
111 for (const Restraints::Chirality& chirality : rt.chirs)
112 if (chirality.sign != ChiralityType::Both) {
113 const Atom* a1 = chirality.id_ctr.get_from(res1, res2, alt);
114 const Atom* a2 = chirality.id1.get_from(res1, res2, alt);
115 const Atom* a3 = chirality.id2.get_from(res1, res2, alt);
116 const Atom* a4 = chirality.id3.get_from(res1, res2, alt);
117 if (a1 && a2 && a3 && a4) {
118 double vol = calculate_chiral_volume(a1->pos, a2->pos,
119 a3->pos, a4->pos);
120 if (chirality.is_wrong(vol))
121 link_score -= 10;
122 }
123 }
124 // check fixed torsion angle (_chem_link_tor.period == 0)
125 for (const Restraints::Torsion& tor : rt.torsions)
126 if (tor.period == 0) {
127 const Atom* a1 = tor.id1.get_from(res1, res2, alt);
128 const Atom* a2 = tor.id2.get_from(res1, res2, alt);
129 const Atom* a3 = tor.id3.get_from(res1, res2, alt);
130 const Atom* a4 = tor.id4.get_from(res1, res2, alt);
131 double z = 10.;
132 if (a1 && a2 && a3 && a4)
133 z = angle_z(calculate_dihedral(a1->pos, a2->pos, a3->pos, a4->pos),
134 tor);
135 link_score -= (int) z;
136 }
137 return link_score;
138 }

162inline Restraints read_link_restraints(const cif::Block& block_) {
163 auto read_aid = [](cif::Table::Row& row, int n) {
164 return Restraints::AtomId{cif::as_int(row[n]), row.str(n+1)};
165 };
166 Restraints rt;
167 cif::Block& block = const_cast<cif::Block&>(block_);
168 for (auto row : block.find("_chem_link_bond.",
169 {"atom_1_comp_id", "atom_id_1",
170 "atom_2_comp_id", "atom_id_2",
171 "type",
172 "value_dist", "value_dist_esd"}))
173 rt.bonds.push_back({read_aid(row, 0), read_aid(row, 2),
174 bond_type_from_string(row[4]), false,
175 cif::as_number(row[5]), cif::as_number(row[6]),
176 NAN, NAN});
177 for (auto row : block.find("_chem_link_angle.",
178 {"atom_1_comp_id", "atom_id_1",
179 "atom_2_comp_id", "atom_id_2",
180 "atom_3_comp_id", "atom_id_3",
181 "value_angle", "value_angle_esd"}))
182 rt.angles.push_back({read_aid(row, 0), read_aid(row, 2), read_aid(row, 4),
183 cif::as_number(row[6]), cif::as_number(row[7])});
184 for (auto row : block.find("_chem_link_tor.",
185 {"id",
186 "atom_1_comp_id", "atom_id_1",
187 "atom_2_comp_id", "atom_id_2",
188 "atom_3_comp_id", "atom_id_3",
189 "atom_4_comp_id", "atom_id_4",
190 "value_angle", "value_angle_esd",
191 "period"}))
192 rt.torsions.push_back({row.str(0),
193 read_aid(row, 1), read_aid(row, 3),
194 read_aid(row, 5), read_aid(row, 7),
195 cif::as_number(row[9]), cif::as_number(row[10]),
196 cif::as_int(row[11])});
197 for (auto row : block.find("_chem_link_chir.",
198 {"atom_centre_comp_id", "atom_id_centre",
199 "atom_1_comp_id", "atom_id_1",
200 "atom_2_comp_id", "atom_id_2",
201 "atom_3_comp_id", "atom_id_3",
202 "volume_sign"}))
203 if (row[4][0] != 'c') // ignore crossN
204 rt.chirs.push_back({read_aid(row, 0), read_aid(row, 2),
205 read_aid(row, 4), read_aid(row, 6),
206 chirality_from_string(row[8])});
207 for (auto row : block.find("_chem_link_plane.",
208 {"plane_id", "atom_comp_id", "atom_id",
209 "dist_esd"})) {
210 Restraints::Plane& plane = rt.get_or_add_plane(row.str(0));
211 if (plane.esd == 0.0)
212 plane.esd = cif::as_number(row[3]);
213 plane.ids.push_back(read_aid(row, 1));
214 }
215 return rt;
216}

218inline ResidueInfo::Kind chemcomp_group_to_kind(const std::string& group) {
219 if (group.size() >= 3) {
220 const char* str = group.c_str();
221 if (group.size() > 4 && (*str == '"' || *str == '\''))
222 ++str;
223 switch (ialpha4_id(str)) {
224 case ialpha4_id("non-"): return ResidueInfo::ELS;
225 case ialpha4_id("pept"): return ResidueInfo::AA;
226 case ialpha4_id("l-pe"): return ResidueInfo::AA;
227 case ialpha4_id("p-pe"): return ResidueInfo::PAA;
228 case ialpha4_id("m-pe"): return ResidueInfo::MAA;
229 case ialpha4_id("dna"): return ResidueInfo::DNA;
230 case ialpha4_id("rna"): return ResidueInfo::RNA;
231 case ialpha4_id("pyra"): return ResidueInfo::PYR;
232 }
233 }
234 return ResidueInfo::UNKNOWN;
235}

284inline Restraints read_restraint_modifications(const cif::Block& block_) {
285 Restraints rt;
286 cif::Block& block = const_cast<cif::Block&>(block_);
287 for (auto row : block.find("_chem_mod_bond.",
288 {"function", "atom_id_1", "atom_id_2",
289 "new_type",
290 "new_value_dist", "new_value_dist_esd",
291 "?new_value_dist_nucleus", "?new_value_dist_nucleus_esd"}))
292 rt.bonds.push_back({{chem_mod_type(row[0]), row.str(1)}, {1, row.str(2)},
293 bond_type_from_string(row[3]), false,
294 cif::as_number(row[4]), cif::as_number(row[5]),
295 row.has(6) ? cif::as_number(row[6]) : NAN,
296 row.has(7) ? cif::as_number(row[7]) : NAN});
297 for (auto row : block.find("_chem_mod_angle.",
298 {"function", "atom_id_1",
299 "atom_id_2", "atom_id_3",
300 "new_value_angle", "new_value_angle_esd"}))
301 rt.angles.push_back({{chem_mod_type(row[0]), row.str(1)},
302 {1, row.str(2)}, {1, row.str(3)},
303 cif::as_number(row[4]), cif::as_number(row[5])});
304 for (auto row : block.find("_chem_mod_tor.",
305 {"function", "id", "atom_id_1",
306 "atom_id_2", "atom_id_3", "atom_id_4",
307 "new_value_angle", "new_value_angle_esd",
308 "new_period"}))
309 rt.torsions.push_back({row.str(1), {chem_mod_type(row[0]), row.str(2)},
310 {1, row.str(3)}, {1, row.str(4)}, {1, row.str(5)},
311 cif::as_number(row[6]), cif::as_number(row[7]),
312 cif::as_int(row[8])});
313 for (auto row : block.find("_chem_mod_chir.",
314 {"function", "atom_id_centre", "atom_id_1",
315 "atom_id_2", "atom_id_3",
316 "new_volume_sign"}))
317 rt.chirs.push_back({{1, row.str(1)}, {chem_mod_type(row[0]), row.str(2)},
318 {1, row.str(3)}, {1, row.str(4)},
319 chirality_from_string(row[5])});
320 for (auto row : block.find("_chem_mod_plane_atom.",
321 {"function", "plane_id", "atom_id" ,
322 "new_dist_esd"})) {
323 Restraints::Plane& plane = rt.get_or_add_plane(row.str(1));
324 if (plane.esd == 0.0)
325 plane.esd = cif::as_number(row[3]);
326 plane.ids.push_back({chem_mod_type(row[0]), row.str(2)});
327 }
328 return rt;
329}

358inline void ChemMod::apply_to(ChemComp& chemcomp) const {
359 // _chem_mod_atom
360 for (const AtomMod& mod : atom_mods) {
361 auto it = chemcomp.find_atom(mod.old_id);
362 switch (mod.func) {
363 case 'a':
364 if (!chemcomp.has_atom(mod.new_id))
365 chemcomp.atoms.push_back({mod.new_id, mod.el,
366 std::isnan(mod.charge) ? mod.charge : 0,
367 mod.chem_type});
368 break;
369 case 'd':
370 if (it != chemcomp.atoms.end()) {
371 chemcomp.atoms.erase(it);
372 // delete restraints containing mod.old_id
373 const std::string& old = mod.old_id;
374 vector_remove_if(chemcomp.rt.bonds, [&](const Restraints::Bond& b) {
375 return b.id1 == old || b.id2 == old;
376 });
377 vector_remove_if(chemcomp.rt.angles, [&](const Restraints::Angle& a) {
378 return a.id1 == old || a.id2 == old || a.id3 == old;
379 });
380 vector_remove_if(chemcomp.rt.torsions,
381 [&](const Restraints::Torsion& t) {
382 return t.id1 == old || t.id2 == old || t.id3 == old ||
383 t.id4 == old;
384 });
385 vector_remove_if(chemcomp.rt.chirs,
386 [&](const Restraints::Chirality& c) {
387 return c.id_ctr == old || c.id1 == old || c.id2 == old ||
388 c.id3 == old;
389 });
390 for (Restraints::Plane& plane : chemcomp.rt.planes)
391 vector_remove_if(plane.ids, [&](const Restraints::AtomId& a) {
392 return a.atom == old;
393 });
394 }
395 break;
396 case 'c':
397 if (it != chemcomp.atoms.end()) {
398 if (!mod.new_id.empty())
399 it->id = mod.new_id;
400 if (mod.el != El::X)
401 it->el = mod.el;
402 if (!std::isnan(mod.charge))
403 it->charge = mod.charge;
404 if (!mod.chem_type.empty())
405 it->chem_type = mod.chem_type;
406 }
407 break;
408 }
409 }
410
411 // _chem_mod_bond
412 for (const Restraints::Bond& mod : rt.bonds) {
413 auto it = chemcomp.rt.find_bond(mod.id1.atom, mod.id2.atom);
414 switch (mod.id1.comp) {
415 case 'a':
416 if (it == chemcomp.rt.bonds.end()) {
417 chemcomp.rt.bonds.push_back(mod);
418 // id1.comp was temporarily set to 'a', set it back to 1
419 chemcomp.rt.bonds.back().id1.comp = 1;
420 }
421 break;
422 case 'd':
423 if (it != chemcomp.rt.bonds.end())
424 chemcomp.rt.bonds.erase(it);
425 break;
426 case 'c':
427 if (it != chemcomp.rt.bonds.end()) {
428 if (mod.type != BondType::Unspec)
429 it->type = mod.type;
430 if (!std::isnan(mod.value))
431 it->value = mod.value;
432 if (!std::isnan(mod.esd))
433 it->esd = mod.esd;
434 if (!std::isnan(mod.value_nucleus))
435 it->value_nucleus = mod.value_nucleus;
436 if (!std::isnan(mod.esd_nucleus))
437 it->esd_nucleus = mod.esd_nucleus;
438 }
439 break;
440 }
441 }
442
443 // _chem_mod_angle
444 for (const Restraints::Angle& mod : rt.angles) {
445 auto it = chemcomp.rt.find_angle(mod.id1.atom, mod.id2.atom, mod.id3.atom);
446 switch (mod.id1.comp) {
447 case 'a':
448 if (it == chemcomp.rt.angles.end()) {
449 chemcomp.rt.angles.push_back(mod);
450 chemcomp.rt.angles.back().id1.comp = 1;
451 }
452 break;
453 case 'd':
454 if (it != chemcomp.rt.angles.end())
455 chemcomp.rt.angles.erase(it);
456 break;
457 case 'c':
458 if (it != chemcomp.rt.angles.end()) {
459 if (!std::isnan(mod.value))
460 it->value = mod.value;
461 if (!std::isnan(mod.esd))
462 it->esd = mod.esd;
463 }
464 break;
465 }
466 }
467
468 // _chem_mod_tor
469 for (const Restraints::Torsion& mod : rt.torsions) {
470 auto it = chemcomp.rt.find_torsion(mod.id1.atom, mod.id2.atom,
471 mod.id3.atom, mod.id4.atom);
472 switch (mod.id1.comp) {
473 case 'a':
474 if (it == chemcomp.rt.torsions.end()) {
475 chemcomp.rt.torsions.push_back(mod);
476 chemcomp.rt.torsions.back().id1.comp = 1;
477 }
478 break;
479 case 'd':
480 if (it != chemcomp.rt.torsions.end())
481 chemcomp.rt.torsions.erase(it);
482 break;
483 case 'c':
484 if (it != chemcomp.rt.torsions.end()) {
485 if (!mod.label.empty())
486 it->label = mod.label;
487 if (!std::isnan(mod.value))
488 it->value = mod.value;
489 if (!std::isnan(mod.esd))
490 it->esd = mod.esd;
491 if (mod.period != -1)
492 it->period = mod.period;
493 }
494 break;
495 }
496 }
497
498 // _chem_mod_chir
499 for (const Restraints::Chirality& mod : rt.chirs) {
500 auto it = chemcomp.rt.find_chir(mod.id_ctr.atom, mod.id1.atom,
501 mod.id2.atom, mod.id3.atom);
502 switch (mod.id1.comp) {
503 case 'a':
504 if (it == chemcomp.rt.chirs.end()) {
505 chemcomp.rt.chirs.push_back(mod);
506 chemcomp.rt.chirs.back().id1.comp = 1;
507 }
508 break;
509 case 'd':
510 if (it != chemcomp.rt.chirs.end())
511 chemcomp.rt.chirs.erase(it);
512 break;
513 case 'c':
514 if (it != chemcomp.rt.chirs.end())
515 it->sign = mod.sign;
516 break;
517 }
518 }
519
520 // _chem_mod_plane_atom
521 for (const Restraints::Plane& mod : rt.planes)
522 for (const Restraints::AtomId& atom_id : mod.ids) {
523 if (atom_id.comp == 'a') {
524 Restraints::Plane& plane = chemcomp.rt.get_or_add_plane(mod.label);
525 if (plane.esd == 0.0 && !std::isnan(mod.esd))
526 plane.esd = mod.esd;
527 auto it = std::find(plane.ids.begin(), plane.ids.end(), atom_id.atom);
528 if (it == plane.ids.end())
529 plane.ids.push_back({1, atom_id.atom});
530 } else if (atom_id.comp == 'd') {
531 auto it = chemcomp.rt.get_plane(mod.label);
532 if (it != chemcomp.rt.planes.end()) {
533 auto item = std::find(it->ids.begin(), it->ids.end(), atom_id.atom);
534 if (item != it->ids.end())
535 it->ids.erase(item);
536 }
537 }
538 }
539}

561 // Returns the most specific link and a flag that is true
562 // if the order is comp2-comp1 in the link definition.
563 // We don't check chirality here (cf. calculate_score).
564 std::pair<const ChemLink*, bool>
565 match_link(const std::string& comp1, const std::string& atom1,
566 const std::string& comp2, const std::string& atom2) const {
567 const ChemLink* best_link = nullptr;
568 int best_score = -1;
569 bool inverted = false;
570 for (auto& ml : links) {
571 const ChemLink& link = ml.second;
572 if (link.rt.bonds.empty())
573 continue;
574 // for now we don't have link definitions with >1 bonds
575 const Restraints::Bond& bond = link.rt.bonds[0];
576 if (bond.id1.atom == atom1 && bond.id2.atom == atom2 &&
577 link_side_matches_residue(link.side1, comp1) &&
578 link_side_matches_residue(link.side2, comp2)) {
579 int score = link.side1.specificity() + link.side2.specificity();
580 if (score > best_score) {
581 best_link = &link;
582 best_score = score;
583 inverted = false;
584 }
585 }
586 if (bond.id1.atom == atom2 && bond.id2.atom == atom1 &&
587 link_side_matches_residue(link.side1, comp2) &&
588 link_side_matches_residue(link.side2, comp1)) {
589 int score = link.side1.specificity() + link.side2.specificity();
590 if (score > best_score) {
591 best_link = &link;
592 best_score = score;
593 inverted = true;
594 }
595 }
596 }
597 return {best_link, inverted};
598 }

638 static std::string relative_monomer_path(const std::string& code) {
639 std::string path(1, std::tolower(code[0]));
640 path += '/'; // works also on Windows
641 path += code;
642 // On Windows several names are reserved (CON, PRN, AUX, ...), see
643 // https://docs.microsoft.com/en-us/windows/win32/fileio/naming-a-file
644 // The workaround in CCP4 monomer library is to use CON_CON.cif, etc.
645 if (code.size() == 3)
646 switch (ialpha3_id(code.c_str())) {
647 case ialpha3_id("AUX"):
648 case ialpha3_id("COM"):
649 case ialpha3_id("CON"):
650 case ialpha3_id("LPT"):
651 case ialpha3_id("PRN"):
652 path += '_';
653 path += code;
654 }
655 path += ".cif";
656 return path;
657 }

699struct BondIndex {
700 const Model& model;
701
702 struct AtomImage {
703 int atom_serial;
704 bool same_image;
705 bool operator==(const AtomImage& o) const {
706 return atom_serial == o.atom_serial && same_image == o.same_image;
707 }
708 };
709 std::map<int, std::vector<AtomImage>> index;
710
711 BondIndex(const Model& model_) : model(model_) {
712 for (const_CRA cra : model.all())
713 if (!index.emplace(cra.atom->serial, std::vector<AtomImage>()).second)
714 fail("duplicated serial numbers");
715 }
716
717 void add_oneway_link(const Atom& a, const Atom& b, bool same_image) {
718 std::vector<AtomImage>& list_a = index.at(a.serial);
719 AtomImage ai{b.serial, same_image};
720 if (!in_vector(ai, list_a))
721 list_a.push_back(ai);
722 }
723
724 void add_link(const Atom& a, const Atom& b, bool same_image) {
725 add_oneway_link(a, b, same_image);
726 add_oneway_link(b, a, same_image);
727 }
728
729 void add_monomer_bonds(MonLib& monlib) {
730 for (const Chain& chain : model.chains)
731 for (const Residue& res : chain.residues) {
732 std::string altlocs;
733 add_distinct_altlocs(res, altlocs);
734 if (altlocs.empty())
735 altlocs += '*';
736 auto monomer = monlib.monomers.find(res.name);
737 if (monomer == monlib.monomers.end())
738 fail("Monomer description not found: " + res.name);
739 for (const Restraints::Bond& bond : monomer->second.rt.bonds)
740 for (char alt : altlocs)
741 if (const Atom* at1 = res.find_atom(bond.id1.atom, alt))
742 if (const Atom* at2 = res.find_atom(bond.id2.atom, alt)) {
743 add_link(*at1, *at2, true);
744 if (!at1->altloc && !at2->altloc)
745 break;
746 }
747 }
748 }
749
750 bool are_linked(const Atom& a, const Atom& b, bool same_image) const {
751 return in_vector({b.serial, same_image}, index.at(a.serial));
752 }
753
754 int graph_distance(const Atom& a, const Atom& b, bool same_image,
755 int max_distance=4) const {
756 std::vector<AtomImage> neighbors(1, {a.serial, true});
757 for (int distance = 1; distance <= max_distance; ++distance) {
758 for (size_t n = neighbors.size(); n--; ) {
759 for (AtomImage ai : index.at(neighbors[n].atom_serial)) {
760 if (!neighbors[n].same_image)
761 ai.same_image = !ai.same_image;
762 if (ai.atom_serial == b.serial && ai.same_image == same_image)
763 return distance;
764 if (!in_vector(ai, neighbors))
765 neighbors.push_back(ai);
766 }
767 }
768 }
769 return max_distance + 1;
770 }
771};
