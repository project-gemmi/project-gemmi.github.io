129 // Returns the most specific link and a flag that is true
130 // if the order is comp2-comp1 in the link definition.
131 std::tuple<const ChemLink*, bool, const ChemComp::Aliasing*, const ChemComp::Aliasing*>
132 match_link(const Residue& res1, const std::string& atom1, char alt1,
133 const Residue& res2, const std::string& atom2, char alt2,
134 double min_bond_sq=0) const {
135 const ChemLink* best_link = nullptr;
136 bool inverted = false;
137 const ChemComp::Aliasing* aliasing1 = nullptr;
138 const ChemComp::Aliasing* aliasing2 = nullptr;
139 const ChemComp::Aliasing* aliasing1_final = nullptr;
140 const ChemComp::Aliasing* aliasing2_final = nullptr;
141 int best_score = -1000;
142 for (const auto& ml : links) {
143 const ChemLink& link = ml.second;
144 if (link.rt.bonds.empty() || starts_with(link.name, "auto-"))
145 continue;
146 // for now we don't have link definitions with >1 bonds
147 const Restraints::Bond& bond = link.rt.bonds[0];
148 if (sq(bond.value) < min_bond_sq)
149 continue;
150 if (link_side_matches_residue(link.side1, res1.name, &aliasing1) &&
151 link_side_matches_residue(link.side2, res2.name, &aliasing2) &&
152 atom_match_with_alias(bond.id1.atom, atom1, aliasing1) &&
153 atom_match_with_alias(bond.id2.atom, atom2, aliasing2)) {
154 int score = link.calculate_score(res1, &res2, alt1, alt2, aliasing1, aliasing2);
155 if (score > best_score) {
156 best_link = &link;
157 best_score = score;
158 aliasing1_final = aliasing1;
159 aliasing2_final = aliasing2;
160 inverted = false;
161 }
162 }
163 if (link_side_matches_residue(link.side1, res2.name, &aliasing2) &&
164 link_side_matches_residue(link.side2, res1.name, &aliasing1) &&
165 atom_match_with_alias(bond.id1.atom, atom2, aliasing2) &&
166 atom_match_with_alias(bond.id2.atom, atom1, aliasing1)) {
167 // NOLINTNEXTLINE(readability-suspicious-call-argument)
168 int score = link.calculate_score(res2, &res1, alt2, alt1, aliasing2, aliasing1);
169 if (score > best_score) {
170 best_link = &link;
171 best_score = score;
172 aliasing1_final = aliasing1;
173 aliasing2_final = aliasing2;
174 inverted = true;
175 }
176 }
177 }
178 return std::make_tuple(best_link, inverted, aliasing1_final, aliasing2_final);
179 }

181 void add_monomer_if_present(const cif::Block& block) {
182 if (block.has_tag("_chem_comp_atom.atom_id")) {
183 ChemComp cc = make_chemcomp_from_block(block);
184 if (cc.group == ChemComp::Group::Null) {
185 auto it = cc_groups.find(cc.name);
186 if (it != cc_groups.end())
187 cc.group = it->second;
188 }
189 std::string name = cc.name;
190 monomers.emplace(name, std::move(cc));
191 }

194 bool link_side_matches_residue(const ChemLink::Side& side,
195 const std::string& res_name,
196 ChemComp::Aliasing const** aliasing) const {
197 assert(aliasing);
198 *aliasing = nullptr;
199 if (!side.comp.empty())
200 return side.comp == res_name;
201 auto it = monomers.find(res_name);
202 if (it != monomers.end()) {
203 if (side.matches_group(it->second.group))
204 return true;
205 for (const ChemComp::Aliasing& a : it->second.aliases)
206 if (side.matches_group(a.group)) {
207 *aliasing = &a;
208 return true;
209 }
210 }
211 return false;
212 }
