109 void read(const cif::Document& doc) {
110 cif::Block& block = const_cast<cif::Block&>(doc.blocks[0]);
111 for (const auto& row : block.find("_lib_atom.",
112 {"type", "hb_type", "vdw_radius", "vdwh_radius",
113 "ion_radius", "element", "valency", "sp"}))
114 atoms.emplace(row[0], Atom{Element(row[5]), row[1][0], cif::as_number(row[2]),

116 cif::as_int(row[6], -1), cif::as_int(row[7], -1)});
117 for (const auto& row : block.find("_lib_bond.",
118 {"atom_type_1", "atom_type_2", "type", "length", "value_esd"}))
119 bonds.emplace(row.str(0), Bond{row.str(1), bond_type_from_string(row[2]),
120 cif::as_number(row[3]), cif::as_number(row[4])});
121 }
122 std::map<std::string, Atom> atoms; // type->Atom
123 std::multimap<std::string, Bond> bonds; // atom_type_1->Bond

143 // Returns the most specific link and a flag that is true
144 // if the order is comp2-comp1 in the link definition.
145 std::tuple<const ChemLink*, bool, const ChemComp::Aliasing*, const ChemComp::Aliasing*>
146 match_link(const Residue& res1, const std::string& atom1, char alt1,
147 const Residue& res2, const std::string& atom2, char alt2,
148 double min_bond_sq=0) const {
149 const ChemLink* best_link = nullptr;
150 bool inverted = false;
151 const ChemComp::Aliasing* aliasing1 = nullptr;
152 const ChemComp::Aliasing* aliasing2 = nullptr;
153 const ChemComp::Aliasing* aliasing1_final = nullptr;
154 const ChemComp::Aliasing* aliasing2_final = nullptr;
155 int best_score = -1000;
156 for (const auto& ml : links) {
157 const ChemLink& link = ml.second;
158 if (link.rt.bonds.empty() || starts_with(link.name, "auto-"))
159 continue;
160 // for now we don't have link definitions with >1 bonds
161 const Restraints::Bond& bond = link.rt.bonds[0];
162 if (sq(bond.value) < min_bond_sq)
163 continue;
164 if (link_side_matches_residue(link.side1, res1.name, &aliasing1) &&
165 link_side_matches_residue(link.side2, res2.name, &aliasing2) &&
166 atom_match_with_alias(bond.id1.atom, atom1, aliasing1) &&
167 atom_match_with_alias(bond.id2.atom, atom2, aliasing2)) {
168 int score = link.calculate_score(res1, &res2, alt1, alt2, aliasing1, aliasing2);
169 if (score > best_score) {
170 best_link = &link;
171 best_score = score;
172 aliasing1_final = aliasing1;
173 aliasing2_final = aliasing2;
174 inverted = false;
175 }
176 }
177 if (link_side_matches_residue(link.side1, res2.name, &aliasing2) &&
178 link_side_matches_residue(link.side2, res1.name, &aliasing1) &&
179 atom_match_with_alias(bond.id1.atom, atom2, aliasing2) &&
180 atom_match_with_alias(bond.id2.atom, atom1, aliasing1)) {
181 // NOLINTNEXTLINE(readability-suspicious-call-argument)
182 int score = link.calculate_score(res2, &res1, alt2, alt1, aliasing2, aliasing1);
183 if (score > best_score) {
184 best_link = &link;
185 best_score = score;
186 aliasing1_final = aliasing1;
187 aliasing2_final = aliasing2;
188 inverted = true;
189 }
190 }
191 }
192 return std::make_tuple(best_link, inverted, aliasing1_final, aliasing2_final);
193 }

208 bool link_side_matches_residue(const ChemLink::Side& side,
209 const std::string& res_name,
210 ChemComp::Aliasing const** aliasing) const {
211 assert(aliasing);
212 *aliasing = nullptr;
213 if (!side.comp.empty())
214 return side.comp == res_name;
215 auto it = monomers.find(res_name);
216 if (it != monomers.end()) {
217 if (side.matches_group(it->second.group))
218 return true;
219 for (const ChemComp::Aliasing& a : it->second.aliases)
220 if (side.matches_group(a.group)) {
221 *aliasing = &a;
222 return true;
223 }
224 }
225 return false;
226 }

233 static std::string relative_monomer_path(const std::string& code) {
234 std::string path;
235 if (!code.empty()) {
236 path += lower(code[0]);
237 path += '/'; // works also on Windows
238 path += code;
239 // On Windows several names are reserved (CON, PRN, AUX, ...), see
240 // https://docs.microsoft.com/en-us/windows/win32/fileio/naming-a-file
241 // The workaround in CCP4 monomer library is to use CON_CON.cif, etc.
242 if (code.size() == 3)
243 switch (ialpha3_id(code.c_str())) {
244 case ialpha3_id("AUX"):
245 case ialpha3_id("COM"):
246 case ialpha3_id("CON"):
247 case ialpha3_id("LPT"):
248 case ialpha3_id("PRN"):
249 path += '_';
250 path += code;
251 }
252 path += ".cif";
253 }
254 return path;
255 }

271 bool read_monomer_lib(const std::string& monomer_dir_,
272 const std::vector<std::string>& resnames,
273 read_cif_func read_cif,
274 std::string* error=nullptr) {
275 if (monomer_dir_.empty())
276 fail("read_monomer_lib: monomer_dir not specified.");
277 set_monomer_dir(monomer_dir_);
278
279 // Only recent versions of CCP4 Monomer Library have links_and_mods.cif
280 try {
281 read_monomer_cif(monomer_dir + "links_and_mods.cif", read_cif);
282 } catch (std::system_error&) {
283 read_monomer_cif(monomer_dir + "list/mon_lib_list.cif", read_cif);
284 }
285 ener_lib.read((*read_cif)(monomer_dir + "ener_lib.cif"));
286
287 bool ok = true;
288 for (const std::string& name : resnames) {
289 if (monomers.find(name) != monomers.end())
290 continue;
291 try {
292 const cif::Document& doc = (*read_cif)(path(name));
293 read_monomer_doc(doc);
294 } catch (std::system_error& err) {
295 if (error) {
296 if (err.code().value() == ENOENT)
297 cat_to(*error, "Monomer not in the library: ", name, ".\n");
298 else
299 cat_to(*error, "Failed to read ", name, ": ", err.what(), ".\n");
300 }
301 ok = false;
302 } catch (std::runtime_error& err) {
303 if (error)
304 cat_to(*error, "Failed to read ", name, ": ", err.what(), ".\n");
305 ok = false;
306 }
307 }
308 return ok;
309 }
