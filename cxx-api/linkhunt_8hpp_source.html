34 void index_chem_links(const MonLib& monlib) {
35 for (const auto& iter : monlib.links) {
36 const ChemLink& link = iter.second;
37 if (link.rt.bonds.empty())
38 continue;
39 if (link.rt.bonds.size() > 1)
40 fprintf(stderr, "Note: considering only the first bond in %s\n",
41 link.id.c_str());
42 if (link.side1.comp.empty() && link.side2.comp.empty())
43 if (link.side1.group == ChemLink::Group::Null ||
44 link.side2.group == ChemLink::Group::Null ||
45 link.id == "SS")
46 continue;
47 const Restraints::Bond& bond = link.rt.bonds[0];
48 if (bond.value > global_max_dist)
49 global_max_dist = bond.value;
50 links.emplace(bond.lexicographic_str(), &link);
51 }
52 for (const auto& ri : monlib.residue_infos)
53 res_group.emplace(ri.first, ChemLink::group_from_residue_info(ri.second));
54 }
55
56 bool match_link_side(const ChemLink::Side& side,
57 const std::string& resname) const {
58 if (!side.comp.empty())
59 return side.comp == resname;
60 if (side.group == ChemLink::Group::Null)
61 return false;
62 auto iter = res_group.find(resname);
63 return iter != res_group.end() && side.matches_group(iter->second);
64 }

66 std::vector<Match> find_possible_links(Structure& st,
67 double bond_margin,
68 double radius_margin,
69 ContactSearch::Ignore ignore) {
70 std::vector<Match> results;
71 Model& model = st.first_model();
72 double search_radius = std::max(global_max_dist * bond_margin,
73 /*max r1+r2 ~=*/3.0 * radius_margin);
74 NeighborSearch ns(model, st.cell, std::max(5.0, search_radius));
75 ns.populate();
76
77 ContactSearch contacts((float) search_radius);
78 contacts.ignore = ignore;
79 contacts.for_each_contact(ns, [&](const CRA& cra1, const CRA& cra2,
80 int image_idx, float dist_sq) {
81 Match match;
82
83 // search for a match in chem_links
84 if (bond_margin > 0) {
85 auto range = links.equal_range(Restraints::lexicographic_str(
86 cra1.atom->name, cra2.atom->name));
87 for (auto iter = range.first; iter != range.second; ++iter) {
88 const ChemLink& link = *iter->second;
89 const Restraints::Bond& bond = link.rt.bonds[0];
90 if (dist_sq > sq(bond.value * bond_margin))
91 continue;
92 bool order1;
93 if (bond.id1.atom == cra1.atom->name &&
94 match_link_side(link.side1, cra1.residue->name) &&
95 match_link_side(link.side2, cra2.residue->name))
96 order1 = true;
97 else if (bond.id2.atom == cra1.atom->name &&
98 match_link_side(link.side2, cra1.residue->name) &&
99 match_link_side(link.side1, cra2.residue->name))
100 order1 = false;
101 else
102 continue;
103 int link_score = link.calculate_score(
104 order1 ? *cra1.residue : *cra2.residue,
105 order1 ? cra2.residue : cra1.residue,
106 cra1.atom->altloc_or(cra2.atom->altloc));
107 match.chem_link_count++;
108 if (link_score > match.score) {
109 match.chem_link = &link;
110 match.score = link_score;
111 if (order1) {
112 match.cra1 = cra1;
113 match.cra2 = cra2;
114 } else {
115 match.cra1 = cra2;
116 match.cra2 = cra1;
117 }
118 }
119 }
120 }
121
122 // potential other links according to covalent radii
123 if (!match.chem_link) {
124 float r1 = cra1.atom->element.covalent_r();
125 float r2 = cra2.atom->element.covalent_r();
126 if (dist_sq > sq((r1 + r2) * radius_margin))
127 return;
128 match.cra1 = cra1;
129 match.cra2 = cra2;
130 }
131
132 // finalize
133 match.same_image = !image_idx;
134 match.bond_length = std::sqrt(dist_sq);
135 results.push_back(match);
136 });
137
138 // add references to st.connections
139 for (Match& match : results)
140 match.conn = st.find_connection_by_cra(match.cra1, match.cra2);
141
142 return results;
143 }
