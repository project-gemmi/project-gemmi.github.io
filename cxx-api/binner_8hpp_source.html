23 int setup_from_1_d2(int nbins, Method method, std::vector<double>&& inv_d2,
24 const UnitCell* cell_) {
25 if (nbins < 1)
26 fail("Binner: nbins argument must be positive");
27 if (inv_d2.empty())
28 fail("Binner: no data");
29 if (cell_)
30 cell = *cell_;
31 if (!cell.is_crystal())
32 fail("Binner: unknown unit cell");
33 limits.resize(nbins);
34 if (method == Method::EqualCount) {
35 std::sort(inv_d2.begin(), inv_d2.end());
36 min_1_d2 = inv_d2.front();
37 max_1_d2 = inv_d2.back();
38 } else {
39 min_1_d2 = max_1_d2 = inv_d2.front();
40 for (double x : inv_d2) {
41 if (x < min_1_d2)
42 min_1_d2 = x;
43 if (x > max_1_d2)
44 max_1_d2 = x;
45 }
46 }
47 switch (method) {
48 case Method::EqualCount: {
49 double avg_count = double(inv_d2.size()) / nbins;
50 for (int i = 1; i < nbins; ++i)
51 limits[i-1] = inv_d2[int(avg_count * i)];
52 break;
53 }
54 case Method::Dstar2: {
55 double step = (max_1_d2 - min_1_d2) / nbins;
56 for (int i = 1; i < nbins; ++i)
57 limits[i-1] = min_1_d2 + i * step;
58 break;
59 }
60 case Method::Dstar: {
61 double min_1_d = std::sqrt(min_1_d2);
62 double max_1_d = std::sqrt(max_1_d2);
63 double step = (max_1_d - min_1_d) / nbins;
64 for (int i = 1; i < nbins; ++i)
65 limits[i-1] = sq(min_1_d + i * step);
66 break;
67 }
68 case Method::Dstar3: {
69 double min_1_d3 = min_1_d2 * std::sqrt(min_1_d2);
70 double max_1_d3 = max_1_d2 * std::sqrt(max_1_d2);
71 double step = (max_1_d3 - min_1_d3) / nbins;
72 for (int i = 1; i < nbins; ++i)
73 limits[i-1] = sq(std::cbrt(min_1_d3 + i * step));
74 break;
75 }
76 }
77 limits.back() = std::numeric_limits<double>::infinity();
78 return (int) limits.size();
79 }

82 int setup(int nbins, Method method, const DataProxy& proxy,
83 const UnitCell* cell_=nullptr) {
84 cell = cell_ ? *cell_ : proxy.unit_cell();
85 std::vector<double> inv_d2(proxy.size() / proxy.stride());

109 // We assume that bins are seeked mostly for sorted reflections,
110 // so it's usually either the same bin as previously, or the next one.
111 int get_bin_from_1_d2_hinted(double inv_d2, int& hint) const {
112 if (inv_d2 <= limits[hint]) {
113 while (hint != 0 && limits[hint-1] > inv_d2)
114 --hint;
115 } else {
116 // limits.back() is +inf, so we won't overrun
117 while (limits[hint] < inv_d2)
118 ++hint;
119 }
120 return hint;
121 }

154inline Correlation combine_two_correlations(const Correlation& a, const Correlation& b) {
155 Correlation r;
156 r.n = a.n + b.n;
157 r.mean_x = (a.n * a.mean_x + b.n * b.mean_x) / r.n;
158 r.mean_y = (a.n * a.mean_y + b.n * b.mean_y) / r.n;
159 r.sum_xx = a.sum_xx + a.n * sq(a.mean_x - r.mean_x)
160 + b.sum_xx + b.n * sq(b.mean_x - r.mean_x);
161 r.sum_yy = a.sum_yy + a.n * sq(a.mean_y - r.mean_y)
162 + b.sum_yy + b.n * sq(b.mean_y - r.mean_y);
163 r.sum_xy = a.sum_xy + a.n * (a.mean_x - r.mean_x) * (a.mean_y - r.mean_y)
164 + b.sum_xy + b.n * (b.mean_x - r.mean_x) * (b.mean_y - r.mean_y);
165 return r;
166}

168inline Correlation combine_correlations(const std::vector<Correlation>& cors) {
169 Correlation result;
170 for (const Correlation& cor : cors)
171 result = combine_two_correlations(result, cor);
172 return result;
173}

179 HklMatch(const Miller* hkl, size_t hkl_size_, const Miller* ref, size_t ref_size)
180 : pos(ref_size, -1), hkl_size(hkl_size_) {
181 // Usually, both datasets are sorted. This make things faster.
182 if (std::is_sorted(hkl, hkl + hkl_size) &&
183 std::is_sorted(ref, ref + ref_size)) {
184 // cf. for_matching_reflections()
185 auto a = hkl;
186 auto b = ref;
187 while (a != hkl + hkl_size && b != ref + ref_size) {
188 if (*a == *b)
189 pos[b++ - ref] = static_cast<int>(a++ - hkl);
190 else if (*a < *b)
191 ++a;
192 else
193 ++b;
194 }
195 } else {
196 std::unordered_map<Miller, int, MillerHash> hkl_index;
197 for (int i = 0; i != (int)hkl_size; ++i)
198 hkl_index.emplace(hkl[i], i);
199 for (size_t i = 0; i != ref_size; ++i) {
200 auto it = hkl_index.find(ref[i]);
201 if (it != hkl_index.end())
202 pos[i] = it->second;
203 }
204 }
205 }

210 template <typename T> std::vector<T> aligned(const std::vector<T>& v, T nan) {
