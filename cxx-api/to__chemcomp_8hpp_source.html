15inline void add_chemcomp_to_block(const ChemComp& cc, cif::Block& block) {
16 {
17 cif::Table tab = block.find_or_add("_chem_comp_atom.",
18 {"comp_id", "atom_id", "type_symbol", "type_energy", "charge"});
19 for (const ChemComp::Atom& a : cc.atoms)
20 tab.append_row({cc.name, a.id, a.el.name(), cif::quote(a.chem_type),
21 std::to_string(iround(a.charge))});
22 }
23 {
24 cif::Table tab = block.find_or_add("_chem_comp_bond.",
25 {"comp_id", "atom_id_1", "atom_id_2", "type", "aromatic",
26 "value_dist", "value_dist_esd", "value_dist_nucleus", "value_dist_nucleus_esd"});
27 for (const Restraints::Bond& a : cc.rt.bonds)
28 tab.append_row({cc.name, a.id1.atom, a.id2.atom, bond_type_to_string(a.type),
29 std::string(1, a.aromatic ? 'y' : 'n'),
30 to_str(a.value), to_str(a.esd),
31 to_str(a.value_nucleus), to_str(a.esd_nucleus)});
32 }
33 {
34 cif::Table tab = block.find_or_add("_chem_comp_angle.",
35 {"comp_id", "atom_id_1", "atom_id_2", "atom_id_3",
36 "value_angle", "value_angle_esd"});
37 for (const Restraints::Angle& a : cc.rt.angles)
38 tab.append_row({cc.name, a.id1.atom, a.id2.atom, a.id3.atom,
39 to_str(a.value), to_str(a.esd)});
40 }
41 {
42 cif::Table tab = block.find_or_add("_chem_comp_tor.",
43 {"comp_id", "id", "atom_id_1", "atom_id_2", "atom_id_3", "atom_id_4",
44 "value_angle", "value_angle_esd", "period"});
45 for (const Restraints::Torsion& a : cc.rt.torsions)
46 tab.append_row({cc.name, a.label, a.id1.atom, a.id2.atom, a.id3.atom, a.id4.atom,
47 to_str(a.value), to_str(a.esd), std::to_string(a.period)});
48 }
49 {
50 cif::Table tab = block.find_or_add("_chem_comp_chir.",
51 {"comp_id", "id", "atom_id_centre", "atom_id_1", "atom_id_2", "atom_id_3",
52 "volume_sign"});
53 for (const Restraints::Chirality& a : cc.rt.chirs) {
54 std::string label = "chir_" + std::to_string(tab.length() + 1);
55 tab.append_row({cc.name, label, a.id_ctr.atom, a.id1.atom, a.id2.atom, a.id3.atom,
56 chirality_to_string(a.sign)});
57 }
58 }
59 {
60 cif::Table tab = block.find_or_add("_chem_comp_plane_atom.",
61 {"comp_id", "plane_id", "atom_id", "dist_esd"});
62 for (const Restraints::Plane& p : cc.rt.planes)
63 for (const Restraints::AtomId& atom_id : p.ids)
64 tab.append_row({cc.name, p.label, atom_id.atom, to_str(p.esd)});
65 }
66}

68inline ChemComp make_chemcomp_with_restraints(const Residue& res) {
69 ChemComp cc;
70 cc.name = res.name;
71 // add atoms
72 cc.atoms.reserve(res.atoms.size());
73 for (const Atom& a : res.atoms) {
74 const std::string& chem_type = a.name;
75 cc.atoms.push_back(ChemComp::Atom{a.name, a.element, float(a.charge), chem_type});
76 }
77 // prepare pairs of atoms
78 struct Pair {
79 size_t n1, n2;
80 double dist;
81 };
82 std::vector<Pair> pairs;
83 for (size_t i = 0; i != res.atoms.size(); ++i) {
84 double r1 = res.atoms[i].element.covalent_r();
85 for (size_t j = i+1; j != res.atoms.size(); ++j) {
86 double d2 = res.atoms[i].pos.dist_sq(res.atoms[j].pos);
87 if (d2 < sq(1.3 * (r1 + res.atoms[j].element.covalent_r())))
88 pairs.push_back(Pair{i, j, std::sqrt(d2)});
89 }
90 }
91 // add bonds
92 for (const Pair& p : pairs) {
93 Restraints::Bond bond;
94 bond.id1 = Restraints::AtomId{1, res.atoms[p.n1].name};
95 bond.id2 = Restraints::AtomId{1, res.atoms[p.n2].name};
96 bond.type = BondType::Unspec;
97 bond.aromatic = false;
98 double rounded_dist = 0.001 * std::round(1000 * p.dist);
99 bond.value = bond.value_nucleus = rounded_dist;
100 bond.esd = bond.esd_nucleus = 0.02;
101 cc.rt.bonds.push_back(bond);
102 }
103 // add angles
104 struct Triple {
105 size_t n1, n2, n3;
106 };
107 std::vector<Triple> triples;
108 for (size_t i = 0; i != pairs.size(); ++i)
109 for (size_t j = i+1; j != pairs.size(); ++j) {
110 if (pairs[i].n1 == pairs[j].n1)
111 triples.push_back(Triple{pairs[i].n2, pairs[i].n1, pairs[j].n2});
112 else if (pairs[i].n1 == pairs[j].n2)
113 triples.push_back(Triple{pairs[i].n2, pairs[i].n1, pairs[j].n1});
114 else if (pairs[i].n2 == pairs[j].n1)
115 triples.push_back(Triple{pairs[i].n1, pairs[i].n2, pairs[j].n2});
116 else if (pairs[i].n2 == pairs[j].n2)
117 triples.push_back(Triple{pairs[i].n1, pairs[i].n2, pairs[j].n1});
118 }
119 for (const Triple& triple : triples) {
120 Restraints::Angle angle;
121 angle.id1 = Restraints::AtomId{1, res.atoms[triple.n1].name};
122 angle.id2 = Restraints::AtomId{1, res.atoms[triple.n2].name};
123 angle.id3 = Restraints::AtomId{1, res.atoms[triple.n3].name};
124 double angle_rad = calculate_angle(res.atoms[triple.n1].pos,
125 res.atoms[triple.n2].pos,
126 res.atoms[triple.n3].pos);
127 angle.value = 0.01 * std::round(100 * deg(angle_rad));
128 angle.esd = 3.0;
129 cc.rt.angles.push_back(angle);
130 }
131 return cc;
132}
