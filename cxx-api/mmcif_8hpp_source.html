36inline std::unordered_map<std::string, SMat33<float>>
37get_anisotropic_u(cif::Block& block) {
38 cif::Table aniso_tab = block.find("_atom_site_anisotrop.",
39 {"id", "U[1][1]", "U[2][2]", "U[3][3]",
40 "U[1][2]", "U[1][3]", "U[2][3]"});
41 std::unordered_map<std::string, SMat33<float>> aniso_map;
42 for (auto ani : aniso_tab)
43 aniso_map.emplace(ani[0], SMat33<float>{
44 (float) cif::as_number(ani[1]),
45 (float) cif::as_number(ani[2]),
46 (float) cif::as_number(ani[3]),
47 (float) cif::as_number(ani[4]),
48 (float) cif::as_number(ani[5]),
49 (float) cif::as_number(ani[6])});
50 return aniso_map;
51}

60inline Transform get_transform_matrix(const cif::Table::Row& r) {
61 Transform t;
62 for (int i = 0; i < 3; ++i) {
63 for (int j = 0; j < 3; ++j)
64 t.mat[i][j] = cif::as_number(r[4*i+j]);
65 t.vec.at(i) = cif::as_number(r[4*i+3]);
66 }
67 return t;
68}

70inline SeqId make_seqid(const std::string& seqid, const std::string* icode) {
71 SeqId ret;
72 if (icode)
73 // the insertion code happens to be always a single letter
74 ret.icode = cif::as_char(*icode, ' ');
75 if (!seqid.empty()) {
76 // old mmCIF files have auth_seq_id as number + icode (e.g. 15A)
77 if (seqid.back() >= 'A') {
78 if (ret.icode == ' ')
79 ret.icode = seqid.back();
80 else if (ret.icode != seqid.back())
81 fail("Inconsistent insertion code in " + seqid);
82 ret.num = cif::as_int(seqid.substr(0, seqid.size() - 1));
83 } else {
84 ret.num = cif::as_int(seqid, Residue::OptionalNum::None);
85 }
86 }
87 return ret;
88}

96inline std::vector<Helix> read_helices(cif::Block& block) {
97 std::vector<Helix> helices;
98 for (const auto row : block.find("_struct_conf.", {
99 "conf_type_id", // 0
100 "beg_auth_asym_id", "beg_label_comp_id", // 1-2
101 "beg_auth_seq_id", "?pdbx_beg_PDB_ins_code", // 3-4
102 "end_auth_asym_id", "end_label_comp_id", // 5-6
103 "end_auth_seq_id", "?pdbx_end_PDB_ins_code", // 7-8
104 "?pdbx_PDB_helix_class", "?pdbx_PDB_helix_length"})) { // 9-10
105 if (alpha_up(row.str(0)[0]) != 'H')
106 continue;
107 Helix h;
108 h.start.chain_name = row.str(1);
109 h.start.res_id = make_resid(row.str(2), row.str(3), row.ptr_at(4));
110 h.end.chain_name = row.str(5);
111 h.end.res_id = make_resid(row.str(6), row.str(7), row.ptr_at(8));
112 if (row.has(9))
113 h.set_helix_class_as_int(cif::as_int(row[9], -1));
114 if (row.has(10))
115 h.length = cif::as_int(row[10], -1);
116 helices.push_back(h);
117 }
118 return helices;
119}

121inline std::vector<Sheet> read_sheets(cif::Block& block) {
122 std::vector<Sheet> sheets;
123 for (const std::string& sheet_id : block.find_values("_struct_sheet.id"))
124 sheets.emplace_back(sheet_id);
125 for (const auto row : block.find("_struct_sheet_range.", {
126 "sheet_id", "id", // 0-1
127 "beg_auth_asym_id", "beg_label_comp_id", // 2-3
128 "beg_auth_seq_id", "?pdbx_beg_PDB_ins_code", // 4-5
129 "end_auth_asym_id", "end_label_comp_id", // 6-7
130 "end_auth_seq_id", "?pdbx_end_PDB_ins_code"})) { // 8-9
131 Sheet& sheet = impl::find_or_add(sheets, row.str(0));
132 sheet.strands.emplace_back();
133 Sheet::Strand& strand = sheet.strands.back();
134 strand.name = row.str(1);
135 strand.start.chain_name = row.str(2);
136 strand.start.res_id = make_resid(row.str(3), row.str(4), row.ptr_at(5));
137 strand.end.chain_name = row.str(6);
138 strand.end.res_id = make_resid(row.str(7), row.str(8), row.ptr_at(9));
139 }
140
141 // below we assume that range_id_1 is the strand preceding range_id_2
142 for (const auto row : block.find("_struct_sheet_order.", {
143 "sheet_id", "range_id_2", "sense"}))
144 if (Sheet* sheet = impl::find_or_null(sheets, row.str(0)))
145 if (Sheet::Strand* ss = impl::find_or_null(sheet->strands, row.str(1)))
146 switch (alpha_up(row.str(2)[0])) {
147 case 'P': ss->sense = 1; break; // parallel
148 case 'A': ss->sense = -1; break; // anti-parallel
149 }
150
151 for (const auto row : block.find("_pdbx_struct_sheet_hbond.", {
152 "sheet_id", "range_id_2",
153 "range_1_auth_asym_id", "range_1_label_comp_id",
154 "range_1_auth_seq_id", "?range_1_PDB_ins_code",
155 "range_1_label_atom_id",
156 "range_2_auth_asym_id", "range_2_label_comp_id",
157 "range_2_auth_seq_id", "?range_2_PDB_ins_code",
158 "range_2_label_atom_id"}))
159 if (Sheet* sheet = impl::find_or_null(sheets, row.str(0)))
160 if (Sheet::Strand* ss = impl::find_or_null(sheet->strands, row.str(1))) {
161 ss->hbond_atom1.chain_name = row.str(2);
162 ss->hbond_atom1.res_id = make_resid(row.str(3), row.str(4),
163 row.ptr_at(5));
164 ss->hbond_atom1.atom_name = row.str(6);
165 ss->hbond_atom2.chain_name = row.str(7);
166 ss->hbond_atom2.res_id = make_resid(row.str(8), row.str(9),
167 row.ptr_at(10));
168 ss->hbond_atom2.atom_name = row.str(11);
169 }
170
171 return sheets;
172}

174inline
175void set_part_of_address_from_label(AtomAddress& a, const Model& model,
176 const std::string& label_asym,
177 const std::string& label_seq_id_raw) {
178 int seq = cif::as_int(label_seq_id_raw, SeqId::OptionalNum::None);
179 for (const Chain& chain : model.chains)
180 if (ConstResidueSpan sub = chain.get_subchain(label_asym)) {
181 a.chain_name = chain.name;
182 for (const Residue& res : sub)
183 if (res.label_seq == seq) {
184 a.res_id.seqid = res.seqid;
185 return;
186 }
187 }

190inline void read_connectivity(cif::Block& block, Structure& st) {
191 enum {
192 kId=0, kConnTypeId=1,
193 kAuthAsymId=2/*-3*/, kLabelAsymId=4/*-5*/, kLabelCompId=6/*-7*/,
194 kLabelAtomId=8/*-9*/, kLabelAltId=10/*-11*/,
195 kAuthSeqId=12/*-13*/, kLabelSeqId=14/*-15*/, kInsCode=16/*-17*/,
196 kSym1=18, kSym2=19, kDistValue=20, kLinkId=21
197 };
198 // label_ identifiers are not sufficient for HOH:
199 // waters have null label_seq_id so we need auth_seq_id+icode.
200 // And since we need auth_seq_id, we also use auth_asym_id for consistency.
201 // Unless only label_*_id are available.
202 for (const auto row : block.find("_struct_conn.", {
203 "id", "conn_type_id", // 0-1
204 "?ptnr1_auth_asym_id", "?ptnr2_auth_asym_id", // 2-3
205 "?ptnr1_label_asym_id", "?ptnr2_label_asym_id", // 4-5
206 "ptnr1_label_comp_id", "ptnr2_label_comp_id", // 6-7
207 "ptnr1_label_atom_id", "ptnr2_label_atom_id", // 8-9
208 "?pdbx_ptnr1_label_alt_id", "?pdbx_ptnr2_label_alt_id", // 10-11
209 "?ptnr1_auth_seq_id", "?ptnr2_auth_seq_id", // 12-13
210 "?ptnr1_label_seq_id", "?ptnr2_label_seq_id", // 14-15
211 "?pdbx_ptnr1_PDB_ins_code", "?pdbx_ptnr2_PDB_ins_code", // 16-17
212 "?ptnr1_symmetry", "?ptnr2_symmetry", // 18-19
213 "?pdbx_dist_value", "?ccp4_link_id"})) { // 20-21
214 Connection c;
215 c.name = row.str(kId);
216 copy_string(row, kLinkId, c.link_id);
217 c.type = connection_type_from_string(row.str(kConnTypeId));
218 if (row.has2(kSym1) && row.has2(kSym2)) {
219 c.asu = (row.str(kSym1) == row.str(kSym2) ? Asu::Same : Asu::Different);
220 }
221 copy_double(row, kDistValue, c.reported_distance);
222 for (int i = 0; i < 2; ++i) {
223 AtomAddress& a = (i == 0 ? c.partner1 : c.partner2);
224 if (row.has(kAuthAsymId+i) && row.has(kAuthSeqId+i)) {
225 a.chain_name = row.str(kAuthAsymId+i);
226 a.res_id = make_resid(row.str(kLabelCompId+i),
227 row.str(kAuthSeqId+i), row.ptr_at(kInsCode+i));
228 } else if (row.has(kLabelAsymId+i) && row.has(kLabelSeqId+i)) {
229 set_part_of_address_from_label(a, st.first_model(),
230 row.str(kLabelAsymId+i),
231 row[kLabelSeqId+i]);
232 a.res_id.name = row.str(kLabelCompId+i);
233 } else {
234 fail("_struct_conn without either _auth_ or _label_ asym_id+seq_id");
235 }
236 a.atom_name = row.str(kLabelAtomId+i);
237 if (row.has2(kLabelAltId+i))
238 a.altloc = cif::as_char(row[kLabelAltId+i], '\0');
239 }
240 st.connections.emplace_back(c);
241 }
242}

244// Operation expression is an item type used for *.oper_expression.
245// Here, to keep it simple, we ignore products such as "(2)(3)".
246// We parse "3", "1,3,5", "one,two", "(3)", "(a)", "(1-60)", "(2,3-8,XY)", etc
247inline std::vector<std::string> parse_operation_expr(const std::string& expr) {
248 std::vector<std::string> result;
249 std::size_t start = 0;
250 std::size_t close_br = std::string::npos;
251 if (expr[0] == '(') {
252 start = 1;
253 close_br = expr.find(')');
254 }
255 for (;;) {
256 std::size_t sep = std::min(expr.find(',', start), close_br);
257 std::size_t minus = expr.find('-', start);
258 if (minus < sep) {
259 int n_min = no_sign_atoi(expr.c_str() + start);
260 int n_max = no_sign_atoi(expr.c_str() + minus + 1);
261 for (int n = n_min; n <= n_max; ++n)
262 result.push_back(std::to_string(n));
263 } else {
264 result.emplace_back(expr, start, sep - start);
265 }
266 if (sep == close_br)
267 break;
268 start = sep + 1;
269 }
270 return result;
271}

273inline std::vector<Assembly> read_assemblies(cif::Block& block) {
274 std::vector<Assembly> assemblies;
275 cif::Table prop_tab = block.find("_pdbx_struct_assembly_prop.",
276 {"biol_id", "type", "value"});
277 cif::Table gen_tab = block.find("_pdbx_struct_assembly_gen.",
278 {"assembly_id", "oper_expression", "asym_id_list"});
279 std::vector<Assembly::Operator> oper_list;
280 std::vector<std::string> oper_list_tags = transform_tags("matrix", "vector");
281 oper_list_tags.emplace_back("id"); // 12
282 oper_list_tags.emplace_back("type"); // 13
283 for (const auto row : block.find("_pdbx_struct_oper_list.", oper_list_tags)) {
284 oper_list.emplace_back();
285 oper_list.back().name = row.str(12);
286 oper_list.back().type = row.str(13);
287 oper_list.back().transform = get_transform_matrix(row);
288 }
289 for (const auto row : block.find("_pdbx_struct_assembly.", {
290 "id", "details", "method_details",
291 "oligomeric_details", "oligomeric_count"})) {
292 assemblies.emplace_back(row.str(0));
293 Assembly& a = assemblies.back();
294 std::string detail = row.str(1);
295 if (detail == "author_and_software_defined_assembly")
296 a.author_determined = a.software_determined = true;
297 else if (detail == "author_defined_assembly")
298 a.author_determined = true;
299 else if (detail == "software_defined_assembly")
300 a.software_determined = true;
301 else if (detail == "complete icosahedral assembly")
302 a.special_kind = Assembly::SpecialKind::CompleteIcosahedral;
303 else if (detail == "representative helical assembly")
304 a.special_kind = Assembly::SpecialKind::RepresentativeHelical;
305 else if (detail == "complete point assembly")
306 a.special_kind = Assembly::SpecialKind::CompletePoint;
307
308 if (!a.author_determined && !a.software_determined &&
309 a.special_kind == Assembly::SpecialKind::NA && !detail.empty()) {
310 assemblies.pop_back();
311 continue;
312 }
313 if (a.software_determined && !cif::is_null(row[2]))
314 a.software_name = row.str(2); // method_details
315 a.oligomeric_details = row.str(3);
316 a.oligomeric_count = cif::as_int(row[4], 0);
317 for (const auto row_p : prop_tab)
318 if (row_p.str(0) == a.name) {
319 std::string type = row_p.str(1);
320 double value = cif::as_number(row_p[2]);
321 if (type == "ABSA (A^2)")
322 a.absa = value;
323 else if (type == "SSA (A^2)")
324 a.ssa = value;
325 else if (type == "MORE")
326 a.more = value;
327 }
328 for (const auto row_g : gen_tab)
329 if (row_g.str(0) == a.name) {
330 a.generators.emplace_back();
331 Assembly::Gen& gen = a.generators.back();
332 split_str_into(row_g.str(2), ',', gen.subchains);
333 for (const std::string& name : parse_operation_expr(row_g.str(1)))
334 if (const Assembly::Operator* oper = impl::find_or_null(oper_list, name))
335 gen.operators.push_back(*oper);
336 }
337 }
338 return assemblies;
339}

341inline void fill_residue_entity_type(Structure& st) {
342 for (Model& model : st.models)
343 for (Chain& chain : model.chains)
344 for (ResidueSpan& sub : chain.subchains()) {
345 EntityType etype = EntityType::Unknown;
346 if (const Entity* ent = st.get_entity_of(sub))
347 etype = ent->entity_type;
348 if (etype == EntityType::Unknown) {
349 if (sub[0].is_water())
350 etype = EntityType::Water;
351 else if (sub.length() > 1)
352 etype = EntityType::Polymer;
353 else
354 etype = EntityType::NonPolymer;
355 }
356 for (Residue& residue : sub)
357 residue.entity_type = etype;
358 }
359}

370inline Structure make_structure_from_block(const cif::Block& block_) {
371 // find() and Table don't have const variants, but we don't change anything.
372 cif::Block& block = const_cast<cif::Block&>(block_);
373 Structure st;
374 st.input_format = CoorFormat::Mmcif;
375 st.name = block.name;
376 set_cell_from_mmcif(block, st.cell);
377 st.spacegroup_hm = cif::as_string(impl::find_spacegroup_hm_value(block));
378
379 auto add_info = [&](const std::string& tag) {
380 bool first = true;
381 for (const std::string& v : block.find_values(tag))
382 if (!cif::is_null(v)) {
383 if (first)
384 st.info[tag] = cif::as_string(v);
385 else
386 st.info[tag] += "; " + cif::as_string(v);
387 first = false;
388 }
389 };
390 add_info("_entry.id");
391 add_info("_cell.Z_PDB");
392 add_info("_exptl.method");
393 add_info("_struct.title");
394 // in pdbx/mmcif v5 date_original was replaced with a much longer tag
395 std::string old_date_tag = "_database_PDB_rev.date_original";
396 std::string new_date_tag
397 = "_pdbx_database_status.recvd_initial_deposition_date";
398 add_info(old_date_tag);
399 add_info(new_date_tag);
400 if (st.info.count(old_date_tag) == 1 && st.info.count(new_date_tag) == 0)
401 st.info[new_date_tag] = st.info[old_date_tag];
402 add_info("_struct_keywords.pdbx_keywords");
403 add_info("_struct_keywords.text");
404
405 for (const std::string& v : block.find_values("_audit_author.name"))
406 if (!cif::is_null(v))
407 st.meta.authors.push_back(cif::as_string(v));
408
409 for (auto row : block.find("_refine.", {"pdbx_refine_id", // 0
410 "?ls_d_res_high", // 1
411 "?ls_d_res_low", // 2
412 "?ls_percent_reflns_obs", // 3
413 "?ls_number_reflns_obs"})) {
414 st.meta.refinement.emplace_back();
415 RefinementInfo& ref = st.meta.refinement.back();
416 ref.id = row.str(0);
417 if (row.has(1)) {
418 ref.resolution_high = cif::as_number(row[1]);
419 if (ref.resolution_high > 0 &&
420 (st.resolution == 0 || ref.resolution_high < st.resolution))
421 st.resolution = ref.resolution_high;
422 }
423 copy_double(row, 2, ref.resolution_low);
424 copy_double(row, 3, ref.completeness);
425 copy_int(row, 4, ref.reflection_count);
426 }
427
428 for (auto row : block.find("_exptl.", {"method", "?crystals_number"})) {
429 st.meta.experiments.emplace_back();
430 st.meta.experiments.back().method = row.str(0);
431 copy_int(row, 1, st.meta.experiments.back().number_of_crystals);
432 }
433
434 for (auto row : block.find("_exptl_crystal.", {"id", "?description"})) {
435 st.meta.crystals.emplace_back();
436 st.meta.crystals.back().id = row.str(0);
437 copy_string(row, 1, st.meta.crystals.back().description);
438 }
439
440 for (auto row : block.find("_diffrn.",
441 {"id", "crystal_id", "?ambient_temp"})) {
442 std::string id = row.str(1);
443 auto cryst = std::find_if(st.meta.crystals.begin(), st.meta.crystals.end(),
444 [&](const CrystalInfo& c) { return c.id == id; });
445 if (cryst != st.meta.crystals.end()) {
446 cryst->diffractions.emplace_back();
447 cryst->diffractions.back().id = row.str(0);
448 copy_double(row, 2, cryst->diffractions.back().temperature);
449 }
450 }
451 for (auto row : block.find("_diffrn_detector.", {"diffrn_id",
452 "?pdbx_collection_date",
453 "?detector",
454 "?type",
455 "?details"}))
456 if (DiffractionInfo* di = find_diffrn(st.meta, row.str(0))) {
457 copy_string(row, 1, di->collection_date);
458 copy_string(row, 2, di->detector);
459 copy_string(row, 3, di->detector_make);
460 copy_string(row, 4, di->optics);
461 }
462 for (auto row : block.find("_diffrn_radiation.",
463 {"diffrn_id",
464 "?pdbx_scattering_type",
465 "?pdbx_monochromatic_or_laue_m_l",
466 "?monochromator"}))
467 if (DiffractionInfo* di = find_diffrn(st.meta, row.str(0))) {
468 copy_string(row, 1, di->scattering_type);
469 if (row.has2(2))
470 di->mono_or_laue = row.str(2)[0];
471 copy_string(row, 3, di->monochromator);
472 }
473 for (auto row : block.find("_diffrn_source.", {"diffrn_id",
474 "?source",
475 "?type",
476 "?pdbx_synchrotron_site",
477 "?pdbx_synchrotron_beamline",
478 "?pdbx_wavelength_list"}))
479 if (DiffractionInfo* di = find_diffrn(st.meta, row.str(0))) {
480 copy_string(row, 1, di->source);
481 copy_string(row, 2, di->source_type);
482 copy_string(row, 3, di->synchrotron);
483 copy_string(row, 4, di->beamline);
484 copy_string(row, 5, di->wavelengths);
485 }
486
487 size_t n = 0;
488 for (auto row : block.find("_reflns.", {"pdbx_diffrn_id", // 0
489 "?number_obs", // 1
490 "?d_resolution_high", // 2
491 "?d_resolution_low", // 3
492 "?percent_possible_obs", // 4
493 "?pdbx_redundancy", // 5
494 "?pdbx_Rmerge_I_obs", // 6
495 "?pdbx_Rsym_value", // 7
496 "?pdbx_netI_over_sigmaI"})) {
497 // In the case of multiple experiments (_exptl), which is rare,
498 // it is not explicit to which experiment which data statistics
499 // (_reflns) corresponds to. We assume they are in the same order.
500 if (n >= st.meta.experiments.size())
501 break;
502 ExperimentInfo& exper = st.meta.experiments[n++];
503 split_str_into(row.str(0), ',', exper.diffraction_ids);
504 copy_int(row, 1, exper.unique_reflections);
505 copy_double(row, 2, exper.reflections.resolution_high);
506 copy_double(row, 3, exper.reflections.resolution_low);
507 copy_double(row, 4, exper.reflections.completeness);
508 copy_double(row, 5, exper.reflections.redundancy);
509 copy_double(row, 6, exper.reflections.r_merge);
510 copy_double(row, 7, exper.reflections.r_sym);

601 if (!atom_table.has_column(kAtomId))
602 fail("Neither _atom_site.label_atom_id nor auth_atom_id found");

604 Model *model = nullptr;
605 Chain *chain = nullptr;
606 Residue *resi = nullptr;
607 if (atom_table.has_column(kModelNum))
608 model = &st.find_or_add_model(atom_table[0].str(kModelNum));
609 else
610 model = &st.find_or_add_model("1");
611 for (auto row : atom_table) {
612 if (row.has(kModelNum) && row[kModelNum] != model->name) {
613 model = &st.find_or_add_model(row.str(kModelNum));
614 chain = nullptr;
615 }
616 if (!chain || cif::as_string(row[kAsymId]) != chain->name) {
617 model->chains.emplace_back(cif::as_string(row[kAsymId]));
618 chain = &model->chains.back();

621 ResidueId rid = make_resid(cif::as_string(row[kCompId]),
622 cif::as_string(row[kAuthSeqId]),
623 row.has(kInsCode) ? &row[kInsCode] : nullptr);
624 if (!resi || !resi->matches(rid)) {
625 resi = chain->find_or_add_residue(rid);
626 if (resi->atoms.empty()) {
627 if (row.has2(kLabelSeqId))
628 resi->label_seq = cif::as_int(row[kLabelSeqId]);
629 resi->subchain = row.str(kLabelAsymId);
630 if (row.has2(kLabelEntityId))
631 resi->entity_id = row.str(kLabelEntityId);
632 // don't check if group_PDB is consistent, it's not that important
633 if (row.has2(kGroupPdb))
634 for (int i = 0; i < 2; ++i) { // first character could be " or '
635 const char c = alpha_up(row[kGroupPdb][i]);
636 if (c == 'A' || c == 'H' || c == '\0')
637 resi->het_flag = c;
638 }
639 }
640 } else if (resi->seqid != rid.seqid) {
641 fail("Inconsistent sequence ID: " + resi->str() + " / " + rid.str());
642 }
643 Atom atom;
644 atom.name = cif::as_string(row[kAtomId]);
645 // altloc is always a single letter (not guaranteed by the mmCIF spec)
646 atom.altloc = cif::as_char(row[kAltId], '\0');
647 atom.charge = row.has2(kCharge) ? cif::as_int(row[kCharge]) : 0;
648 atom.element = gemmi::Element(cif::as_string(row[kSymbol]));
649 // According to the PDBx/mmCIF spec _atom_site.id can be a string,
650 // but in all the files it is a serial number; its value is not essential,
651 // so we just ignore non-integer ids.
652 atom.serial = string_to_int(row[kId], false);
653 if (row.has2(kCalcFlag)) {
654 const std::string& cf = row[kCalcFlag];
655 if (cf[0] == 'c')
656 atom.calc_flag = CalcFlag::Calculated;
657 if (cf[0] == 'd')
658 atom.calc_flag = cf[1] == 'u' ? CalcFlag::Dummy
659 : CalcFlag::Determined;
660 }

662 const char* str = row[kTlsGroupId].c_str();
663 const char* endptr;
664 int tls_id = no_sign_atoi(str, &endptr);
665 if (endptr != str)
666 atom.tls_group_id = (short) tls_id;
667 }
668 atom.pos.x = cif::as_number(row[kX]);
669 atom.pos.y = cif::as_number(row[kY]);
670 atom.pos.z = cif::as_number(row[kZ]);
671 atom.occ = (float) cif::as_number(row[kOcc], 1.0);
672 atom.b_iso = (float) cif::as_number(row[kBiso], 50.0);
673
674 if (!aniso_map.empty()) {
675 auto ani = aniso_map.find(row[kId]);
676 if (ani != aniso_map.end())
677 atom.aniso = ani->second;

679 resi->atoms.emplace_back(atom);
680 }
