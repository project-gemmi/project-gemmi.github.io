26 struct Bond {
27 const Restraints::Bond* restr;
28 std::array<Atom*, 2> atoms;
29 double calculate() const { return atoms[0]->pos.dist(atoms[1]->pos); }
30 double calculate_z() const {
31 return std::abs(calculate() - restr->value) / restr->esd;
32 }
33 };
34 struct Angle {
35 const Restraints::Angle* restr;
36 std::array<Atom*, 3> atoms;
37 double calculate() const {
38 return calculate_angle(atoms[0]->pos, atoms[1]->pos, atoms[2]->pos);
39 }
40 double calculate_z() const { return angle_z(calculate(), *restr); }
41 };
42 struct Torsion {
43 const Restraints::Torsion* restr;
44 std::array<Atom*, 4> atoms;
45 double calculate() const {
46 return calculate_dihedral(atoms[0]->pos, atoms[1]->pos,
47 atoms[2]->pos, atoms[3]->pos);
48 }
49 double calculate_z() const {
50 return angle_z(calculate(), *restr, 360. / std::max(1, restr->period));
51 }
52 };
53 struct Chirality {
54 const Restraints::Chirality* restr;
55 std::array<Atom*, 4> atoms;
56 double calculate() const {
57 return calculate_chiral_volume(atoms[0]->pos, atoms[1]->pos,
58 atoms[2]->pos, atoms[3]->pos);
59 }
60 double calculate_z(double ideal_abs_vol, double esd) const {
61 double calc = calculate();
62 if (restr->sign == ChiralityType::Negative ||
63 (restr->sign == ChiralityType::Both && calc < 0))
64 ideal_abs_vol *= -1;
65 return std::abs(calc - ideal_abs_vol) / esd;
66 }
67 bool check() const { return !restr->is_wrong(calculate()); }
68 };

72 bool has(const Atom* atom) const {

168 const Restraints::Bond* take_bond(const Atom* a, const Atom* b) const {
169 auto range = bond_index.equal_range(a);
170 for (auto i = range.first; i != range.second; ++i) {
171 const Bond* bond = i->second;
172 if ((bond->atoms[0] == b && bond->atoms[1] == a) ||
173 (bond->atoms[1] == b && bond->atoms[0] == a))
174 return bond->restr;
175 }
176 return nullptr;
177 }

179 const Restraints::Angle* take_angle(const Atom* a,
180 const Atom* b,
181 const Atom* c) const {
182 auto range = angle_index.equal_range(b);
183 for (auto i = range.first; i != range.second; ++i) {
184 const Angle* ang = i->second;
185 if ((ang->atoms[0] == a && ang->atoms[2] == c) ||
186 (ang->atoms[0] == c && ang->atoms[2] == a))
187 return ang->restr;
188 }
189 return nullptr;

199 double ideal_chiral_abs_volume(const Chirality &ch) const {
200 const Restraints::Bond* bond_c1 = take_bond(ch.atoms[0], ch.atoms[1]);
201 const Restraints::Bond* bond_c2 = take_bond(ch.atoms[0], ch.atoms[2]);
202 const Restraints::Bond* bond_c3 = take_bond(ch.atoms[0], ch.atoms[3]);
203 const Restraints::Angle* angle_1c2 = take_angle(ch.atoms[1], ch.atoms[0], ch.atoms[2]);
204 const Restraints::Angle* angle_2c3 = take_angle(ch.atoms[2], ch.atoms[0], ch.atoms[3]);
205 const Restraints::Angle* angle_3c1 = take_angle(ch.atoms[3], ch.atoms[0], ch.atoms[1]);
206 if (bond_c1 && bond_c2 && bond_c3 && angle_1c2 && angle_2c3 && angle_3c1)
207 return chiral_abs_volume(bond_c1->value, bond_c2->value, bond_c3->value,
208 angle_1c2->value, angle_2c3->value, angle_3c1->value);
209 return std::numeric_limits<double>::quiet_NaN();
210 }

212 std::vector<Rule> apply_restraints(const Restraints& rt,
213 Residue& res, Residue* res2,
214 char altloc='*') {
215 std::string altlocs;
216 if (altloc == '*') {
217 // find all distinct altlocs
218 add_distinct_altlocs(res, altlocs);
219 if (res2)
220 add_distinct_altlocs(*res2, altlocs);
221 }
222 if (altlocs.empty())
223 altlocs += altloc;
224
225 std::vector<Rule> rules;
226 for (const Restraints::Bond& bond : rt.bonds)
227 for (char alt : altlocs)
228 if (Atom* at1 = bond.id1.get_from(res, res2, alt))
229 if (Atom* at2 = bond.id2.get_from(res, res2, alt)) {
230 rules.push_back({RKind::Bond, bonds.size()});
231 bonds.push_back({&bond, {{at1, at2}}});
232 if (!at1->altloc && !at2->altloc)
233 break;
234 }
235 for (const Restraints::Angle& angle : rt.angles)
236 for (char alt : altlocs)
237 if (Atom* at1 = angle.id1.get_from(res, res2, alt))
238 if (Atom* at2 = angle.id2.get_from(res, res2, alt))
239 if (Atom* at3 = angle.id3.get_from(res, res2, alt)) {
240 rules.push_back({RKind::Angle, angles.size()});
241 angles.push_back({&angle, {{at1, at2, at3}}});
242 if (!at1->altloc && !at2->altloc && !at3->altloc)
243 break;
244 }
245 for (const Restraints::Torsion& tor : rt.torsions)
246 for (char alt : altlocs)
247 if (Atom* at1 = tor.id1.get_from(res, res2, alt))
248 if (Atom* at2 = tor.id2.get_from(res, res2, alt))
249 if (Atom* at3 = tor.id3.get_from(res, res2, alt))
250 if (Atom* at4 = tor.id4.get_from(res, res2, alt)) {
251 rules.push_back({RKind::Torsion, torsions.size()});
252 torsions.push_back({&tor, {{at1, at2, at3, at4}}});
253 if (!at1->altloc && !at2->altloc &&
254 !at3->altloc && !at4->altloc)
255 break;
256 }
257 for (const Restraints::Chirality& chir : rt.chirs)
258 for (char alt : altlocs)
259 if (Atom* at1 = chir.id_ctr.get_from(res, res2, alt))
260 if (Atom* at2 = chir.id1.get_from(res, res2, alt))
261 if (Atom* at3 = chir.id2.get_from(res, res2, alt))
262 if (Atom* at4 = chir.id3.get_from(res, res2, alt)) {
263 rules.push_back({RKind::Chirality, chirs.size()});
264 chirs.push_back({&chir, {{at1, at2, at3, at4}}});
265 if (!at1->altloc && !at2->altloc &&
266 !at3->altloc && !at4->altloc)
267 break;
268 }
269 for (const Restraints::Plane& plane : rt.planes)
270 for (char alt : altlocs) {
271 std::vector<Atom*> atoms;
272 for (const Restraints::AtomId& id : plane.ids)
273 if (Atom* atom = id.get_from(res, res2, alt))
274 atoms.push_back(atom);
275 if (atoms.size() >= 4) {
276 rules.push_back({RKind::Plane, planes.size()});
277 planes.push_back({&plane, atoms});
278 }
279 if (std::all_of(atoms.begin(), atoms.end(),
280 [](Atom* a) { return !a->altloc; }))
281 break;
282 }
283 return rules;
284 }

286 void apply_restraints_to_residue(ResInfo& ri, const MonLib& monlib) {
287 // link restraints
288 for (Link& prev : ri.prev)
289 if (const ChemLink* link = monlib.get_link(prev.link_id)) {
290 auto rules = apply_restraints(link->rt, *prev.res1, ri.res);
291 vector_move_extend(prev.link_rules, std::move(rules));
292 }
293 // monomer restraints
294 auto rules = apply_restraints(ri.chemcomp.rt, *ri.res, nullptr);
295 vector_move_extend(ri.monomer_rules, std::move(rules));
296 }

298 void apply_restraints_to_extra_link(Link& link, const MonLib& monlib) {
299 const ChemLink* cl = monlib.get_link(link.link_id);
300 if (!cl) {
301 err("ignoring link '" + link.link_id + "' as it is not in the monomer library");
302 return;
303 }
304 if (link.alt1 && link.alt2 && link.alt1 != link.alt2)
305 err(cat("LINK between different conformers ", link.alt1, " and ", link.alt2, '.'));
306 char alt = link.alt1 ? link.alt1 : link.alt2;
307 auto rules = apply_restraints(cl->rt, *link.res1, link.res2, alt);
308 vector_move_extend(link.link_rules, std::move(rules));
309 }

338 Link* find_polymer_link(const AtomAddress& a1, const AtomAddress& a2) {
339 for (ChainInfo& ci : chain_infos)
340 if (a1.chain_name == ci.chain_ref.name && a2.chain_name == ci.chain_ref.name) {
341 for (ResInfo& ri : ci.res_infos)
342 for (Link& link : ri.prev) {
343 assert(link.res1 && link.res2);
344 if ((a1.res_id.matches_noseg(*link.res1) &&
345 a2.res_id.matches_noseg(*link.res2)) ||
346 (a2.res_id.matches_noseg(*link.res1) &&
347 a1.res_id.matches_noseg(*link.res2)))
348 return &link;
349 }
350 }
351 return nullptr;

382inline void Topo::ChainInfo::setup_polymer_links() {
383 if (!polymer || res_infos.empty())
384 return;
385 //ResidueGroup residue_groups()
386 RGroup prev_group = group_from(res_infos.begin());
387 while (prev_group.end != res_infos.end()) {
388 RGroup group = group_from(prev_group.end);
389 for (auto ri = group.begin; ri != group.end; ++ri)
390 for (auto prev_ri = prev_group.begin; prev_ri != prev_group.end; ++prev_ri) {
391 Link p;
392 p.res1 = prev_ri->res;
393 p.res2 = ri->res;
394 assert(prev_ri - ri == p.res1 - p.res2);
395 if (are_connected(*prev_ri->res, *ri->res, polymer_type)) {
396 if (is_polypeptide(polymer_type) && ri->chemcomp.is_peptide_group() &&
397 prev_ri->chemcomp.is_peptide_group()) {
398 int id = ialpha4_id(ri->chemcomp.group.c_str());
399 if (id == ialpha4_id("p-pe"))
400 p.link_id = "P"; // PCIS, PTRANS
401 else if (id == ialpha4_id("m-pe"))
402 p.link_id = "NM"; // NMCIS, NMTRANS
403 p.link_id += prev_ri->res->is_cis ? "CIS" : "TRANS";
404 } else if (is_polynucleotide(polymer_type) && ri->chemcomp.is_nucleotide_group() &&
405 prev_ri->chemcomp.is_nucleotide_group()) {
406 p.link_id = "p";
407 } else {
408 p.link_id = "?";
409 }
410 } else {
411 p.link_id = "gap";
412 }
413 ri->prev.push_back(p);
414 }
415 prev_group = group;
416 }

420inline void Topo::initialize_refmac_topology(Structure& st, Model& model0,
421 MonLib& monlib, bool ignore_unknown_links) {
422 // initialize chains and residues
423 for (Chain& chain : model0.chains)
424 for (ResidueSpan& sub : chain.subchains()) {
425 const Entity* ent = st.get_entity_of(sub);
426 chain_infos.emplace_back(sub, chain, ent);
427 }
428 for (ChainInfo& ci : chain_infos) {
429 // copy monomer description
430 for (ResInfo& ri : ci.res_infos) {
431 auto it = monlib.monomers.find(ri.res->name);
432 if (it != monlib.monomers.end())
433 ri.chemcomp = it->second;
434 else
435 err("unknown chemical component " + ri.res->name
436 + " in chain " + ci.chain_ref.name);
437 }
438 ci.setup_polymer_links();
439 }
440
441 // add extra links
442 for (Connection& conn : st.connections)
443 if (conn.type != Connection::Hydrog) // ignoring hydrogen bonds
444 setup_connection(conn, model0, monlib, ignore_unknown_links);
445
446 // Add modifications from standard links. We do it here b/c polymer links
447 // could be disabled (link_id = "?") in setup_connection().
448 for (ChainInfo& ci : chain_infos)
449 for (ResInfo& ri : ci.res_infos)
450 for (Link& prev : ri.prev)
451 if (const ChemLink* link = monlib.get_link(prev.link_id)) {
452 ResInfo* ri_prev = &ri + (prev.res1 - prev.res2);
453 ri_prev->add_mod(link->side1.mod);
454 ri.add_mod(link->side2.mod);
455 }
456
457 for (ChainInfo& chain_info : chain_infos)
458 for (ResInfo& ri : chain_info.res_infos) {
459 // apply modifications
460 for (const std::string& modif : ri.mods) {
461 if (const ChemMod* chem_mod = monlib.find_mod(modif))
462 try {
463 chem_mod->apply_to(ri.chemcomp);
464 } catch(std::runtime_error& e) {
465 err("failed to apply modification " + chem_mod->id
466 + " to " + ri.res->name + ": " + e.what());
467 }
468 else
469 err("modification not found: " + modif);
470 }
471 }
472}

474// it has side-effects: may modifies conn.link_id and add to monlib.links
475inline void Topo::setup_connection(Connection& conn, Model& model0, MonLib& monlib,
476 bool ignore_unknown_links) {
477 Link extra;
478 CRA cra1 = model0.find_cra(conn.partner1, true);
479 CRA cra2 = model0.find_cra(conn.partner2, true);
480 if (!cra1.atom || !cra2.atom)
481 return;
482 extra.res1 = cra1.residue;
483 extra.res2 = cra2.residue;
484 extra.alt1 = conn.partner1.altloc;
485 extra.alt2 = conn.partner2.altloc;
486 extra.asu = conn.asu;
487
488 const ChemLink* match = nullptr;
489
490 // If we have link_id find ChemLink by name (and check if it matches).
491 if (!conn.link_id.empty()) {
492 match = monlib.get_link(conn.link_id);
493 if (!match) {
494 err("link not found in monomer library: " + conn.link_id);
495 return;
496 }
497 if (match->rt.bonds.empty() ||
498 match->rt.bonds[0].id1.atom != conn.partner1.atom_name ||
499 match->rt.bonds[0].id2.atom != conn.partner2.atom_name ||
500 !monlib.link_side_matches_residue(match->side1, extra.res1->name) ||
501 !monlib.link_side_matches_residue(match->side2, extra.res2->name)) {
502 err("link from the monomer library does not match: " + conn.link_id);
503 return;
504 }
505 } else {
506 // we don't have link_id - use the best matching link (if any)
507 auto r = monlib.match_link(*extra.res1, conn.partner1.atom_name,
508 *extra.res2, conn.partner2.atom_name,
509 extra.alt1 ? extra.alt1 : extra.alt2);
510 match = r.first;
511 if (match && r.second) {
512 std::swap(extra.res1, extra.res2);
513 std::swap(extra.alt1, extra.alt2);
514 }
515 }
516
517 // If a polymer link is also given in LINK/struct_conn,
518 // use only one of them. If LINK has explicit name (ccp4_link_id),
519 // or if it matches residue-specific link from monomer library, use it;
520 // otherwise, LINK is repetition of TRANS/CIS, so ignore LINK.
521 if (Link* polymer_link = find_polymer_link(conn.partner1, conn.partner2)) {
522 if (conn.link_id.empty() && !cif::is_null(polymer_link->link_id) &&
523 (!match || (match->side1.comp.empty() && match->side2.comp.empty())))
524 return;
525 polymer_link->link_id = "?"; // disable polymer link
526 }
527
528 if (match) {
529 extra.link_id = match->id;
530 // add modifications from the link
531 find_resinfo(extra.res1)->add_mod(match->side1.mod);
532 find_resinfo(extra.res2)->add_mod(match->side2.mod);
533 } else {
534 if (ignore_unknown_links)
535 return;
536 // create a new ChemLink and add it to the monomer library
537 ChemLink cl;
538 cl.side1.comp = extra.res1->name;
539 cl.side2.comp = extra.res2->name;
540 cl.id = cl.side1.comp + cl.side2.comp;
541 bool use_ion = cra1.atom->element.is_metal() || cra2.atom->element.is_metal();
542 double ideal_dist = monlib.find_radius(cra1, use_ion) +
543 monlib.find_radius(cra2, use_ion);
544 cl.rt.bonds.push_back({Restraints::AtomId{1, conn.partner1.atom_name},
545 Restraints::AtomId{2, conn.partner2.atom_name},
546 BondType::Unspec, false,
547 ideal_dist, 0.02,
548 ideal_dist, 0.02});
549 monlib.ensure_unique_link_name(cl.id);
550 monlib.links.emplace(cl.id, cl);
551 extra.link_id = cl.id;
552 }
553 if (conn.link_id.empty())
554 conn.link_id = extra.link_id;
555 extras.push_back(extra);
556}
