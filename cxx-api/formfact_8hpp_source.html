16// precalculated density of an isotropic atom
17template<int N, typename Real>
18struct ExpSum {
19 Real a[N], b[N];
20
21 Real calculate(Real r2) const {
22 Real density = 0;
23 for (int i = 0; i < N; ++i)
24 density += a[i] * std::exp(b[i] * r2);
25 return density;
26 }
27
28 std::pair<Real,Real> calculate_with_derivative(Real r) const {
29 Real density = 0;
30 Real derivative = 0;
31 for (int i = 0; i < N; ++i) {
32 Real y = a[i] * std::exp(b[i] * r * r);
33 density += y;
34 derivative += 2 * b[i] * r * y;
35 }
36 return std::make_pair(density, derivative);
37 }
38};

54// Gaussian coefficients with functions to calculate sf and density.
55template<int N, int WithC, typename Real>
56struct GaussianCoef {
57 using coef_type = Real;
58 std::array<Real, 2*N+WithC> coefs;
59 Real a(int n) const { return coefs[n]; }
60 Real b(int n) const { return coefs[N+n]; }
61 Real c() const { return WithC ? coefs[2*N] : 0; }
62
63 void set_coefs(const std::array<Real, 2*N+WithC>& c) { coefs = c; }
64
65 // argument: (sin(theta)/lambda)^2
66 Real calculate_sf(Real stol2) const {
67 Real sf = c();
68 for (int i = 0; i < N; ++i)
69 sf += a(i) * std::exp(-b(i)*stol2);
70 return sf;
71 }

73 static constexpr Real pow15(Real x) { return x * std::sqrt(x); }
74
75 Real calculate_density_iso(Real r2, Real B) const {
76 constexpr Real _4pi = 4 * pi();
77 Real r2pi = r2 * pi();
78 Real density = c() * pow15(_4pi / B) * std::exp(-(_4pi / B) * r2pi);
79 for (int i = 0; i < N; ++i) {
80 Real t = _4pi / (b(i)+B);
81 density += a(i) * pow15(t) * std::exp(-t*r2pi);
82 }
83 return density;
84 }
85
86 // note: addend is considered only if WithC (addend is usually dispersion f')
87 ExpSum<N+WithC,Real> precalculate_density_iso(Real B, Real addend=0) const {
88 ExpSum<N+WithC,Real> prec;
89 constexpr Real _4pi = 4 * pi();
90 for (int i = 0; i < N; ++i) {
91 Real t = _4pi / (b(i)+B);
92 prec.a[i] = a(i) * pow15(t);
93 prec.b[i] = -t * pi();
94 }
95 if (WithC) {
96 Real t = _4pi / B;
97 prec.a[N] = (c() + addend) * pow15(t);
98 prec.b[N] = -t * pi();
99 }
100 return prec;
101 }
102
103 Real calculate_density_aniso(const Vec3& r, const SMat33<float>& U) const {
104 constexpr Real pi2 = sq(pi());
105 const SMat33<Real> B = U.scaled(8 * pi2);
106 Real density = c() * pow15(4 * pi()) / std::sqrt(B.determinant()) *
107 std::exp(-4 * pi2 * B.inverse().r_u_r(r));
108 for (int i = 0; i < N; ++i) {
109 SMat33<Real> Bb = B.added_kI(b(i));
110 density += a(i) * pow15(4 * pi()) / std::sqrt(Bb.determinant()) *
111 std::exp(-4 * pi2 * Bb.inverse().r_u_r(r));
112 }
113 return density;
114 }
115
116 ExpAnisoSum<N+WithC,Real> precalculate_density_aniso_b(const SMat33<Real>& B,
117 Real addend=0) const {
118 constexpr Real m4pi2 = -4 * sq(pi());
119 ExpAnisoSum<N+WithC,Real> prec;
120 for (int i = 0; i < N; ++i) {
121 SMat33<Real> Bb = B.added_kI(b(i));
122 prec.a[i] = a(i) * pow15(4 * pi()) / std::sqrt(Bb.determinant());
123 prec.b[i] = Bb.inverse().scaled(m4pi2);
124 }
125 if (WithC) {
126 prec.a[N] = (c() + addend) * pow15(4 * pi()) / std::sqrt(B.determinant());
127 prec.b[N] = B.inverse().scaled(m4pi2);
128 }
129 return prec;
130 }

132 ExpAnisoSum<N+WithC,Real> precalculate_density_aniso_u(const SMat33<float>& U,
133 Real addend=0) const {
134 constexpr Real UtoB = 8 * sq(pi());
135 return precalculate_density_aniso_b(U.scaled(UtoB), addend);
136 }
