15inline std::string read_pir_or_fasta(std::istream& is) {
16 std::string sequence;
17 std::string line;
18 std::getline(is, line);
19 if (line[0] != '>')
20 fail("PIR/FASTA files start with '>'");
21 const bool pir_format = (line.size() > 3 && line[3] == ';');
22 bool ensure_pir_format = false;
23 bool desc_line = pir_format;
24 size_t discard_n = 0;
25 while (is.peek() != '>' && std::getline(is, line)) {
26 for (char c : line) {
27 if (('A' <= c && c <= 'Z') || ('a' <= c && c <= 'z')) {
28 sequence += c;
29 } else if (!std::isspace(c) && c != '-') {
30 if (desc_line) {
31 ensure_pir_format = true;
32 sequence.clear();
33 break;
34 }
35 if (pir_format && c == '*')
36 return sequence.substr(discard_n);
37 fail("unexpected character in sequence file: ", c);
38 }
39 }
40 if (desc_line)
41 discard_n = sequence.size();
42 desc_line = false;
43 }
44 if (ensure_pir_format)
45 fail("PIR sequence must end with '*'");
46 return sequence;
47}
