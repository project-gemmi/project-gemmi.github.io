31 ReflnBlock() = default;
32 ReflnBlock(ReflnBlock&& rblock_) = default;
33 ReflnBlock(cif::Block&& block_) : block(std::move(block_)) {
34 entry_id = cif::as_string(block.find_value("_entry.id"));
35 impl::set_cell_from_mmcif(block, cell);
36 if (const std::string* hm = impl::find_spacegroup_hm_value(block))
37 spacegroup = find_spacegroup_by_name(cif::as_string(*hm),
38 cell.alpha, cell.gamma);
39 cell.set_cell_images_from_spacegroup(spacegroup);
40 const char* wave_tag = "_diffrn_radiation_wavelength.wavelength";
41 cif::Column wave_col = block.find_values(wave_tag);
42 wavelength_count = wave_col.length();
43 wavelength = wavelength_count == 1 ? cif::as_number(wave_col[0]) : 0.;
44 refln_loop = block.find_loop("_refln.index_h").get_loop();
45 diffrn_refln_loop = block.find_loop("_diffrn_refln.index_h").get_loop();
46 default_loop = refln_loop ? refln_loop : diffrn_refln_loop;
47 }
48 ReflnBlock& operator=(ReflnBlock&&) = default;
49 ReflnBlock& operator=(const ReflnBlock& o) {
50 if (this == &o)
51 return *this;
52 block = o.block;
53 entry_id = o.entry_id;
54 cell = o.cell;
55 spacegroup = o.spacegroup;
56 wavelength = o.wavelength;
57 wavelength_count = o.wavelength_count;
58 if (o.refln_loop)
59 refln_loop = block.find_loop("_refln.index_h").get_loop();
60 if (o.diffrn_refln_loop)
61 diffrn_refln_loop = block.find_loop("_diffrn_refln.index_h").get_loop();
62 default_loop = refln_loop ? refln_loop : diffrn_refln_loop;
63 return *this;
64 }

85 int find_column_index(const std::string& tag) const {
86 if (!ok())
87 return -1;
88 size_t name_pos = tag_offset();
89 for (int i = 0; i != (int) default_loop->tags.size(); ++i)

102 template<typename T>
103 std::vector<T> make_vector(const std::string& tag, T null) const {
104 size_t n = get_column_index(tag);
105 std::vector<T> v(default_loop->length());
106 for (size_t j = 0; j != v.size(); n += default_loop->width(), ++j)
107 v[j] = cif::as_any(default_loop->values[n], null);

112 return {{get_column_index("index_h"),
113 get_column_index("index_k"),

117 std::vector<Miller> make_miller_vector() const {
118 auto hkl_idx = get_hkl_column_indices();
119 std::vector<Miller> v(default_loop->length());
120 for (size_t j = 0, n = 0; j != v.size(); j++, n += default_loop->width())
121 for (int i = 0; i != 3; ++i)
122 v[j][i] = cif::as_int(default_loop->values[n + hkl_idx[i]]);
123 return v;
124 }
125
126 std::vector<double> make_1_d2_vector() const {
127 if (!cell.is_crystal() || cell.a <= 0)
128 fail("Unit cell is not known");
129 auto hkl_idx = get_hkl_column_indices();
130 std::vector<double> r(default_loop->length());
131 for (size_t j = 0, n = 0; j != r.size(); j++, n += default_loop->width()) {
132 Miller hkl;
133 for (int i = 0; i != 3; ++i)
134 hkl[i] = cif::as_int(default_loop->values[n + hkl_idx[i]]);
135 r[j] = cell.calculate_1_d2(hkl);
136 }
137 return r;
138 }
139
140 std::vector<double> make_d_vector() const {
141 std::vector<double> vec = make_1_d2_vector();
142 for (double& d : vec)
143 d = 1.0 / std::sqrt(d);
144 return vec;
145 }

148// moves blocks from the argument to the return value
149inline
150std::vector<ReflnBlock> as_refln_blocks(std::vector<cif::Block>&& blocks) {
151 std::vector<ReflnBlock> rvec;
152 rvec.reserve(blocks.size());
153 for (cif::Block& block : blocks)
154 rvec.emplace_back(std::move(block));
155 blocks.clear();
156 // Some blocks miss space group or unit cell, try to fill it in.
157 const SpaceGroup* first_sg = nullptr;
158 const UnitCell* first_cell = nullptr;
159 for (ReflnBlock& rblock : rvec) {
160 if (!first_sg)
161 first_sg = rblock.spacegroup;
162 else if (!rblock.spacegroup)
163 rblock.spacegroup = first_sg;
164 if (rblock.cell.is_crystal()) {
165 if (!first_cell)
166 first_cell = &rblock.cell;
167 } else if (first_cell) {
168 rblock.cell = *first_cell;
169 }
170 }
171 return rvec;
172}

174// Get the first (merged) block with required labels.
175// Optionally, block name can be specified.
176inline ReflnBlock get_refln_block(std::vector<cif::Block>&& blocks,
177 const std::vector<std::string>& labels,
178 const char* block_name=nullptr) {
179 const SpaceGroup* first_sg = nullptr;
180 for (cif::Block& block : blocks) {
181 if (!first_sg)
182 if (const std::string* hm = impl::find_spacegroup_hm_value(block)) {
183 first_sg = find_spacegroup_by_name(cif::as_string(*hm));
184 if (first_sg && first_sg->ext == 'H') {
185 UnitCell cell;
186 impl::set_cell_from_mmcif(block, cell);
187 first_sg = find_spacegroup_by_name(cif::as_string(*hm),
188 cell.alpha, cell.gamma);
189 }
190 }
191 if (block_name && block.name != block_name)
192 continue;
193 if (cif::Loop* loop = block.find_loop("_refln.index_h").get_loop())
194 if (std::all_of(labels.begin(), labels.end(),
195 [&](const std::string& s) { return loop->has_tag("_refln."+s); })) {
196 ReflnBlock rblock(std::move(block));
197 if (!rblock.spacegroup && first_sg)
198 rblock.spacegroup = first_sg;
199 return rblock;
200 }
201 }
202 fail("Required block or tags not found in the SF-mmCIF file.");
203}

205inline ReflnBlock hkl_cif_as_refln_block(cif::Block& block) {
206 ReflnBlock rb;
207 rb.block.swap(block);
208 rb.entry_id = rb.block.name;
209 impl::set_cell_from_mmcif(rb.block, rb.cell, /*mmcif=*/false);
210 const char* hm_tag = "_symmetry_space_group_name_H-M";
211 if (const std::string* hm = rb.block.find_value(hm_tag))
212 rb.spacegroup = find_spacegroup_by_name(cif::as_string(*hm),
213 rb.cell.alpha, rb.cell.gamma);
214 rb.cell.set_cell_images_from_spacegroup(rb.spacegroup);
215 rb.refln_loop = rb.block.find_loop("_refln_index_h").get_loop();
216 rb.default_loop = rb.refln_loop;
217 return rb;
218}
