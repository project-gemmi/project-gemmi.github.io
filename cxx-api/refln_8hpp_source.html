20struct ReflnBlock {
21 cif::Block block;

23 UnitCell cell;
24 const SpaceGroup* spacegroup = nullptr;

30 ReflnBlock() = default;
31 ReflnBlock(ReflnBlock&& rblock_) = default;
32 ReflnBlock(cif::Block&& block_) : block(block_) {
33 entry_id = cif::as_string(block.find_value("_entry.id"));
34 impl::set_cell_from_mmcif(block, cell);
35 if (const std::string* hm = impl::find_spacegroup_hm_value(block))
36 spacegroup = find_spacegroup_by_name(cif::as_string(*hm),
37 cell.alpha, cell.gamma);
38 cell.set_cell_images_from_spacegroup(spacegroup);
39 const char* wave_tag = "_diffrn_radiation_wavelength.wavelength";
40 cif::Column wave_col = block.find_values(wave_tag);
41 wavelength = wave_col.length() == 1 ? cif::as_number(wave_col[0]) : 0.;
42 refln_loop = block.find_loop("_refln.index_h").get_loop();
43 diffrn_refln_loop = block.find_loop("_diffrn_refln.index_h").get_loop();
44 default_loop = refln_loop ? refln_loop : diffrn_refln_loop;
45 }
46 ReflnBlock& operator=(ReflnBlock&&) = default;
47 ReflnBlock& operator=(const ReflnBlock& o) {
48 block = o.block;
49 entry_id = o.entry_id;
50 cell = o.cell;
51 spacegroup = o.spacegroup;
52 wavelength = o.wavelength;
53 if (o.refln_loop)
54 refln_loop = block.find_loop("_refln.index_h").get_loop();
55 if (o.diffrn_refln_loop)
56 diffrn_refln_loop = block.find_loop("_diffrn_refln.index_h").get_loop();
57 default_loop = refln_loop ? refln_loop : diffrn_refln_loop;

59 }

97 template<typename T>
98 std::vector<T> make_vector(const std::string& tag, T null) const {
99 size_t n = get_column_index(tag);
100 std::vector<T> v(default_loop->length());
101 for (size_t j = 0; j != v.size(); n += default_loop->width(), ++j)
102 v[j] = cif::as_any(default_loop->values[n], null);
103 return v;

112 std::vector<Miller> make_miller_vector() const {
113 auto hkl_idx = get_hkl_column_indices();
114 std::vector<Miller> v(default_loop->length());
115 for (size_t j = 0, n = 0; j != v.size(); j++, n += default_loop->width())
116 for (int i = 0; i != 3; ++i)
117 v[j][i] = cif::as_int(default_loop->values[n + hkl_idx[i]]);
118 return v;
119 }
120
121 std::vector<double> make_1_d2_vector() const {
122 if (!cell.is_crystal() || cell.a <= 0)
123 fail("Unit cell is not known");
124 auto hkl_idx = get_hkl_column_indices();
125 std::vector<double> r(default_loop->length());
126 for (size_t j = 0, n = 0; j != r.size(); j++, n += default_loop->width()) {
127 Miller hkl;
128 for (int i = 0; i != 3; ++i)
129 hkl[i] = cif::as_int(default_loop->values[n + hkl_idx[i]]);
130 r[j] = cell.calculate_1_d2(hkl);
131 }
132 return r;
133 }
134
135 std::vector<double> make_d_vector() const {
136 std::vector<double> vec = make_1_d2_vector();
137 for (double& d : vec)
138 d = 1.0 / std::sqrt(d);
139 return vec;
140 }

143// moves blocks from the argument to the return value
144inline
145std::vector<ReflnBlock> as_refln_blocks(std::vector<cif::Block>&& blocks) {
146 std::vector<ReflnBlock> rvec;
147 rvec.reserve(blocks.size());
148 for (cif::Block& block : blocks)
149 rvec.emplace_back(std::move(block));
150 blocks.clear();
151 // Some blocks miss space group or unit cell, try to fill it in.
152 const SpaceGroup* first_sg = nullptr;
153 const UnitCell* first_cell = nullptr;
154 for (ReflnBlock& rblock : rvec) {
155 if (!first_sg)
156 first_sg = rblock.spacegroup;
157 else if (!rblock.spacegroup)
158 rblock.spacegroup = first_sg;
159 if (rblock.cell.is_crystal()) {
160 if (!first_cell)
161 first_cell = &rblock.cell;
162 } else if (first_cell) {
163 rblock.cell = *first_cell;
164 }
165 }
166 return rvec;
167}

169// Get the first (merged) block with required labels.
170// Optionally, block name can be specified.
171inline ReflnBlock get_refln_block(std::vector<cif::Block>&& blocks,
172 const std::vector<std::string>& labels,
173 const char* block_name=nullptr) {
174 const SpaceGroup* first_sg = nullptr;
175 for (cif::Block& block : blocks) {
176 if (!first_sg)
177 if (const std::string* hm = impl::find_spacegroup_hm_value(block)) {
178 first_sg = find_spacegroup_by_name(cif::as_string(*hm));
179 if (first_sg && first_sg->ext == 'H') {
180 UnitCell cell;
181 impl::set_cell_from_mmcif(block, cell);
182 first_sg = find_spacegroup_by_name(cif::as_string(*hm),
183 cell.alpha, cell.gamma);
184 }
185 }
186 if (block_name && block.name != block_name)
187 continue;
188 if (cif::Loop* loop = block.find_loop("_refln.index_h").get_loop())
189 if (std::all_of(labels.begin(), labels.end(),
190 [&](const std::string& s) { return loop->has_tag("_refln."+s); })) {
191 ReflnBlock rblock(std::move(block));
192 if (!rblock.spacegroup && first_sg)
193 rblock.spacegroup = first_sg;
194 return rblock;
195 }
196 }
197 fail("Required block or tags not found in the SF-mmCIF file.");
198}

200inline ReflnBlock hkl_cif_as_refln_block(cif::Block& block) {
201 ReflnBlock rb;
202 rb.block.swap(block);
203 rb.entry_id = rb.block.name;
204 impl::set_cell_from_mmcif(rb.block, rb.cell, /*mmcif=*/false);
205 const char* hm_tag = "_symmetry_space_group_name_H-M";
206 if (const std::string* hm = rb.block.find_value(hm_tag))
207 rb.spacegroup = find_spacegroup_by_name(cif::as_string(*hm),
208 rb.cell.alpha, rb.cell.gamma);
209 rb.cell.set_cell_images_from_spacegroup(rb.spacegroup);
210 rb.refln_loop = rb.block.find_loop("_refln_index_h").get_loop();
211 rb.default_loop = rb.refln_loop;
212 return rb;
213}
