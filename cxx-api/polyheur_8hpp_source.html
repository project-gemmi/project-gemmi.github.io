16// A simplistic classification. It may change in the future.
17// It returns PolymerType which corresponds to _entity_poly.type,
18// but here we use only PeptideL, Rna, Dna, DnaRnaHybrid and Unknown.
19inline PolymerType check_polymer_type(const ConstResidueSpan& polymer) {
20 if (polymer.empty())
21 return PolymerType::Unknown;
22 size_t counts[ResidueInfo::ELS+1] = {0};
23 size_t aa = 0;
24 size_t na = 0;
25 for (const Residue& r : polymer)
26 if (r.entity_type == EntityType::Unknown ||
27 r.entity_type == EntityType::Polymer) {
28 ResidueInfo info = find_tabulated_residue(r.name);
29 if (info.found())
30 counts[info.kind]++;
31 else if (r.get_ca())
32 ++aa;
33 else if (r.get_p())
34 ++na;
35 }
36 aa += counts[ResidueInfo::AA] + counts[ResidueInfo::AAD] +
37 counts[ResidueInfo::PAA] + counts[ResidueInfo::MAA];
38 na += counts[ResidueInfo::RNA] + counts[ResidueInfo::DNA];
39 if (2 * aa > polymer.size())
40 return counts[ResidueInfo::AA] >= counts[ResidueInfo::AAD]
41 ? PolymerType::PeptideL : PolymerType::PeptideD;
42 if (2 * na > polymer.size()) {
43 if (counts[ResidueInfo::DNA] == 0)
44 return PolymerType::Rna;
45 if (counts[ResidueInfo::RNA] == 0)
46 return PolymerType::Dna;
47 return PolymerType::DnaRnaHybrid;
48 }
49 return PolymerType::Unknown;
50}

52inline double calculate_sequence_weight(const std::vector<std::string>& seq,
53 double unknown=0.) {
54 double weight = 0.;
55 for (const std::string& item : seq) {
56 ResidueInfo res_info = find_tabulated_residue(Entity::first_mon(item));
57 weight += res_info.found() ? res_info.weight : unknown;
58 }
59 return weight - (seq.size() - 1) * h2o_weight();
60}

76inline bool is_polymer_residue(const Residue& res, PolymerType ptype) {
77 ResidueInfo info = find_tabulated_residue(res.name);
78 // If a standard residue is HETATM we assume that it is in the buffer.
79 if (info.found() && info.is_standard() && res.het_flag == 'H')
80 return false;
81 switch (ptype) {
82 case PolymerType::PeptideL:
83 case PolymerType::PeptideD:
84 // here we don't mind mixing D- and L- peptides
85 return info.found() ? info.is_amino_acid() : !!res.get_ca();
86 case PolymerType::Dna:
87 return info.found() ? info.is_dna() : !!res.get_p();
88 case PolymerType::Rna:
89 return info.found() ? info.is_rna() : !!res.get_p();
90 case PolymerType::DnaRnaHybrid:
91 return info.found() ? info.is_nucleic_acid() : !!res.get_p();
92 default:
93 return false;
94 }
95}

107inline bool are_connected(const Residue& r1, const Residue& r2, PolymerType ptype) {
108 if (is_polypeptide(ptype)) {
109 const Atom* a1 = r1.get_c();
110 const Atom* a2 = r2.get_n();
111 return a1 && a2 && a1->pos.dist_sq(a2->pos) < sq(1.341 * 1.5);
112 }
113 if (is_polynucleotide(ptype)) {
114 const Atom* a1 = r1.get_o3prim();
115 const Atom* a2 = r2.get_p();
116 return a1 && a2 && a1->pos.dist_sq(a2->pos) < sq(1.6 * 1.5);
117 }
118 return false;
119}

121// are_connected2() is less exact, but requires only CA (or P) atoms.
122inline bool are_connected2(const Residue& r1, const Residue& r2, PolymerType ptype) {
123 auto this_or_first = [](const Atom* a, const Residue& r) -> const Atom* {
124 return a || r.atoms.empty() ? a : &r.atoms.front();
125 };
126 if (is_polypeptide(ptype)) {
127 const Atom* a1 = this_or_first(r1.get_ca(), r1);
128 const Atom* a2 = this_or_first(r2.get_ca(), r2);
129 return a1 && a2 && a1->pos.dist_sq(a2->pos) < sq(5.0);
130 }
131 if (is_polynucleotide(ptype)) {
132 const Atom* a1 = this_or_first(r1.get_p(), r1);
133 const Atom* a2 = this_or_first(r2.get_p(), r2);
134 return a1 && a2 && a1->pos.dist_sq(a2->pos) < sq(7.5);
135 }
136 return false;
137}

139// are_connected3() = are_connected() + fallback to are_connected2()
140inline bool are_connected3(const Residue& r1, const Residue& r2, PolymerType ptype) {
141 if (is_polypeptide(ptype)) {
142 if (const Atom* a1 = r1.get_c())
143 if (const Atom* a2 = r2.get_n())
144 return a1->pos.dist_sq(a2->pos) < sq(1.341 * 1.5);
145 if (const Atom* a1 = r1.get_ca())
146 if (const Atom* a2 = r2.get_ca())
147 return a1->pos.dist_sq(a2->pos) < sq(5.0);
148 } else if (is_polynucleotide(ptype)) {
149 if (const Atom* a1 = r1.get_o3prim())
150 if (const Atom* a2 = r2.get_p())
151 return a1->pos.dist_sq(a2->pos) < sq(1.6 * 1.5);
152 if (const Atom* a1 = r1.get_p())
153 if (const Atom* a2 = r2.get_p())
154 return a1->pos.dist_sq(a2->pos) < sq(7.5);
155 }
156 return false;
157}

159inline std::string make_one_letter_sequence(const ConstResidueSpan& polymer) {
160 std::string seq;
161 const Residue* prev = nullptr;
162 PolymerType ptype = check_polymer_type(polymer);
163 for (const Residue& residue : polymer.first_conformer()) {
164 ResidueInfo info = find_tabulated_residue(residue.name);
165 if (prev && !are_connected2(*prev, residue, ptype))
166 seq += '-';
167 seq += (info.one_letter_code != ' ' ? info.one_letter_code : 'X');
168 prev = &residue;
169 }
170 return seq;
171}

178inline void add_entity_types(Chain& chain, bool overwrite) {
179 PolymerType ptype = check_polymer_type(chain.whole());
180 auto it = chain.residues.begin();
181 for (; it != chain.residues.end(); ++it)
182 if (overwrite || it->entity_type == EntityType::Unknown) {
183 if (!is_polymer_residue(*it, ptype))
184 break;
185 it->entity_type = EntityType::Polymer;
186 } else if (it->entity_type != EntityType::Polymer) {
187 break;
188 }
189 for (; it != chain.residues.end(); ++it)
190 if (overwrite || it->entity_type == EntityType::Unknown)
191 it->entity_type = it->is_water() ? EntityType::Water
192 : EntityType::NonPolymer;
193}

198 add_entity_types(chain, overwrite);

201// The subchain field in the residue is where we store_atom_site.label_asym_id
202// from mmCIF files. As of 2018 wwPDB software splits author's chains
203// (auth_asym_id) into label_asym_id units:
204// * linear polymer,
205// * non-polymers (each residue has different separate label_asym_id),
206// * and waters.
207// Refmac/makecif is doing similar thing but using different naming and
208// somewhat different rules (it was written in 1990's before PDBx/mmCIF).
209//
210// Here we use naming and rules different from both wwPDB and makecif.
211inline void assign_subchain_names(Chain& chain) {
212 for (Residue& res : chain.residues) {
213 res.subchain = chain.name;
214 switch (res.entity_type) {
215 case EntityType::Polymer: res.subchain += "poly"; break;
216 case EntityType::NonPolymer: res.subchain += res.seqid.str(); break;
217 case EntityType::Water: res.subchain += "wat"; break;
218 case EntityType::Branched: // FIXME
219 case EntityType::Unknown: break; // should not happen
220 }
221 }
222}

224inline void assign_subchains(Structure& st, bool force) {

227 if (force || !has_subchains_assigned(chain)) {

229 assign_subchain_names(chain);

233inline void ensure_entities(Structure& st) {
234 for (Model& model : st.models)
235 for (Chain& chain : model.chains)
236 for (ResidueSpan& sub : chain.subchains()) {
237 Entity* ent = st.get_entity_of(sub);
238 if (!ent) {
239 EntityType etype = sub[0].entity_type;
240 std::string name;
241 if (etype == EntityType::Polymer)
242 name = chain.name;
243 else if (etype == EntityType::NonPolymer)
244 name = sub[0].name + "!";
245 else if (etype == EntityType::Water)
246 name = "water";
247 if (!name.empty()) {
248 ent = &impl::find_or_add(st.entities, name);
249 ent->entity_type = etype;
250 ent->subchains.push_back(sub.subchain_id());
251 }
252 }
253 // ensure we have polymer_type set where needed
254 if (ent && ent->entity_type == EntityType::Polymer &&
255 ent->polymer_type == PolymerType::Unknown)
256 ent->polymer_type = check_polymer_type(sub);
257 }
258}

268inline void deduplicate_entities(Structure& st) {
269 for (auto i = st.entities.begin(); i != st.entities.end(); ++i)
270 if (!i->full_sequence.empty())
271 for (auto j = i + 1; j != st.entities.end(); ++j)
272 if (j->polymer_type == i->polymer_type &&
273 j->full_sequence == i->full_sequence &&
274 j->dbrefs == i->dbrefs) {
275 vector_move_extend(i->subchains, std::move(j->subchains));
276 st.entities.erase(j--);
277 }
278}

289 for (auto& child : obj.children())

297// Remove ligands and waters. It may leave empty chains.
298template<class T> void remove_ligands_and_waters(T& obj) {
299 for (auto& child : obj.children())
300 remove_ligands_and_waters(child);
301}
302template<> inline void remove_ligands_and_waters(Chain& ch) {
303 PolymerType ptype = check_polymer_type(ch.whole());
304 vector_remove_if(ch.residues, [&](const Residue& res) {
305 if (res.entity_type == EntityType::Unknown) {
306 // TODO: check connectivity
307 return !is_polymer_residue(res, ptype);
308 }
309 return res.entity_type != EntityType::Polymer;
310 });
311}

313// Trim to alanine. Returns true if trimmed, false if it's (likely) not AA.
314inline bool trim_to_alanine(Residue& res) {
315 static const std::pair<std::string, El> ala_atoms[6] = {
316 {"N", El::N}, {"CA", El::C}, {"C", El::C}, {"O", El::O}, {"CB", El::C},
317 {"OXT", El::O}
318 };
319 if (res.get_ca() == nullptr)
320 return false;
321 vector_remove_if(res.atoms, [](const Atom& a) {
322 for (const auto& name_el : ala_atoms)
323 if (a.name == name_el.first && a.element == name_el.second)
324 return false;
325 return true;
326 });
327 return true;
328}
329
330inline void trim_to_alanine(Chain& chain) {
331 for (Residue& res : chain.residues)
332 trim_to_alanine(res);
333}
