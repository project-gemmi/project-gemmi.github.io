51template<class T> CenterOfMass calculate_center_of_mass(const T& obj) {
52 CenterOfMass total{{}, 0.};
53 for (const auto& child : obj.children()) {
54 CenterOfMass part = calculate_center_of_mass(child);
55 total = {total.weighted_sum + part.weighted_sum, total.mass + part.mass};
56 }
57 return total;
58}
59template<> inline CenterOfMass calculate_center_of_mass(const Atom& atom) {
60 double w_mass = atom.element.weight() * atom.occ;
61 return CenterOfMass{Position(atom.pos * w_mass), w_mass};
62}

64template<class T> std::pair<float,float> calculate_b_iso_range(const T& obj) {
65 std::pair<float, float> range{INFINITY, -INFINITY};
66 for (const auto& child : obj.children()) {
67 auto r = calculate_b_iso_range(child);
68 range.first = std::min(range.first, r.first);
69 range.second = std::max(range.second, r.second);
70 }
71 return range;

78inline std::pair<double, double> calculate_b_aniso_range(const Model& model) {
79 std::pair<double, double> range{INFINITY, -INFINITY};
80 for (const Chain& chain : model.chains)
81 for (const Residue& residue : chain.residues)
82 for (const Atom& atom : residue.atoms) {
83 if (atom.occ == 0)
84 continue;
85 if (atom.aniso.nonzero()) {
86 std::array<double,3> eig = atom.aniso.calculate_eigenvalues();
87 auto u = std::minmax({eig[0], eig[1], eig[2]});
88 range.first = std::min(range.first, u.first * u_to_b());
89 range.second = std::max(range.second, u.second * u_to_b());
90 } else {
91 range.first = std::min(range.first, (double) atom.b_iso);
92 range.second = std::max(range.second, (double) atom.b_iso);
93 }
94 }
95 return range;
96}

116inline Box<Fractional> calculate_fractional_box(const Structure& st, double margin=0.) {
117 if (!st.cell.is_crystal())
118 fail("calculate_fractional_box(): Structure has no unit cell for fractionalization");
119 Box<Fractional> box;
120 for (const Model& model : st.models)
121 for (const Chain& chain : model.chains)
122 for (const Residue& res : chain.residues)
123 for (const Atom& atom : res.atoms)
124 box.extend(st.cell.fractionalize(atom.pos));
125 if (margin != 0.)
126 box.add_margins({margin * st.cell.ar, margin * st.cell.br, margin * st.cell.cr});
127 return box;
128}

131// Calculate B_est from E. Merritt, Some B_eq are more equivalent than others,
132// Acta Cryst. A67, 512 (2011)
133// http://skuld.bmsc.washington.edu/parvati/ActaA_67_512.pdf
134inline double calculate_b_est(const Atom& atom) {
135 auto eig = atom.aniso.calculate_eigenvalues();
136 return u_to_b() * std::sqrt((eig[0] + eig[1] + eig[2]) /
137 (1/eig[0] + 1/eig[1] + 1/eig[2]));
138}

145// discussion: https://stackoverflow.com/questions/20305272/
146inline double calculate_dihedral(const Position& p0, const Position& p1,
147 const Position& p2, const Position& p3) {
148 Vec3 b0 = p1 - p0;
149 Vec3 b1 = p2 - p1;
150 Vec3 b2 = p3 - p2;
151 Vec3 u = b1.cross(b0);
152 Vec3 w = b2.cross(b1);
153 double y = u.cross(w).dot(b1);
154 double x = u.dot(w) * b1.length();
155 return std::atan2(y, x);
156}

183inline std::array<double, 2> calculate_phi_psi(const Residue* prev,
184 const Residue& res,
185 const Residue* next) {
186 std::array<double, 2> phi_psi{{NAN, NAN}};
187 if (prev || next) {
188 const Atom* CA = res.get_ca();
189 const Atom* C = res.get_c();
190 const Atom* N = res.get_n();
191 if (prev)
192 phi_psi[0] = calculate_dihedral_from_atoms(prev->get_c(), N, CA, C);
193 if (next)
194 phi_psi[1] = calculate_dihedral_from_atoms(N, CA, C, next->get_n());
195 }
196 return phi_psi;
197}
