59template<class T> CenterOfMass calculate_center_of_mass(const T& obj) {
60 CenterOfMass total{{}, 0.};
61 for (const auto& child : obj.children()) {
62 CenterOfMass part = calculate_center_of_mass(child);
63 total = {total.weighted_sum + part.weighted_sum, total.mass + part.mass};
64 }
65 return total;
66}
67template<> inline CenterOfMass calculate_center_of_mass(const Atom& atom) {
68 double w_mass = atom.element.weight() * atom.occ;
69 return CenterOfMass{Position(atom.pos * w_mass), w_mass};
70}

72inline Box<Position> calculate_box(const Structure& st, double margin=0.) {
73 Box<Position> box;
74 for (const Model& model : st.models)
75 for (const Chain& chain : model.chains)
76 for (const Residue& res : chain.residues)
77 for (const Atom& atom : res.atoms)
78 box.extend(atom.pos);
79 if (margin != 0.)
80 box.add_margin(margin);

97// Calculate B_est from E. Merritt, Some B_eq are more equivalent than others,
98// Acta Cryst. A67, 512 (2011)
99// http://skuld.bmsc.washington.edu/parvati/ActaA_67_512.pdf
100inline double calculate_b_est(const Atom& atom) {
101 auto eig = atom.aniso.calculate_eigenvalues();
102 return u_to_b() * std::sqrt((eig[0] + eig[1] + eig[2]) /
103 (1/eig[0] + 1/eig[1] + 1/eig[2]));
104}

111// discussion: https://stackoverflow.com/questions/20305272/
112inline double calculate_dihedral(const Position& p0, const Position& p1,
113 const Position& p2, const Position& p3) {
114 Vec3 b0 = p1 - p0;
115 Vec3 b1 = p2 - p1;
116 Vec3 b2 = p3 - p2;
117 Vec3 u = b1.cross(b0);
118 Vec3 w = b2.cross(b1);
119 double y = u.cross(w).dot(b1);
120 double x = u.dot(w) * b1.length();
121 return std::atan2(y, x);
122}

141inline std::array<double, 2> calculate_phi_psi(const Residue* prev,
142 const Residue& res,
143 const Residue* next) {
144 std::array<double, 2> phi_psi{{NAN, NAN}};
145 if (prev || next) {
146 const Atom* CA = res.get_ca();
147 const Atom* C = res.get_c();
148 const Atom* N = res.get_n();
149 if (prev)
150 phi_psi[0] = calculate_dihedral_from_atoms(prev->get_c(), N, CA, C);
151 if (next)
152 phi_psi[1] = calculate_dihedral_from_atoms(N, CA, C, next->get_n());
153 }
154 return phi_psi;
155}

157inline std::array<double, 4> find_best_plane(const std::vector<Atom*>& atoms) {
158 Vec3 mean;
159 for (const Atom* atom : atoms)
160 mean += atom->pos;
161 mean /= (double) atoms.size();
162 SMat33<double> m{0, 0, 0, 0, 0, 0};
163 for (const Atom* atom : atoms) {
164 Vec3 p = Vec3(atom->pos) - mean;
165 m.u11 += p.x * p.x;
166 m.u22 += p.y * p.y;
167 m.u33 += p.z * p.z;
168 m.u12 += p.x * p.y;
169 m.u13 += p.x * p.z;
170 m.u23 += p.y * p.z;
171 }
172 auto eig = m.calculate_eigenvalues();
173 double smallest = eig[0];
174 for (double d : {eig[1], eig[2]})
175 if (std::fabs(d) < std::fabs(smallest))
176 smallest = d;
177 Vec3 eigvec = m.calculate_eigenvector(smallest);
178 if (eigvec.x < 0)
179 eigvec *= -1;
180 return {{eigvec.x, eigvec.y, eigvec.z, -eigvec.dot(mean)}};
181}
