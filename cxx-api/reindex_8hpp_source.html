14inline void reindex_mtz(Mtz& mtz, const Op& op, std::ostream* out) {
15 mtz.switch_to_original_hkl();
16 Op real_space_op{op.transposed_rot(), {0, 0, 0}};
17 if (out)
18 *out << "Real space transformation: " << real_space_op.triplet() << '\n';
19 size_t replace_row = size_t(-1);
20 // change Miller indices
21 for (size_t n = 0; n < mtz.data.size(); n += mtz.columns.size()) {
22 Miller hkl_den = real_space_op.apply_to_hkl_without_division(mtz.get_hkl(n));
23 Miller hkl = Op::divide_hkl_by_DEN(hkl_den);
24 if (hkl[0] * Op::DEN == hkl_den[0] &&
25 hkl[1] * Op::DEN == hkl_den[1] &&
26 hkl[2] * Op::DEN == hkl_den[2]) {
27 mtz.set_hkl(n, hkl);
28 } else {
29 if (replace_row == size_t(-1))
30 replace_row = n;
31 mtz.data[n] = NAN; // mark for removal
32 }
33 }
34 // remove reflections that came out fractional
35 if (replace_row != size_t(-1)) {
36 size_t ncol = mtz.columns.size();
37 for (size_t n = replace_row + ncol; n < mtz.data.size(); n += ncol)
38 if (!std::isnan(mtz.data[n])) {
39 std::memcpy(mtz.data.data() + replace_row,
40 mtz.data.data() + n,
41 ncol * sizeof(float));
42 replace_row += ncol;
43 }
44 if (out)
45 *out << "Reflections removed (because of fractional indices): "
46 << (mtz.data.size() - replace_row) / ncol << '\n';
47 mtz.data.resize(replace_row);
48 mtz.nreflections = int(replace_row / ncol);
49 }
50 // change space group
51 const SpaceGroup* sg_before = mtz.spacegroup;
52 if (sg_before) {
53 GroupOps gops = sg_before->operations();
54 gops.change_basis_backward(op);
55 mtz.spacegroup = find_spacegroup_by_ops(gops);
56 }
57
58 if (mtz.spacegroup) {
59 if (mtz.spacegroup != sg_before) {
60 if (out)
61 *out << "Space group changed from " << sg_before->xhm() << " to "
62 << mtz.spacegroup->xhm() << ".\n";
63 mtz.spacegroup_number = mtz.spacegroup->ccp4;
64 mtz.spacegroup_name = mtz.spacegroup->hm;
65 } else {
66 if (out)
67 *out << "Space group stays the same:" << sg_before->xhm() << ".\n";
68 }
69 } else {
70 fail("reindexing: failed to determine new space group name");
71 }
72 // change unit cell parameters
73 mtz.cell = mtz.cell.changed_basis_backward(real_space_op, false);
74 for (Mtz::Dataset& ds : mtz.datasets)
75 ds.cell = ds.cell.changed_basis_backward(real_space_op, false);
76 for (Mtz::Batch& batch : mtz.batches)
77 batch.set_cell(batch.get_cell().changed_basis_backward(real_space_op, false));
78
79 if (mtz.is_merged())
80 mtz.ensure_asu();
81 else
82 mtz.switch_to_asu_hkl();
83}
