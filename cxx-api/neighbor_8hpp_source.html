134 for_each_cell(pos, [&](std::vector<Mark>& marks, const Fractional& fr) {
135 Position p = grid.unit_cell.orthogonalize(fr);
136 for (Mark& m : marks) {

172 void set_bounding_cell(const UnitCell& cell) {
173 if (cell.is_crystal()) {
174 grid.unit_cell = cell;
175 } else {
176 // cf. calculate_box()
177 Box<Position> box;
178 for (CRA cra : model->all())
179 box.extend(cra.atom->pos);
180 // The box needs to include all NCS images (strict NCS from MTRIXn).
181 // To avoid additional function parameter that would pass Structure::ncs,
182 // here we obtain NCS transformations from UnitCell::images.
183 std::vector<FTransform> ncs = cell.get_ncs_transforms();
184 if (!ncs.empty()) {
185 for (CRA cra : model->all())
186 // images store fractional transforms, but for non-crystal
187 // it should be the same as Cartesian transform.
188 for (const Transform& tr : ncs)
189 box.extend(Position(tr.apply(cra.atom->pos)));
190 }
191 box.add_margin(1.5 * radius_specified); // much more than needed
192 Position size = box.get_size();
193 grid.unit_cell.set(size.x, size.y, size.z, 90, 90, 90);
194 for (const Transform& tr : ncs) {
195 UnitCell& c = grid.unit_cell;
196 // cf. add_ncs_images_to_cs_images()
197 c.images.push_back(c.frac.combine(tr.combine(c.orth)));
198 }
199 }

203inline NeighborSearch& NeighborSearch::populate(bool include_h_) {
204 include_h = include_h_;
205 if (model) {
206 for (int n_ch = 0; n_ch != (int) model->chains.size(); ++n_ch)
207 add_chain_n(model->chains[n_ch], n_ch);
208 } else if (small_structure) {
209 for (int n = 0; n != (int) small_structure->sites.size(); ++n) {
210 SmallStructure::Site& site = small_structure->sites[n];
211 if (include_h || !site.element.is_hydrogen())
212 add_site(site, n);
213 }
214 } else {
215 fail("NeighborSearch not initialized");
216 }
217 return *this;
218}

220inline void NeighborSearch::add_chain(const Chain& chain, bool include_h_) {
221 if (!model)
222 fail("NeighborSearch.add_chain(): model not initialized yet");
223 // to be safe avoid (&chain - model.chains[0]) which could be UB
224 for (int n_ch = 0; n_ch != (int) model->chains.size(); ++n_ch)
225 if (&model->chains[n_ch] == &chain) {
226 include_h = include_h_;

244inline void NeighborSearch::add_atom(const Atom& atom,
245 int n_ch, int n_res, int n_atom) {
246 const UnitCell& gcell = grid.unit_cell;
247 Fractional frac0 = gcell.fractionalize(atom.pos);
248 {
249 Fractional frac = frac0.wrap_to_unit();
250 Position pos = gcell.orthogonalize(frac);
251 get_subcell(frac).emplace_back(pos, atom.altloc, atom.element.elem,
252 0, n_ch, n_res, n_atom);
253 }
254 for (int n_im = 0; n_im != (int) gcell.images.size(); ++n_im) {
255 Fractional frac = gcell.images[n_im].apply(frac0).wrap_to_unit();
256 Position pos = gcell.orthogonalize(frac);
257 get_subcell(frac).emplace_back(pos, atom.altloc, atom.element.elem,
258 n_im + 1, n_ch, n_res, n_atom);
259 }
260}

262// We exclude special position images of atoms here, but not in add_atom.
263// This choice is somewhat arbitrary, but it also reflects the fact that
264// in MX files occupances of atoms on special positions are (almost always)
265// fractional and all images are to be taken into account.
266inline void NeighborSearch::add_site(const SmallStructure::Site& site, int n) {
267 const double SPECIAL_POS_TOL = 0.4;
268 const UnitCell& gcell = grid.unit_cell;
269 std::vector<Fractional> others;
270 others.reserve(gcell.images.size());
271 Fractional frac0 = site.fract.wrap_to_unit();
272 {
273 Position pos = gcell.orthogonalize(frac0);
274 get_subcell(frac0).emplace_back(pos, '\0', site.element.elem, 0, -1, -1, n);
275 }
276 for (int n_im = 0; n_im != (int) gcell.images.size(); ++n_im) {
277 Fractional frac = gcell.images[n_im].apply(site.fract).wrap_to_unit();
278 if (gcell.distance_sq(frac, frac0) < sq(SPECIAL_POS_TOL) ||
279 std::any_of(others.begin(), others.end(), [&](const Fractional& f) {
280 return gcell.distance_sq(frac, f) < sq(SPECIAL_POS_TOL);
281 }))
282 continue;
283 Position pos = gcell.orthogonalize(frac);
284 get_subcell(frac).emplace_back(pos, '\0', site.element.elem,
285 n_im + 1, -1, -1, n);
286 others.push_back(frac);
287 }

290template<typename Func>
291void NeighborSearch::for_each_cell(const Position& pos, const Func& func) {
292 Fractional fr = grid.unit_cell.fractionalize(pos).wrap_to_unit();
293 const int u0 = int(fr.x * grid.nu);
294 const int v0 = int(fr.y * grid.nv);
295 const int w0 = int(fr.z * grid.nw);
296 const int uend = u0 + std::min(3, grid.nu) - 1;
297 const int vend = v0 + std::min(3, grid.nv) - 1;
298 const int wend = w0 + std::min(3, grid.nw) - 1;
299 for (int w = w0 - 1; w < wend; ++w) {
300 int dw = w >= grid.nw ? -1 : w < 0 ? 1 : 0;
301 for (int v = v0 - 1; v < vend; ++v) {
302 int dv = v >= grid.nv ? -1 : v < 0 ? 1 : 0;
303 for (int u = u0 - 1; u < uend; ++u) {
304 int du = u >= grid.nu ? -1 : u < 0 ? 1 : 0;
305 size_t idx = grid.index_q(u + du * grid.nu,
306 v + dv * grid.nv,
307 w + dw * grid.nw);
308 func(grid.data[idx], Fractional(fr.x + du, fr.y + dv, fr.z + dw));
309 }
310 }
311 }

314template<typename Func>
315void NeighborSearch::for_each(const Position& pos, char alt, float radius,
316 const Func& func) {
317 if (radius <= 0.f)
318 return;
319 for_each_cell(pos, [&](std::vector<Mark>& marks, const Fractional& fr) {
320 Position p = grid.unit_cell.orthogonalize(fr);
321 for (Mark& m : marks) {
322 float dist_sq = m.dist_sq_(p);
323 if (dist_sq < sq(radius) && is_same_conformer(alt, m.altloc))
324 func(m, dist_sq);
325 }
326 });
327}

330inline void remove_cras(Model& model, std::vector<CRA>& vec) {
331 // sort in reverse order, so items can be erased without invalidating pointers
332 std::sort(vec.begin(), vec.end(), [](const CRA& a, const CRA& b) {
333 return std::tie(a.chain, a.residue, a.atom) > std::tie(b.chain, b.residue, b.atom);
334 });
335 const Atom* prev_a = nullptr;
336 for (CRA& cra : vec) {
337 if (cra.atom == prev_a)
338 continue;
339 prev_a = cra.atom;
340 auto atom_idx = cra.atom - cra.residue->atoms.data();
341 cra.residue->atoms.erase(cra.residue->atoms.begin() + atom_idx);
342 if (cra.residue->atoms.empty()) {
343 auto res_idx = cra.residue - cra.chain->residues.data();
344 cra.chain->residues.erase(cra.chain->residues.begin() + res_idx);
345 if (cra.chain->residues.empty()) {
346 auto chain_idx = cra.chain - model.chains.data();
347 model.chains.erase(model.chains.begin() + chain_idx);
348 }
349 }
350 }
351}

353// To be used after expand_ncs() and make_assembly().
354// Searches and merges overlapping equivalent atoms from different chains.
355inline void merge_atoms_in_expanded_model(Model& model, const UnitCell& cell,
356 double max_dist=0.2) {
357 using Mark = NeighborSearch::Mark;
358 NeighborSearch ns(model, cell, 4.0);
359 ns.populate(true);
360 std::vector<CRA> to_be_deleted;
361 for (int n_ch = 0; n_ch != (int) model.chains.size(); ++n_ch) {
362 Chain& chain = model.chains[n_ch];
363 for (int n_res = 0; n_res != (int) chain.residues.size(); ++n_res) {
364 Residue& res = chain.residues[n_res];
365 for (int n_atom = 0; n_atom != (int) res.atoms.size(); ++n_atom) {
366 Atom& atom = res.atoms[n_atom];
367 std::vector<std::pair<CRA, int>> equiv;
368 ns.for_each_cell(atom.pos, [&](std::vector<Mark>& marks, const Fractional& fr) {
369 for (Mark& m : marks) {
370 // We look for the same atoms, but copied to a different chain.
371 // First quick check that filters out most of non-matching pairs.
372 if (m.altloc != atom.altloc || m.element != atom.element ||
373 m.chain_idx == n_ch || m.atom_idx != n_atom)
374 continue;
375 // Now check if everything else matches.
376 CRA cra = m.to_cra(model);
377 if (cra.atom &&
378 cra.atom->serial == atom.serial &&
379 cra.atom->name == atom.name &&
380 cra.atom->b_iso == atom.b_iso &&
381 cra.residue->matches_noseg(res) &&
382 m.dist_sq_(ns.grid.unit_cell.orthogonalize(fr)) < sq(max_dist))
383 equiv.emplace_back(cra, m.image_idx);
384 }
385 });
386 if (!equiv.empty()) {
387 Position pos_sum = atom.pos;
388 for (auto& t : equiv) {
389 CRA& cra = t.first;
390 pos_sum += ns.grid.unit_cell.find_nearest_pbc_position(
391 atom.pos, cra.atom->pos, t.second);
392 // The atoms in equiv are to be discarded later.
393 // Deleting now would invalidate indices in NeighborSearch.
394 to_be_deleted.push_back(cra);
395 // Modify the atoms to avoid processing them again.
396 cra.atom->serial = -1; // this should be enough
397 cra.atom->name.clear(); // this is just in case
398 }
399 size_t n = 1 + equiv.size();
400 atom.pos = pos_sum / n;
401 atom.occ = std::min(1.f, n * atom.occ);
402 }
403 }
404 }
405 }
406 remove_cras(model, to_be_deleted);
407}

411template<typename T>
412void interpolate_grid_of_aligned_model(Grid<T>& dest, const Grid<T>& src,
413 const Transform& tr, NeighborSearch& ns,
414 double radius=0.) {
415 if (radius <= 0.)
416 radius = ns.radius_specified;
417 else if (radius > ns.radius_specified)
418 fail("set_grid_values_interpolated_from(): radius exceeds NeighborSearch radius");
419 // mask model or its part that was used for NeighborSearch
420 std::vector<bool> mask(dest.data.size(), false);
421 for (const std::vector<NeighborSearch::Mark>& marks : ns.grid.data)
422 for (const NeighborSearch::Mark& mark : marks)
423 if (mark.image_idx == 0) // leave out symmetry mates
424 dest.template use_points_around<true>(
425 ns.grid.unit_cell.fractionalize(mark.pos()),
426 radius,
427 [&](T& point, double) { mask[&point - dest.data.data()] = true; });
428 size_t idx = 0;
429 for (int w = 0; w != dest.nw; ++w)
430 for (int v = 0; v != dest.nv; ++v)
431 for (int u = 0; u != dest.nu; ++u, ++idx) {
432 if (!mask[idx])
433 continue;
434 Position pos2 = dest.get_position(u, v, w);
435 // in contacts use only nodes that are nearer to the original molecule
436 NeighborSearch::Mark* mark = ns.find_nearest_atom(pos2);
437 if (mark && mark->image_idx == 0) {
438 Position delta = mark->to_cra(*ns.model).atom->pos - mark->pos();
439 Position pos1 = Position(tr.apply(pos2 + delta));
440 dest.data[idx] = src.interpolate_value(pos1);
441 }
442 }
443}
