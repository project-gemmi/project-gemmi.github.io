42template<typename T, typename S>
43T* find_or_null(std::vector<T>& vec, const S& name) {
44 auto it = find_iter_(vec, name);
45 return it != vec.end() ? &*it : nullptr;
46}

48template<typename T, typename S>
49T& find_or_add(std::vector<T>& vec, const S& name) {
50 if (T* ret = find_or_null(vec, name))
51 return *ret;
52 vec.emplace_back(name);
53 return vec.back();
54}

56template<typename Span, typename S>
57typename Span::iterator find_iter(Span& span, const S& name) {
58 using T = typename Span::value_type;
59 auto it = find_iter_(span, name);
60 if (it == span.end())
61 throw std::invalid_argument(cat(
62 T::what(), ' ', name, " not found (only [",
63 join_str(span.begin(), span.end(), ' ', [](const T& item) { return cat(get_id(item)); }),
64 "])"));
65 return it;
66}

122 char altloc = '\0'; // 0 if not set
123 signed char charge = 0; // [-8, +8]
124 Element element = El::X;
125 CalcFlag calc_flag = CalcFlag::NotSet; // mmCIF _atom_site.calc_flag
126 char flag = '\0'; // a custom flag
127 short tls_group_id = -1;
128 int serial = 0;
129 float fraction = 0.f; // custom value, one use is Refmac's ccp4_deuterium_fraction
130 Position pos;
131 float occ = 1.0f;
132 // ADP - in MX it's usual to give isotropic ADP as B and anisotropic as U
133 float b_iso = 20.0f; // arbitrary default value

136 char altloc_or(char null_char) const { return altloc ? altloc : null_char; }

145 bool has_altloc() const { return altloc != '\0'; }

150 // the first two characters make the element name.

152 std::string s;

154 if (el[1] == '\0' &&
155 (el[0] == alpha_up(name[0]) || (is_hydrogen() && alpha_up(name[0]) == 'H')) &&

157 s += ' ';

340 // We assume residues are ordered. It works (approximately) also with
341 // missing numbers which can be present in DBREF.
342 SeqId label_seq_id_to_auth(SeqId::OptionalNum label_seq_id) const {
343 if (empty())
344 throw std::out_of_range("label_seq_id_to_auth(): empty span");
345 const auto* it = std::lower_bound(begin(), end(), label_seq_id,
346 [](const Residue& r, SeqId::OptionalNum v){ return r.label_seq < v; });
347 if (it == end())
348 --it;
349 else if (it->label_seq == label_seq_id)
350 return it->seqid;
351 else if (it != begin() &&
352 label_seq_id - (it-1)->label_seq < it->label_seq - label_seq_id)
353 --it;
354 return {it->seqid.num + (label_seq_id - it->label_seq), ' '};
355 }
356 // The residue numbers (auth) are sometimes not ordered.
357 // That is why we use this multi-step heuristic.
358 SeqId::OptionalNum auth_seq_id_to_label(SeqId auth_seq_id) const {
359 if (empty())
360 throw std::out_of_range("auth_seq_id_to_label(): empty span");
361 for (const Residue& r : *this)
362 if (r.seqid == auth_seq_id)
363 return r.label_seq;
364 const_iterator it;
365 if (auth_seq_id.num < front().seqid.num) {
366 it = begin();
367 } else if (back().seqid.num < auth_seq_id.num) {
368 it = end() - 1;
369 } else {
370 it = std::lower_bound(begin(), end(), auth_seq_id.num,
371 [](const Residue& r, SeqId::OptionalNum v){ return r.seqid.num < v; });
372 while (it != end() && it->seqid.num == auth_seq_id.num &&
373 it->seqid.icode != auth_seq_id.icode)
374 ++it;
375 if (it == end())
376 --it;
377 }
378 return it->label_seq + (auth_seq_id.num - it->seqid.num);
379 }

646template<typename CraT>
647class CraIterPolicy {
648public:
649 using value_type = CraT;
650 using reference = const CraT;
651 CraIterPolicy() : chains_end(nullptr), cra{nullptr, nullptr, nullptr} {}
652 CraIterPolicy(const Chain* end, CraT cra_) : chains_end(end), cra(cra_) {}
653 void increment() {
654 if (cra.atom == nullptr)
655 return;
656 if (++cra.atom == vector_end_ptr(cra.residue->atoms)) {
657 do {
658 if (++cra.residue == vector_end_ptr(cra.chain->residues)) {
659 do {
660 if (++cra.chain == chains_end) {
661 cra.atom = nullptr;
662 return;
663 }
664 } while (cra.chain->residues.empty());
665 cra.residue = &cra.chain->residues[0];
666 }
667 } while (cra.residue->atoms.empty());
668 cra.atom = &cra.residue->atoms[0];
669 }
670 }
671 void decrement() {
672 while (cra.atom == nullptr || cra.atom == cra.residue->atoms.data()) {
673 while (cra.residue == nullptr || cra.residue == cra.chain->residues.data()) {
674 // iterating backward beyond begin() will have undefined effects
675 while ((--cra.chain)->residues.empty()) {}
676 cra.residue = vector_end_ptr(cra.chain->residues);
677 }
678 --cra.residue;
679 cra.atom = vector_end_ptr(cra.residue->atoms);
680 }
681 --cra.atom;
682 }
683 bool equal(const CraIterPolicy& o) const { return cra.atom == o.cra.atom; }
684 CraT dereference() { return cra; } // make copy b/c increment() modifies cra
685 using const_policy = CraIterPolicy<const_CRA>;
686 operator const_policy() const { return const_policy(chains_end, cra); }
687private:
688 const Chain* chains_end;
689 CraT cra;
690};

693struct CraProxy_ {
694 ChainsRefT chains;
695 using iterator = BidirIterator<CraIterPolicy<CraT>>;

698 for (auto& residue : chain.residues)
699 for (auto& atom : residue.atoms)
700 return CraIterPolicy<CraT>{vector_end_ptr(chains), CraT{&chain, &residue, &atom}};
701 return {};
702 }
703 iterator end() {
704 auto* chains_end = vector_end_ptr(chains);
