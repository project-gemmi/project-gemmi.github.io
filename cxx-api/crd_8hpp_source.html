18inline bool has_anisou(const Model& model) {
19 for (const Chain& chain : model.chains)
20 for (const Residue& res : chain.residues)
21 for (const Atom& a : res.atoms)
22 if (a.aniso.nonzero())
23 return true;
24 return false;
25}

27inline std::string refmac_calc_flag(const Atom& a) {
28 switch (a.calc_flag) {
29 case CalcFlag::NotSet: return ".";
30 // Refmac seems to be using only flags R and M
31 case CalcFlag::Calculated: return a.is_hydrogen() ? "R" : "M";
32 // I think these are not used
33 case CalcFlag::Determined: return "d";
34 case CalcFlag::Dummy: return "dum";
35 }
36 unreachable();
37}

39// Get value of _entity_poly_seq.ccp4_mod_id compatible with makecif.
40inline std::string get_ccp4_mod_id(const std::vector<std::string>& mods) {
41 for (const std::string& m : mods)
42 if (!starts_with(m, "DEL-OXT") &&
43 !starts_with(m, "DEL-HN") &&
44 m != "DEL-NMH")
45 return m;
46 return ".";
47}

49inline cif::Block prepare_crd(const Structure& st, const Topo& topo) {
50 auto e_id = st.info.find("_entry.id");
51 std::string id = cif::quote(e_id != st.info.end() ? e_id->second : st.name);
52 cif::Block block("structure_" + id);
53 auto& items = block.items;
54
55 items.emplace_back("_entry.id", id);
56 items.emplace_back("_database_2.code_PDB", id);
57 auto keywords = st.info.find("_struct_keywords.pdbx_keywords");
58 if (keywords != st.info.end())
59 items.emplace_back("_struct_keywords.text", cif::quote(keywords->second));
60 auto title = st.info.find("_struct.title");
61 if (title != st.info.end())
62 items.emplace_back(title->first, cif::quote(title->second));
63 auto initial_date =
64 st.info.find("_pdbx_database_status.recvd_initial_deposition_date");
65 if (initial_date != st.info.end())
66 items.emplace_back("_audit.creation_date", initial_date->second);
67 items.emplace_back("_software.name", "gemmi");
68
69 items.emplace_back(cif::CommentArg{"############\n"
70 "## ENTITY ##\n"
71 "############"});
72 cif::Loop& entity_loop = block.init_mmcif_loop("_entity.", {"id", "type"});
73 for (const Entity& ent : st.entities)
74 entity_loop.add_row({ent.name, entity_type_to_string(ent.entity_type)});
75 items.emplace_back(cif::CommentArg{"#####################\n"
76 "## ENTITY_POLY_SEQ ##\n"
77 "#####################"});
78 cif::Loop& poly_loop = block.init_mmcif_loop("_entity_poly_seq.", {
79 "mon_id", "ccp4_auth_seq_id", "entity_id",
80 "ccp4_back_connect_type", "ccp4_num_mon_back", "ccp4_mod_id"});
81 for (const Topo::ChainInfo& chain_info : topo.chain_infos) {
82 if (!chain_info.polymer)
83 continue;
84 for (const Topo::ResInfo& ri : chain_info.res_infos) {
85 const Topo::Link* prev = ri.prev.empty() ? nullptr : &ri.prev[0];
86 poly_loop.add_row({ri.res->name,
87 ri.res->seqid.str(),
88 chain_info.entity_id,
89 prev ? prev->link_id : ".",
90 prev ? prev->res1->seqid.str() : "n/a",
91 get_ccp4_mod_id(ri.mods)});
92 }
93 }
94 items.emplace_back(cif::CommentArg{"##########\n"
95 "## CELL ##\n"
96 "##########"});
97 items.emplace_back("_cell.entry_id", id);
98 items.emplace_back("_cell.length_a", to_str(st.cell.a));
99 items.emplace_back("_cell.length_b", to_str(st.cell.b));
100 items.emplace_back("_cell.length_c", to_str(st.cell.c));
101 items.emplace_back("_cell.angle_alpha", to_str(st.cell.alpha));
102 items.emplace_back("_cell.angle_beta", to_str(st.cell.beta));
103 items.emplace_back("_cell.angle_gamma", to_str(st.cell.gamma));
104 bool write_fract_matrix = true;
105 if (write_fract_matrix) {
106 items.emplace_back(cif::CommentArg{"##############################\n"
107 "## FRACTIONALISATION MATRIX ##\n"
108 "##############################"});
109 std::string prefix = "_atom_sites.fract_transf_";
110 for (int i = 0; i < 3; ++i) {
111 std::string start = prefix + "matrix[" + std::to_string(i+1) + "][";
112 const auto& row = st.cell.frac.mat[i];
113 for (int j = 0; j < 3; ++j)
114 items.emplace_back(start + std::to_string(j+1) + "]", to_str(row[j]));
115 }
116 for (int i = 0; i < 3; ++i)
117 items.emplace_back(prefix + "vector[" + std::to_string(i + 1) + "]",
118 to_str(st.cell.frac.vec.at(i)));
119 }
120
121 items.emplace_back(cif::CommentArg{"##############\n"
122 "## SYMMETRY ##\n"
123 "##############"});
124 items.emplace_back("_symmetry.entry_id", id);
125 const std::string& hm = st.spacegroup_hm;
126 items.emplace_back("_symmetry.space_group_name_H-M", cif::quote(hm));
127 if (const SpaceGroup* sg = st.find_spacegroup())
128 items.emplace_back("_symmetry.Int_Tables_number",
129 std::to_string(sg->number));
130 const Model& model0 = st.first_model();
131 items.emplace_back(cif::CommentArg{"#################\n"
132 "## STRUCT_ASYM ##\n"
133 "#################"});
134 cif::Loop& asym_loop = block.init_mmcif_loop("_struct_asym.",
135 {"id", "entity_id"});
136 for (const Chain& chain : model0.chains)
137 for (ConstResidueSpan sub : chain.subchains())
138 if (!sub.subchain_id().empty()) {
139 const Entity* ent = st.get_entity_of(sub);
140 asym_loop.add_row({sub.subchain_id(), (ent ? ent->name : "?")});
141 }
142 if (!st.connections.empty()) {
143 items.emplace_back(cif::CommentArg{"#################\n"
144 "## STRUCT_CONN ##\n"
145 "#################"});
146 impl::write_struct_conn(st, block);
147 }
148
149 items.emplace_back(cif::CommentArg{"###############\n"
150 "## ATOM_SITE ##\n"
151 "###############"});
152 cif::Loop& atom_loop = block.init_mmcif_loop("_atom_site.", {
153 "group_PDB",
154 "id",
155 "label_atom_id",
156 "label_alt_id",
157 "label_comp_id",
158 "label_asym_id",
159 "auth_seq_id",
160 //"pdbx_PDB_ins_code",
161 "Cartn_x",
162 "Cartn_y",
163 "Cartn_z",
164 "occupancy",
165 "B_iso_or_equiv",
166 "type_symbol",
167 "calc_flag",
168 "label_seg_id",
169 "auth_atom_id",
170 "label_chem_id"});
171 bool write_anisou = has_anisou(model0);
172 if (write_anisou)
173 for (const char* idx : {"[1][1]", "[2][2]", "[3][3]",
174 "[1][2]", "[1][3]", "[2][3]"})
175 atom_loop.tags.push_back(std::string("_atom_site.aniso_U") + idx);
176 std::vector<std::string>& vv = atom_loop.values;
177 vv.reserve(count_atom_sites(st) * atom_loop.tags.size());
178
179
180 for (const Topo::ChainInfo& chain_info : topo.chain_infos)
181 for (const Topo::ResInfo& ri : chain_info.res_infos) {
182 const ChemComp& cc = ri.chemcomp;
183 const Residue& res = *ri.res;
184 std::string auth_seq_id = res.seqid.num.str();
185 //std::string ins_code(1, res.icode != ' ' ? res.icode : '?');
186 for (const Atom& a : res.atoms) {
187 vv.emplace_back("ATOM");
188 vv.emplace_back(std::to_string(a.serial));
189 vv.emplace_back(a.name);
190 vv.emplace_back(1, a.altloc ? a.altloc : '.');
191 vv.emplace_back(res.name);
192 vv.emplace_back(cif::quote(chain_info.chain_ref.name));
193 vv.emplace_back(auth_seq_id);
194 //vv.emplace_back(ins_code);
195 vv.emplace_back(to_str(a.pos.x));
196 vv.emplace_back(to_str(a.pos.y));
197 vv.emplace_back(to_str(a.pos.z));
198 vv.emplace_back(to_str(a.occ));
199 vv.emplace_back(to_str(a.b_iso));
200 vv.emplace_back(a.element.uname());
201 vv.emplace_back(refmac_calc_flag(a));
202 vv.emplace_back(1, '.'); // label_seg_id
203 vv.emplace_back(a.name); // again
204 vv.emplace_back(cc.get_atom(a.name).chem_type); // label_chem_id
205 if (write_anisou) {
206 if (a.aniso.nonzero()) {
207 for (float u : {a.aniso.u11, a.aniso.u22, a.aniso.u33,
208 a.aniso.u12, a.aniso.u13, a.aniso.u23})
209 vv.push_back(to_str(u));
210 } else {
211 vv.resize(vv.size() + 6, ".");
212 }
213 }
214 }
215 }
216 return block;
217}

219inline void add_restraint_row(cif::Loop& restr_loop,
220 const char* record, int counter,
221 const std::string& label, const std::string& period,
222 std::initializer_list<const Atom*> atoms,
223 double value, double dev,
224 double value_nucleus, double dev_nucleus,
225 double obs) {
226 // Ignore restraints with zero-occupancy atoms (NoZeroOccRestr)
227 // Perhaps it could be done earlier, in Topo::apply_restraints().
228 for (const Atom* a : atoms)
229 if (a->occ == 0)
230 return;
231
232 auto& values = restr_loop.values;
233 auto to_str_dot = [&](double x) { return std::isnan(x) ? "." : to_str_prec<3>(x); };
234 values.emplace_back(record); // record
235 values.emplace_back(std::to_string(counter)); // number
236 values.emplace_back(label); // label
237 values.emplace_back(period); // period
238 for (const Atom* a : atoms)
239 values.emplace_back(std::to_string(a->serial)); // atom_id_i
240 for (size_t i = atoms.size(); i < 4; ++i)
241 values.emplace_back(".");
242 values.emplace_back(to_str_dot(value)); // value
243 values.emplace_back(to_str_dot(dev)); // dev
244 values.emplace_back(to_str_dot(value_nucleus)); // value_nucleus
245 values.emplace_back(to_str_dot(dev_nucleus)); // dev_nucleus
246 values.emplace_back(to_str_prec<3>(obs)); // val_obs
247 std::string& last = values.back();
248 last += " #";
249 for (const Atom* a : atoms) {
250 last += ' ';
251 last += a->name;
252 }
253}

255inline void add_restraints(const Topo::Rule rule, const Topo& topo,
256 cif::Loop& restr_loop, int (&counters)[6],
257 const UnitCell* cell=nullptr) {
258 if (rule.rkind == Topo::RKind::Bond) {
259 const Topo::Bond& t = topo.bonds[rule.index];
260 if (cell == nullptr) { // don't use symmetry
261 add_restraint_row(restr_loop, "BOND", ++counters[0],
262 bond_type_to_string(t.restr->type), ".",
263 {t.atoms[0], t.atoms[1]},
264 t.restr->value, t.restr->esd,
265 t.restr->value_nucleus, t.restr->esd_nucleus,
266 t.calculate());
267 } else {
268 NearestImage im = cell->find_nearest_image(t.atoms[0]->pos, t.atoms[1]->pos,
269 Asu::Different);
270 add_restraint_row(restr_loop, "BNDS", ++counters[5],
271 im.symmetry_code(true), ".",
272 {t.atoms[0], t.atoms[1]},
273 t.restr->value, t.restr->esd,
274 t.restr->value_nucleus, t.restr->esd_nucleus,
275 std::sqrt(im.dist_sq));
276 }
277 } else if (rule.rkind == Topo::RKind::Angle) {
278 const Topo::Angle& t = topo.angles[rule.index];
279 add_restraint_row(restr_loop, "ANGL", ++counters[1], ".", ".",
280 {t.atoms[0], t.atoms[1], t.atoms[2]},
281 t.restr->value, t.restr->esd, NAN, NAN,
282 deg(t.calculate()));
283 } else if (rule.rkind == Topo::RKind::Torsion) {
284 const Topo::Torsion& t = topo.torsions[rule.index];
285 add_restraint_row(restr_loop, "TORS", ++counters[2],
286 t.restr->label, std::to_string(t.restr->period),
287 {t.atoms[0], t.atoms[1], t.atoms[2], t.atoms[3]},
288 t.restr->value, t.restr->esd, NAN, NAN,
289 deg(t.calculate()));
290 } else if (rule.rkind == Topo::RKind::Chirality) {
291 const Topo::Chirality& t = topo.chirs[rule.index];
292 add_restraint_row(restr_loop, "CHIR", ++counters[3],
293 chirality_to_string(t.restr->sign), ".",
294 {t.atoms[0], t.atoms[1], t.atoms[2], t.atoms[3]},
295 topo.ideal_chiral_abs_volume(t), 0.02, NAN, NAN,
296 t.calculate());
297 } else if (rule.rkind == Topo::RKind::Plane) {
298 const Topo::Plane& t = topo.planes[rule.index];
299 ++counters[4];
300 auto coeff = find_best_plane(t.atoms);
301 for (const Atom* atom : t.atoms)
302 add_restraint_row(restr_loop, "PLAN", counters[4], t.restr->label, ".",
303 {atom},
304 t.restr->esd, NAN, NAN, NAN,
305 get_distance_from_plane(atom->pos, coeff));
306 }
307}

309inline cif::Block prepare_rst(const Topo& topo, const MonLib& monlib, const UnitCell& cell) {
310 cif::Block block("restraints");
311 cif::Loop& restr_loop = block.init_mmcif_loop("_restr.", {
312 "record", "number", "label", "period",
313 "atom_id_1", "atom_id_2", "atom_id_3", "atom_id_4",
314 "value", "dev", "value_nucleus", "dev_nucleus", "val_obs"});
315 int counters[6] = {0, 0, 0, 0, 0, 0};
316 for (const Topo::ChainInfo& chain_info : topo.chain_infos) {
317 for (const Topo::ResInfo& ri : chain_info.res_infos) {
318 // write link
319 for (const Topo::Link& prev : ri.prev) {
320 const ChemLink* link = monlib.get_link(prev.link_id);
321 if (link && !prev.link_rules.empty()) {
322 std::string comment = " link " + prev.link_id + " " +
323 prev.res1->seqid.str() + " " +
324 prev.res1->name + " - " +
325 ri.res->seqid.str() + " " + ri.res->name;
326 restr_loop.add_comment_and_row({comment, "LINK", ".", cif::quote(prev.link_id), ".",
327 ".", ".", ".", ".", ".", ".", ".", ".", "."});
328 for (const Topo::Rule& rule : prev.link_rules)
329 add_restraints(rule, topo, restr_loop, counters);
330 }
331 }
332 // write monomer
333 if (!ri.monomer_rules.empty()) {
334 std::string res_info = " monomer " + chain_info.subchain_name + " " +
335 ri.res->seqid.str() + " " + ri.res->name;
336 if (!ri.mods.empty())
337 res_info += " modified by " + join_str(ri.mods, ", ");
338
339 // need to revisit it later on
340 std::string group = cif::quote(ri.chemcomp.group.substr(0, 8));
341 if (group == "peptide" || group == "P-peptid" || group == "M-peptid")
342 group = "L-peptid";
343 else if (group == "NON-POLY")
344 group = ".";
345
346 restr_loop.add_comment_and_row({res_info, "MONO", ".", group, ".",
347 ".", ".", ".", ".", ".", ".", ".", ".", "."});
348 for (const Topo::Rule& rule : ri.monomer_rules)
349 add_restraints(rule, topo, restr_loop, counters);
350 }
351 }
352 }
353 // explicit links
354 for (const Topo::Link& extra : topo.extras) {
355 if (extra.asu == Asu::Different) // symmetry links are left for later
356 continue;
357 const ChemLink* chem_link = monlib.get_link(extra.link_id);
358 assert(chem_link);
359 std::string comment = " link " + chem_link->id;
360 restr_loop.add_comment_and_row({comment, "LINK", ".", cif::quote(chem_link->id), ".",
361 ".", ".", ".", ".", ".", ".", ".", ".", "."});
362 for (const Topo::Rule& rule : extra.link_rules)
363 add_restraints(rule, topo, restr_loop, counters);
364 }
365
366 // special symmetry related links
367 for (const Topo::Link& extra : topo.extras) {
368 if (extra.asu != Asu::Different)
369 continue;
370 const ChemLink* chem_link = monlib.get_link(extra.link_id);
371 assert(chem_link);
372 std::string comment = " link (symmetry) " + chem_link->id;
373 restr_loop.add_comment_and_row({comment, "LINK", ".", "symmetry", ".",
374 ".", ".", ".", ".", ".", ".", ".", ".", "."});
375 for (const Topo::Rule& rule : extra.link_rules)
376 if (rule.rkind == Topo::RKind::Bond)
377 add_restraints(rule, topo, restr_loop, counters, &cell);
378 }
379
380 return block;
381}
